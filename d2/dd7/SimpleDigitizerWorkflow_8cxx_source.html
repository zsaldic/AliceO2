81 // customize the completion policy
82 void customize(std::vector<o2::framework::CompletionPolicy>& policies)
83 {
84  using o2::framework::CompletionPolicy;
85  // we customize the completion policy for the writer since it should stream immediately
86  auto matcher = [](DeviceSpec const& device) {
87  bool matched = device.name == "TPCDigitWriter";
88  if (matched) {
89  LOG(INFO) << "DPL completion policy for " << device.name << " customized";
90  }
91  return matched;
92  };
93 
94  auto policy = [](gsl::span<o2::framework::PartRef const> const& inputs) {
95  return CompletionPolicy::CompletionOp::Consume;
96  };
97 
98  policies.push_back({ CompletionPolicy{ "process-any", matcher, policy } });
99 }

140 // extract num TPC lanes, a lane is a streaming line of processors (digitizer-clusterizer-etc)
141 // by default this will be std::max(the number of physical cores, numberofsectors)
142 // as a temporary means to fully use a machine and as a way to play with different topologies
143 int getNumTPCLanes(std::vector<int> const& sectors, ConfigContext const& configcontext)
144 {
145  auto lanes = configcontext.options().get<int>("tpc-lanes");
146  if (lanes < 0) {
147  LOG(FATAL) << "tpc-lanes needs to be positive\n";
148  return 0;
149  }
150  // crosscheck with sectors
151  return std::min(lanes, (int)sectors.size());
152 }

156 void initTPC()
157 {
158  // We only want to do this for the DPL master
159  // I am not aware of an easy way to query if "I am DPL master" so
160  // using for the moment a mechanism defining/setting an environment variable
161  // with the parent ID and query inside forks if this environment variable exists
162  // (it assumes fundamentally that the master executes this function first)
163  std::stringstream streamthis;
164  std::stringstream streamparent;
165 
166  streamthis << "TPCGEMINIT_PID" << getpid();
167  streamparent << "TPCGEMINIT_PID" << getppid();
168  if (getenv(streamparent.str().c_str())) {
169  LOG(DEBUG) << "GEM ALREADY INITIALIZED ... SKIPPING HERE";
170  return;
171  }
172 
173  LOG(DEBUG) << "INITIALIZING TPC GEMAmplification";
174  setenv(streamthis.str().c_str(), "ON", 1);
175 
176  auto& cdb = o2::TPC::CDBInterface::instance();
177  cdb.setUseDefaults();
178  // by invoking this constructor we make sure that a common file will be created
179  // in future we should take this from OCDB and just forward per message
180  const static auto& ampl = o2::TPC::GEMAmplification::instance();
181 }

209 // Split a given string on a separator character
210 std::vector<std::string> splitString(std::string src, char sep)
211 {
212  std::vector<std::string> fields;
213  std::string token;
214  std::istringstream ss(src);
215 
216  while (std::getline(ss, token, sep)) {
217  if (!token.empty()) {
218  fields.push_back(token);
219  }
220  }
221 
222  return fields;
223 }

226 // Filters detectors based on a white/black list provided via the onlyDet/skipDet CLI args
227 struct DetFilterer {
228  // detlist: A character-separated list of detectors
229  // unsetVal: The value when the option is unset
230  // separator: The character that separates the list of detectors defined in option
231  // mustContain: The nature of this DetFilterer. If true, it is a white lister
232  // i.e. option defines the list of allowed detectors. If false
233  // it is a black lister i.e defines the list of disallowed detectors.
234  DetFilterer(std::string detlist, std::string unsetVal, char separator, bool doWhiteListing)
235  {
236  // option is not set, nothing to do
237  if (detlist.compare(unsetVal) == 0) {
238  return;
239  }
240 
241  std::vector<std::string> tokens = splitString(detlist, separator);
242 
243  // Convert a vector of strings to one of o2::detectors::DetID
244  for (auto token : tokens) {
245  ids.emplace_back(token.c_str());
246  }
247 
248  isWhiteLister = doWhiteListing;
249  }
250 
251  // isSet determines if a detector list was provided
252  // against which to filter
253  bool isSet()
254  {
255  return ids.size() > 0;
256  }
257 
258  // accept determines if a given detector should be accepted
259  bool accept(o2::detectors::DetID id)
260  {
261  bool found = std::find(ids.begin(), ids.end(), id) != ids.end();
262  return found == isWhiteLister;
263  }
264 
265  private:
266  std::vector<o2::detectors::DetID> ids;
267  bool isWhiteLister; // true = accept only detectors in the ids vector
268 };
269 
270 // Helper function to define a white listing DetFilterer
271 DetFilterer whitelister(std::string optionVal, std::string unsetValue, char separator)
272 {
273  return DetFilterer(optionVal, unsetValue, separator, true);
274 }
275 
276 // Helper function to define a black listing DetFilterer
277 DetFilterer blacklister(std::string optionVal, std::string unsetValue, char separator)
278 {
279  return DetFilterer(optionVal, unsetValue, separator, false);
280 }

286 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
287 {
288  // Reserve one entry which fill be filled with the SimReaderSpec
289  // at the end. This places the processor at the beginning of the
290  // workflow in the upper left corner of the GUI.
291  WorkflowSpec specs(1);
292 
293  // Update the (declared) parameters if changed from the command line
294  // Note: In the future this should be done only on a dedicated processor managing
295  // the parameters and then propagated automatically to all devices
296  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
297  // write the configuration used for the digitizer workflow
298  o2::conf::ConfigurableParam::writeINI("o2digitizerworkflow_configuration.ini");
299 
300  // First, read the GRP to detect which components need instantiations
301  // (for the moment this assumes the file o2sim_grp.root to be in the current directory)
302  const auto grp = readGRP();
303  if (!grp) {
304  return specs;
305  }
306 
307  // onlyDet takes precedence on skipDet
308  DetFilterer filterers[2] = {
309  whitelister(configcontext.options().get<std::string>("onlyDet"), "none", ','),
310  blacklister(configcontext.options().get<std::string>("skipDet"), "none", ',')
311  };
312 
313  auto accept = [&configcontext, &filterers](o2::detectors::DetID id) {
314  for (auto& f : filterers) {
315  if (f.isSet()) {
316  return f.accept(id);
317  }
318  }
319 
320  // accept all if neither onlyDet/skipDet are provided
321  return true;
322  };
323 
324  // lambda to extract detectors which are enabled in the workflow
325  // will complain if user gave wrong input in construction of DetID
326  auto isEnabled = [&configcontext, &filterers, accept, grp](o2::detectors::DetID id) {
327  auto accepted = accept(id);
328  bool is_ingrp = grp->isDetReadOut(id);
329  LOG(INFO) << id.getName()
330  << " is in grp? " << (is_ingrp ? "yes" : "no") << ";"
331  << " is skipped? " << (!accepted ? "yes" : "no");
332  return accepted && is_ingrp;
333  };
334 
335  std::vector<o2::detectors::DetID> detList; // list of participating detectors
336  int fanoutsize = 0;
337  if (wantCollisionTimePrinter()) {
338  specs.emplace_back(o2::steer::getCollisionTimePrinter(fanoutsize++));
339  }
340 
341  // the TPC part
342  // we need to init this anyway since TPC is treated a bit special (for the moment)
343  initTPC();
344  // keeps track of which subchannels correspond to tpc channels
345  auto tpclanes = std::make_shared<std::vector<int>>();
346  // keeps track of which tpc sectors to process
347  std::vector<int> tpcsectors;
348 
349  if (isEnabled(o2::detectors::DetID::TPC)) {
350  tpcsectors = o2::RangeTokenizer::tokenize<int>(configcontext.options().get<std::string>("tpc-sectors"));
351  auto lanes = getNumTPCLanes(tpcsectors, configcontext);
352  detList.emplace_back(o2::detectors::DetID::TPC);
353 
354  for (int l = 0; l < lanes; ++l) {
355  specs.emplace_back(o2::TPC::getTPCDigitizerSpec(fanoutsize, (l == 0)));
356  tpclanes->emplace_back(fanoutsize); // this records that TPC is "listening under this subchannel"
357  fanoutsize++;
358  }
359 
360  auto tpcRecoOutputType = configcontext.options().get<std::string>("tpc-reco-type");
361  if (tpcRecoOutputType.empty()) {
362  // for writing digits to disc
363  specs.emplace_back(o2::TPC::getTPCDigitRootWriterSpec(lanes));
364  } else {
365  // attach the TPC reco workflow
366  auto tpcRecoWorkflow = o2::TPC::RecoWorkflow::getWorkflow(tpcsectors, true, lanes, "digitizer", tpcRecoOutputType.c_str());
367  specs.insert(specs.end(), tpcRecoWorkflow.begin(), tpcRecoWorkflow.end());
368  }
369  }
370 
371  // the ITS part
372  if (isEnabled(o2::detectors::DetID::ITS)) {
373  detList.emplace_back(o2::detectors::DetID::ITS);
374  // connect the ITS digitization
375  specs.emplace_back(o2::ITSMFT::getITSDigitizerSpec(fanoutsize++));
376  // connect ITS digit writer
377  specs.emplace_back(o2::ITSMFT::getITSDigitWriterSpec());
378  }
379 
380  // the MFT part
381  if (isEnabled(o2::detectors::DetID::MFT)) {
382  detList.emplace_back(o2::detectors::DetID::MFT);
383  // connect the MFT digitization
384  specs.emplace_back(o2::ITSMFT::getMFTDigitizerSpec(fanoutsize++));
385  // connect MFT digit writer
386  specs.emplace_back(o2::ITSMFT::getMFTDigitWriterSpec());
387  }
388 
389  // the TOF part
390  if (isEnabled(o2::detectors::DetID::TOF)) {
391  detList.emplace_back(o2::detectors::DetID::TOF);
392  // connect the TOF digitization
393  specs.emplace_back(o2::tof::getTOFDigitizerSpec(fanoutsize++));
394  // add TOF digit writer
395  specs.emplace_back(o2::tof::getTOFDigitWriterSpec());
396  // add TOF clusterer
397  specs.emplace_back(o2::tof::getTOFClusterizerSpec());
398  // add TOF cluster writer
399  specs.emplace_back(o2::tof::getTOFClusterWriterSpec());
400  }
401 
402  // the FIT part
403  if (isEnabled(o2::detectors::DetID::FIT)) {
404  detList.emplace_back(o2::detectors::DetID::FIT);
405  // connect the FIT digitization
406  specs.emplace_back(o2::fit::getFITDigitizerSpec(fanoutsize++));
407  // connect the FIT digit writer
408  specs.emplace_back(o2::fit::getFITDigitWriterSpec());
409  }
410 
411  // the EMCal part
412  if (isEnabled(o2::detectors::DetID::EMC)) {
413  detList.emplace_back(o2::detectors::DetID::EMC);
414  // connect the EMCal digitization
415  specs.emplace_back(o2::emcal::getEMCALDigitizerSpec(fanoutsize++));
416  // connect the EMCal digit writer
417  specs.emplace_back(o2::emcal::getEMCALDigitWriterSpec());
418  }
419 
420  // add HMPID
421  if (isEnabled(o2::detectors::DetID::HMP)) {
422  detList.emplace_back(o2::detectors::DetID::HMP);
423  // connect the HMP digitization
424  specs.emplace_back(o2::hmpid::getHMPIDDigitizerSpec(fanoutsize++));
425  // connect the HMP digit writer
426  specs.emplace_back(o2::hmpid::getHMPIDDigitWriterSpec());
427  }
428 
429  // add TRD
430  if (isEnabled(o2::detectors::DetID::TRD)) {
431  detList.emplace_back(o2::detectors::DetID::TRD);
432  // connect the TRD digitization
433  specs.emplace_back(o2::trd::getTRDDigitizerSpec(fanoutsize++));
434  // connect the TRD digit writer
435  specs.emplace_back(o2::trd::getTRDDigitWriterSpec());
436  }
437 
438  //add MUON MCH
439  if (isEnabled(o2::detectors::DetID::MCH)) {
440  detList.emplace_back(o2::detectors::DetID::MCH);
441  //connect the MUON MCH digitization
442  specs.emplace_back(o2::mch::getMCHDigitizerSpec(fanoutsize++));
443  //connect the MUON MCH digit writer
444  specs.emplace_back(o2::mch::getMCHDigitWriterSpec());
445  }
446 
447  // add MID
448  if (isEnabled(o2::detectors::DetID::MID)) {
449  detList.emplace_back(o2::detectors::DetID::MID);
450  // connect the MID digitization
451  specs.emplace_back(o2::mid::getMIDDigitizerSpec(fanoutsize++));
452  // connect the MID digit writer
453  specs.emplace_back(o2::mid::getMIDDigitWriterSpec());
454  }
455 
456  // GRP updater: must come after all detectors since requires their list
457  specs.emplace_back(o2::parameters::getGRPUpdaterSpec(detList));
458 
459  // The SIM Reader. NEEDS TO BE LAST
460  specs[0] = o2::steer::getSimReaderSpec(fanoutsize, tpcsectors, tpclanes);
461 
462  return specs;
463 }
