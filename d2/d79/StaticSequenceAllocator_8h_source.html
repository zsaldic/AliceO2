67  constexpr size_t insert(BufferT* buffer) const noexcept {
68  static_assert(sizeof(BufferT) == 1, "buffer required to be of byte-type");
69  size_t length = 0;
70  memcpy(buffer + length, &header, sizeof(HeaderType));
71  length += sizeof(HeaderType);
72  memcpy(buffer + length, data, dataLength);
73  length += dataLength;
74  if (trailerLength > 0) {
75  memcpy(buffer + length, &trailer, trailerLength);
76  length += trailerLength;
77  }
