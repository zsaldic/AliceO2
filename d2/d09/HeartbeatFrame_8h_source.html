48 struct HeartbeatHeader
49 {
50  union {
51  // the complete 64 bit header word, initialize with blockType 1 and size 1
52  uint64_t headerWord = 0x1100000000000000;
53  struct {
54  // bit 0 to 31: orbit number
55  uint32_t orbit;
56  // bit 32 to 43: bunch crossing id
57  uint16_t bcid:12;
58  // bit 44 to 47: reserved
59  uint16_t reserved:4;
60  // bit 48 to 51: trigger type
61  uint8_t triggerType:4;
62  // bit 52 to 55: reserved
63  uint8_t reservedTriggerType:4;
64  // bit 56 to 59: header length
65  uint8_t headerLength:4;
66  // bit 60 to 63: block type (=1 HBF/trigger Header)
67  uint8_t blockType:4;
68  };
69  }; // end union
70 
71  operator bool() const {return headerWord != 0 && blockType == 0x1;}
72 
73  bool operator<(const HeartbeatHeader& other) const {
74  return this->orbit < other.orbit;
75  }
76 
77  operator uint64_t() const {return headerWord;}
78 };
79 static_assert(sizeof(HeartbeatHeader) == 8, "Heartbeat header must be 64bit");

81 struct HeartbeatTrailer
82 {
83  union {
84  // the complete 64 bit trailer word, initialize with blockType 5 and size 1
85  uint64_t trailerWord = 0x5100000000000000;
86  struct {
87  // bit 0 to 31: data length in words
88  uint32_t dataLength;
89  // bit 32 to 52: detector specific status words
90  uint32_t status:21;
91  // bit 53: =1 in case a new physics trigger arrived within read-out period
92  uint16_t hbfTruncated:1;
93  // bit 54: =0 HBF correctly transmitted
94  uint16_t hbfStatus:1;
95  // bit 55: =1 HBa/0 HBr received
96  uint16_t hbAccept:1;
97  // bit 56 to 59: trailer length
98  uint16_t trailerLength:4;
99  // bit 60 to 63: block type (=5 HBF Trailer)
100  uint8_t blockType:4;
101  };
102  }; // end union
103 
104  operator bool() const {return trailerWord != 0 && blockType == 0x5;}
105 
106  operator uint64_t() const {return trailerWord;}
107 };
108 static_assert(sizeof(HeartbeatTrailer) == 8, "Heartbeat trailer must be 64bit");
