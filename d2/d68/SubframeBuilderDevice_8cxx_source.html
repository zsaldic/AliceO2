40 void o2::DataFlow::SubframeBuilderDevice::InitTask()
41 {
42  mOrbitDuration = GetConfig()->GetValue<uint32_t>(OptionKeyOrbitDuration);
43  mOrbitsPerTimeframe = GetConfig()->GetValue<uint32_t>(OptionKeyOrbitsPerTimeframe);
44  mInputChannelName = GetConfig()->GetValue<std::string>(OptionKeyInputChannelName);
45  mOutputChannelName = GetConfig()->GetValue<std::string>(OptionKeyOutputChannelName);
46  mFLPId= GetConfig()->GetValue<size_t>(OptionKeyFLPId);
47  mStripHBF= GetConfig()->GetValue<bool>(OptionKeyStripHBF);
48 
49  LOG(INFO) << "Obtaining data from DataPublisher\n";
50  // Now that we have all the information lets create the policies to do the
51  // payload extraction and merging and create the actual PayloadMerger.
52 
53  // We extract the timeframeId from the number of orbits.
54  // FIXME: handle multiple socket ids
55  Merger::IdExtractor makeId = [this](std::unique_ptr<FairMQMessage> &msg) -> SubframeId {
56  HeartbeatHeader *hbh = reinterpret_cast<HeartbeatHeader*>(msg->GetData());
57  SubframeId id = {.timeframeId = hbh->orbit / this->mOrbitsPerTimeframe,
58  .socketId = 0 };
59  return id;
60  };
61 
62  // We extract the payload differently depending on wether we want to strip
63  // the header or not.
64  Merger::PayloadExtractor payloadExtractor = [this](char **out, char *in, size_t inSize) -> size_t {
65  if (!this->mStripHBF) {
66  return Merger::fullPayloadExtractor(out, in, inSize);
67  }
68  return o2::dataflow::extractDetectorPayloadStrip(out, in, inSize);
69  };
70 
71  // Whether a given timeframe is complete depends on how many orbits per
72  // timeframe we want.
73  Merger::MergeCompletionCheker checkIfComplete =
74  [this](Merger::MergeableId id, Merger::MessageMap &map) {
75  return map.count(id) < this->mOrbitsPerTimeframe;
76  };
77 
78  mMerger.reset(new Merger(makeId, checkIfComplete, payloadExtractor));
79  OnData(mInputChannelName.c_str(), &o2::DataFlow::SubframeBuilderDevice::HandleData);
80 }

82 bool o2::DataFlow::SubframeBuilderDevice::BuildAndSendFrame(FairMQParts &inParts)
83 {
84  auto id = mMerger->aggregate(inParts.At(1));
85 
86  char **outBuffer;
87  size_t outSize = mMerger->finalise(outBuffer, id);
88  // In this case we do not have enough subtimeframes for id,
89  // so we simply return.
90  if (outSize == 0)
91  return true;
92  // If we reach here, it means we do have enough subtimeframes.
93 
94  // top level subframe header, the DataHeader is going to be used with
95  // description "SUBTIMEFRAMEMD"
96  // this should be defined in a common place, and also the origin
97  // the origin can probably name a detector identifier, but not sure if
98  // all CRUs of a FLP in all cases serve a single detector
99  o2::header::DataHeader dh;
100  dh.dataDescription = o2::header::DataDescription("SUBTIMEFRAMEMD");
101  dh.dataOrigin = o2::header::DataOrigin("FLP");
102  dh.subSpecification = mFLPId;
103  dh.payloadSize = sizeof(SubframeMetadata);
104 
105  DataHeader payloadheader(*o2::header::get<DataHeader*>((byte*)inParts.At(0)->GetData()));
106 
107  // subframe meta information as payload
108  SubframeMetadata md;
109  // id is really the first orbit in the timeframe.
110  md.startTime = id.timeframeId * mOrbitsPerTimeframe * static_cast<uint64_t>(mOrbitDuration);
111  md.duration = mOrbitDuration*mOrbitsPerTimeframe;
112  LOG(INFO) << "Start time for subframe (" << md.startTime << ", "
113  << md.duration
114  << ")";
115 
116  // Add the metadata about the merged subtimeframes
117  // FIXME: do we really need this?
118  O2Message outgoing;
119  o2::Base::addDataBlock(outgoing, dh, NewSimpleMessage(md));
120 
121  // Add the actual merged payload.
122  o2::Base::addDataBlock(outgoing, payloadheader,
123  NewMessage(*outBuffer, outSize,
124  [](void* data, void* hint) { delete[] reinterpret_cast<char*>(hint); }, *outBuffer));
125  // send message
126  Send(outgoing, mOutputChannelName.c_str());
127  // FIXME: do we actually need this? outgoing should go out of scope
128  outgoing.fParts.clear();
129 
130  return true;
131 }

133 bool o2::DataFlow::SubframeBuilderDevice::HandleData(FairMQParts& msgParts, int /*index*/)
134 {
135  // loop over header payload pairs in the incoming multimessage
136  // for each pair
137  // - check timestamp
138  // - create new subtimeframe if none existing where the timestamp of the data fits
139  // - add pair to the corresponding subtimeframe
140 
141  // check for completed subtimeframes and send all completed frames
142  // the builder does not implement the routing to the EPN, this is done in the
143  // specific FLP-EPN setup
144  // to fit into the simple emulation of event/frame ids in the flpSender the order of
145  // subtimeframes needs to be preserved
146  BuildAndSendFrame(msgParts);
147  return true;
148 }
