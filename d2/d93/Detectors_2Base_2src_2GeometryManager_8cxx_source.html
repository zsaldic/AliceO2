50 Bool_t GeometryManager::getOriginalMatrix(const char* symname, TGeoHMatrix& m)
51 {
52  m.Clear();
53 
54  if (!gGeoManager || !gGeoManager->IsClosed()) {
55  LOG(ERROR) << "No active geometry or geometry not yet closed!" << FairLogger::endl;
56  return kFALSE;
57  }
58 
59  if (!gGeoManager->GetListOfPhysicalNodes()) {
60  LOG(WARNING) << "gGeoManager doesn't contain any aligned nodes!" << FairLogger::endl;
61 
62  if (!gGeoManager->cd(symname)) {
63  LOG(ERROR) << "Volume path " << symname << " not valid!" << FairLogger::endl;
64  return kFALSE;
65  } else {
66  m = *gGeoManager->GetCurrentMatrix();
67  return kTRUE;
68  }
69  }
70

72  const char* path = nullptr;
73 
74  if (pne) {
75  m = *pne->GetGlobalOrig();
76  return kTRUE;
77  } else {
78  LOG(WARNING) << "The symbolic volume name " << symname
79  << "does not correspond to a physical entry. Using it as a volume path!" << FairLogger::endl;
80  path = symname;
81  }
82

87 Bool_t GeometryManager::getOriginalMatrixFromPath(const char* path, TGeoHMatrix& m)
88 {
89  m.Clear();
90 
91  if (!gGeoManager || !gGeoManager->IsClosed()) {
92  LOG(ERROR) << "Can't get the original global matrix! gGeoManager doesn't exist or it is still opened!"
93  << FairLogger::endl;
94  return kFALSE;
95  }
96 
97  if (!gGeoManager->CheckPath(path)) {
98  LOG(ERROR) << "Volume path " << path << " not valid!" << FairLogger::endl;
99  return kFALSE;
100  }
101 
102  TIter next(gGeoManager->GetListOfPhysicalNodes());
103  gGeoManager->cd(path);
104 
105  while (gGeoManager->GetLevel()) {
106  TGeoPhysicalNode* physNode = nullptr;
107  next.Reset();
108  TGeoNode* node = gGeoManager->GetCurrentNode();
109 
110  while ((physNode = (TGeoPhysicalNode*)next())) {
111  if (physNode->GetNode() == node) {
112  break;
113  }
114  }
115 
116  TGeoMatrix* lm = nullptr;
117  if (physNode) {
118  lm = physNode->GetOriginalMatrix();
119  if (!lm) {
120  lm = node->GetMatrix();
121  }
122  } else {
123  lm = node->GetMatrix();
124  }
125 
126  m.MultiplyLeft(lm);
127 
128  gGeoManager->CdUp();
129  }
130  return kTRUE;
131 }

243 void GeometryManager::MatBudget::accountMaterial(const TGeoMaterial* material)
244 {
245  meanRho = material->GetDensity();
246  meanX2X0 = material->GetRadLen();
247  meanA = material->GetA();
248  meanZ = material->GetZ();
249  if (material->IsMixture()) {
250  TGeoMixture* mixture = (TGeoMixture*)material;
251  Double_t norm = 0.;
252  meanZ2A = 0.;
253  for (Int_t iel = 0; iel < mixture->GetNelements(); iel++) {
254  norm += mixture->GetWmixt()[iel];
255  meanZ2A += mixture->GetZmixt()[iel] * mixture->GetWmixt()[iel] / mixture->GetAmixt()[iel];
256  }
257  meanZ2A /= norm;
258  } else {
259  meanZ2A = meanZ / meanA;
260  }
261 }

264 GeometryManager::MatBudget GeometryManager::MeanMaterialBudget(float x0, float y0, float z0, float x1, float y1,
265  float z1)
266 {
267  //
268  // Calculate mean material budget and material properties between
269  // the points "0" and "1".
270  //
271  // "mparam" - parameters used for the energy and multiple scattering
272  // corrections:
273  //
274  // see MatBudget data members for provided information
275  //
276  // Origin: Marian Ivanov, Marian.Ivanov@cern.ch
277  //
278  // Corrections and improvements by
279  // Andrea Dainese, Andrea.Dainese@lnl.infn.it,
280  // Andrei Gheata, Andrei.Gheata@cern.ch
281  //
282  // Ported to O2: ruben.shahoyan@cern.ch
283  //
284 
285  double length, startD[3] = { x0, y0, z0 };
286  double dir[3] = { x1 - x0, y1 - y0, z1 - z0 };
287  if ((length = dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]) <
288  TGeoShape::Tolerance() * TGeoShape::Tolerance()) {
289  return MatBudget(); // return empty struct
290  }
291  length = TMath::Sqrt(length);
292  double invlen = 1. / length;
293  for (int i = 3; i--;) {
294  dir[i] *= invlen;
295  }
296 
297  // Initialize start point and direction
298  TGeoNode* currentnode = gGeoManager->InitTrack(startD, dir);
299  if (!currentnode) {
300  LOG(ERROR) << "start point out of geometry: " << x0 << ':' << y0 << ':' << z0 << FairLogger::endl;
301  return MatBudget(); // return empty struct
302  }
303 
304  MatBudget budTotal, budStep;
305  budStep.accountMaterial(currentnode->GetVolume()->GetMedium()->GetMaterial());
306  budStep.length = length;
307 
308  // Locate next boundary within length without computing safety.
309  // Propagate either with length (if no boundary found) or just cross boundary
310  gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
311  Double_t stepTot = 0.0; // Step made
312  Double_t step = gGeoManager->GetStep();
313  // If no boundary within proposed length, return current step data
314  if (!gGeoManager->IsOnBoundary()) {
315  budStep.meanX2X0 = budStep.length / budStep.meanX2X0;
316  return MatBudget(budStep);
317  }
318  // Try to cross the boundary and see what is next
319  Int_t nzero = 0;
320  while (length > TGeoShape::Tolerance()) {
321  if (step < 2. * TGeoShape::Tolerance()) {
322  nzero++;
323  } else {
324  nzero = 0;
325  }
326  if (nzero > 3) {
327  // This means navigation has problems on one boundary
328  // Try to cross by making a small step
329  const double* curPos = gGeoManager->GetCurrentPoint();
330  LOG(ERROR) << "Cannot cross boundary at (" << curPos[0] << ',' << curPos[1] << ',' << curPos[2] << ')'
331  << FairLogger::endl;
332  budTotal.normalize(stepTot);
333  budTotal.nCross = -1; // flag failed navigation
334  return MatBudget(budTotal);
335  }
336  stepTot += step;
337 
338  budTotal.meanRho += step * budStep.meanRho;
339  budTotal.meanX2X0 += step / budStep.meanX2X0;
340  budTotal.meanA += step * budStep.meanA;
341  budTotal.meanZ += step * budStep.meanZ;
342  budTotal.meanZ2A += step * budStep.meanZ2A;
343  budTotal.nCross++;
344 
345  if (step >= length) {
346  break;
347  }
348  currentnode = gGeoManager->GetCurrentNode();
349  if (!currentnode) {
350  break;
351  }
352  length -= step;
353  budStep.accountMaterial(currentnode->GetVolume()->GetMedium()->GetMaterial());
354  gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
355  step = gGeoManager->GetStep();
356  }
357  budTotal.normalize(stepTot);
358  return MatBudget(budTotal);
359 }
