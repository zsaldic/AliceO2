47 void Clusterer::MakeClusters(const std::vector<Digit>* digits, std::vector<Cluster>* clusters)
48 {
49  // A cluster is defined as a list of neighbour digits
50  const double kClusteringThreshold = 0.050; // TODO: To be read from RecoParam
51  const double kDigitMinEnergy = 0.010; // TODO: to be implemented as a digit energy cut
52 
53  // Mark all digits as unused yet
54  const int maxNDigits = 3584; // There is no clusters larger than PHOS module ;)
55  int nDigits = digits->size();
56  bool digitsUsed[maxNDigits];
57 
58  for (int i = 0; i < nDigits; i++) {
59  digitsUsed[i] = false;
60  }
61  int iFirst = 0; // first index of digit which potentially can be a part of cluster
62  // e.g. first digit in this module, first CPV digit etc.
63 
64  for (int i = 0; i < nDigits; i++) {
65  if (digitsUsed[i])
66  continue;
67 
68  const Digit* digitSeed = &(digits->at(i));
69 
70  // is this digit so energetic that start cluster?
71  Cluster* clu = nullptr;
72  int iDigitInCluster = 0;
73  if (digitSeed->getAmplitude() > kClusteringThreshold) {
74  // start a new EMC RecPoint
75  Cluster cluTmp(digitSeed->getAbsId(), digitSeed->getAmplitude(), digitSeed->getTime());
76  clusters->push_back(cluTmp);
77  clu = &(clusters->back());
78 
79  digitsUsed[i] = true;
80  iDigitInCluster = 1;
81  } else {
82  continue;
83  }
84 
85  // Now scan remaining digits in list to find neigbours of our seed
86  int index = 0;
87  while (index < iDigitInCluster) { // scan over digits already in cluster
88  int digitSeedAbsId = clu->GetDigitAbsId(index);
89  index++;
90  for (Int_t j = iFirst; j < nDigits; j++) {
91  if (digitsUsed[j])
92  continue; // look through remaining digits
93  const Digit* digitN = &(digits->at(j));
94  ;
95  // call (digit,digitN) in THAT oder !!!!!
96  Int_t ineb = mPHOSGeom->AreNeighbours(digitSeedAbsId, digitN->getAbsId());
97  switch (ineb) {
98  case -1: // too early (e.g. previous module), do not look before j at subsequent passes
99  iFirst = j;
100  break;
101  case 0: // not a neighbour
102  break;
103  case 1: // are neighbours
104  clu->AddDigit(digitN->getAbsId(), digitN->getAmplitude(), digitN->getTime());
105  iDigitInCluster++;
106  digitsUsed[j] = true;
107  break;
108  case 2: // too far from each other
109  default:
110  break;
111  } // switch
112  }
113  } // loop over cluster
114  } // energy theshold
115 }
