51  reset();
52  mEvent = event;
53  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {

55  const size_t clustersNum{ currentLayer.size() };
56  if (clustersNum > 0) {
57  mClusters[iLayer].clear();
58  if (clustersNum > mClusters[iLayer].capacity()) {
59  mClusters[iLayer].reserve(clustersNum);
60  }
61  for (auto iCluster{ 0 }; iCluster < clustersNum; ++iCluster) {
62  mClusters[iLayer].emplace_back(iLayer, currentLayer.at(iCluster));
63  mAverageClustersRadii[iLayer] += mClusters[iLayer].back().rCoordinate;
64  }
65  mAverageClustersRadii[iLayer] *= 1.f / clustersNum;
66 
67  std::sort(mClusters[iLayer].begin(), mClusters[iLayer].end(), [](Cluster& cluster1, Cluster& cluster2) {
68  return cluster1.indexTableBinIndex < cluster2.indexTableBinIndex;
69  });
70  int previousBinIndex{ 0 };
71  mIndexTables[iLayer][0] = 0;
72  for (int iCluster{ 0 }; iCluster < clustersNum; ++iCluster) {
73  const int currentBinIndex{ mClusters[iLayer][iCluster].indexTableBinIndex };
74  if (currentBinIndex > previousBinIndex) {
75  for (int iBin{ previousBinIndex + 1 }; iBin <= currentBinIndex; ++iBin) {
76  mIndexTables[iLayer][iBin] = iCluster;
77  }
78  previousBinIndex = currentBinIndex;
79  }
80  }
81  for (int iBin{ previousBinIndex + 1 }; iBin <= ZBins * PhiBins; iBin++) {
82  mIndexTables[iLayer][iBin] = static_cast<int>(clustersNum);

87  // }
88  }

90  mDeltaRadii21 = mAverageClustersRadii[2] - mAverageClustersRadii[1];
91  mMaxDirectorCosine3 =

142  iCluster0 < std::get<0>(clusters0[iRow0]) + std::get<1>(clusters0[iRow0]); ++iCluster0) {
143  if (usedCluster0Flags[iCluster0])
144  continue;

149  (mAverageClustersRadii[2] - mClusters[0][iCluster0].rCoordinate) *
150  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
151  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate) };
152  float ZProjectionInner{ mClusters[0][iCluster0].zCoordinate +
153  (mClusters[0][iCluster0].rCoordinate) *
154  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
155  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate) };
156  if (std::abs(ZProjection) > (LayersZCoordinate()[0] + mVrtParams.mZCut))
157  continue;
158  int ZProjectionBin{ (ZProjection < -LayersZCoordinate()[0])
159  ? 0
160  : (ZProjection > LayersZCoordinate()[0]) ? ZBins - 1
161  : getZBinIndex(2, ZProjection) };
162  int ZBinLow2{ (ZProjectionBin - mVrtParams.mZSpan < 0) ? 0 : ZProjectionBin - mVrtParams.mZSpan };
163  int ZBinHigh2{ (ZProjectionBin + mVrtParams.mZSpan > ZBins - 1) ? ZBins - 1 : ZProjectionBin + mVrtParams.mZSpan };

170  clusters2 =

172  for (int iRow2{ 0 }; iRow2 < clusters2.size(); ++iRow2) {
173  for (int iCluster2{ std::get<0>(clusters2[iRow2]) };
174  iCluster2 < std::get<0>(clusters2[iRow2]) + std::get<1>(clusters2[iRow2]); ++iCluster2) {
175  if (usedCluster2Flags[iCluster2])
176  continue;
177  float ZProjectionRefined{
178  mClusters[0][iCluster0].zCoordinate +
179  (mClusters[2][iCluster2].rCoordinate - mClusters[0][iCluster0].rCoordinate) *
180  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
181  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate)

221 void VertexerTraits::computeVertices()
222 {
223  const int numTracklets{ static_cast<int>(mTracklets.size()) };
224  std::vector<bool> usedTracklets{};
225  usedTracklets.resize(mTracklets.size(), false);
226  for (int tracklet1{ 0 }; tracklet1 < numTracklets; ++tracklet1) {
227  if (usedTracklets[tracklet1])
228  continue;
229  for (int tracklet2{ tracklet1 + 1 }; tracklet2 < numTracklets; ++tracklet2) {
230  if (usedTracklets[tracklet2])
231  continue;
232  if (Line::getDCA(mTracklets[tracklet1], mTracklets[tracklet2]) <= mVrtParams.mPairCut) {
233  mTrackletClusters.emplace_back(tracklet1, mTracklets[tracklet1], tracklet2, mTracklets[tracklet2]);
234  std::array<float, 3> tmpVertex{ mTrackletClusters.back().getVertex() };
235  if (tmpVertex[0] * tmpVertex[0] + tmpVertex[1] * tmpVertex[1] > 4.f) {
236  mTrackletClusters.pop_back();
237  break;
238  }
239  usedTracklets[tracklet1] = true;
240  usedTracklets[tracklet2] = true;
241  for (int tracklet3{ 0 }; tracklet3 < numTracklets; ++tracklet3) {
242  if (usedTracklets[tracklet3])
243  continue;
244  if (Line::getDistanceFromPoint(mTracklets[tracklet3], tmpVertex) < mVrtParams.mPairCut) {
245  mTrackletClusters.back().add(tracklet3, mTracklets[tracklet3]);
246  usedTracklets[tracklet3] = true;
247  tmpVertex = mTrackletClusters.back().getVertex();
248  }
249  }
250  break;
251  }
252  }
253  }
254  std::sort(mTrackletClusters.begin(), mTrackletClusters.end(),
255  [](ClusterLines& cluster1, ClusterLines& cluster2) { return cluster1.getSize() > cluster2.getSize(); });
256  int noClusters{ static_cast<int>(mTrackletClusters.size()) };
257  for (int iCluster1{ 0 }; iCluster1 < noClusters; ++iCluster1) {
258  std::array<float, 3> vertex1{ mTrackletClusters[iCluster1].getVertex() };
259  std::array<float, 3> vertex2{};
260  for (int iCluster2{ iCluster1 + 1 }; iCluster2 < noClusters; ++iCluster2) {
261  vertex2 = mTrackletClusters[iCluster2].getVertex();
262  if (std::abs(vertex1[2] - vertex2[2]) < mVrtParams.mClusterCut) {
263 
264  float distance{ (vertex1[0] - vertex2[0]) * (vertex1[0] - vertex2[0]) +
265  (vertex1[1] - vertex2[1]) * (vertex1[1] - vertex2[1]) +
266  (vertex1[2] - vertex2[2]) * (vertex1[2] - vertex2[2]) };
267  if (distance <= mVrtParams.mPairCut * mVrtParams.mPairCut) {
268  for (auto label : mTrackletClusters[iCluster2].getLabels()) {
269  mTrackletClusters[iCluster1].add(label, mTracklets[label]);
270  vertex1 = mTrackletClusters[iCluster1].getVertex();
271  }
272  }
273  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster2);
274  --iCluster2;
275  --noClusters;
276  }
277  }
278  }
279  for (int iCluster{ 0 }; iCluster < noClusters; ++iCluster) {
280  if (mTrackletClusters[iCluster].getSize() < mVrtParams.mClusterContributorsCut && noClusters > 1) {
281  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster);
282  noClusters--;
283  continue;
284  }
285  float dist{ 0. };
286  for (auto& line : mTrackletClusters[iCluster].mLines) {
287  dist += Line::getDistanceFromPoint(line, mTrackletClusters[iCluster].getVertex()) /
288  mTrackletClusters[iCluster].getSize();
289  }
290  if (mTrackletClusters[iCluster].getVertex()[0] * mTrackletClusters[iCluster].getVertex()[0] +
291  mTrackletClusters[iCluster].getVertex()[1] * mTrackletClusters[iCluster].getVertex()[1] <
292  1.98 * 1.98) {
293  mVertices.emplace_back(mTrackletClusters[iCluster].getVertex()[0],
294  mTrackletClusters[iCluster].getVertex()[1],
295  mTrackletClusters[iCluster].getVertex()[2],
296  mTrackletClusters[iCluster].getRMS2(), // Symm matrix. Diagonal: RMS2 components,
297  // off-diagonal: square mean of projections on planes.
298  mTrackletClusters[iCluster].getSize(), // Contributors
299  mTrackletClusters[iCluster].getAvgDistance2(), // In place of chi2
300  mEvent->getROFrameId());
301  mEvent->addPrimaryVertex(mVertices.back().mX, mVertices.back().mY, mVertices.back().mZ);
302  }
303  }
