97 void TrackExtrap::linearExtrapToZCov(TrackParam* trackParam, double zEnd, bool updatePropagator)
98 {
101 
102  if (trackParam->getZ() == zEnd) {
103  return; // nothing to be done if same z
104  }
105 
106  // No need to propagate the covariance matrix if it does not exist
107  if (!trackParam->hasCovariances()) {
108  LOG(WARNING) << "Covariance matrix does not exist";
109  // Extrapolate linearly track parameters to "zEnd"
110  linearExtrapToZ(trackParam, zEnd);
111  return;
112  }
113 
114  // Compute track parameters
115  double dZ = zEnd - trackParam->getZ();
116  trackParam->setNonBendingCoor(trackParam->getNonBendingCoor() + trackParam->getNonBendingSlope() * dZ);
117  trackParam->setBendingCoor(trackParam->getBendingCoor() + trackParam->getBendingSlope() * dZ);
118  trackParam->setZ(zEnd);
119 
120  // Calculate the jacobian related to the track parameters linear extrapolation to "zEnd"
121  TMatrixD jacob(5, 5);
122  jacob.UnitMatrix();
123  jacob(0, 1) = dZ;
124  jacob(2, 3) = dZ;
125 
126  // Extrapolate track parameter covariances to "zEnd"
127  TMatrixD tmp(trackParam->getCovariances(), TMatrixD::kMultTranspose, jacob);
128  TMatrixD tmp2(jacob, TMatrixD::kMult, tmp);
129  trackParam->setCovariances(tmp2);
130 
131  // Update the propagator if required
132  if (updatePropagator) {
133  trackParam->updatePropagator(jacob);
134  }
135 }

151 bool TrackExtrap::extrapToZCov(TrackParam* trackParam, double zEnd, bool updatePropagator)
152 {
155 
156  if (trackParam->getZ() == zEnd) {
157  return true; // nothing to be done if same z
158  }
159 
160  if (!sFieldON) { // linear extrapolation if no magnetic field
161  linearExtrapToZCov(trackParam, zEnd, updatePropagator);
162  return true;
163  }
164 
165  // No need to propagate the covariance matrix if it does not exist
166  if (!trackParam->hasCovariances()) {
167  LOG(WARNING) << "Covariance matrix does not exist";
168  // Extrapolate track parameters to "zEnd"
169  return extrapToZ(trackParam, zEnd);
170  }
171 
172  // Save the actual track parameters
173  TrackParam trackParamSave(*trackParam);
174  TMatrixD paramSave(trackParamSave.getParameters());
175  double zBegin = trackParamSave.getZ();
176 
177  // Get reference to the parameter covariance matrix
178  const TMatrixD& kParamCov = trackParam->getCovariances();
179 
180  // Extrapolate track parameters to "zEnd"
181  // Do not update the covariance matrix if the extrapolation failed
182  if (!extrapToZ(trackParam, zEnd)) {
183  return false;
184  }
185 
186  // Get reference to the extrapolated parameters
187  const TMatrixD& extrapParam = trackParam->getParameters();
188 
189  // Calculate the jacobian related to the track parameters extrapolation to "zEnd"
190  bool extrapStatus = true;
191  TMatrixD jacob(5, 5);
192  jacob.Zero();
193  TMatrixD dParam(5, 1);
194  double direction[5] = { -1., -1., 1., 1., -1. };
195  for (int i = 0; i < 5; i++) {
196  // Skip jacobian calculation for parameters with no associated error
197  if (kParamCov(i, i) <= 0.) {
198  continue;
199  }
200 
201  // Small variation of parameter i only
202  for (int j = 0; j < 5; j++) {
203  if (j == i) {
204  dParam(j, 0) = TMath::Sqrt(kParamCov(i, i));
205  dParam(j, 0) *= TMath::Sign(1., direction[j] * paramSave(j, 0)); // variation always in the same direction
206  } else {
207  dParam(j, 0) = 0.;
208  }
209  }
210 
211  // Set new parameters
212  trackParamSave.setParameters(paramSave);
213  trackParamSave.addParameters(dParam);
214  trackParamSave.setZ(zBegin);
215 
216  // Extrapolate new track parameters to "zEnd"
217  if (!extrapToZ(&trackParamSave, zEnd)) {
218  LOG(WARNING) << "Bad covariance matrix";
219  extrapStatus = false;
220  }
221 
222  // Calculate the jacobian
223  TMatrixD jacobji(trackParamSave.getParameters(), TMatrixD::kMinus, extrapParam);
224  jacobji *= 1. / dParam(i, 0);
225  jacob.SetSub(0, i, jacobji);
226  }
227 
228  // Extrapolate track parameter covariances to "zEnd"
229  TMatrixD tmp(kParamCov, TMatrixD::kMultTranspose, jacob);
230  TMatrixD tmp2(jacob, TMatrixD::kMult, tmp);
231  trackParam->setCovariances(tmp2);
232 
233  // Update the propagator if required
234  if (updatePropagator) {
235  trackParam->updatePropagator(jacob);
236  }
237 
238  return extrapStatus;
239 }

242 double TrackExtrap::getMCSAngle2(const TrackParam& param, double dZ, double x0)
243 {
247  double bendingSlope = param.getBendingSlope();
248  double nonBendingSlope = param.getNonBendingSlope();
249  double inverseTotalMomentum2 = param.getInverseBendingMomentum() * param.getInverseBendingMomentum() *
250  (1.0 + bendingSlope * bendingSlope) /
251  (1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
252  // Path length in the material
253  double pathLength =
254  TMath::Abs(dZ) * TMath::Sqrt(1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
255  // relativistic velocity
256  double velo = 1.;
257  // Angular dispersion square of the track (variance) in a plane perpendicular to the trajectory
258  double theta02 = 0.0136 / velo * (1 + 0.038 * TMath::Log(pathLength / x0));
259  return theta02 * theta02 * inverseTotalMomentum2 * pathLength / x0;
260 }

262 //__________________________________________________________________________
263 void TrackExtrap::addMCSEffect(TrackParam* trackParam, double dZ, double x0)
264 {
270 
271  double bendingSlope = trackParam->getBendingSlope();
272  double nonBendingSlope = trackParam->getNonBendingSlope();
273  double inverseBendingMomentum = trackParam->getInverseBendingMomentum();
274  double inverseTotalMomentum2 = inverseBendingMomentum * inverseBendingMomentum * (1.0 + bendingSlope * bendingSlope) /
275  (1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
276  // Path length in the material
277  double signedPathLength = dZ * TMath::Sqrt(1.0 + bendingSlope * bendingSlope + nonBendingSlope * nonBendingSlope);
278  double pathLengthOverX0 = (x0 > 0.) ? TMath::Abs(signedPathLength) / x0 : TMath::Abs(signedPathLength);
279  // relativistic velocity
280  double velo = 1.;
281  // Angular dispersion square of the track (variance) in a plane perpendicular to the trajectory
282  double theta02 = 0.0136 / velo * (1 + 0.038 * TMath::Log(pathLengthOverX0));
283  theta02 *= theta02 * inverseTotalMomentum2 * pathLengthOverX0;
284 
285  double varCoor = (x0 > 0.) ? signedPathLength * signedPathLength * theta02 / 3. : 0.;
286  double varSlop = theta02;
287  double covCorrSlope = (x0 > 0.) ? signedPathLength * theta02 / 2. : 0.;
288 
289  // Set MCS covariance matrix
290  TMatrixD newParamCov(trackParam->getCovariances());
291  // Non bending plane
292  newParamCov(0, 0) += varCoor;
293  newParamCov(0, 1) += covCorrSlope;
294  newParamCov(1, 0) += covCorrSlope;
295  newParamCov(1, 1) += varSlop;
296  // Bending plane
297  newParamCov(2, 2) += varCoor;
298  newParamCov(2, 3) += covCorrSlope;
299  newParamCov(3, 2) += covCorrSlope;
300  newParamCov(3, 3) += varSlop;
301 
302  // Set momentum related covariances if B!=0
303  if (sFieldON) {
304  // compute derivative d(q/Pxy) / dSlopeX and d(q/Pxy) / dSlopeY
305  double dqPxydSlopeX =
306  inverseBendingMomentum * nonBendingSlope / (1. + nonBendingSlope * nonBendingSlope + bendingSlope * bendingSlope);
307  double dqPxydSlopeY = -inverseBendingMomentum * nonBendingSlope * nonBendingSlope * bendingSlope /
308  (1. + bendingSlope * bendingSlope) /
309  (1. + nonBendingSlope * nonBendingSlope + bendingSlope * bendingSlope);
310  // Inverse bending momentum (due to dependences with bending and non bending slopes)
311  newParamCov(4, 0) += dqPxydSlopeX * covCorrSlope;
312  newParamCov(0, 4) += dqPxydSlopeX * covCorrSlope;
313  newParamCov(4, 1) += dqPxydSlopeX * varSlop;
314  newParamCov(1, 4) += dqPxydSlopeX * varSlop;
315  newParamCov(4, 2) += dqPxydSlopeY * covCorrSlope;
316  newParamCov(2, 4) += dqPxydSlopeY * covCorrSlope;
317  newParamCov(4, 3) += dqPxydSlopeY * varSlop;
318  newParamCov(3, 4) += dqPxydSlopeY * varSlop;
319  newParamCov(4, 4) += (dqPxydSlopeX * dqPxydSlopeX + dqPxydSlopeY * dqPxydSlopeY) * varSlop;
320  }
321 
322  // Set new covariances
323  trackParam->setCovariances(newParamCov);
324 }

327 void TrackExtrap::convertTrackParamForExtrap(TrackParam* trackParam, double forwardBackward, double* v3)
328 {
332  v3[0] = trackParam->getNonBendingCoor(); // X
333  v3[1] = trackParam->getBendingCoor(); // Y
334  v3[2] = trackParam->getZ(); // Z
335  double pYZ = TMath::Abs(1.0 / trackParam->getInverseBendingMomentum());
336  double pZ = pYZ / TMath::Sqrt(1.0 + trackParam->getBendingSlope() * trackParam->getBendingSlope());
337  v3[6] = TMath::Sqrt(pYZ * pYZ + pZ * pZ * trackParam->getNonBendingSlope() * trackParam->getNonBendingSlope()); // P
338  v3[5] = -forwardBackward * pZ / v3[6]; // PZ/P spectro. z<0
339  v3[3] = trackParam->getNonBendingSlope() * v3[5]; // PX/P
340  v3[4] = trackParam->getBendingSlope() * v3[5]; // PY/P
341 }

359 bool TrackExtrap::extrapToZRungekutta(TrackParam* trackParam, double zEnd)
360 {
363  if (trackParam->getZ() == zEnd) {
364  return true; // nothing to be done if same Z
365  }
366  double forwardBackward; // +1 if forward, -1 if backward
367  if (zEnd < trackParam->getZ()) {
368  forwardBackward = 1.0; // spectro. z<0
369  } else {
370  forwardBackward = -1.0;
371  }
372  // sign of charge (sign of fInverseBendingMomentum if forward motion)
373  // must be changed if backward extrapolation
374  double chargeExtrap = forwardBackward * TMath::Sign(double(1.0), trackParam->getInverseBendingMomentum());
375  double v3[7] = { 0. }, v3New[7] = { 0. };
376  double dZ(0.), step(0.);
377  int stepNumber = 0;
378 
379  // Extrapolation loop (until within tolerance or the track turn around)
380  double residue = zEnd - trackParam->getZ();
381  bool uturn = false;
382  bool trackingFailed = false;
383  bool tooManyStep = false;
384  while (TMath::Abs(residue) > SRungeKuttaMaxResidue && stepNumber <= SMaxStepNumber) {
385 
386  dZ = zEnd - trackParam->getZ();
387  // step lenght assuming linear trajectory
388  step = dZ * TMath::Sqrt(1.0 + trackParam->getBendingSlope() * trackParam->getBendingSlope() +
389  trackParam->getNonBendingSlope() * trackParam->getNonBendingSlope());
390  convertTrackParamForExtrap(trackParam, forwardBackward, v3);
391 
392  do { // reduce step lenght while zEnd oversteped
393  if (stepNumber > SMaxStepNumber) {
394  LOG(WARNING) << "Too many trials: " << stepNumber;
395  tooManyStep = true;
396  break;
397  }
398  stepNumber++;
399  step = TMath::Abs(step);
400  if (!extrapOneStepRungekutta(chargeExtrap, step, v3, v3New)) {
401  trackingFailed = true;
402  break;
403  }
404  residue = zEnd - v3New[2];
405  step *= dZ / (v3New[2] - trackParam->getZ());
406  } while (residue * dZ < 0 && TMath::Abs(residue) > SRungeKuttaMaxResidue);
407 
408  if (trackingFailed) {
409  break;
410  } else if (v3New[5] * v3[5] < 0) { // the track turned around
411  LOG(WARNING) << "The track turned around";
412  uturn = true;
413  break;
414  } else {
415  recoverTrackParam(v3New, chargeExtrap * forwardBackward, trackParam);
416  }
417  }
418 
419  // terminate the extropolation with a straight line up to the exact "zEnd" value
420  if (trackingFailed || uturn) {
421 
422  // track ends +-100 meters away in the bending direction
423  dZ = zEnd - v3[2];
424  double bendingSlope = TMath::Sign(1.e4, -sSimpleBValue * trackParam->getInverseBendingMomentum()) / dZ;
425  double pZ =
426  TMath::Abs(1. / trackParam->getInverseBendingMomentum()) / TMath::Sqrt(1.0 + bendingSlope * bendingSlope);
427  double nonBendingSlope = TMath::Sign(TMath::Abs(v3[3]) * v3[6] / pZ, trackParam->getNonBendingSlope());
428  trackParam->setNonBendingCoor(trackParam->getNonBendingCoor() + dZ * nonBendingSlope);
429  trackParam->setNonBendingSlope(nonBendingSlope);
430  trackParam->setBendingCoor(trackParam->getBendingCoor() + dZ * bendingSlope);
431  trackParam->setBendingSlope(bendingSlope);
432  trackParam->setZ(zEnd);
433 
434  return false;
435 
436  } else {
437 
438  // track extrapolated normally
439  trackParam->setNonBendingCoor(trackParam->getNonBendingCoor() + residue * trackParam->getNonBendingSlope());
440  trackParam->setBendingCoor(trackParam->getBendingCoor() + residue * trackParam->getBendingSlope());
441  trackParam->setZ(zEnd);
442 
443  return !tooManyStep;
444  }
445 }

476  double secxs[4] = { 0. }, secys[4] = { 0. }, seczs[4] = { 0. }, hxp[3] = { 0. };

507  // *. ------------------------------------------------------------------
508  // *.
509  // * this constant is for units cm,gev/c and kgauss
510  // *
511  int iter = 0;
512  int ncut = 0;
513  for (int j = 0; j < 7; j++) {

515  }

521

531  // * start of integration
532  // *

534  y = vout[1];

538  c = vout[5];

548  if (ang2 > kpisqua) {

550  }

553  dyt = h2 * b + h4 * secys[0];
554  dzt = h2 * c + h4 * seczs[0];

556  yt = y + dyt;
557  zt = z + dzt;

559  // * second intermediate point
560  // *
561

563  if (est > h) {
564  if (ncut++ > maxcut) {
565  break;
566  }
567  h *= khalf;
568  continue;
569  }
570 
571  xyzt[0] = xt;
572  xyzt[1] = yt;
573  xyzt[2] = zt;
574 
575  // cmodif: call gufld(xyzt,f) changed into:

577 
578  at = a + secxs[0];
579  bt = b + secys[0];
580  ct = c + seczs[0];
581

583  secys[1] = (ct * f[0] - at * f[2]) * ph2;
584  seczs[1] = (at * f[1] - bt * f[0]) * ph2;
585  at = a + secxs[1];
586  bt = b + secys[1];
587  ct = c + seczs[1];
588  secxs[2] = (bt * f[2] - ct * f[1]) * ph2;
589  secys[2] = (ct * f[0] - at * f[2]) * ph2;
590  seczs[2] = (at * f[1] - bt * f[0]) * ph2;
591  dxt = h * (a + secxs[2]);
592  dyt = h * (b + secys[2]);
593  dzt = h * (c + seczs[2]);
594  xt = x + dxt;
595  yt = y + dyt;
596  zt = z + dzt;
597  at = a + 2. * secxs[2];
598  bt = b + 2. * secys[2];
599  ct = c + 2. * seczs[2];
600

602  if (est > 2. * TMath::Abs(h)) {
603  if (ncut++ > maxcut) {
604  break;
605  }
606  h *= khalf;
607  continue;
608  }
609 
610  xyzt[0] = xt;
611  xyzt[1] = yt;
612  xyzt[2] = zt;
613 
614  // cmodif: call gufld(xyzt,f) changed into:
