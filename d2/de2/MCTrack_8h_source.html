177 {
178  const auto mass = GetMass();
179  return std::sqrt(mass * mass + mStartVertexMomentumX * mStartVertexMomentumX +

269 template <typename T>
270 inline Double_t MCTrackT<T>::GetRapidity() const
271 {
272  const auto e = GetEnergy();
273  Double_t y =
274  0.5 * std::log((e + static_cast<double>(mStartVertexMomentumZ)) / (e - static_cast<double>(mStartVertexMomentumZ)));
