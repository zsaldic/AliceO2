40  struct inputSpecHasher {
41  size_t operator()(const InputSpec& i) const
42  {
43  if (auto matcher = std::get_if<ConcreteDataMatcher>(&i.matcher)) {
44  // 'Compressing' dataOrigin and dataDescription to 64 bits. SubSpecification is not taken into account,
45  // because sometimes we want to have subSpec-agnostic match.
46  return (static_cast<size_t>(matcher->description.itg[0]) << 32 |
47  static_cast<size_t>(matcher->description.itg[1])) ^
48  static_cast<size_t>(matcher->origin.itg[0]);
49  } else {
50  throw std::runtime_error("Unsupported InputSpec type");
51  }
52  }
53  };
54  struct inputSpecEqual {
55  bool operator()(const InputSpec& a, const InputSpec& b) const
56  {
57  auto matcherA = std::get_if<ConcreteDataMatcher>(&a.matcher);
58  auto matcherB = std::get_if<ConcreteDataMatcher>(&b.matcher);
59  if (matcherA && matcherB) {
60  // -1 means 'match all subSpec'
61  if (matcherA->subSpec == -1 || matcherB->subSpec == -1) {
62  return matcherA->description == matcherB->description && matcherA->origin == matcherB->origin;
63  } else {
64  return *matcherA == *matcherB;
65  }
66  } else {
67  return a == b;
68  }
69  }
70  };
