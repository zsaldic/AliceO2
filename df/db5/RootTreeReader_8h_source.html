136  bool next()
137  {
138  if ((mReadEntry + 1) >= mNEntries || mNEntries == 0) {
139  if (mPublishingMode == PublishingMode::Single) {
140  // stop here

192  template <typename F>
193  bool process(F&& snapshot) const
194  {
195  if (mReadEntry >= mNEntries || mNEntries == 0 || (mMaxEntries > 0 && mNofPublished >= mMaxEntries)) {
196  return false;
197  }
198 
199  for (auto& spec : mBranchSpecs) {
200  char* data = nullptr;
201  spec.second->branch->SetAddress(&data);
202  spec.second->branch->GetEntry(mReadEntry);
203  if (spec.second->sizebranch == nullptr) {
204  snapshot(spec.first, std::move(ROOTSerializedByClass(*data, spec.second->classinfo)));
205  } else {
206  size_t datasize = 0;
207  spec.second->sizebranch->SetAddress(&datasize);
208  spec.second->sizebranch->GetEntry(mReadEntry);
209  auto* buffer = reinterpret_cast<BinaryDataStoreType*>(data);
210  if (buffer->size() == datasize) {
211  LOG(INFO) << "branch " << spec.second->name << ": publishing binary chunk of " << datasize << " bytes(s)";
212  snapshot(spec.first, std::move(*buffer));
213  } else {
214  LOG(ERROR) << "branch " << spec.second->name << ": inconsitent size of binary chunk "
215  << buffer->size() << " vs " << datasize;
216  BinaryDataStoreType empty;
217  snapshot(spec.first, empty);
218  }
219  }
220  auto* delfunc = spec.second->classinfo->GetDelete();
221  if (delfunc) {
222  (*delfunc)(data);
223  }
224  spec.second->branch->DropBaskets("all");
225  }
226  return true;
227  }

245  void addBranchSpec(KeyType key, const char* branchName)
246  {
247  // right now we allow the same key to appear for multiple branches
248  auto branch = mInput.GetBranch(branchName);
249  if (branch) {
250  mBranchSpecs.emplace_back(key, std::make_unique<BranchSpec>(BranchSpec{ branchName }));
251  mBranchSpecs.back().second->branch = branch;
252  std::string sizebranchName = std::string(branchName) + "Size";
253  auto sizebranch = mInput.GetBranch(sizebranchName.c_str());
254  auto* classinfo = TClass::GetClass(branch->GetClassName());
255  if (!sizebranch) {
256  if (classinfo == nullptr) {
257  throw std::runtime_error(std::string("can not find class description for branch ") + branchName);
258  }
259  LOG(INFO) << "branch set up: " << branchName;
260  } else {
261  if (classinfo == nullptr || classinfo != TClass::GetClass(typeid(BinaryDataStoreType))) {
262  throw std::runtime_error("mismatching class type, expecting std::vector<char> for binary branch");
263  }
264  mBranchSpecs.back().second->sizebranch = sizebranch;
265  LOG(INFO) << "binary branch set up: " << branchName;
266  }
267  mBranchSpecs.back().second->classinfo = classinfo;
268  } else {
269  throw std::runtime_error(std::string("can not find branch ") + branchName);
270  }
271  }
