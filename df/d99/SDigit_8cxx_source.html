105 void SDigit::addSignal(Int_t track, Int_t hit, Double_t signal)
106 {
107  // Adds this track number and signal to the pList and orders them
108  // Inputs:
109  // Int_t track The track number which produced this signal
110  // Int_t hit The hit number which produced this signal
111  // Int_t chip The chip where this signal occurred
112  // Int_t index The cell index where this signal occurred
113  // Double_t signal The value of the signal (ionization)
114  Int_t i, j;
115  Bool_t flg = kFALSE;
116  //
117  if (TMath::Abs(signal) > 2147483647.0) {
118  // PH 2147483647 is the max. integer
119  // PH This apparently is a problem which needs investigation
120  LOG(WARNING) << "Too big or too small signal value " << signal << FairLogger::endl;
121  signal = TMath::Sign((Double_t)2147483647, signal);
122  }
123  //
124  mTotalSignal += signal; // Keep track of sum signal.
125  for (i = mNumOfTracks; i--;) {
126  if (track == mTrack[i]) {
127  mSignal[i] += signal;
128  flg = kTRUE;
129  break;
130  } // end for i & if.
131  }
132  //
133  if (flg) {
134  if (mNumOfTracks > 1) { // resort arrays.
135  for (i = 1; i < mNumOfTracks; i++) {
136  j = i;
137  while (j > 0 && mSignal[j] > mSignal[j - 1]) {
138  std::swap(mTrack[j - 1], mTrack[j]);
139  std::swap(mHits[j - 1], mHits[j]);
140  std::swap(mSignal[j - 1], mSignal[j]);
141  j--;
142  } // end while
143  } // end if i
144  } // end if added to existing and resorted array
145  return;
146  }
147  //
148  // new entry add it in order.
149  if (mNumOfTracks == (kBuffSize - 1) && signal <= mSignal[kBuffSize - 1])
150  return; // if this signal is <= smallest then don't add it.
151  //
152  for (i = mNumOfTracks; i--;) {
153  if (signal > mSignal[i]) { // shift smaller signals to the end of the list
154  if (i < kBuffSize - 2) { // (if there is a space...)
155  mSignal[i + 1] = mSignal[i];
156  mTrack[i + 1] = mTrack[i];
157  mHits[i + 1] = mHits[i];
158  }
159  } else {
160  mSignal[i + 1] = signal;
161  mTrack[i + 1] = track;
162  mHits[i + 1] = hit;
163  if (mNumOfTracks < kBuffSize - 1)
164  mNumOfTracks++;
165  return; // put it in the right place, now exit.
166  } // end if
167  } // end if; end for i
168  //
169  // Still haven't found the right place. Must be at top of list.
170  mSignal[0] = signal;
171  mTrack[0] = track;
172  mHits[0] = hit;
173  mNumOfTracks++;
174  return;
175 }

178 void SDigit::add(const SDigit* pl)
179 {
180  // Adds the contents of pl to this
181  // pl could come from different chip and index
182  Double_t sigT = 0.0;
183  for (int i = pl->getNumOfTracks(); i--;) {
184  double sig = pl->getSignal(i);
185  addSignal(pl->getTrack(i), pl->getHit(i), sig);
186  sigT += sig;
187  } // end for i
188  mTotalSignal += (pl->mTotalSignal - sigT);
189  mNoise += pl->mNoise;
190  return;
191  //
192 }

196 {
197  // Adds the contents of pl to this with track number off set given by
198  // fileIndex.
199  // Inputs:

201  // SDigit *pl an SDigit to be added to this class.
202  //
203  for (int i = pl->getNumOfTracks(); i--;)
204  addSignal(pl->getTrack(i) + fileIndex, pl->getHit(i), pl->getSignal(i));
205  mSignalAfterElect += (pl->mSignalAfterElect + pl->mNoise - mNoise);
206  mNoise = pl->mNoise;
