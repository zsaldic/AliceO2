65 void EPNReceiverDevice::DiscardIncompleteTimeframes()
66 {
67  auto it = mTimeframeBuffer.begin();
68 
69  while (it != mTimeframeBuffer.end()) {
70  if (duration_cast<milliseconds>(steady_clock::now() - (it->second).start).count() > mBufferTimeoutInMs) {
71  LOG(WARN) << "Timeframe #" << it->first << " incomplete after " << mBufferTimeoutInMs << " milliseconds, discarding";
72  mDiscardedSet.insert(it->first);
73  mTimeframeBuffer.erase(it++);
74  LOG(WARN) << "Number of discarded timeframes: " << mDiscardedSet.size();
75  } else {
76  // LOG(INFO) << "Timeframe #" << it->first << " within timeout, buffering...";
77  ++it;
78  }

82 void EPNReceiverDevice::Run()
83 {
84  uint16_t id = 0; // holds the timeframe id of the currently arrived sub-timeframe.
85 
86  FairMQChannel& ackOutChannel = fChannels.at(mAckChannelName).at(0);
87 
88  // Simple multi timeframe index
89  using TimeframeId = int;
90  using FlpId = int;
91  std::multimap<TimeframeId, IndexElement> index;
92  std::multimap<TimeframeId, FlpId> flpIds;
93 
94  while (CheckCurrentState(RUNNING)) {
95  FairMQParts subtimeframeParts;
96  if (Receive(subtimeframeParts, mInChannelName, 0, 100) <= 0)
97  continue;
98 
99  assert(subtimeframeParts.Size() >= 2);
100 
101  const auto* dh = o2::header::get<header::DataHeader*>(subtimeframeParts.At(0)->GetData());
102  assert(strncmp(dh->dataDescription.str, "SUBTIMEFRAMEMD", 16) == 0);
103  SubframeMetadata* sfm = reinterpret_cast<SubframeMetadata*>(subtimeframeParts.At(1)->GetData());
104  id = o2::DataFlow::timeframeIdFromTimestamp(sfm->startTime, sfm->duration);
105  auto flpId = sfm->flpIndex;
106 
107  if (mDiscardedSet.find(id) == mDiscardedSet.end())
108  {
109  if (mTimeframeBuffer.find(id) == mTimeframeBuffer.end())
110  {
111  // if this is the first part with this ID, save the receive time.
112  mTimeframeBuffer[id].start = steady_clock::now();
113  }
114  flpIds.insert(std::make_pair(id, flpId));
115  LOG(INFO) << "Timeframe ID " << id << " for startTime " << sfm->startTime << "\n";
116  // If the received ID has not previously been discarded, store
117  // the data part in the buffer For the moment we just concatenate
118  // the subtimeframes and add an index for their description at
119  // the end. Given every second part is a data header we skip
120  // every two parts to populate the index. Moreover we know that
121  // the SubframeMetadata is always in the second part, so we can
122  // extract the flpId from there.
123  for (size_t i = 0; i < subtimeframeParts.Size(); ++i)
124  {
125  if (i % 2 == 0)
126  {
127  const auto* adh = o2::header::get<header::DataHeader*>(subtimeframeParts.At(i)->GetData());
128  auto ie = std::make_pair(*adh, index.count(id)*2);
129  index.insert(std::make_pair(id, ie));
130  }
131  mTimeframeBuffer[id].parts.AddPart(move(subtimeframeParts.At(i)));
132  }
133  }
134  else
135  {
136  // if received ID has been previously discarded.
137  LOG(WARN) << "Received part from an already discarded timeframe with id " << id;
138  }
139 
140  if (flpIds.count(id) == mNumFLPs) {
141  LOG(INFO) << "Timeframe " << id << " complete. Publishing.\n";
142  o2::header::DataHeader tih;
143  std::vector<IndexElement> flattenedIndex;
144 
145  tih.dataDescription = o2::header::DataDescription("TIMEFRAMEINDEX");
146  tih.dataOrigin = o2::header::DataOrigin("EPN");
147  tih.subSpecification = 0;
148  tih.payloadSize = index.count(id) * sizeof(flattenedIndex.front());
149  void *indexData = malloc(tih.payloadSize);
150  auto indexRange = index.equal_range(id);
151  for (auto ie = indexRange.first; ie != indexRange.second; ++ie)
152  {
153  flattenedIndex.push_back(ie->second);
154  }
155  memcpy(indexData, flattenedIndex.data(), tih.payloadSize);
156 
157  mTimeframeBuffer[id].parts.AddPart(NewSimpleMessage(tih));
158  mTimeframeBuffer[id].parts.AddPart(NewMessage(indexData, tih.payloadSize,
159  [](void* data, void* hint){ free(data); }, nullptr));
160  // LOG(INFO) << "Collected all parts for timeframe #" << id;
161  // when all parts are collected send then to the output channel
162  Send(mTimeframeBuffer[id].parts, mOutChannelName);
163  LOG(INFO) << "Index count for " << id << " " << index.count(id) << "\n";
164  index.erase(id);
165  LOG(INFO) << "Index count for " << id << " " << index.count(id) << "\n";
166  flpIds.erase(id);
167 
168  if (mTestMode > 0) {
169  // Send an acknowledgement back to the sampler to measure the round trip time
170  unique_ptr<FairMQMessage> ack(NewMessage(sizeof(uint16_t)));
171  memcpy(ack->GetData(), &id, sizeof(uint16_t));
172 
173  if (ackOutChannel.Send(ack, 0) <= 0) {
174  LOG(ERROR) << "Could not send acknowledgement without blocking";
175  }
176  }
177 
178  mTimeframeBuffer.erase(id);
179  }
180 
181  // LOG(WARN) << "Buffer size: " << fTimeframeBuffer.size();
182 
183  // Check if any incomplete timeframes in the buffer are older than
184  // timeout period, and discard them if they are
185  // QUESTION: is this really what we want to do?
186  DiscardIncompleteTimeframes();
187  }
188 }
