83 void Initializer::setupGeometry()
84 {
85  // read path to geometry files from config file
86  TEnv settings;
87  ConfigurationManager::getInstance().getConfig(settings);
88 
89  // get geometry from Geometry Manager and register in multiview
90  auto multiView = MultiView::getInstance();
91 
92  for(int iDet=0;iDet<NvisualisationGroups;++iDet){
93  EVisualisationGroup det = static_cast<EVisualisationGroup>(iDet);
94  string detName = gVisualisationGroupName[det];
95  if(settings.GetValue((detName+".draw").c_str(), false))
96  {
97  if( detName=="TPC" || detName=="MCH" || detName=="MTR"
98  || detName=="MID" || detName=="MFT" || detName=="AD0"
99  || detName=="FMD"){// don't load MUON+MFT and AD and standard TPC to R-Phi view
100 
101  multiView->drawGeometryForDetector(detName,true,false);
102  }
103  else if(detName=="RPH"){// special TPC geom from R-Phi view
104 
105  multiView->drawGeometryForDetector(detName,false,true,false);
106  }
107  else{// default
108  multiView->drawGeometryForDetector(detName);
109  }
110  }
111  }

114 void Initializer::setupCamera()
115 {
116  // move and rotate sub-views
117  TEnv settings;
118  ConfigurationManager::getInstance().getConfig(settings);
119 
120  // read settings from config file
121  const double angleHorizontal = settings.GetValue("camera.3D.rotation.horizontal",-0.4);
122  const double angleVertical = settings.GetValue("camera.3D.rotation.vertical",1.0);
123 
124  double zoom[MultiView::NumberOfViews];
125  zoom[MultiView::View3d] = settings.GetValue("camera.3D.zoom",1.0);
126  zoom[MultiView::ViewRphi] = settings.GetValue("camera.R-Phi.zoom",1.0);
127  zoom[MultiView::ViewZrho] = settings.GetValue("camera.Rho-Z.zoom",1.0);
128 
129  // get necessary elements of the multiview and set camera position
130  auto multiView = MultiView::getInstance();
131 
132  for(int viewIter=0;viewIter<MultiView::NumberOfViews;++viewIter){
133  TGLViewer *glView = multiView->getView(static_cast<MultiView::EViews>(viewIter))->GetGLViewer();
134  glView->CurrentCamera().Reset();
135 
136  if(viewIter==0){
137  glView->CurrentCamera().RotateRad(angleHorizontal, angleVertical);
138  }
139  glView->CurrentCamera().Dolly(zoom[viewIter], kFALSE, kTRUE);
140  }
141 }
