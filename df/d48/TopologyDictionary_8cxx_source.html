63  int TopologyDictionary::ReadBinaryFile(string fname)
64  {
65  mVectorOfGroupIDs.clear();
66  mFinalMap.clear();
67  for (auto& p : mSmallTopologiesLUT)
68  p = -1;
69  std::ifstream in(fname.data(), std::ios::in | std::ios::binary);
70  GroupStruct gr;
71  int groupID = 0;
72  if (!in.is_open()) {
73  LOG(ERROR) << "The file coud not be opened" << FairLogger::endl;
74  throw std::runtime_error("No fired pixels in small topology");
75  } else {
76  while (in.read(reinterpret_cast<char*>(&gr.mHash), sizeof(unsigned long))) {
77  in.read(reinterpret_cast<char*>(&gr.mErrX), sizeof(float));
78  in.read(reinterpret_cast<char*>(&gr.mErrZ), sizeof(float));
79  in.read(reinterpret_cast<char*>(&gr.mXCOG), sizeof(float));
80  in.read(reinterpret_cast<char*>(&gr.mZCOG), sizeof(float));
81  in.read(reinterpret_cast<char*>(&gr.mNpixels), sizeof(int));
82  in.read(reinterpret_cast<char*>(&gr.mFrequency), sizeof(double));
83  in.read(reinterpret_cast<char*>(&gr.mPattern.mBitmap), sizeof(unsigned char) * (ClusterPattern::kExtendedPatternBytes));
84  mVectorOfGroupIDs.push_back(gr);
85  if (gr.mPattern.getUsedBytes() == 1)
86  mSmallTopologiesLUT[(gr.mPattern.getColumnSpan() - 1) * 255 + (int)gr.mPattern.mBitmap[2]] = groupID;
87  if (((gr.mHash) & 0xffffffff) != 0)
88  mFinalMap.insert(std::make_pair(gr.mHash, groupID));
89  groupID++;
90  }
91  }
92  in.close();
93  return 0;
94  }
