72  void run(framework::ProcessingContext& pc)
73  {
74  if (mFinished) {
75  return;
76  }
77  std::string detStr = mID.getName();
78  std::string detStrL = mID.getName();
79  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
80 
81  // retrieve the digits from the input
82  auto inDigits = pc.inputs().get<std::vector<o2::ITSMFT::Digit>>((detStr + "digits").c_str());
83  auto inROFs = pc.inputs().get<std::vector<o2::ITSMFT::ROFRecord>>((detStr + "digitsROF").c_str());
84  auto inMC2ROFs = pc.inputs().get<std::vector<o2::ITSMFT::MC2ROFRecord>>((detStr + "digitsMC2ROF").c_str());
85  auto inLabels = pc.inputs().get<MCCont*>((detStr + "digitsMCTR").c_str());
86  LOG(INFO) << "RECEIVED DIGITS SIZE " << inDigits.size();
87 
88  auto digitsP = &inDigits;
89  auto labelsRaw = inLabels.get();
90  // connect this to a particular branch
91 
92  auto brDig = getOrMakeBranch(*mOutTreeDig.get(), (detStr + "Digit").c_str(), &digitsP);
93  auto brLbl = getOrMakeBranch(*mOutTreeDig.get(), (detStr + "DigitMCTruth").c_str(), &labelsRaw);
94  mOutTreeDig->Fill();
95 
96  auto rofP = &inROFs;
97  auto brROF = getOrMakeBranch(*mOutTreeROF.get(), (detStr + "DigitROF").c_str(), &rofP);
98  mOutTreeROF->Fill();
99 
100  auto mc2rofP = &inMC2ROFs;
101  auto brMC2ROF = getOrMakeBranch(*mOutTreeMC2ROF.get(), (detStr + "DigitMC2ROF").c_str(), &mc2rofP);
102  mOutTreeMC2ROF->Fill();
103 
104  // mOutFile->WriteObjectAny(&inROFs, "std::vector<o2::ITSMFT::ROFRecord>", (detStr + "DigitROF").c_str());
105  // mOutFile->WriteObjectAny(&inMC2ROFs, "std::vector<o2::ITSMFT::MC2ROFRecord>", (detStr + "DigitMC2ROF").c_str());
106  mOutTreeDig->Write();
107  mOutTreeROF->Write();
108  mOutTreeMC2ROF->Write();
109  mOutTreeDig.reset(); // delete the trees before closing the file
110  mOutTreeROF.reset();
111  mOutTreeMC2ROF.reset();
112 
113  mOutFile->Close();
114  mFinished = true;
115  pc.services().get<ControlService>().readyToQuit(false);
116  }

118  protected:
119  ITSMFTDPLDigitWriter() = default;
120  template <typename T>
121  TBranch* getOrMakeBranch(TTree& tree, std::string brname, T* ptr)
122  {
123  if (auto br = tree.GetBranch(brname.c_str())) {
124  br->SetAddress(static_cast<void*>(ptr));
125  return br;
126  }
127  // otherwise make it
128  return tree.Branch(brname.c_str(), ptr);
129  }

189  auto detOrig = ITSDPLDigitWriter::DETOR;
190 
191  std::vector<InputSpec> inputs;

194  inputs.emplace_back(InputSpec{ (detStr + "digitsMC2ROF").c_str(), detOrig, "DIGITSMC2ROF", 0, Lifetime::Timeframe });
195  inputs.emplace_back(InputSpec{ (detStr + "digitsMCTR").c_str(), detOrig, "DIGITSMCTR", 0, Lifetime::Timeframe });
196 
197  return DataProcessorSpec{
198  (detStr + "DigitWriter").c_str(),
199  inputs,
200  {}, // no output
201  AlgorithmSpec(adaptFromTask<ITSDPLDigitWriter>()),
202  Options{
203  { (detStrL + "-digit-outfile").c_str(), VariantType::String, (detStrL + "digits.root").c_str(), { "Name of the input file" } },
