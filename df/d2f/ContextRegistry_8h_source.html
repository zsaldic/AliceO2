64  template <typename T>
65  void set(T* instance)
66  {
67  static_assert(std::is_void<T>::value == false, "can not register a void object");
68  size_t i = 0;
69  for (i = 0; i < mRegistryCount; ++i) {
70  if (typeid(T*).hash_code() == mRegistryKey[i]) {
71  return;
72  }
73  }
74  if (i == MAX_REGISTRY_SIZE) {
75  throw std::runtime_error("Too many entries in ContextRegistry");
76  }
77  mRegistryCount = i + 1;
78  mRegistryKey[i] = typeid(T*).hash_code();
79  mRegistryValue[i] = instance;
80  }
