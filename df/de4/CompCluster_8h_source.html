26 class CompCluster
27 {
28  private:
29  static constexpr int NBitsRow = 9; // number of bits for the row
30  static constexpr int NBitsCol = 10; // number of bits for the column
31  static constexpr int NBitsPattID = 11; // number of bits for the pattern ID
32  static constexpr UInt_t RowMask = (0x1 << NBitsRow) - 1;
33  static constexpr UInt_t ColMask = (0x1 << NBitsCol) - 1;
34  static constexpr UInt_t PattIDMask = (0x1 << NBitsPattID) - 1;
35  static constexpr UInt_t FlagBit = 0x1 << (NBitsRow + NBitsCol + NBitsPattID);
36  //
38  UInt_t mData;
39 
40  void sanityCheck();
41 
42  public:
43  CompCluster(UShort_t row = 0, UShort_t col = 0, UShort_t patt = 0)
44  {
45  set(row, col, patt);
46  }
47 
48  void set(UShort_t row, UShort_t col, UShort_t patt)
49  {
50  mData = (row & RowMask) | ((col & ColMask) << NBitsRow) | ((patt & PattIDMask) << (NBitsRow + NBitsCol));
51  }
52 
53  UShort_t getRow() const { return mData & RowMask; }
54  UShort_t getCol() const { return (mData >> NBitsRow) & ColMask; }
55  UShort_t getPatternID() const { return (mData >> (NBitsRow + NBitsCol)) & PattIDMask; }
56  bool getFlag() const { return (mData & FlagBit) == FlagBit; }
57 
58  void setRow(UShort_t r)
59  {
60  mData &= ~RowMask;
61  mData |= (r & RowMask);
62  }
63  void setCol(UShort_t c)
64  {
65  mData &= ~(ColMask << NBitsRow);
66  mData |= (c & ColMask) << NBitsRow;
67  };
68  void setPatternID(UShort_t p)
69  {
70  mData &= ~(PattIDMask << (NBitsRow + NBitsCol));
71  mData |= (p & PattIDMask) << (NBitsRow + NBitsCol);
72  }
73  void setFlag(bool v)
74  {
75  mData &= ~FlagBit;
76  if (v) {
77  mData |= FlagBit;
78  }
79  }
