37 DataProcessorSpec CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
38  std::vector<InputSpec>& unmatched)
39 {
40  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
41  auto filename = ic.options().get<std::string>("outfile");
42  auto keepString = ic.options().get<std::string>("keep");
43 
44  if (filename.empty()) {
45  throw std::runtime_error("output file missing");
46  }
47 
48  bool hasOutputsToWrite = false;
49  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
50  VariableContext context;
51  for (auto& spec : danglingOutputInputs) {
52  auto concrete = DataSpecUtils::asConcreteDataMatcher(spec);
53  if (outputMatcher->match(concrete, context)) {
54  hasOutputsToWrite = true;
55  }
56  }
57  if (hasOutputsToWrite == false) {
58  return std::move([](ProcessingContext& pc) mutable -> void {
59  static bool once = false;
62  if (!once) {
63  LOG(INFO) << "No dangling output to be dumped.";
64  once = true;
65  }
66  sleep(1);
67  });
68  }
69  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
70  return std::move([ output, matcher = outputMatcher ](ProcessingContext & pc) mutable->void {
71  VariableContext matchingContext;
72  LOG(INFO) << "processing data set with " << pc.inputs().size() << " entries";
73  for (const auto& entry : pc.inputs()) {
74  LOG(INFO) << " " << *(entry.spec);
75  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
76  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
77  if (matcher->match(*header, matchingContext) == false) {
78  continue;
79  }
80  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
81  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
82  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
83  LOG(INFO) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
84  }
85  });
86  };
87 
88  std::vector<InputSpec> validBinaryInputs;
89  auto onlyTimeframe = [](InputSpec const& input) {
90  return input.lifetime == Lifetime::Timeframe;
91  };
92 
93  auto noTimeframe = [](InputSpec const& input) {
94  return input.lifetime != Lifetime::Timeframe;
95  };
96 
97  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
98  std::back_inserter(validBinaryInputs), onlyTimeframe);
99  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
100  std::back_inserter(unmatched), noTimeframe);
101 
102  DataProcessorSpec spec{
103  "dpl-global-binary-file-sink",
104  validBinaryInputs,
105  Outputs{},
106  AlgorithmSpec(writerFunction),
107  { { "outfile", VariantType::String, "dpl-out.bin", { "Name of the output file" } },
108  { "keep", VariantType::String, "", { "Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile" } } }
109  };
110 
111  return spec;
112 }
