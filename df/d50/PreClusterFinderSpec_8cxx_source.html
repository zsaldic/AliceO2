74  void run(framework::ProcessingContext& pc)
75  {
77 
78  // prepare to receive new data
79  mPreClusterFinder.reset();
80 
81  // get the input buffer
82  auto msgIn = pc.inputs().get<gsl::span<char>>("digits");
83  auto bufferPtrIn = msgIn.data();
84  auto sizeIn = msgIn.size();
85 
86  // get header info and check message consistency
87  if (sizeIn < SSizeOfDigitBlock) {
88  throw out_of_range("missing DigitBlock");
89  }
90  auto digitBlock(reinterpret_cast<const DigitBlock*>(bufferPtrIn));
91  bufferPtrIn += SSizeOfDigitBlock;
92  sizeIn -= SSizeOfDigitBlock;
93  if (digitBlock->header.fRecordWidth != SSizeOfDigitStruct) {
94  throw length_error("incorrect size of digits. Corrupted message?");
95  }
96  if (sizeIn != digitBlock->header.fNrecords * SSizeOfDigitStruct) {
97  throw length_error("incorrect payload");
98  }
99 
100  // load the digits to get the fired pads
101  auto digits(reinterpret_cast<const DigitStruct*>(bufferPtrIn));
102  mPreClusterFinder.loadDigits(digits, digitBlock->header.fNrecords);
103 
104  // preclusterize
105  int nPreClusters = mPreClusterFinder.run();
106 
107  // number of DEs with preclusters and total number of pads used
108  int nUsedDigits(0);
109  int nDEWithPreClusters = mPreClusterFinder.getNDEWithPreClusters(nUsedDigits);
110 
111  // create the output message of the exactly needed buffer size
112  auto sizeOut = SSizeOfInt + nDEWithPreClusters * 2 * SSizeOfInt +
113  PreClusterBlock::sizeOfPreClusterBlocks(nDEWithPreClusters, nPreClusters, nUsedDigits);
114  auto msgOut = pc.outputs().make<char>(Output{ "MCH", "PRECLUSTERS", 0, Lifetime::Timeframe }, sizeOut);
115  auto bufferPtrOut = msgOut.data();
116  if (msgOut.size() != sizeOut) {
117  throw length_error("incorrect message payload");
118  }
119 
120  // store the number of DE with preclusters
121  memcpy(bufferPtrOut, &nDEWithPreClusters, SSizeOfInt);
122  bufferPtrOut += SSizeOfInt;
123  sizeOut -= SSizeOfInt;
124 
125  // store preclusters
126  try {
127  storePreClusters(bufferPtrOut, sizeOut);
128  } catch (exception const& e) {
129  throw length_error(std::string("fail to store preclusters: ") + e.what());
130  }
131  }

134  //_________________________________________________________________________________________________
135  void storePreClusters(char* buffer, uint32_t size)
136  {
138 
139  const PreClusterFinder::PreCluster* cluster(nullptr);
140  const DigitStruct* digit(nullptr);
141  uint32_t* bytesUsed(nullptr);
142  uint32_t totalBytesUsed(0);
143 
144  for (int iDE = 0, nDEs = mPreClusterFinder.getNDEs(); iDE < nDEs; ++iDE) {
145 
146  if (!mPreClusterFinder.hasPreClusters(iDE)) {
147  continue;
148  }
149 
150  // store the DE ID
151  if (size - totalBytesUsed >= SSizeOfInt) {
152  auto deId(reinterpret_cast<int*>(buffer + totalBytesUsed));
153  *deId = mPreClusterFinder.getDEId(iDE);
154  totalBytesUsed += SSizeOfInt;
155  } else {
156  throw length_error("cannot store DE ID");
157  }
158 
159  // prepare to store the size of the PreClusterBlock
160  if (size - totalBytesUsed >= SSizeOfInt) {
161  bytesUsed = reinterpret_cast<uint32_t*>(buffer + totalBytesUsed);
162  totalBytesUsed += SSizeOfInt;
163  } else {
164  throw length_error("cannot store size of the PreClusterBlock");
165  }
166 
167  // prepare to store the preclusters of this DE
168  if (mPreClusterBlock.reset(buffer + totalBytesUsed, size - totalBytesUsed, true) < 0) {
169  throw length_error("cannot reset the cluster block");
170  }
171 
172  for (int iPlane = 0; iPlane < 2; ++iPlane) {
173  for (int iCluster = 0, nClusters = mPreClusterFinder.getNPreClusters(iDE, iPlane);
174  iCluster < nClusters; ++iCluster) {
175 
176  cluster = mPreClusterFinder.getPreCluster(iDE, iPlane, iCluster);
177  if (!cluster->storeMe) {
178  continue;
179  }
180 
181  // add the precluster with its first digit
182  digit = mPreClusterFinder.getDigit(iDE, cluster->firstPad);
183  if (mPreClusterBlock.startPreCluster(*digit) < 0) {
184  throw length_error("cannot store a new precluster");
185  }
186 
187  // loop over other pads and add corresponding digits
188  for (uint16_t iOrderedPad = cluster->firstPad + 1; iOrderedPad <= cluster->lastPad; ++iOrderedPad) {
189  digit = mPreClusterFinder.getDigit(iDE, iOrderedPad);
190  if (mPreClusterBlock.addDigit(*digit) < 0) {
191  throw length_error("cannot store a new digit");
192  }
193  }
194  }
195  }
196 
197  // store the size of the PreClusterBlock
198  *bytesUsed = mPreClusterBlock.getCurrentSize();
199  totalBytesUsed += *bytesUsed;
200 
201  if (mPrint) {
202  LOG(INFO) << "block: " << mPreClusterBlock;
203  }
204  }
205 
206  if (totalBytesUsed != size) {
207  throw length_error("incorrect payload");
208  }
209  }

212  static constexpr uint32_t SSizeOfDigitBlock = sizeof(DigitBlock);
213  static constexpr uint32_t SSizeOfDigitStruct = sizeof(DigitStruct);
214 
215  bool mPrint = false;
216  PreClusterFinder mPreClusterFinder{};
217  PreClusterBlock mPreClusterBlock{};
218 };
