65 const PadPos PadRegionInfo::findPad(const float localX, const float localY, const Side side/*=Side::A*/) const
66 {
67  if (!isInRegion(localX)) return PadPos(255, 255);
68 
69  // the pad coordinate system is for pad-side view.
70  // on the A-Side one looks from the back-side, therefore
71  // the localY-sign must be changed
72  const float localYfactor=(side==Side::A)?-1.f:1.f;
73  const unsigned int row = std::floor((localX-mRadiusFirstRow)/mPadHeight);
74  if (row>=mNumberOfPadRows) return PadPos(255, 255);
75 
76  const unsigned int npads=getPadsInRowRegion(row);
77  const unsigned int pad =int((npads/2*mPadWidth-localYfactor*localY)/mPadWidth);
78 
79  if (pad>=npads) return PadPos(255, 255);
80 
81  return PadPos(row, pad);
82 }
