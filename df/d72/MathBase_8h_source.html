55  Double_t fitGaus(const size_t nBins, const T *arr, const T xMin, const T xMax, std::vector<T>& param)
56  {
57  static TLinearFitter fitter(3,"pol2");
58  static TMatrixD mat(3,3);
59  static Double_t kTol = mat.GetTol();
60  fitter.StoreData(kFALSE);
61  fitter.ClearPoints();
62  TVectorD par(3);
63  TVectorD sigma(3);
64  TMatrixD A(3,3);
65  TMatrixD b(3,1);
66  T rms = TMath::RMS(nBins,arr);
67  T max = TMath::MaxElement(nBins,arr);
68  T binWidth = (xMax-xMin)/T(nBins);
69 
70  Float_t meanCOG = 0;
71  Float_t rms2COG = 0;
72  Float_t sumCOG = 0;
73 
74  Float_t entries = 0;
75  Int_t nfilled=0;
76 
77  param.resize(4);
78  param[0] = 0.;
79  param[1] = 0.;
80  param[2] = 0.;
81  param[3] = 0.;
82 
83  for (Int_t i=0; i<nBins; i++){
84  entries+=arr[i];
85  if (arr[i]>0) nfilled++;
86  }
87 
88  // TODO: Check why this is needed
89  if (max<4) return -4;
90  if (entries<12) return -4;
91 
92  if (rms<kTol) return -4;
93 
94  param[3] = entries;
95 
96  Int_t npoints=0;
97  for (Int_t ibin=0;ibin<nBins; ibin++){
98  Float_t entriesI = arr[ibin];
99  if (entriesI>1){
100  Double_t xcenter = xMin+(ibin+0.5)*binWidth;
101  Double_t error = 1./TMath::Sqrt(entriesI);
102  Double_t val = TMath::Log(Float_t(entriesI));
103  fitter.AddPoint(&xcenter,val,error);
104  if (npoints<3){
105  A(npoints,0)=1;
106  A(npoints,1)=xcenter;
107  A(npoints,2)=xcenter*xcenter;
108  b(npoints,0)=val;
109  meanCOG+=xcenter*entriesI;
110  rms2COG +=xcenter*entriesI*xcenter;
111  sumCOG +=entriesI;
112  }
113  npoints++;
114  }
115  }
116 
117  Double_t chi2 = 0;
118  if (npoints>=3){
119  if ( npoints == 3 ){
120  //analytic calculation of the parameters for three points
121  A.Invert();
122  TMatrixD res(1,3);
123  res.Mult(A,b);
124  par[0]=res(0,0);
125  par[1]=res(0,1);
126  par[2]=res(0,2);
127  chi2 = -3.;
128  } else {
129  // use fitter for more than three points
130  fitter.Eval();
131  fitter.GetParameters(par);
132  fitter.GetCovarianceMatrix(mat);
133  chi2 = fitter.GetChisquare()/Double_t(npoints);
134  }
135  if (TMath::Abs(par[1])<kTol) return -4;
136  if (TMath::Abs(par[2])<kTol) return -4;
137 
138  param[1] = T(par[1]/(-2.*par[2]));
139  param[2] = T(1./TMath::Sqrt(TMath::Abs(-2.*par[2])));
140  Double_t lnparam0 = par[0]+ par[1]* param[1] + par[2]*param[1]*param[1];
141  if ( lnparam0>307 ) return -4;
142  param[0] = TMath::Exp(lnparam0);
143 
144  return chi2;
145  }
146 
147  if (npoints == 2){
148  //use center of gravity for 2 points
149  meanCOG/=sumCOG;
150  rms2COG /=sumCOG;
151  param[0] = max;
152  param[1] = meanCOG;
153  param[2] = TMath::Sqrt(TMath::Abs(meanCOG*meanCOG-rms2COG));
154  chi2=-2.;
155  }
156  if ( npoints == 1 ){
157  meanCOG/=sumCOG;
158  param[0] = max;
159  param[1] = meanCOG;
160  param[2] = binWidth/TMath::Sqrt(12);
161  chi2=-1.;
162  }
163  return chi2;
164 
165  }

184  template <typename T>
185  StatisticsData getStatisticsData(const T* arr, const size_t nBins, const double xMin, const double xMax)
186  {
187  double mean = 0;
188  double rms2 = 0;
189  double sum = 0;
190  size_t npoints = 0;
191 
192  double binWidth = (xMax-xMin)/(double)nBins;
193 
194  StatisticsData data;
195  // in case something went wrong the COG is the histogram lower limit
196  data.mCOG = xMin;
197 
198  for (size_t ibin=0; ibin<nBins; ++ibin) {
199  double entriesI = (double)arr[ibin];
200  double xcenter = xMin+(ibin+0.5)*binWidth; // +0.5 to shift to bin centre
201  if ( entriesI>0 ) {
202  mean += xcenter*entriesI;
203  rms2 += xcenter*entriesI*xcenter;
204  sum += entriesI;
205  ++npoints;
206  }
207  }
208  if ( sum == 0 ) return data;
209  mean/=sum;
210 
211  data.mCOG = mean;
212  // exception in case of only one bin is filled
213  // set the standard deviation to bin width over sqrt(12)
214  rms2 /= sum;
215  if (npoints == 1) {
216  data.mStdDev = binWidth/std::sqrt(12.);
217  }
218  else {
219  data.mStdDev = std::sqrt(std::abs(rms2-mean*mean));
220  }
221 
222  data.mSum = sum;
223 
224  return data;
225  }

283  bool LTMUnbinned(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeep)
284  {
285  int nPoints = data.size();
286  std::vector<float> w(2 * nPoints);
287  int nKeep = nPoints * fracKeep;
288  if (nKeep > nPoints) {
289  nKeep = nPoints;
290  }
291  if (nKeep < 2) {
292  return false;
293  }
294  // sort in increasing order
295  SortData(data, index);
296  // build cumulants
297  double sum1 = 0.0;
298  double sum2 = 0.0;
299  for (int i = 0; i < nPoints; i++) {
300  double x = data[index[i]];
301  sum1 += x;
302  sum2 += x * x;
303  w[i] = sum1;
304  w[i + nPoints] = sum2;
305  }
306  double maxRMS = sum2 + 1e6;
307  params[0] = nKeep;
308  int limI = nPoints - nKeep + 1; // lowest possible bin to accept
309  for (int i = 0; i < limI; i++) {
310  const int limJ = i + nKeep - 1; // highest accepted bin
311  sum1 = static_cast<double>(w[limJ]) - static_cast<double>(i ? w[i - 1] : 0.);
312  sum2 = static_cast<double>(w[nPoints + limJ]) - static_cast<double>(i ? w[nPoints + i - 1] : 0.);
313  const double mean = sum1 / nKeep;
314  const double rms2 = sum2 / nKeep - mean * mean;
315  if (rms2 > maxRMS) {
316  continue;
317  }
318  maxRMS = rms2;
319  params[1] = mean;
320  params[2] = rms2;
321  params[5] = i;
322  params[6] = limJ;
323  }
324  //
325  if (params[2] < 0) {
326  LOG(error) << "Rounding error: RMS = " << params[2] << " < 0";
327  return false;
328  }
329  params[2] = std::sqrt(params[2]);
330  params[3] = params[2] / std::sqrt(params[0]); // error on mean
331  params[4] = params[3] / std::sqrt(2.0); // error on RMS
332  return true;
333  }

366 
367  if (!sorted) {
368  // sort in increasing order

370  } else {
371  // array is already sorted
372  for (int i = 0; i < nPoints; ++i) {
373  index[i] = i;
374  }
375  }
376  // build cumulants
377  double sum1 = 0.0;
378  double sum2 = 0.0;
379  for (int i = 0; i < nPoints; i++) {
380  double x = data[index[i]];
381  sum1 += x;
382  sum2 += x * x;
383  w[i] = sum1;
384  w[i + nPoints] = sum2;
385  }

387  int keepMin = fracKeepMin * nPoints;
388  if (keepMin > keepMax) {

390  }

392  //

403  sum1 = w[limJ] - (i ? w[i - 1] : 0.);
404  sum2 = w[limJ + nPoints] - (i ? w[nPoints + i - 1] : 0.);
405  const double mean = sum1 / keepN;
406  const double rms2 = sum2 / keepN - mean * mean;

412  params[2] = rms2;

414  params[6] = limJ;
415  }
416  if (maxRMS < sigTgt2) {
417  keepMin = keepN;
418  } else {
419  keepMax = keepN;
420  }
421  if (keepMin >= keepMax - 1) {
422  break;
423  }
424  }

426  params[3] = params[2] / std::sqrt(params[0]); // error on mean
