108 template <typename T>
109 const T* attachBranch(std::string const& name, FairMQChannel& channel, FairMQParts& parts)
110 {
111  auto mgr = FairRootManager::Instance();
112  // check if branch is present
113  if (mgr->GetBranchId(name) == -1) {
114  LOG(ERROR) << "Branch " << name << " not found";
115  return nullptr;
116  }
117  auto data = mgr->InitObjectAs<const T*>(name.c_str());
118  if (data) {
119  o2::Base::attachTMessage(*data, channel, parts);
120  }
121  return data;
122 }
