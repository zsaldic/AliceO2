92 void ShishKebabTrd1Module::Init(Double_t A, Double_t B)
93 {
94  // Define parameter module from parameters A,B from previous.
95  Double_t yl = (sb / 2) * TMath::Sin(mTheta) + (sa / 2) * TMath::Cos(mTheta) + sr, y = yl;
96  Double_t xl = (yl - B) / A; // y=A*x+B
97 
98  // Double_t xp1 = (fga/2. + fgb/2.*fgtanBetta)/(TMath::Sin(fTheta) + fgtanBetta*TMath::Cos(fTheta));
99  // printf(" xp1 %9.3f \n ", xp1);
100  // xp1 == xp => both methods give the same results - 3-feb-05
101  Double_t alpha = TMath::Pi() / 2. + sangle / 2;
102  Double_t xt =
103  (sa + sa2) * TMath::Tan(mTheta) * TMath::Tan(alpha) / (4. * (1. - TMath::Tan(mTheta) * TMath::Tan(alpha)));
104  Double_t yt = xt / TMath::Tan(mTheta), xp = TMath::Sqrt(xt * xt + yt * yt);
105  Double_t x = xl + xp;
106  mOK.Set(x, y);
107  // printf(" yl %9.3f | xl %9.3f | xp %9.3f \n", yl, xl, xp);
108 
109  // have to define A and B;
110  Double_t yCprev = sr + sa * TMath::Cos(mTheta);
111  Double_t xCprev = (yCprev - B) / A;
112  Double_t xA = xCprev + sa * TMath::Sin(mTheta), yA = sr;
113 
114  mThetaA = mTheta - sangle / 2.;
115  mA = TMath::Tan(mThetaA); // !!
116  mB = yA - mA * xA;
117 
118  DefineAllStuff();
119 }

121 void ShishKebabTrd1Module::DefineAllStuff()
122 {
123  // Define some parameters
124  // DefineName(mTheta);
125  // Centers of cells - 2X2 case
126  Double_t kk1 = (sa + sa2) / (2. * 4.); // kk1=kk2
127 
128  Double_t xk1 = mOK.X() - kk1 * TMath::Sin(mTheta);
129  Double_t yk1 = mOK.Y() + kk1 * TMath::Cos(mTheta) - sr;
130  mOK1.Set(xk1, yk1);
131 
132  Double_t xk2 = mOK.X() + kk1 * TMath::Sin(mTheta);
133  Double_t yk2 = mOK.Y() - kk1 * TMath::Cos(mTheta) - sr;
134  mOK2.Set(xk2, yk2);
135 
136  // Centers of cells - 3X3 case; Nov 9,2006
137  mOK3X3[1].Set(mOK.X(), mOK.Y() - sr); // coincide with module center
138 
139  kk1 = ((sa + sa2) / 4. + sa / 6.) / 2.;
140 
141  xk1 = mOK.X() - kk1 * TMath::Sin(mTheta);
142  yk1 = mOK.Y() + kk1 * TMath::Cos(mTheta) - sr;
143  mOK3X3[0].Set(xk1, yk1);
144 
145  xk2 = mOK.X() + kk1 * TMath::Sin(mTheta);
146  yk2 = mOK.Y() - kk1 * TMath::Cos(mTheta) - sr;
147  mOK3X3[2].Set(xk2, yk2);
148 
149  // May 15, 2006; position of module(cells) center face
150  mOB.Set(mOK.X() - sb / 2. * TMath::Cos(mTheta), mOK.Y() - sb / 2. * TMath::Sin(mTheta) - sr);
151  mOB1.Set(mOB.X() - sa / 4. * TMath::Sin(mTheta), mOB.Y() + sa / 4. * TMath::Cos(mTheta));
152  mOB2.Set(mOB.X() + sa / 4. * TMath::Sin(mTheta), mOB.Y() - sa / 4. * TMath::Cos(mTheta));
153  // Jul 30, 2007 - for taking into account a position of shower maximum
154  mThetaOB1 = mTheta - sangle / 4.; // ??
155  mThetaOB2 = mTheta + sangle / 4.;
156 
157  // Position of right/top point of module
158  // Gives the posibility to estimate SM size in z direction
159  Double_t xBottom = (sr - mB) / mA;
160  Double_t yBottom = sr;
161  mORB.Set(xBottom, yBottom);
162 
163  Double_t l = sb / TMath::Cos(sangle / 2.); // length of lateral module side
164  Double_t xTop = xBottom + l * TMath::Cos(TMath::ATan(mA));
165  Double_t yTop = mA * xTop + mB;
166  mORT.Set(xTop, yTop);
167 }

169 void ShishKebabTrd1Module::DefineFirstModule(const Int_t key)
170 {
171  // Define first module
172  if (key == 0) {
173  // theta in radians ; first object theta=pi/2.
174  mTheta = TMath::PiOver2();
175  mOK.Set(sa2 / 2., sr + sb / 2.); // position the center of module vs o
176 
177  // parameters of right line : y = A*z + B in system where zero point is IP.
178  mThetaA = mTheta - sangle / 2.;
179  mA = TMath::Tan(mThetaA);
180  Double_t xA = sa / 2. + sa2 / 2.;
181  Double_t yA = sr;
182  mB = yA - mA * xA;
183  } else if (key == 1) {
184  // theta in radians ; first object theta = 90-0.75 = 89.25 degree
185  mTheta = 89.25 * TMath::DegToRad();
186  Double_t al1 = sangle / 2.;
187  Double_t x = 0.5 * (sa * TMath::Cos(al1) + sb * TMath::Sin(al1));
188  Double_t y = 0.5 * (sb + sa * TMath::Sin(al1)) * TMath::Cos(al1);
189  mOK.Set(x, sr + y);
190  // parameters of right line : y = A*z + B in system where zero point is IP.
191  mThetaA = mTheta - sangle / 2.;
192  mA = TMath::Tan(mThetaA);
193  Double_t xA = sa * TMath::Cos(al1);
194  Double_t yA = sr;
195  mB = yA - mA * xA;
196  } else {
197  LOG(ERROR) << "key=" << key << " : wrong case \n";
198  assert(0);
199  }
200 
201  DefineAllStuff();
202 }
