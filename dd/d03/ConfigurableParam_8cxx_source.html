165  // Remove leading whitespace
166  auto ltrim = [](std::string src) {
167  return src.erase(0, src.find_first_not_of(' '));
168  };
169 
170  // Remove trailing whitespace
171  auto rtrim = [](std::string src) {
172  return src.erase(src.find_last_not_of(' ') + 1);
173  };
174 
175  // Remove leading/trailing whitespace
176  auto trimSpace = [ltrim, rtrim](std::string src) {
177  return ltrim(rtrim(src));
178  };
179 
180  // Split a given string on a delim character, return vector of tokens
181  // If trim is true, then also remove leading/trailing whitespace of each token.
182  auto splitString = [trimSpace](const std::string& src, char delim, bool trim = false) {
183  std::stringstream ss(src);
184  std::string token;
185  std::vector<std::string> tokens;
186 
187  while (std::getline(ss, token, delim)) {
188  token = (trim ? trimSpace(token) : token);
189  if (!token.empty()) {
190  tokens.push_back(std::move(token));
191  }
192  }
193 
194  return tokens;
195  };
196 
197  // Take a vector of strings with elements of form a=b, and
198  // return a vector of pairs with each pair of form <a, b>
199  auto getKeyValPairs = [trimSpace, splitString](std::vector<std::string>& tokens) {
200  std::vector<std::pair<std::string, std::string>> pairs;
201 
202  for (auto& token : tokens) {
203  auto keyval = splitString(token, '=');
204  if (keyval.size() != 2) {
205  LOG(FATAL) << "Illegal command-line key/value string: " << token;
206  continue;
207  }
208 
209  std::pair<std::string, std::string> pair = std::make_pair(keyval[0], trimSpace(keyval[1]));
210  pairs.push_back(pair);
211  }
212 
213  return pairs;
214  };
215 
216  // Simple check that the string starts/ends with an open square bracket
217  auto isArray = [](std::string& el) {
218  return (el.at(0) == '[') && (el.at(el.size() - 1) == ']');
219  };
220 
221  // Does the given string key exist in the boost property tree?
222  auto keyExists = [](std::string key) {
223  return sPtree->get_optional<std::string>(key).is_initialized();
224  };
225 
226  // Find the maximum index of a given key with array value.
227  // We store string keys for arrays as a[0]...a[size_of_array]
228  auto maxIndex = [keyExists](std::string baseName) {
229  bool isFound = true;
230  int index = -1;
231  do {
232  index++;
233  std::string key = baseName + "[" + std::to_string(index) + "]";
234  isFound = keyExists(key);
235  } while (isFound);
236 
237  return index;
238  };
239 
240  // ---- end of helper functions --------------------
241 
242  // Command-line string is a ;-separated list of key=value params
243  auto params = splitString(configstring, ';', true);
244 
245  // Now split each key=value string into its <key, value> parts
246  auto keyValues = getKeyValPairs(params);
247 
248  // Take an array of param key/value pairs
249  // and update the storage map for each of them by calling setValue.
250  // For string/scalar types this is simple.
251  // For array values we need to iterate over each array element
252  // and call setValue on the element, using an appropriately constructed key.
253  for (auto& keyValue : keyValues) {
254  std::string key = keyValue.first;
255  std::string value = trimSpace(keyValue.second);
256 
257  if (isArray(value)) {
258  // We need to check how big the array is, by increasing key[0], key[1],...
259  // and checking for this key in the sPtree until we no longer find a match.
260 
261  // We remove the lead/trailing square bracket
262  value.erase(0, 1); // remove leading bracket
263  value.pop_back(); // remove trailing bracket
264 
265  auto elems = splitString(value, ',', true);
266  /*
267  auto expectedLen = maxIndex(key);
268  int elemsLen = elems.size();
269 
270  if (expectedLen != elemsLen) {
271  LOG(FATAL) << "Cannot configure " << key
272  << " with a different array size (expected length "
273  << expectedLen << ", found " << elemsLen << ")";
274  continue;
275  }
276  */
277  // TODO:
278  // 1. Should not assume each array element is a scalar/string. We may need to recurse.
279  // 2. Should not assume each array element - even if not complex - is correctly written. Validate.
280  // 3. Validation should include finding same types as in provided defaults.
281 
282  for (int i = 0; i < elems.size(); ++i) {
283  std::string indexKey = key + "[" + std::to_string(i) + "]";
284  setValue(indexKey, elems[i]);
285  }
286 
287  continue;
288  }
289 
290  // Value is not an array, presume a string/scalar type
291  if (!keyExists(key)) {
292  LOG(FATAL) << "Inexistant ConfigurableParam key: " << key;
293  continue;
294  }
295 
296  assert(sKeyToStorageMap->find(key) != sKeyToStorageMap->end());
297 
298  // TODO: this will trap complex types like maps and structs.
299  // These need to be broken into their own cases, so that we only
300  // get strings and scalars here.
301  setValue(key, value);
302  }
