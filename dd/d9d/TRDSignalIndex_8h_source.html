134 void TRDSignalIndex::addIndexRC(const Int_t row, const Int_t col)
135 {
136  //
137  // Adds RC combination to array
138  //
139 
140  const Int_t num = row * mNcols + col;
141  if (mBoolIndex[num])
142  return;
143  mBoolIndex[num] = kTRUE;
144  mSortedIndex[mCountRC].s.col = col;
145  mSortedIndex[mCountRC].s.row = row;
146  mCountRC++;
147 }

149 Bool_t TRDSignalIndex::nextRCIndex(Int_t& row, Int_t& col)
150 {
151  //
152  // Returns next used RC combination
153  //
154 
155  if (!isAllocated())
156  return kFALSE;
157 
158  if (mSortedIndex[mPositionRC].rc > -1) {
159  row = mCurrRow = mSortedIndex[mPositionRC].s.row;
160  col = mCurrCol = mSortedIndex[mPositionRC].s.col;
161  mPositionRC++;
162  return kTRUE;
163  } else
164  return checkSorting(row, col);
165 }
