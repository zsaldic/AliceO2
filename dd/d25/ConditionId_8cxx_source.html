49 ConditionId *ConditionId::makeFromString(const TString &idString)
50 {
51  // constructor from string
52  // string has the format as the output of ConditionId::ToString:
53  // path: "TRD/Calib/PIDLQ"; run range: [0,999999999]; version: v0_s0
54 
55  ConditionId *id = new ConditionId("a/b/c", -1, -1, -1, -1);
56 
57  TObjArray *arr1 = idString.Tokenize(';');
58  TIter iter1(arr1);
59  TObjString *objStr1 = nullptr;
60  while ((objStr1 = dynamic_cast<TObjString *>(iter1.Next()))) {
61  TString buff(objStr1->GetName());
62 
63  if (buff.Contains("path:")) {
64  TString path(buff(buff.First('\"') + 1, buff.Length() - buff.First('\"') - 2));
65  id->setPath(path.Data());
66 
67  } else if (buff.Contains("run range:")) {
68  TString firstRunStr(buff(buff.Index('[') + 1, buff.Index(',') - buff.Index('[') - 1));
69  TString lastRunStr(buff(buff.Index(',') + 1, buff.Index(']') - buff.Index(',') - 1));
70  id->setIdRunRange(firstRunStr.Atoi(), lastRunStr.Atoi());
71 
72  } else if (buff.Contains("version:")) {
73  if (buff.Contains("_s")) {
74  TString versStr(buff(buff.Last('v') + 1, buff.Index('_') - buff.Last('v') - 1));
75  TString subVersStr(buff(buff.Last('s') + 1, buff.Length() - buff.Last('s') - 1));
76  id->setVersion(versStr.Atoi());
77  id->setSubVersion(subVersStr.Atoi());
78  } else {
79  TString versStr(buff(buff.Last('v') + 1, buff.Length() - buff.Last('v') - 1));
80  id->setVersion(versStr.Atoi());
81  }
82  }
83  }
84 
85  delete arr1;
86 
87  return id;
88 }

104 Bool_t ConditionId::isEqual(const TObject *obj) const
105 {
106  // check if this id is equal to other id (compares path, run range, versions)
107 
108  if (this == obj) {
