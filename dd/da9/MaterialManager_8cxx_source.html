93 void MaterialManager::Mixture(const char* modname, Int_t imat, const char* name, Float_t* a, Float_t* z, Float_t dens,
94  Int_t nlmat, Float_t* wmat)
95 {
96  TString uniquename = modname;
97  uniquename.Append("_");
98  uniquename.Append(name);
99 
100  if (TVirtualMC::GetMC()) {
101  // Check this!!!
102  int kmat = -1;
103  TVirtualMC::GetMC()->Mixture(kmat, uniquename.Data(), a, z, dens * mDensityFactor, nlmat, wmat);
104  mMaterialMap[modname][imat] = kmat;
105  insertMaterialName(uniquename.Data(), kmat);
106 
107  } else {
108  auto uid = gGeoManager->GetListOfMaterials()->GetSize();
109  if (nlmat < 0) {
110  nlmat = -nlmat;
111  Double_t amol = 0;
112  Int_t i;
113  for (i = 0; i < nlmat; i++) {
114  amol += a[i] * wmat[i];
115  }
116  for (i = 0; i < nlmat; i++) {
117  wmat[i] *= a[i] / amol;
118  }
119  }
120  auto mix = gGeoManager->Mixture(uniquename.Data(), a, z, dens * mDensityFactor, nlmat, wmat, uid);
121  mMaterialMap[modname][imat] = uid;
122  insertMaterialName(uniquename.Data(), uid);
123  }
124 }

169 void MaterialManager::Cut(ESpecial special, int globalindex, ECut cut, Float_t val)
170 {
171  // this check is needed, in principal only for G3, otherwise SegFault
172  if (val < 0.) {
173  return;
174  }
175  auto it = mCutIDToName.find(cut);
176  if (it == mCutIDToName.end()) {
177  return;
178  }
179  if (special == ESpecial::kFALSE) {
180  mDefaultCutMap[cut] = val;
182  TVirtualMC::GetMC()->SetCut(it->second, val);
183  } else if (mApplySpecialCuts) {
184  mMediumCutMap[globalindex][cut] = val;
185  TVirtualMC::GetMC()->Gstpar(globalindex, it->second, val);
186  }
187 }
