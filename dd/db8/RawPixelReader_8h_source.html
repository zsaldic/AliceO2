272  int digits2raw(const std::vector<o2::ITSMFT::Digit>& digiVec, int from, int ndig, const o2::InteractionRecord& bcData,
273  PayLoadCont& sink, uint8_t ruSWMin = 0, uint8_t ruSWMax = 0xff)
274  {
275  // convert vector of digits to binary vector (no reset is applied)
276  constexpr uint16_t DummyChip = 0xffff;
277  int nDigTot = digiVec.size();
278  assert(from < nDigTot);
279  int oldsize = sink.getSize(), last = (from + ndig <= nDigTot) ? from + ndig : nDigTot;
280  int curChipID = DummyChip; // currently processed SW chip id
281 
282  o2::ITSMFT::ChipPixelData chipData; // data of currently processed chip
283 
284  mRUEncode.clear();
285  chipData.clear();
286  chipData.setChipID(DummyChip);
287  int emptyRUID = 0;
288  bool ignore = false;
289 
290  mRUEncode.bcData = bcData;
291  for (int id = from; id < last; id++) {
292  const auto& dig = digiVec[id];
293  if (dig.getChipIndex() != curChipID) { // new chip data start
294 
295  // since a new chip data start, convert digits of previously processed chip to Alpide format in the temp. buffer
296  if (curChipID != DummyChip && !ignore) {
297  convertChipData(mRUEncode, chipData);
298  }
299  curChipID = dig.getChipIndex();
300  MAP.getChipInfoSW(curChipID, mRUEncode.chInfo);
301 
302  if (mRUEncode.chInfo.ru != mRUEncode.id) { // new RU data starts, flush the previous one
303  if (mRUEncode.id != -1) {
304  for (; emptyRUID < mRUEncode.id; emptyRUID++) { // flush empty raw data for RUs w/o digits
305  if (emptyRUID >= ruSWMin && emptyRUID <= ruSWMax) { // do we want this RU?
306  flushEmptyRU(emptyRUID, mRUEncode.bcData, sink);
307  }
308  }
309  emptyRUID = mRUEncode.id + 1; // flushing of next empty RU will start from this one
310  if (!ignore) {
311  flushRUData(mRUEncode, sink); // flush already converted RU data
312  }
313  }
314  mRUEncode.id = mRUEncode.chInfo.ru; // and update with new RU identifier
315  mRUEncode.type = mRUEncode.chInfo.ruType;
316  mRUEncode.nChips = MAP.getNChipsOnRUType(mRUEncode.chInfo.ruType);
317  mRUEncode.nCables = MAP.getNCablesOnRUType(mRUEncode.chInfo.ruType);
318  ignore = mRUEncode.id < ruSWMin || mRUEncode.id > ruSWMax;
319  }
320  chipData.setChipID(mRUEncode.chInfo.chOnRU->chipOnModuleHW); // HW chip ID in module
321  }
322  if (!ignore) { // add only if the RU is not to be rejected
323  chipData.getData().emplace_back(&dig); // add new digit to the container
324  }
325  }
326  if (!ignore) {
327  convertChipData(mRUEncode, chipData); // finish what is left in the buffer
328  }
329  for (; emptyRUID < mRUEncode.id; emptyRUID++) { // flush empty raw data for RUs w/o digits
330  if (emptyRUID >= ruSWMin && emptyRUID <= ruSWMax) { // do we want this RU?
331  flushEmptyRU(emptyRUID, mRUEncode.bcData, sink);
332  }
333  }
334  if (!ignore) {
335  flushRUData(mRUEncode, sink);
336  }
337  // flush empty raw data for RUs w/o digits
338  for (emptyRUID = mRUEncode.chInfo.ru + 1; emptyRUID < MAP.getNRUs(); emptyRUID++) {
339  if (emptyRUID >= ruSWMin && emptyRUID <= ruSWMax) { // do we want this RU?
340  flushEmptyRU(emptyRUID, mRUEncode.bcData, sink);
341  }
342  }
343 
344  return last - from;
345  }

442  void flushRUData(RUEncodeData& ruData, PayLoadCont& sink)
443  {
444  constexpr uint8_t zero16[o2::ITSMFT::GBTPaddedWordLength] = { 0 }; // to speedup padding
445  if (ruData.id < 0) {
446  return;
447  }
448  convertEmptyChips(ruData, ruData.nChips); // fill empty chips up to the last chip of the RU
449 
450  // calculate number of GBT words needed to store this data (with 9 bytes per GBT word)
451  int nGBTWordsNeeded = 0;
452  for (int i = 0; i < ruData.nCables; i++) {
453  int start = ruData.cableEntryInRaw[i];
454  int end = ruData.cableEndInRaw[i];
455  int nb = end > start ? end - start : 0;
456  nGBTWordsNeeded += nb ? 1 + (nb - 1) / 9 : 0;
457  }
458  //
459  // prepare RDH
460  // 4x128 bit, represented as 8 64-bit words
461  o2::header::RAWDataHeader rdh;
462  rdh.headerSize = 0x40; // 4*128 bits;
463  rdh.feeId = MAP.RUSW2FEEId(ruData.id, 0); // write on link 0 always
464  rdh.triggerOrbit = rdh.heartbeatOrbit = ruData.bcData.orbit;
465  rdh.triggerBC = rdh.heartbeatBC = ruData.bcData.bc;
466  rdh.triggerType = o2::trigger::PhT;
467  rdh.detectorField = MAP.getRUDetectorField();
468  rdh.par = 0;
469  o2::ITSMFT::GBTDataHeader gbtHeader(0, MAP.getCablesOnRUType(ruData.type));
470  o2::ITSMFT::GBTDataTrailer gbtTrailer;
471 
472  // max real payload words (accounting for GBT header and trailer) per packet
473  // Note: for this estimate we use GBTPaddedWordLength rather than the actual mGBTWordSize
474  int maxGBTWordsPerPacket = (MaxGBTPacketBytes - rdh.headerSize) / o2::ITSMFT::GBTPaddedWordLength - 2;
475 
476  if (sink.getFreeCapacity() < 2 * MaxGBTPacketBytes) { // make sure there is enough capacity
477  sink.expand(sink.getCapacity() + 10 * MaxGBTPacketBytes);
478  }
479 
480  rdh.blockLength = 0xffff; //MaxGBTPacketBytes; ITS does not fill it // total packet size: always use max size ? in bits ?
481  rdh.memorySize = rdh.headerSize + (nGBTWordsNeeded + 2) * mGBTWordSize; // update remaining size
482  if (rdh.memorySize > MaxGBTPacketBytes) {
483  rdh.memorySize = MaxGBTPacketBytes;
484  }
485  rdh.offsetToNext = mImposeMaxPage ? MaxGBTPacketBytes : rdh.memorySize;
486  rdh.stop = 0;
487 
488  sink.addFast(reinterpret_cast<uint8_t*>(&rdh), rdh.headerSize); // write RDH for current packet
489 
490  gbtHeader.setPacketID(rdh.pageCnt);
491  sink.addFast(gbtHeader.getW8(), mGBTWordSize); // write GBT header for current packet
492 
493  if (mVerbose) {
494  LOG(INFO) << "Flushing RU data";
495  printRDH(rdh);
496  gbtHeader.printX(mPadding128);
497  }
498 
499  int nGBTWordsInPacket = 0;
500  do {
501  for (int icab = 0; icab < ruData.nCables; icab++) {
502  int &start(ruData.cableEntryInRaw[icab]), &end(ruData.cableEndInRaw[icab]), nb = end - start;
503  if (nb > 0) { // write 80b word only if there is something to write
504  if (nb > 9) {
505  nb = 9;
506  }
507  int gbtWordStart = sink.getSize(); // beginning of the current GBT word in the sink
508  sink.add(ruData.buffer.data() + start, nb);
509  // fill the rest of the GBT word by 0
510  sink.add(zero16, mGBTWordSize - nb);
511  sink[gbtWordStart + 9] = MAP.getGBTHeaderRUType(ruData.type, ruData.cableHWID[icab]); // set cable flag
512  start += nb;
513  nGBTWordsNeeded--;
514  if (mVerbose) {
515  ((GBTData*)(sink.data() + gbtWordStart))->printX(mPadding128);
516  }
517  if (++nGBTWordsInPacket == maxGBTWordsPerPacket) { // check if new GBT packet must be created
518  break;
519  }
520  } // storing data of single cable
521  } // loop over cables
522 
523  if (nGBTWordsNeeded && nGBTWordsInPacket >= maxGBTWordsPerPacket) {
524  // more data to write, write trailer and add new GBT packet
525  sink.add(gbtTrailer.getW8(), mGBTWordSize); // write GBT trailer for current packet
526  if (mVerbose) {
527  gbtTrailer.printX(mPadding128);
528  }
529  rdh.pageCnt++; // flag new page
530  rdh.stop = nGBTWordsNeeded < maxGBTWordsPerPacket; // flag if this is the last packet of multi-packet
531  rdh.blockLength = 0xffff; // (nGBTWordsNeeded % maxGBTWordsPerPacket + 2) * mGBTWordSize; // record payload size
532  // update remaining size, using padded GBT words (as CRU writes)
