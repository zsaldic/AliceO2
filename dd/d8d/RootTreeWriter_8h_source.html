298  // specialization for binary buffers indicated by const char*
299  // a variable binary data branch is written, the size of each entry is stored in a size branch
300  template <typename T, typename std::enable_if_t<std::is_same<T, const char*>::value, int> = 0>
301  TBranch* createBranch(TTree* tree, const char* name, size_t branchIdx)
302  {
303  // size variable to write the size of the data branch
304  std::get<2>(mStore.at(branchIdx)) = 1;
305  // branch name of the size branch
306  std::string sizeBranchName = std::string(name) + "Size";
307  // leaf list: one leaf of type unsinged int
308  std::string leafList = sizeBranchName + "/i";
309  std::get<1>(mStore.at(branchIdx)) = tree->Branch(sizeBranchName.c_str(), &(std::get<2>(mStore.at(branchIdx))), leafList.c_str());
310  return tree->Branch(name, &(std::get<0>(mStore.at(branchIdx))));

314  void setupInstance(std::vector<BranchSpec>& specs, TTree* tree)
315  {
316  // recursing through the tree structure by simply using method of the previous type,
317  // i.e. the base class method.
318  PrevT::setupInstance(specs, tree);
319  constexpr size_t SpecIndex = STAGE - 1;
320  specs[SpecIndex].classinfo = TClass::GetClass(typeid(value_type));
321  if (std::is_same<value_type, const char*>::value == false && std::is_fundamental<value_type>::value == false &&
322  specs[SpecIndex].classinfo == nullptr) {
323  // for all non-fundamental types but the special case for binary chunks, a dictionary is required
324  // FIXME: find a reliable way to check that the type has been specified in the LinkDef
325  // Only then the required functionality for streaming the type to the branch is available.
326  // If e.g. a standard container of some ROOT serializable type has not been specified in
327  // LinkDef, the functionality is not available and the attempt to stream will simply crash.
328  // Unfortunately, a class info object can be extracted for the type, so this check does not help
329  throw std::runtime_error(std::to_string(SpecIndex) + ": no dictionary available for non-fundamental type " + typeid(value_type).name());
330  }
331  size_t branchIdx = 0;
332  mStore.resize(specs[SpecIndex].names.size());
333  for (auto const& name : specs[SpecIndex].names) {
334  specs[SpecIndex].branches.at(branchIdx) = createBranch<value_type>(tree, name.c_str(), branchIdx);
335  LOG(INFO) << SpecIndex << ": branch " << name << " set up";
336  branchIdx++;
337  }
338  }

364  // specialization for binary buffers using const char*
365  // this writes both the data branch and a size branch
366  template <typename T, typename std::enable_if_t<std::is_same<T, BinaryBranchStoreType<char>>::value, int> = 0>
367  void fillData(InputContext& context, const char* key, TBranch* branch, size_t branchIdx)
368  {
369  auto data = context.get<gsl::span<char>>(key);
370  std::get<2>(mStore.at(branchIdx)) = data.size();
371  std::get<1>(mStore.at(branchIdx))->Fill();
372  std::get<0>(mStore.at(branchIdx)).resize(data.size());
373  memcpy(std::get<0>(mStore.at(branchIdx)).data(), data.data(), data.size());

377  // process previous stage and this stage
378  void process(InputContext& context, std::vector<BranchSpec>& specs)
379  {
380  // recursing through the tree structure by simply using method of the previous type,
381  // i.e. the base class method.
382  PrevT::process(context, specs);
383  constexpr size_t SpecIndex = STAGE - 1;
384  BranchSpec const& spec = specs[SpecIndex];
385  // loop over all defined inputs
386  for (auto const& key : spec.keys) {
387  auto dataref = context.get(key.c_str());
388  size_t branchIdx = 0;
389  if (spec.getIndex) {
390  branchIdx = spec.getIndex(dataref);
391  if (branchIdx == ~(size_t)0) {
392  // this indicates skipping
393  continue;
394  }
395  }
396  fillData<store_type>(context, key.c_str(), spec.branches.at(branchIdx), branchIdx);
397  }
398  }

406  template <typename BASE, typename T, typename... Args>
407  std::enable_if_t<std::is_base_of<BranchDef<typename T::type, typename T::key_type, typename T::key_extractor>, T>::value, std::unique_ptr<TreeStructureInterface>>
408  createTreeStructure(T def, Args&&... args)
409  {
410  mBranchSpecs.push_back({ {}, { def.branchName } });
411  auto& spec = mBranchSpecs.back();
412 
413  // extract the internal keys for the list of provided input definitions
414  size_t idx = 0;
415  spec.keys.resize(def.keys.size());
416  std::generate(spec.keys.begin(), spec.keys.end(), [&def, &idx] { return T::key_extractor::asString(def.keys[idx++]); });
417  mBranchSpecs.back().branches.resize(def.nofBranches, nullptr);
418  // the number of branches has to match the number of inputs but can be larger depending
419  // on the exact functionality provided with the getIndex callback. In any case, the
420  // callbacks only need to be propagated if multiple branches are defined
421  assert(def.nofBranches >= spec.keys.size());
422  // a getIndex function makes only sense if there are multiple branches
423  assert(def.nofBranches == 1 || def.getIndex);
424  if (def.nofBranches > 1) {
425  assert(def.getIndex && def.getName);
426  mBranchSpecs.back().getIndex = def.getIndex;
427  mBranchSpecs.back().getName = def.getName;
428  mBranchSpecs.back().names.resize(def.nofBranches);
429 
430  // fill the branch names by calling the getName callback
431  idx = 0;
432  std::generate(mBranchSpecs.back().names.begin(), mBranchSpecs.back().names.end(),
433  [&def, &idx]() { return def.getName(def.branchName, idx++); });
434  }
435  using type = TreeStructureElement<typename T::type, BASE>;
436  return std::move(createTreeStructure<type>(std::forward<Args>(args)...));
437  }
