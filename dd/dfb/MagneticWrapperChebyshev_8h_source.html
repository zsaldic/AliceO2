427 inline void MagneticWrapperChebyshev::cylindricalToCartesianCylB(const Double_t *rphiz, const Double_t *brphiz,
428  Double_t *bxyz)
429 {
430  Double_t btr = TMath::Sqrt(brphiz[0] * brphiz[0] + brphiz[1] * brphiz[1]);
431  Double_t psiPLUSphi = TMath::ATan2(brphiz[1], brphiz[0]) + rphiz[1];
432  bxyz[0] = btr * TMath::Cos(psiPLUSphi);
433  bxyz[1] = btr * TMath::Sin(psiPLUSphi);
434  bxyz[2] = brphiz[2];
435 }

439  Double_t *bxyz)

442  Double_t phiPLUSpsi = TMath::ATan2(xyz[1], xyz[0]) + TMath::ATan2(brphiz[1], brphiz[0]);
443  bxyz[0] = btr * TMath::Cos(phiPLUSpsi);

450  Double_t *brphiz)
451 {
452  Double_t btr = TMath::Sqrt(bxyz[0] * bxyz[0] + bxyz[1] * bxyz[1]);
453  Double_t psiMINphi = TMath::ATan2(bxyz[1], bxyz[0]) - TMath::ATan2(xyz[1], xyz[0]);
454 
455  brphiz[0] = btr * TMath::Cos(psiMINphi);
