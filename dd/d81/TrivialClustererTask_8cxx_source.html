55 {
56  FairRootManager* mgr = FairRootManager::Instance();
57  if (!mgr) {
58  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
59  return kERROR;
60  }
61 
62  mDigitsArray = mgr->InitObjectAs<const std::vector<o2::ITSMFT::Digit>*>("ITSDigit");
63  if (!mDigitsArray) {
64  LOG(ERROR) << "ITS points not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
65  return kERROR;
66  }
67 
68  // Register output container
69  mgr->RegisterAny("ITSCluster", mClustersArray, kTRUE);
70 
71  // Register MC Truth container
72  if (mClsLabels)

74 
75  GeometryTGeo* geom = GeometryTGeo::Instance();
76  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L)); // make sure T2L matrices are loaded
77  mGeometry = geom;
78  mTrivialClusterer.setGeometry(geom);
79  mTrivialClusterer.setMCTruthContainer(mClsLabels);
