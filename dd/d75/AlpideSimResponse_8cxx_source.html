31 void AlpideSimResponse::initData()
32 {
33  /*
34  * read grid parameters and load data
35  */
36  if (mData.size()) {
37  cout << "Object already initialized" << endl;
38  print();
39  return;
40  }
41 
42  const float kTiny = 1e-6; // to check 0 values
43 
44  // if needed, append path with slash
45  if (mDataPath.length() && mDataPath.back() != '/') {
46  mDataPath.push_back('/');
47  }
48  mDataPath = gSystem->ExpandPathName(mDataPath.data());
49  string inpfname = mDataPath + mGridColName;
50  std::ifstream inpGrid;
51 
52  // read X grid
53  inpGrid.open(inpfname, std::ifstream::in);
54  if (inpGrid.fail()) {
55  LOG(FATAL) << "Failed to open file " << inpfname << FairLogger::endl;
56  }
57 
58  while (inpGrid >> mStepInvCol && inpGrid.good()) {
59  mNBinCol++;
60  }
61 
62  if (!mNBinCol || mStepInvCol < kTiny) {
63  LOG(FATAL) << "Failed to read X(col) binning from " << inpfname << FairLogger::endl;
64  }
65  mMaxBinCol = mNBinCol - 1;
66  mStepInvCol = mMaxBinCol / mStepInvCol; // inverse of the X bin width
67  inpGrid.close();
68 
69  // read Y grid
70  inpfname = mDataPath + mGridRowName;
71  inpGrid.open(inpfname, std::ifstream::in);
72  if (inpGrid.fail()) {
73  LOG(FATAL) << "Failed to open file " << inpfname << FairLogger::endl;
74  }
75 
76  while (inpGrid >> mStepInvRow && inpGrid.good())
77  mNBinRow++;
78  if (!mNBinRow || mStepInvRow < kTiny) {
79  LOG(FATAL) << "Failed to read Y(row) binning from " << inpfname << FairLogger::endl;
80  }
81  mMaxBinRow = mNBinRow - 1;
82  mStepInvRow = mMaxBinRow / mStepInvRow; // inverse of the Row bin width
83  inpGrid.close();
84 
85  // load response data
86  int nz = 0;
87  size_t cnt = 0;
88  float val, gx, gy, gz;
89  int lost, untrck, dead, nele;
90  size_t dataSize = 0;
91  mDptMax = -2.e9;
92  mDptMin = 2.e9;
93  const int npix = AlpideRespSimMat::getNPix();
94 
95  for (int ix = 0; ix < mNBinCol; ix++) {
96  for (int iy = 0; iy < mNBinRow; iy++) {
97  inpfname = composeDataName(ix, iy);
98  inpGrid.open(inpfname, std::ifstream::in);
99  if (inpGrid.fail()) {
100  LOG(FATAL) << "Failed to open file " << inpfname << FairLogger::endl;
101  }
102  inpGrid >> nz;
103  if (cnt == 0) {
104  mNBinDpt = nz;
105  dataSize = mNBinCol * mNBinRow * mNBinDpt;
106  mData.reserve(dataSize); // reserve space for data
107  } else if (nz != mNBinDpt) {
108  LOG(FATAL) << "Mismatch in Nz slices of bin X(col): " << ix << " Y(row): " << iy
109  << " wrt bin 0,0. File " << inpfname << FairLogger::endl;
110  }
111 
112  // load data
113  for (int iz = 0; iz < nz; iz++) {
114  AlpideRespSimMat mat;
115 
116  std::array<float, AlpideRespSimMat::MatSize>* arr = mat.getArray();
117  for (int ip = 0; ip < npix * npix; ip++) {
118  inpGrid >> val;
119  (*arr)[ip] = val;
120  cnt++;
121  }
122  inpGrid >> lost >> dead >> untrck >> nele >> gx >> gy >> gz;
123 
124  if (inpGrid.bad()) {
125  LOG(FATAL) << "Failed reading data for depth(Z) slice " << iz << " from "
126  << inpfname << FairLogger::endl;
127  }
128  if (!nele) {
129  LOG(FATAL) << "Wrong normalization Nele=" << nele << "for depth(Z) slice "
130  << iz << " from " << inpfname
131  << FairLogger::endl;
132  }
133 
134  if (mDptMax < -1e9)
135  mDptMax = gz;
136  if (mDptMin > gz)
137  mDptMin = gz;
138 
139  // normalize
140  float norm = 1.f / nele;
141  for (int ip = npix * npix; ip--;)
142  (*arr)[ip] *= norm;
143  mData.push_back(mat); // store in the final container
144  } // loop over z
145 
146  inpGrid.close();
147 
148  } // loop over y
149  } // loop over x
150 
151  // final check
152  if (dataSize != mData.size()) {
153  LOG(FATAL) << "Mismatch between expected " << dataSize << " and loaded " << mData.size()
154  << " number of bins" << FairLogger::endl;
155  }
156 
157  // normalize Dpt boundaries
158 
159  mStepInvCol /= micron2cm;
160  mStepInvRow /= micron2cm;
161 
162  mDptMin *= micron2cm;
163  mDptMax *= micron2cm;
164  mStepInvDpt = (mNBinDpt - 1) / (mDptMax - mDptMin);
165  mDptMin -= 0.5 / mStepInvDpt;
166  mDptMax += 0.5 / mStepInvDpt;
167  mDptShift = 0.5 * (mDptMax + mDptMin);
168  print();
169 }

185 string AlpideSimResponse::composeDataName(int colBin, int rowBin)
186 {
187  /*
188  * compose the file-name to read data for bin colBin,rowBin
189  */
190 
191  // ugly but safe way to compose the file name
192  float vcol = colBin / mStepInvCol, vrow = rowBin / mStepInvRow;
193  size_t size = snprintf(nullptr, 0, mColRowDataFmt.data(), vcol, vrow) + 1;
194  unique_ptr<char[]> tmp(new char[size]);
195  snprintf(tmp.get(), size, mColRowDataFmt.data(), vcol, vrow);
196  return mDataPath + string(tmp.get(), tmp.get() + size - 1);
197 }

200 bool AlpideSimResponse::getResponse(float vRow, float vCol, float vDepth, AlpideRespSimMat& dest) const
201 {
202  /*
203  * get linearized NPix*NPix matrix for response at point vRow(sensor local X, along row)
204  * vCol(sensor local Z, along columns) and vDepth (sensor local Y, i.e. depth)
205  */
206  if (!mNBinDpt) {
207  LOG(FATAL) << "response object is not initialized" << FairLogger::endl;
208  }
209  bool flipCol = false, flipRow = true;
210  if (vDepth < mDptMin || vDepth > mDptMax) return false;
211  if (vCol < 0) {
212  vCol = -vCol;
213  flipCol = true;
214  }
215  if (vCol > mColMax)
216  return false;
217  if (vRow < 0) {
218  vRow = -vRow;
219  flipRow = false;
220  }
221  if (vRow > mRowMax)
222  return false;
223 
224  size_t bin = getDepthBin(vDepth) + mNBinDpt * (getRowBin(vRow) + mNBinRow * getColBin(vCol));
225  if (bin >= mData.size()) {
226  // this should not happen
227  LOG(FATAL) << "requested bin " << bin << "row/col/depth: " << getRowBin(vRow) << ":" << getColBin(vCol)
228  << ":" << getDepthBin(vDepth) << ")"
229  << ">= maxBin " << mData.size()
230  << " for X(row)=" << vRow << " Z(col)=" << vCol << " Y(depth)=" << vDepth << FairLogger::endl;
231  }
232  // printf("bin %d %d %d\n",getColBin(vCol),getRowBin(vRow),getDepthBin(vDepth));
233  // return &mData[bin];
234  dest.adopt(mData[bin], flipRow, flipCol);
235  return true;
236 }

240 {
241  /*
242  * get linearized NPix*NPix matrix for response at point vRow(sensor local X, along row)
243  * vCol(sensor local Z, along columns) and vDepth (sensor local Y, i.e. depth)

245  if (!mNBinDpt) {

249  return nullptr;
250  if (vCol < 0) {
251  vCol = -vCol;
252  flipCol = true;
253  } else {
254  flipCol = false;
255  }
256  if (vCol > mColMax)
257  return nullptr;
258  if (vRow < 0) {
259  vRow = -vRow;
260  flipRow = false;
261  } else {
262  flipRow = true;
263  }
264  if (vRow > mRowMax)
265  return nullptr;
266 
267  size_t bin = getDepthBin(vDepth) + mNBinDpt * (getRowBin(vRow) + mNBinRow * getColBin(vCol));
