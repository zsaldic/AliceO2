64 void XmlHandler::OnStartElement(const char *name, const TList *attributes)
65 {
66  // when a new XML element is found, it is processed here
67 
68  // set the current system if necessary
69  TString strName(name);
70  LOG(DEBUG) << "name = " << strName.Data();
71  Int_t startRun = -1;
72  Int_t endRun = -1;
73  TXMLAttr *attr;
74  TIter next(attributes);
75  while ((attr = (TXMLAttr *) next())) {
76  TString attrName = attr->GetName();
77  LOG(DEBUG) << "Name = " << attrName.Data();
78  if (attrName == "StartIdRunRange") {
79  startRun = (Int_t) (((TString) (attr->GetValue())).Atoi());
80  LOG(DEBUG) << "startRun = " << startRun;
81  }
82  if (attrName == "EndIdRunRange") {
83  endRun = (Int_t) (((TString) (attr->GetValue())).Atoi());
84  LOG(DEBUG) << "endRun = " << endRun;
85  }
86  if (attrName == "OCDBFolder") {
87  if (mRun >= startRun && mRun <= endRun && startRun != -1 && endRun != -1) {
88  mOCDBFolder = (TString) (attr->GetValue());
89  LOG(DEBUG) << "OCDBFolder = " << mOCDBFolder.Data();
90  mStartIdRunRange = startRun;
91  mEndIdRunRange = endRun;
92  }
93  }
94  }
95  return;
96 }
