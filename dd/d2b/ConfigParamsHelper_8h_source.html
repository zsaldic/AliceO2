45  template <typename ContainerType>
46  static boost::program_options::options_description
47  prepareOptionDescriptions(const ContainerType& workflow,
48  const std::vector<ConfigParamSpec>& workflowOptions,
49  options_description vetos = options_description(),
50  std::string mode = "full")
51  {
52  boost::program_options::options_description toplevel;
53  boost::program_options::options_description wo("Global workflow options");
54  if (prepareOptionsDescription(workflowOptions, wo, vetos)) {
55  toplevel.add(wo);
56  }
57  std::string specOptionsDescription = "Available data processors";
58  if (mode == "short") {
59  specOptionsDescription += " (full info with '--help full')";
60  }
61  options_description specOptions(specOptionsDescription);
62  for (const auto & spec : workflow) {
63  std::string name = "Data processor options: " + spec.name;
64  boost::program_options::options_description processorOptions(name);
65  if (prepareOptionsDescription(spec.options, processorOptions, vetos)) {
66  // if vetos have been provided to the function we also need to make
67  // sure that there are no duplicate option definitions for the individual
68  // processor specs, so we add in order to be vetos for all subsequent specs.
69  // Note: this only concerns the main parser, all individual options are
70  // handled when starting individual processors.
71  if (vetos.options().size() > 0) {
72  vetos.add(processorOptions);
73  }
74  if (mode == "full") {
75  specOptions.add(processorOptions);
76  } else if (mode == spec.name) {
77  toplevel.add(processorOptions);
78  break;
79  }
80  }
81  if (mode == "full" || mode == "short") {
82  std::string help;
83  if (mode == "full") {
84  help = "Option groups by process name: --" + spec.name + R"( "<processor options>")";
85  } else if (mode == "short" && processorOptions.options().size() > 0) {
86  help = "Use '--help " + spec.name + "' to display processor options";
87  } else if (mode == "short" && processorOptions.options().size() == 0) {
88  help = "No processor options";
89  }
90  specOptions.add_options()(spec.name.c_str(),
91  boost::program_options::value<std::string>(),
92  help.c_str());
93  }
94  }
95  if (workflow.size() > 0 && (mode == "full" || mode == "short")) {
96  toplevel.add(specOptions);
97  }
98  return toplevel;
99  }

101  template<VariantType V>
102  static void addConfigSpecOption(const ConfigParamSpec & spec,
103  boost::program_options::options_description& options)
104  {
105  const char *name = spec.name.c_str();
106  const char *help = spec.help.c_str();
107  using Type = typename variant_type<V>::type;
108  using BoostType = typename std::conditional<V == VariantType::String, std::string, Type>::type;
109  auto value = boost::program_options::value<BoostType>();
110  if (spec.defaultValue.type() != VariantType::Empty) {
111  // set the default value if provided in the config spec
112  value = value->default_value(spec.defaultValue.get<Type>());
113  }
114  if (V == VariantType::Bool) {
115  // for bool values we also support the zero_token option to make
116  // the option usable as a single switch
117  value = value->zero_tokens();
118  }
119  options.add_options()(name, value, help);
120  }
121 };
