13 // Container class for ADC values of the TRD Pads.
14 // originally from aliroot
15 //
16 // The interface will remain constant.
17 // The underlying storage mechanism remains to be optimised as of 2/2/2019
18 //
19 // The array is a 3d array mapped to a linear array.
20 // 2 space coordinates 1 time cordinate
21 // [row][columb][30 time bins]
22 //
23 // There are always 30 times per pad.
24 //
25 // This will be converted to a sparse matrix after
26 // due testing. This will be transparent to users.
27 //
28 // Compression and Expansion is a run length encoding
29 // with encodings in the negative numbers, real data is positive.
30 //
31 //
32 // NB
33 // Do Not use the getDataAddress as the pointer
34 // can not be guaranteed to remain constant.
35 // It will cause a fatal log error
36 //

122 void TRDArrayADC::allocate(Int_t nrow, Int_t ncol, Int_t ntime)
123 {
124  //
125  // Allocate memory for an TRDArrayADC array with dimensions
126  // Row*NumberOfNecessaryMCMs*ADCchannelsInMCM*Time
127  //
128 
129  mNrow = nrow;
130  mNcol = ncol;
131  mNtime = ntime;
132  Int_t adcchannelspermcm = TRDFeeParam::getNadcMcm();
133  Int_t padspermcm = TRDFeeParam::getNcolMcm();
134  Int_t numberofmcms = mNcol / padspermcm;
135  mNumberOfChannels = numberofmcms * adcchannelspermcm;
136  mNAdim = nrow * mNumberOfChannels * ntime;
137 
138  if (mADC) {
139  delete[] mADC;
140  }
141 
142  mADC = new Short_t[mNAdim];
143  //std::fill(mADC.begin(), myVector.end(), 0);
144  memset(mADC, 0, sizeof(Short_t) * mNAdim);

215 UChar_t TRDArrayADC::getPadStatus(Int_t row, Int_t col, Int_t time) const
216 {
217  //
218  // Returns the pad status stored in the pad signal
219  //
220  // Output is a UChar_t value
221  // Status Codes:
222  // Noisy Masking: 2
223  // Bridged Left Masking 8
224  // Bridged Right Masking 8
225  // Not Connected Masking Digits
226  //
227  // Adapted from code of the class TRDclusterizer
228  //
229 
230  UChar_t padstatus = 0;
231  Short_t signal = getData(row, col, time);
232  if (signal > 0 && TESTBIT(signal, 10)) {
233  if (signal & 0x800) //TESTBIT(signal, 11))
234  if (signal & 0x1000) //TESTBIT(signal, 12))
235  padstatus = TRDCalPadStatus::kPadBridgedRight;
236  else
237  padstatus = TRDCalPadStatus::kNotConnected;
238  else if (signal & 0x1000) //TESTBIT(signal, 12))
239  padstatus = TRDCalPadStatus::kPadBridgedLeft;
240  else
241  padstatus = TRDCalPadStatus::kMasked;
242  }
243 
244  return padstatus;
245 }

248 void TRDArrayADC::setPadStatus(Int_t row, Int_t col, Int_t time, UChar_t status)
249 {
250  //
251  // Setting the pad status into the signal using the Bits 10 to 14
252  // (currently used: 10 to 12)
253  //
254  // Input codes (Unsigned char):
255  // Noisy Masking: 2
256  // Bridged Left Masking 8
257  // Bridged Right Masking 8
258  // Not Connected Masking 32
259  //
260  // Status codes: Any masking: Bit 10(1)
261  // Noisy masking: Bit 11(0), Bit 12(0)
262  // No Connection masking: Bit 11(1), Bit 12(0)
263  // Bridged Left masking: Bit 11(0), Bit 12(1)
264  // Bridged Right masking: Bit 11(1), Bit 12(1)
265  //
266  // Adapted from code of the class TRDclusterizer
267  //
268 
269  Short_t signal = getData(row, col, time);
270 
271  // Only set the Pad Status if the signal is > 0
272  if (signal > 0) {
273  switch (status) {
274  case TRDCalPadStatus::kMasked:
275  SETBIT(signal, 10);
276  CLRBIT(signal, 11);
277  CLRBIT(signal, 12);
278  break;
279  case TRDCalPadStatus::kNotConnected:
280  SETBIT(signal, 10);
281  SETBIT(signal, 11);
282  CLRBIT(signal, 12);
283  break;
284  case TRDCalPadStatus::kPadBridgedLeft:
285  SETBIT(signal, 10);
286  CLRBIT(signal, 11);
287  SETBIT(signal, 12);
288  break;
289  case TRDCalPadStatus::kPadBridgedRight:
290  SETBIT(signal, 10);
291  SETBIT(signal, 11);
292  SETBIT(signal, 12);
293  break;
294  default:
295  CLRBIT(signal, 10);
296  CLRBIT(signal, 11);
297  CLRBIT(signal, 12);
298  }
299  setData(row, col, time, signal);
300  }
301 }

317 void TRDArrayADC::compress()
318 {
319  //
320  // Compress the array
321  //
322 
323  if (mNAdim != mNrow * mNumberOfChannels * mNtime) {
324  LOG(info) << "The ADC array is already compressed";
325  return;
326  }
327 
328  Int_t counter = 0;
329  Int_t newDim = 0;
330  Int_t j;
331  Int_t l;
332  Int_t r = 0;
333  Int_t s = 0;
334  Int_t k = 0;
335 
336  Int_t* longm = new Int_t[mNAdim];
337  Int_t* longz = new Int_t[mNAdim];
338 
339  if (longz && longm && mADC) {
340 
341  memset(longz, 0, sizeof(Int_t) * mNAdim);
342  memset(longm, 0, sizeof(Int_t) * mNAdim);
343 
344  for (Int_t i = 0; i < mNAdim; i++) {
345  j = 0;
346  if (mADC[i] == -1) {
347  for (k = i; k < mNAdim; k++) {
348  if ((mADC[k] == -1) && (j < 16000)) {
349  j = j + 1;
350  longm[r] = j;
351  } else {
352  break;
353  }
354  }
355  r = r + 1;
356  }
357  l = 16001;
358  if (mADC[i] == 0) {
359  for (k = i; k < mNAdim; k++) {
360  if ((mADC[k] == 0) && (l < 32767)) {
361  l = l + 1;
362  longz[s] = l;
363  } else {
364  break;
365  }
366  }
367  s = s + 1;
368  }
369  if (mADC[i] > 0) {
370  i = i + 1;
371  }
372  i = i + j + (l - 16001 - 1);
373  }
374 
375  //Calculate the size of the compressed array
376  for (Int_t i = 0; i < mNAdim; i++) {
377  if (longm[i] != 0) {
378  counter = counter + longm[i] - 1;
379  }
380  if (longz[i] != 0) {
381  counter = counter + (longz[i] - 16001) - 1;
382  }
383  }
384 
385  Int_t counterTwo = 0;
386  newDim = mNAdim - counter; //Dimension of the compressed array
387  Short_t* buffer = new Short_t[newDim];
388 
389  if (buffer) {
390 
391  //Fill the buffer of the compressed array
392  Int_t g = 0;
393  Int_t h = 0;
394  for (Int_t i = 0; i < newDim; i++) {
395  if (counterTwo < mNAdim) {
396  if (mADC[counterTwo] > 0) {
397  buffer[i] = mADC[counterTwo];
398  }
399  if (mADC[counterTwo] == -1) {
400  buffer[i] = -(longm[g]);
401  counterTwo = counterTwo + longm[g] - 1;
402  g++;
403  }
404  if (mADC[counterTwo] == 0) {
405  buffer[i] = -(longz[h]);
406  counterTwo = counterTwo + (longz[h] - 16001) - 1;
407  h++;
408  }
409  counterTwo++;
410  }
411  }
412 
413  //Copy the buffer
414  delete[] mADC;
415  mADC = nullptr;
416  mADC = new Short_t[newDim];
417  mNAdim = newDim;
418  for (Int_t i = 0; i < newDim; i++) {
419  mADC[i] = buffer[i];
420  }
421 
422  //Delete auxiliary arrays
423  delete[] buffer;
424  buffer = nullptr;
425  }
426  }
427 
428  if (longz) {
429  delete[] longz;
430  longz = nullptr;
431  }
432  if (longm) {
433  delete[] longm;
434  longm = nullptr;
435  }
436 }

439 void TRDArrayADC::expand()
440 {
441  //
442  // Expand the array
443  //
444 
445  if (mADC) {
446 
447  //Check if the array has not been already expanded
448  Int_t verif = 0;
449  for (Int_t i = 0; i < mNAdim; i++) {
450  if (mADC[i] < -1) {
451  verif++;
452  }
453  }
454 
455  if (verif == 0) {
456  LOG(info) << "Nothing to expand";
457  return;
458  }
459 
460  Int_t dimexp = 0;
461  Int_t* longz = new Int_t[mNAdim];
462  Int_t* longm = new Int_t[mNAdim];
463 
464  if (longz && longm) {
465 
466  //Initialize arrays
467  memset(longz, 0, sizeof(Int_t) * mNAdim);
468  memset(longm, 0, sizeof(Int_t) * mNAdim);
469  Int_t r2 = 0;
470  Int_t r3 = 0;
471  for (Int_t i = 0; i < mNAdim; i++) {
472  if ((mADC[i] < 0) && (mADC[i] >= -16000)) {
473  longm[r2] = -mADC[i];
474  r2++;
475  }
476  if (mADC[i] < -16000) {
477  longz[r3] = -mADC[i] - 16001;
478  r3++;
479  }
480  }
481 
482  //Calculate the new dimensions of the array
483  for (Int_t i = 0; i < mNAdim; i++) {
484  if (longm[i] != 0) {
485  dimexp = dimexp + longm[i] - 1;
486  }
487  if (longz[i] != 0) {
488  dimexp = dimexp + longz[i] - 1;
489  }
490  }
491  dimexp = dimexp + mNAdim;
492 
493  //Write in the buffer the new array
494  Int_t contaexp = 0;
495  Int_t h = 0;
496  Int_t l = 0;
497  Short_t* bufferE = new Short_t[dimexp];
498  if (bufferE) {
499  for (Int_t i = 0; i < dimexp; i++) {
500  if (mADC[contaexp] > 0) {
501  bufferE[i] = mADC[contaexp];
502  }
503  if ((mADC[contaexp] < 0) && (mADC[contaexp] >= -16000)) {
504  for (Int_t j = 0; j < longm[h]; j++) {
505  bufferE[i + j] = -1;
506  }
507  i = i + longm[h] - 1;
508  h++;
509  }
510  if (mADC[contaexp] < -16000) {
511  for (Int_t j = 0; j < longz[l]; j++) {
512  bufferE[i + j] = 0;
513  }
514  i = i + longz[l] - 1;
515  l++;
516  }
517  contaexp++;
518  }
519  //Copy the buffer
520  delete[] mADC;
521  mADC = new Short_t[dimexp];
522  mNAdim = dimexp;
523  for (Int_t i = 0; i < dimexp; i++) {
524  mADC[i] = bufferE[i];
525  }
526 
527  delete[] bufferE;
528  }
529 
530  //Delete auxiliary arrays
531  delete[] longm;
532  delete[] longz;
533  }
534  }
