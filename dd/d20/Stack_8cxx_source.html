249 // calculates a hash based on particle properties
250 // hash may serve as seed for this track
251 ULong_t getHash(TParticle const& p)
252 {
253  auto asLong = [](double x) {
254  return (ULong_t) * (reinterpret_cast<ULong_t*>(&x));
255  };
256 
257  ULong_t hash;
258  o2::MCTrackT<double> track(p);
259 
260  hash = asLong(track.GetStartVertexCoordinatesX());
261  hash ^= asLong(track.GetStartVertexCoordinatesY());
262  hash ^= asLong(track.GetStartVertexCoordinatesZ());
263  hash ^= asLong(track.GetStartVertexCoordinatesT());
264  hash ^= asLong(track.GetStartVertexMomentumX());
265  hash ^= asLong(track.GetStartVertexMomentumY());
266  hash ^= asLong(track.GetStartVertexMomentumZ());
267  hash += (ULong_t)track.GetPdgCode();
268  return hash;

340 void Stack::finishCurrentPrimary()
341 {
342  // Here transport of a primary and all its secondaries is finished
343  // we can do some cleanup of the memory structures
344  LOG(DEBUG) << "STACK: Cleaning up" << FairLogger::endl;
345  auto selected = selectTracks();
346  // loop over current particle buffer
347  int index = 0;
348  int indexoffset = mTracks->size();
349  int neglected = 0;
350  std::vector<int> indicesKept;
351  for (const auto& particle : mParticles) {
352  if (particle.getStore() || !mPruneKinematics) {
353  // map the global track index to the new persistent index
354  // FIXME: only fill the map for non-trivial mappings in which mTransportedIDs[index]!=mTracks->size();
355  mIndexMap[mTransportedIDs[index]] = mTracks->size();
356  auto mother = particle.getMotherTrackId();
357  assert(mother < index);
358  mTracks->emplace_back(particle);
359  indicesKept.emplace_back(index);
360  if (mother != -1) {
361  auto iter = std::find_if(indicesKept.begin(), indicesKept.end(), [mother](int x) { return x == mother; });
362  if (iter != indicesKept.end()) {
363  // complexity should be constant
364  auto newmother = std::distance(indicesKept.begin(), iter);
365  mTracks->back().SetMotherTrackId(newmother + indexoffset);
366  }
367  }
368  // LOG(INFO) << "Adding to map " << mTransportedIDs[index] << " to " << mIndexMap[mTransportedIDs[index]];
369  } else {
370  neglected++;
371  }
372  index++;
373  mTracksDone++;
374  }
375  // we can now clear the particles buffer!
376  mParticles.clear();
377  mTransportedIDs.clear();
378  mTrackIDtoParticlesEntry.clear();
379  mIndexOfPrimaries.clear();
380 }

539 bool Stack::selectTracks()
540 {
541  bool tracksdiscarded = false;
542  // Check particles in the fParticle array
543  int prim = -1; // counter how many primaries seen (mainly to constrain search in motherindex remapping)
544  LOG(DEBUG) << "Stack: Entering track selection on " << mParticles.size() << " tracks";
545  for (auto& thisPart : mParticles) {
546  Bool_t store = kTRUE;
547 
548  // Get track parameters
549  Int_t iMother = thisPart.getMotherTrackId();
550  if (iMother < 0) {
551  prim++;
552  // for primaries we are done quickly
553  store = kTRUE;
554  } else {
555  // for other particles we potentially need to correct the mother indices
556  thisPart.SetMotherTrackId(mTrackIDtoParticlesEntry[iMother]);
557 
558  // no secondaries; also done
559  if (!mStoreSecondaries) {
560  store = kFALSE;
561  tracksdiscarded = true;
562  } else {
563  // Calculate number of hits created by this track
564  // Note: we only distinguish between no hit and more than 0 hits
565  int nHits = thisPart.hasHits();
566 
567  // Check for cuts (store primaries in any case)
568  if (nHits < mMinHits) {
569  store = kFALSE;
570  tracksdiscarded = true;
571  }
572  // only if we have non-trival energy cut
573  if (mEnergyCut > 0.) {
574  Double_t energy = thisPart.GetEnergy();
575  Double_t mass = thisPart.GetMass();
576  Double_t eKin = energy - mass;
577 
578  if (eKin < mEnergyCut) {
579  store = kFALSE;
580  tracksdiscarded = true;
581  }
582  }
583  }
584  }
585  // LOG(INFO) << "storing " << store << FairLogger::endl;
586  thisPart.setStore(store);
587  }
588 
589  // If flag is set, flag recursively mothers of selected tracks
590  if (mStoreMothers) {
591  for (auto& particle : mParticles) {
592  if (particle.getStore()) {
593  Int_t iMother = particle.getMotherTrackId();
594  while (iMother >= 0) {
595  auto& mother = mParticles[iMother];
596  mother.setStore(true);
597  iMother = mother.getMotherTrackId();
598  }
599  }
600  }
601  }
602 
603  return !tracksdiscarded;
604 }
