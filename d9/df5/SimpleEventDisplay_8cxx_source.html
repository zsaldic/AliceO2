78 Int_t SimpleEventDisplay::updateROC(const Int_t roc,
79  const Int_t row,
80  const Int_t pad,
81  const Int_t timeBin,
82  const Float_t signal)
83 {
84  //
85  // Signal filling methode on the fly pedestal and time offset correction if necessary.
86  // no extra analysis necessary. Assumes knowledge of the signal shape!
87  // assumes that it is looped over consecutive time bins of one pad
88  //
89  //printf("update called: %d, %d, %d, %d, %.3f\n", roc, row, pad, timeBin, signal);
90  if (row < 0)
91  return 0;
92  if (pad < 0)
93  return 0;
94  if (timeBin < 0)
95  return 0;
96  if ((timeBin > mLastTimeBin) || (timeBin < mFirstTimeBin))
97  return 0;
98  if (mSectorLoop && roc % 36 != mSelectedSector % 36)
99  return 0;
100 
101  if (row < 0 || pad < 0) {
102  printf("Wrong Pad or Row number, skipping!");
103  return 0;
104  }
105 
106  float corrSignal = signal;
107 
108  // ===| get pedestal |========================================================
109  if (mPedestals) {
110  corrSignal -= mPedestals->getValue(ROC(roc), row, pad);
111  }
112 
113  const int iChannel = mTPCmapper.getPadNumberInROC(PadROCPos(roc, row, pad));
114  //Int_t iChannel = mTPCmapper.globalPadNumber(PadPos(row, pad)) - (roc >= mTPCmapper.getNumberOfIROCs()) * mTPCmapper.getPadsInIROC();
115 
116  //init first pad and roc in this event
117  if (mCurrentChannel == -1) {
118  mCurrentChannel = iChannel;
119  mCurrentROC = roc;
120  mCurrentRow = row;
121  mCurrentPad = pad;
122  }
123 
124  //process last pad if we change to a new one
125  if (iChannel != mCurrentChannel) {
126  //ProcessPad();
127  mLastSector = mCurrentROC;
128  mCurrentChannel = iChannel;
129  mCurrentROC = roc;
130  mCurrentRow = row;
131  mCurrentPad = pad;
132  mMaxPadSignal = 0;
133  }
134  // if (signal>0) printf("%02d:%03d:%03d:%05d: %.3f\n",mCurrentROC,mCurrentRow,mCurrentPad,mCurrentChannel,signal);
135  //fill signals for current pad
136  if (mCurrentROC % 36 == mSelectedSector % 36) {
137  const Int_t nbins = mLastTimeBin - mFirstTimeBin;
138  const Int_t offset = (nbins + 2) * (iChannel + 1) + timeBin + 1;
139 
140  if ((UInt_t)roc < mTPCmapper.getNumberOfIROCs()) {
141  mHSigIROC->GetArray()[offset] = corrSignal;
142  } else {
143  mHSigOROC->GetArray()[offset] = corrSignal;
144  }
145  }
146 
147  CalROC& calROC = mPadMax.getCalArray(mCurrentROC);
148  //auto val = calROC.getValue(mCurrentRow, mCurrentPad);
149  auto val = calROC.getValue(row, pad);
150 
151  if (corrSignal > val) {
152  //printf("sec: %2d, row: %2d, pad: %3d, val: %.2f, sig: (%.2f) %.2f\n", mCurrentROC, mCurrentRow, mCurrentPad, val, corrSignal, signal);
153  //calROC.setValue(mCurrentRow,mCurrentPad,signal);
154  calROC.setValue(row, pad, corrSignal);
155  mMaxPadSignal = corrSignal;
156  mMaxTimeBin = timeBin;
157  }
158  //printf("update done\n");
159  return 0;
160 }
