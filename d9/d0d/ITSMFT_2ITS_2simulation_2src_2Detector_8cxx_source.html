72 static double radii2Turbo(double rMin, double rMid, double rMax, double sensW)
73 {
74  // compute turbo angle from radii and sensor width
75  return TMath::ASin((rMax * rMax - rMin * rMin) / (2 * rMid * sensW)) * TMath::RadToDeg();
76 }

78 static void configITS(Detector* its)
79 {
80  // build ITS upgrade detector
81  const double kSiThickIB = 50e-4;
82  const double kSiThickOB = 50e-4;
83  //
84  const int kNLr = 7;
85  const int kNLrInner = 3;
86  const int kBuildLevel = 0;
87  const int kSensTypeID = 0; // dummy id for Alpide sensor
88 
89  enum { kRmn, kRmd, kRmx, kNModPerStave, kPhi0, kNStave, kNPar };
90  // Radii are from last TDR (ALICE-TDR-017.pdf Tab. 1.1, rMid is mean value)
91  const double tdr5dat[kNLr][kNPar] = {
92  { 2.24, 2.34, 2.67, 9., 16.42, 12 }, // for each inner layer: rMin,rMid,rMax,NChip/Stave, phi0, nStaves
93  { 3.01, 3.15, 3.46, 9., 12.18, 16 }, { 3.78, 3.93, 4.21, 9., 9.55, 20 },
94  { -1, 19.6, -1, 4., 0., 24 }, // for others: -, rMid, -, NMod/HStave, phi0, nStaves // 24 was 49
95  { -1, 24.55, -1, 4., 0., 30 }, // 30 was 61
96  { -1, 34.39, -1, 7., 0., 42 }, // 42 was 88
97  { -1, 39.34, -1, 7., 0., 48 } // 48 was 100
98  };
99  const int nChipsPerModule = 7; // For OB: how many chips in a row
100  const double zChipGap = 0.01; // For OB: gap in Z between chips
101  const double zModuleGap = 0.01; // For OB: gap in Z between modules
102 
103  double dzLr, rLr, phi0, turbo;
104  int nStaveLr, nModPerStaveLr;
105 
106  its->setStaveModelIB(o2::ITS::Detector::kIBModel4);
107  its->setStaveModelOB(o2::ITS::Detector::kOBModel2);
108 
109  const int kNWrapVol = 3;
110  const double wrpRMin[kNWrapVol] = { 2.1, 15.0, 32.0 };
111  const double wrpRMax[kNWrapVol] = { 14.0, 30.0, 46.0 };
112  const double wrpZSpan[kNWrapVol] = { 70., 95., 200. };
113 
114  for (int iw = 0; iw < kNWrapVol; iw++) {
115  its->defineWrapperVolume(iw, wrpRMin[iw], wrpRMax[iw], wrpZSpan[iw]);
116  }
117 
118  for (int idLr = 0; idLr < kNLr; idLr++) {
119  rLr = tdr5dat[idLr][kRmd];
120  phi0 = tdr5dat[idLr][kPhi0];
121 
122  nStaveLr = TMath::Nint(tdr5dat[idLr][kNStave]);
123  nModPerStaveLr = TMath::Nint(tdr5dat[idLr][kNModPerStave]);
124  int nChipsPerStaveLr = nModPerStaveLr;
125  if (idLr >= kNLrInner) {
126  its->defineLayer(idLr, phi0, rLr, nStaveLr, nModPerStaveLr, kSiThickOB, Segmentation::SensorThickness,
127  kSensTypeID, kBuildLevel);
128  } else {
129  turbo = -radii2Turbo(tdr5dat[idLr][kRmn], rLr, tdr5dat[idLr][kRmx], Segmentation::SensorSizeRows);
130  its->defineLayerTurbo(idLr, phi0, rLr, nStaveLr, nChipsPerStaveLr, Segmentation::SensorSizeRows, turbo,
131  kSiThickIB, Segmentation::SensorThickness, kSensTypeID, kBuildLevel);
132  }
133  }
134 }

261 Bool_t Detector::ProcessHits(FairVolume* vol)
262 {
263  // This method is called from the MC stepping
264  if (!(fMC->TrackCharge())) {
265  return kFALSE;
266  }
267 
268  Int_t lay = 0, volID = vol->getMCid();
269 
270  // FIXME: Determine the layer number. Is this information available directly from the FairVolume?
271  bool notSens = false;
272  while ((lay < sNumberLayers) && (notSens = (volID != mLayerID[lay]))) {
273  ++lay;
274  }
275  if (notSens)
276  return kFALSE; // RS: can this happen? This method must be called for sensors only?
277 
278  // Is it needed to keep a track reference when the outer ITS volume is encountered?
279  auto stack = (o2::Data::Stack*)fMC->GetStack();
280  if (fMC->IsTrackExiting() && (lay == 0 || lay == 6)) {
281  // Keep the track refs for the innermost and outermost layers only
282  o2::TrackReference tr(*fMC, GetDetId());
283  tr.setTrackID(stack->GetCurrentTrackNumber());
284  tr.setUserId(lay);
285  stack->addTrackReference(tr);
286  }
287  bool startHit = false, stopHit = false;
288  unsigned char status = 0;
289  if (fMC->IsTrackEntering()) {
290  status |= Hit::kTrackEntering;
291  }
292  if (fMC->IsTrackInside()) {
293  status |= Hit::kTrackInside;
294  }
295  if (fMC->IsTrackExiting()) {
296  status |= Hit::kTrackExiting;
297  }
298  if (fMC->IsTrackOut()) {
299  status |= Hit::kTrackOut;
300  }
301  if (fMC->IsTrackStop()) {
302  status |= Hit::kTrackStopped;
303  }
304  if (fMC->IsTrackAlive()) {
305  status |= Hit::kTrackAlive;
306  }
307 
308  // track is entering or created in the volume
309  if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
310  startHit = true;
311  } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
312  stopHit = true;
313  }
314 
315  // increment energy loss at all steps except entrance
316  if (!startHit)
317  mTrackData.mEnergyLoss += fMC->Edep();
318  if (!(startHit | stopHit))
319  return kFALSE; // do noting
320 
321  if (startHit) {
322  mTrackData.mEnergyLoss = 0.;
323  fMC->TrackMomentum(mTrackData.mMomentumStart);
324  fMC->TrackPosition(mTrackData.mPositionStart);
325  mTrackData.mTrkStatusStart = status;
326  mTrackData.mHitStarted = true;
327  }
328  if (stopHit) {
329  TLorentzVector positionStop;
330  fMC->TrackPosition(positionStop);
331  // Retrieve the indices with the volume path
332  int stave(0), halfstave(0), chipinmodule(0), module;
333  fMC->CurrentVolOffID(1, chipinmodule);
334  fMC->CurrentVolOffID(2, module);
335  fMC->CurrentVolOffID(3, halfstave);
336  fMC->CurrentVolOffID(4, stave);
337  int chipindex = mGeometryTGeo->getChipIndex(lay, stave, halfstave, module, chipinmodule);
338 
339  Hit* p = addHit(stack->GetCurrentTrackNumber(), chipindex, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
340  mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
341  mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
342  // p->SetTotalEnergy(vmc->Etot());
343 
344  // RS: not sure this is needed
345  // Increment number of Detector det points in TParticle
346  stack->addHit(GetDetId());
347  }
348 
349  return kTRUE;
350 }

707 void Detector::constructDetectorGeometry()
708 {
709  // Create the geometry and insert it in the mother volume ITSV
710  TGeoManager* geoManager = gGeoManager;
711 
712  TGeoVolume* vALIC = geoManager->GetVolume("cave");
713 
714  if (!vALIC) {
715  LOG(FATAL) << "Could not find the top volume";
716  }
717 
718  new TGeoVolumeAssembly(GeometryTGeo::getITSVolPattern());
719  TGeoVolume* vITSV = geoManager->GetVolume(GeometryTGeo::getITSVolPattern());
720  vALIC->AddNode(vITSV, 2, nullptr); // Copy number is 2 to cheat AliGeoManager::CheckSymNamesLUT
721 
722  const Int_t kLength = 100;
723  Char_t vstrng[kLength] = "xxxRS"; //?
724  vITSV->SetTitle(vstrng);
725 
726  // Check that we have all needed parameters
727  for (Int_t j = 0; j < sNumberLayers; j++) {
728  if (mLayerRadii[j] <= 0) {
729  LOG(FATAL) << "Wrong layer radius for layer " << j << "(" << mLayerRadii[j] << ")";
730  }
731  if (mStavePerLayer[j] <= 0) {
732  LOG(FATAL) << "Wrong number of staves for layer " << j << "(" << mStavePerLayer[j] << ")";
733  }
734  if (mUnitPerStave[j] <= 0) {
735  LOG(FATAL) << "Wrong number of chips for layer " << j << "(" << mUnitPerStave[j] << ")";
736  }
737  if (mChipThickness[j] < 0) {
738  LOG(FATAL) << "Wrong chip thickness for layer " << j << "(" << mChipThickness[j] << ")";
739  }
740  if (mTurboLayer[j] && mStaveWidth[j] <= 0) {
741  LOG(FATAL) << "Wrong stave width for layer " << j << "(" << mStaveWidth[j] << ")";
742  }
743  if (mDetectorThickness[j] < 0) {
744  LOG(FATAL) << "Wrong Sensor thickness for layer " << j << "(" << mDetectorThickness[j] << ")";
745  }
746 
747  if (j > 0) {
748  if (mLayerRadii[j] <= mLayerRadii[j - 1]) {
749  LOG(FATAL) << "Layer " << j << " radius (" << mLayerRadii[j] << ") is smaller than layer " << j - 1
750  << " radius (" << mLayerRadii[j - 1] << ")";
751  }
752  }
753 
754  if (mChipThickness[j] == 0) {
755  LOG(INFO) << "Chip thickness for layer " << j << " not set, using default";
756  }
757  if (mDetectorThickness[j] == 0) {
758  LOG(INFO) << "Sensor thickness for layer " << j << " not set, using default";
759  }
760  }
761 
762  // Create the wrapper volumes
763  TGeoVolume** wrapVols = nullptr;
764 
765  if (sNumberOfWrapperVolumes) {
766  wrapVols = new TGeoVolume*[sNumberOfWrapperVolumes];
767  for (int id = 0; id < sNumberOfWrapperVolumes; id++) {
768  wrapVols[id] = createWrapperVolume(id);
769  vITSV->AddNode(wrapVols[id], 1, nullptr);
770  }
771  }
772 
773  // Now create the actual geometry
774  for (Int_t j = 0; j < sNumberLayers; j++) {
775  TGeoVolume* dest = vITSV;
776  mWrapperLayerId[j] = -1;
777 
778  if (mTurboLayer[j]) {
779  mGeometry[j] = new V3Layer(j, kTRUE, kFALSE);
780  mGeometry[j]->setStaveWidth(mStaveWidth[j]);
781  mGeometry[j]->setStaveTilt(mStaveTilt[j]);
782  } else {
783  mGeometry[j] = new V3Layer(j, kFALSE);
784  }
785 
786  mGeometry[j]->setPhi0(mLayerPhi0[j]);
787  mGeometry[j]->setRadius(mLayerRadii[j]);
788  mGeometry[j]->setNumberOfStaves(mStavePerLayer[j]);
789  mGeometry[j]->setNumberOfUnits(mUnitPerStave[j]);
790  mGeometry[j]->setChipType(mChipTypeID[j]);
791  mGeometry[j]->setBuildLevel(mBuildLevel[j]);
792 
793  if (j < sNumberInnerLayers) {
794  mGeometry[j]->setStaveModel(mStaveModelInnerBarrel);
795  } else {
796  mGeometry[j]->setStaveModel(mStaveModelOuterBarrel);
797  }
798 
799  LOG(DEBUG1) << "mBuildLevel: " << mBuildLevel[j];
800 
801  if (mChipThickness[j] != 0) {
802  mGeometry[j]->setChipThick(mChipThickness[j]);
803  }
804  if (mDetectorThickness[j] != 0) {
805  mGeometry[j]->setSensorThick(mDetectorThickness[j]);
806  }
807 
808  for (int iw = 0; iw < sNumberOfWrapperVolumes; iw++) {
809  if (mLayerRadii[j] > mWrapperMinRadius[iw] && mLayerRadii[j] < mWrapperMaxRadius[iw]) {
810  LOG(DEBUG) << "Will embed layer " << j << " in wrapper volume " << iw;
811 
812  dest = wrapVols[iw];
813  mWrapperLayerId[j] = iw;
814  break;
815  }
816  }
817  mGeometry[j]->createLayer(dest);
818  }
819  createServiceBarrel(kTRUE, wrapVols[0]);
820  createServiceBarrel(kFALSE, wrapVols[2]);
821 
822  delete[] wrapVols; // delete pointer only, not the volumes
823 }
