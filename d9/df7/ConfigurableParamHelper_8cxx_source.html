32 // a generic looper of data members of a TClass; calling a callback
33 // reused in various functions below
34 void loopOverMembers(TClass* cl, void* obj,
35  std::function<void(const TDataMember*, int, int)>&& callback)
36 {
37  auto memberlist = cl->GetListOfDataMembers();
38  for (int i = 0; i < memberlist->GetEntries(); ++i) {
39  auto dm = (TDataMember*)memberlist->At(i);
40 
41  auto isValidComplex = [dm]() {
42  return isString(*dm);
43  };
44 
45  // filter out static members for now
46  if (dm->Property() & kIsStatic) {
47  continue;
48  }
49  if (dm->IsaPointer()) {
50  LOG(WARNING) << "Pointer types not supported in ConfigurableParams";
51  continue;
52  }
53  if (!dm->IsBasic() && !isValidComplex()) {
54  LOG(WARNING) << "Generic complex types not supported in ConfigurableParams";
55  continue;
56  }
57  const auto dim = dm->GetArrayDim();
58  // we support very simple vectored data in 1D for now
59  if (dim > 1) {
60  LOG(WARNING) << "We support at most 1 dimensional arrays in ConfigurableParams";
61  continue;
62  }
63  const auto size = (dim == 1) ? dm->GetMaxIndex(dim - 1) : 1; // size of array (1 if scalar)
64  for (int index = 0; index < size; ++index) {
65  callback(dm, index, size);
66  }
67  }
68 }

81 const char* asString(TDataMember const& dm, char* pointer)
82 {
83  // first check if this is a basic data type, in which case
84  // we let ROOT do the work
85  if (auto dt = dm.GetDataType()) {
86  return dt->AsString(pointer);
87  }
88 
89  // if data member is a std::string just return
90  else if (isString(dm)) {
91  return ((std::string*)pointer)->c_str();
92  }
93  // potentially other cases to be added here
94 
95  LOG(ERROR) << "COULD NOT REPRESENT AS STRING";
96  return nullptr;
97 }

99 void _ParamHelper::printParametersImpl(std::string mainkey, TClass* cl, void* obj,
100  std::map<std::string, ConfigurableParam::EParamProvenance> const* provmap)
101 {
102  auto printMembers = [&mainkey, obj, provmap](const TDataMember* dm, int index, int size) {
103  // pointer to object
104  auto dt = dm->GetDataType();
105  auto TS = dt ? dt->Size() : 0;
106  char* pointer = ((char*)obj) + dm->GetOffset() + index * TS;
107  const auto name = getName(dm, index, size);
108  std::cout << name << " : " << asString(*dm, pointer);
109  if (provmap != nullptr) {
110  auto iter = provmap->find(mainkey + "." + name);

112  std::cout << "\t\t[ " << ConfigurableParam::toString(iter->second) << " ]";
113  }
114  }
115  std::cout << "\n";
116  };
117  loopOverMembers(cl, obj, printMembers);
118 }

120 // a function converting a string representing a type to the type_info
121 // because unfortunately typeid(double) != typeid("double")
122 // but we can take the TDataType (if it exists) as a hint in order to
123 // minimize string comparisons
124 std::type_info const& nameToTypeInfo(const char* tname, TDataType const* dt)
125 {
126  if (dt) {
127  switch (dt->GetType()) {
128  case kChar_t: {
129  return typeid(char);
130  }
131  case kUChar_t: {
132  return typeid(unsigned char);
133  }
134  case kShort_t: {
135  return typeid(short);
136  }
137  case kUShort_t: {
138  return typeid(unsigned short);
139  }
140  case kInt_t: {
141  return typeid(int);
142  }
143  case kUInt_t: {
144  return typeid(unsigned int);
145  }
146  case kLong_t: {
147  return typeid(long);
148  }
149  case kULong_t: {
150  return typeid(unsigned long);
151  }
152  case kFloat_t: {
153  return typeid(float);
154  }
155  case kDouble_t: {
156  return typeid(double);
157  }
158  case kDouble32_t: {
159  return typeid(double);
160  }
161  case kBool_t: {
162  return typeid(bool);
163  }
164  case kLong64_t: {
165  return typeid(long long);
166  }
167  case kULong64_t: {
168  return typeid(unsigned long long);
169  }
170  default: {
171  break;
172  }
173  }
174  }
175  // if we get here none of the above worked
176  if (strcmp(tname, "string") == 0 || strcmp(tname, "std::string")) {
177  return typeid(std::string);
178  }
179  LOG(ERROR) << "ENCOUNTERED AN UNSUPPORTED TYPE " << tname << "IN A CONFIGURABLE PARAMETER";
180  return typeid("ERROR");
181 }

183 void _ParamHelper::fillKeyValuesImpl(std::string mainkey, TClass* cl, void* obj, boost::property_tree::ptree* tree,
184  std::map<std::string, std::pair<std::type_info const&, void*>>* keytostoragemap)
185 {
186  boost::property_tree::ptree localtree;
187  auto fillMap = [obj, &mainkey, &localtree, &keytostoragemap](const TDataMember* dm, int index, int size) {
188  const auto name = getName(dm, index, size);
189  auto dt = dm->GetDataType();
190  auto TS = dt ? dt->Size() : 0;
191  char* pointer = ((char*)obj) + dm->GetOffset() + index * TS;
192  localtree.put(name, asString(*dm, pointer));
193 
194  auto key = mainkey + "." + name;
195  using mapped_t = std::pair<std::type_info const&, void*>;
196  auto& ti = nameToTypeInfo(dm->GetTrueTypeName(), dt);
197  keytostoragemap->insert(std::pair<std::string, mapped_t>(key, mapped_t(ti, pointer)));
198  };
199  loopOverMembers(cl, obj, fillMap);
200  tree->add_child(mainkey, localtree);
201 }

214 void _ParamHelper::assignmentImpl(std::string mainkey, TClass* cl, void* to, void* from,
215  std::map<std::string, ConfigurableParam::EParamProvenance>* provmap)

218  const auto name = getName(dm, index, size);
219  auto dt = dm->GetDataType();
220  auto TS = dt ? dt->Size() : 0;
221  char* pointerto = ((char*)to) + dm->GetOffset() + index * TS;
222  char* pointerfrom = ((char*)from) + dm->GetOffset() + index * TS;
223 
224  // lambda to update the provenance

226  auto key = mainkey + "." + name;
227  auto iter = provmap->find(key);
228  if (iter != provmap->end()) {
229  iter->second = ConfigurableParam::EParamProvenance::kCCDB; // TODO: change to "current STATE"??
230  } else {
231  LOG(WARN) << "KEY " << key << " NOT FOUND WHILE UPDATING PARAMETER PROVENANCE";
232  }
233  };
234 
235  // TODO: this could dispatch to the same method used in ConfigurableParam::setValue
236  // but will be slower
237 
238  // test if a complicated case
239  if (isString(*dm)) {
240  std::string& target = *(std::string*)pointerto;
241  std::string const& origin = *(std::string*)pointerfrom;
242  if (target.compare(origin) != 0) {
243  updateProv();
244  target = origin;
245  }
246  return;
247  }
248 
249  //
250  if (!isMemblockDifferent(pointerto, pointerfrom, TS)) {
251  updateProv();
252  // actually copy
253  std::memcpy(pointerto, pointerfrom, dt->Size());
254  }
255  };
256  loopOverMembers(cl, to, assignifchanged);
