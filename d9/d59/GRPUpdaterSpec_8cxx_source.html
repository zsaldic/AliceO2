46  void run(framework::ProcessingContext& pc)
47  {
48  const std::string inputGRP = "o2sim_grp.root";
49  const std::string grpName = "GRP";
50  if (mFinished) {
51  return;
52  }
53 
54  TFile flGRP(inputGRP.c_str(), "update");
55  if (flGRP.IsZombie()) {
56  LOG(ERROR) << "Failed to open in update mode " << inputGRP << FairLogger::endl;
57  return;
58  }
59  std::unique_ptr<GRP> grp(static_cast<GRP*>(flGRP.GetObjectChecked(grpName.c_str(), GRP::Class())));
60  for (auto det : sDetList) { // get readout mode data from different detectors
61  auto roMode = pc.inputs().get<o2::parameters::GRPObject::ROMode>(det.getName());
62  if (!(roMode & o2::parameters::GRPObject::PRESENT)) {
63  LOG(ERROR) << "Detector " << det.getName() << " is read out while processor set ABSENT";
64  continue;
65  }
66  grp->setDetROMode(det, roMode);
67  }
68  LOG(INFO) << "Updated GRP in " << inputGRP << " for detectors RO mode" << FairLogger::endl;
69  grp->print();
70  flGRP.WriteObjectAny(grp.get(), grp->Class(), grpName.c_str());
71  flGRP.Close();
72  mFinished = true;
73 
74  pc.services().get<ControlService>().readyToQuit(false);
75  }
76 
77  private:
78  bool mFinished = false;
79 };

82 o2::framework::DataProcessorSpec getGRPUpdaterSpec(const std::vector<o2::detectors::DetID>& detList)
83 {
84  sDetList = detList;
85  static constexpr std::array<o2::header::DataOrigin, o2::detectors::DetID::nDetectors> sOrigins = {
86  o2::header::gDataOriginITS, o2::header::gDataOriginTPC, o2::header::gDataOriginTRD,
87  o2::header::gDataOriginTOF, o2::header::gDataOriginPHS, o2::header::gDataOriginCPV,
88  o2::header::gDataOriginEMC, o2::header::gDataOriginHMP, o2::header::gDataOriginMFT,
89  o2::header::gDataOriginMCH, o2::header::gDataOriginMID, o2::header::gDataOriginZDC,
90  o2::header::gDataOriginFIT, o2::header::gDataOriginACO
91  };
92 
93  // prepare specs
94  std::vector<InputSpec> inputs;
95  for (const auto det : detList) {
96  inputs.emplace_back(InputSpec{ det.getName(), sOrigins[det], "ROMode",
97  static_cast<SubSpecificationType>(0 /*det.second*/), Lifetime::Timeframe });
98  }
99 
100  return DataProcessorSpec{
101  "GRPUpdater",
102  inputs, // input status from each detector
103  {}, // no output
104  AlgorithmSpec{ adaptFromTask<GRPDPLUpdatedTask>() },
105  Options{ /* for the moment no options */ }
106  };
107 }
