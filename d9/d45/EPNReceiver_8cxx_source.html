85 {

88  while (it != mTimeframeBuffer.end()) {
89  if (duration_cast<milliseconds>(steady_clock::now() - (it->second).start).count() > mBufferTimeoutInMs) {
90  LOG(WARN) << "Timeframe #" << it->first << " incomplete after " << mBufferTimeoutInMs << " milliseconds, discarding";
91  mDiscardedSet.insert(it->first);
92  mTimeframeBuffer.erase(it++);

94  } else {
