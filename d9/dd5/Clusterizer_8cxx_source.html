46 bool Clusterizer::makeClusters(PreClusters& pcs)
47 {
49  int deIndex = pcs.getDEId();
50 
51  LOG(DEBUG) << "Clusterizing " << deIndex;
52 
53  // loop over pre-clusters in the non-bending plane
54  for (int inb = 0; inb < pcs.getNPreClustersNBP(); ++inb) {
55  PreClusters::PreClusterNBP& pcNB = pcs.getPreClusterNBP(inb);
56  // Try to find matching pre-clusters in the bending plane
57  int icolumn = pcNB.firstColumn;
58  if (pcNB.firstColumn == pcNB.lastColumn) {
59  // This is the most simple and common case: the NBP pre-cluster is on
60  // one single column. So it can be easily matched with the BP
61  // since the corresponding contours are both rectangles
62  for (int ib = 0; ib < pcs.getNPreClustersBP(icolumn); ++ib) {
63  makeCluster(pcs.getPreClusterBP(icolumn, ib), pcNB, deIndex);
64  }
65  } else {
66  // The NBP pre-cluster spans different columns.
67  // The BP contour is therefore a serie of neighbour rectangles
68  // Here we could build a giant cluster with all neighbour pre-clusters
69  // in the bending plane.
70  // However, the a pre-cluster in the BP might be close to more than one pre-clusters
71  // in the BP of the close column, thus leading to a pattern with non-contiguous
72  // pre-clusters along the y axis...which are treated as separate clusters
73  // in the standard configuration.
74  // All in all, since the strips are huge, we decided to pair the clusters only between
75  // two adjacent columns.
76  // In this way, a huge cluster spanning three columns will be transformed in
77  // two clusters centered at the border between two columns.
78  // This leads to a redundancy. Still, the sigma of the resulting cluster will be so large
79  // to have no impact whatsoever in the track reconstruction.
80  LOG(DEBUG) << "Spanning non-bend: " << icolumn << " -> " << pcNB.lastColumn;
81  int pairId = 10 + inb;
82  for (; icolumn <= pcNB.lastColumn; ++icolumn) {
83  for (int ib = 0; ib < pcs.getNPreClustersBP(icolumn); ++ib) {
84  PreClusters::PreClusterBP& pcB = pcs.getPreClusterBP(icolumn, ib);
85  // This function checks for the neighbours only on icolumn+1
86  std::vector<int> neighbours = pcs.getNeighbours(icolumn, ib);
87  // It can happen that the NBP spans two columns...but there are fired strips only on
88  // one column of the BP. In this case we only consider the column with a hit in the BP.
89  // Of course, we need to check that the current pre-cluster was not already paired
90  // with the pre-cluster in the previous column.
91  if (neighbours.empty() && pcB.paired != pairId) {
92  makeCluster(pcB, pcNB, deIndex);
93  } else {
94  for (auto& jb : neighbours) {
95  PreClusters::PreClusterBP& pcBneigh = pcs.getPreClusterBP(icolumn + 1, jb);
96  makeCluster(pcB, pcBneigh, pcNB, deIndex);
97  // Here we set the paired flag to a custom ID of the pre-cluster in the NBP
98  // So that, when we move to the next column, we do not add it twice.
99  pcBneigh.paired = pairId;
100  }
101  }
102  } // loop on pre-clusters in the BP
103  } // loop on columns
104  }
105 
106  if (pcNB.paired == 0) {
107  // If it is not paired, it means that we have
108  // a monocathodic cluster in the NBP
109  makeCluster(pcNB, deIndex);
110  }
111  } // loop on pre-clusters in the NBP
112 
114  for (int icolumn = 0; icolumn <= 6; ++icolumn) {
115  for (int ib = 0; ib < pcs.getNPreClustersBP(icolumn); ++ib) {
116  PreClusters::PreClusterBP& pcB = pcs.getPreClusterBP(icolumn, ib);
117  if (pcB.paired == 0) {
118  makeCluster(pcB, deIndex);
119  }
120  }
121  }
122 
123  return true;
124 }

150 void Clusterizer::makeCluster(PreClusters::PreClusterBP& clBend, PreClusters::PreClusterNBP& clNonBend, const int& deIndex)
151 {
153  Cluster2D& cl = nextCluster();
154  int icolumn = clBend.column;
155  cl.deId = (uint8_t)deIndex;
156  cl.xCoor = 0.5 * (clNonBend.area[icolumn].getXmax() + clNonBend.area[icolumn].getXmin());
157  cl.yCoor = 0.5 * (clBend.area.getYmax() + clBend.area.getYmin());
158  double deltaX = clNonBend.area[icolumn].getXmax() - clNonBend.area[icolumn].getXmin();
159  double deltaY = clBend.area.getYmax() - clBend.area.getYmin();
160  cl.sigmaX2 = deltaX * deltaX / 12;
161  cl.sigmaY2 = deltaY * deltaY / 12;
162  clBend.paired = 1;
163  clNonBend.paired = 1;
164 
165  LOG(DEBUG) << "pos: (" << cl.xCoor << ", " << cl.yCoor << ") err2: (" << cl.sigmaX2 << ", " << cl.sigmaY2 << ")";
166 }

189 void Clusterizer::makeCluster(PreClusters::PreClusterBP& clBend, PreClusters::PreClusterBP& clBendNeigh, PreClusters::PreClusterNBP& clNonBend, const int& deIndex)
190 {
192  // This is the general case:
193  // perform the full calculation assuming a uniform charge distribution
194 
195  Cluster2D& cl = nextCluster();
196  cl.deId = (uint8_t)deIndex;
197 
198  double x2[2][2] = { { 0., 0. }, { 0., 0. } };
199  double x3[2][2] = { { 0., 0. }, { 0., 0. } };
200  double dim[2][2];
201  double delta[2];
202  double sumArea = 0.;
203 
204  std::vector<PreClusters::PreClusterBP*> pcBlist = { &clBend, &clBendNeigh };
205 
206  for (auto* pcBP : pcBlist) {
207  int icolumn = pcBP->column;
208  dim[0][0] = clNonBend.area[icolumn].getXmin();
209  dim[0][1] = clNonBend.area[icolumn].getXmax();
210  dim[1][0] = pcBP->area.getYmin();
211  dim[1][1] = pcBP->area.getYmax();
212  for (int iplane = 0; iplane < 2; ++iplane) {
213  delta[iplane] = dim[iplane][1] - dim[iplane][0];
214  }
215  // area = dx * dy
216  sumArea += delta[0] * delta[1];
217  LOG(DEBUG) << "Area += " << delta[0] * delta[1] << " => " << sumArea;
218  for (int iplane = 0; iplane < 2; ++iplane) {
219  for (int ip = 0; ip < 2; ++ip) {
220  // second momentum = x_i * x_i * dy
221  double currX2 = dim[iplane][ip] * dim[iplane][ip] * delta[1 - iplane];
222  x2[iplane][ip] += currX2;
223  // third momentum = x_i * x_i * x_i * dy
224  x3[iplane][ip] += currX2 * dim[iplane][ip];
225  LOG(DEBUG) << "x[" << iplane << "][" << ip << "] => val " << dim[iplane][ip] << " delta " << delta[1 - iplane]
226  << " => x2 " << x2[iplane][ip] << " x3 " << x3[iplane][ip];
227  }
228  }
229  pcBP->paired = 1;
230  clNonBend.paired = 1;
231  } // loop on column
232 
233  double coor[2], sigma2[2];
234  for (int iplane = 0; iplane < 2; ++iplane) {
235  coor[iplane] = (x2[iplane][1] - x2[iplane][0]) / sumArea / 2.;
236  sigma2[iplane] = (x3[iplane][1] - x3[iplane][0]) / sumArea / 3. - coor[iplane] * coor[iplane];
237  }
238 
239  cl.xCoor = (float)coor[0];
240  cl.yCoor = (float)coor[1];
241  cl.sigmaX2 = (float)sigma2[0];
242  cl.sigmaY2 = (float)sigma2[1];
243 
244  LOG(DEBUG) << "pos: (" << cl.xCoor << ", " << cl.yCoor << ") err2: (" << cl.sigmaX2 << ", " << cl.sigmaY2 << ")";
245 }
