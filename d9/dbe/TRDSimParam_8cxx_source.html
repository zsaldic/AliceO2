169 void TRDSimParam::ReInit()
170 {
171  //
172  // Reinitializes the parameter class after a change
173  //
174 
175  if (TRDCommonParam::Instance()->IsXenon()) {
176  // The range and the binwidth for the sampled TRF
177  mTRFbin = 200;
178  // Start 0.2 mus before the signal
179  mTRFlo = -0.4;
180  // End the maximum drift time after the signal
181  mTRFhi = 3.58;
182  // Standard gas gain
183  mGasGain = 4000.0;
184  } else if (TRDCommonParam::Instance()->IsArgon()) {
185  // The range and the binwidth for the sampled TRF
186  mTRFbin = 50;
187  // Start 0.2 mus before the signal
188  mTRFlo = 0.02;
189  // End the maximum drift time after the signal
190  mTRFhi = 1.98;
191  // Higher gas gain
192  mGasGain = 8000.0;
193  } else {
194  LOG(FATAL) << "Not a valid gas mixture!\n";
195  }
196  mTRFwid = (mTRFhi - mTRFlo) / ((float)mTRFbin);
197 
198  // Create the sampled TRF
199  SampleTRF();
200 }

300 double TRDSimParam::TimeResponse(double time) const
301 {
302  //
303  // Applies the preamp shaper time response
304  // (We assume a signal rise time of 0.2us = fTRFlo/2.
305  //
306 
307  double rt = (time - .5 * mTRFlo) / mTRFwid;
308  int iBin = (int)rt;
309  double dt = rt - iBin;
310  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
311  return mTRFsmp[iBin] + (mTRFsmp[iBin + 1] - mTRFsmp[iBin]) * dt;
312  } else {
313  return 0.0;
314  }
315 }
