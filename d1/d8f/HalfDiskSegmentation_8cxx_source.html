88 void HalfDiskSegmentation::createLadders(TXMLEngine* xml, XMLNodePointer_t node)
89 {
90  Int_t iladder;
91  Int_t nsensor;
92  Double_t pos[3];
93  Double_t ang[3] = { 0., 0., 0. };
94 
95  Geometry* mftGeom = Geometry::instance();
96 
97  TString nodeName = xml->GetNodeName(node);
98  if (!nodeName.CompareTo("ladder")) {
99  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
100  while (attr != nullptr) {
101  TString attrName = xml->GetAttrName(attr);
102  TString attrVal = xml->GetAttrValue(attr);
103  if (!attrName.CompareTo("iladder")) {
104  iladder = attrVal.Atoi();
105  if (iladder >= getNLadders() || iladder < 0) {
106  LOG(FATAL) << "Wrong ladder number : " << iladder << FairLogger::endl;
107  }
108  } else if (!attrName.CompareTo("nsensor")) {
109  nsensor = attrVal.Atoi();
110  } else if (!attrName.CompareTo("xpos")) {
111  pos[0] = attrVal.Atof();
112  } else if (!attrName.CompareTo("ypos")) {
113  pos[1] = attrVal.Atof();
114  } else if (!attrName.CompareTo("zpos")) {
115  pos[2] = attrVal.Atof();
116  } else if (!attrName.CompareTo("phi")) {
117  ang[0] = attrVal.Atof();
118  } else if (!attrName.CompareTo("theta")) {
119  ang[1] = attrVal.Atof();
120  } else if (!attrName.CompareTo("psi")) {
121  ang[2] = attrVal.Atof();
122  } else {
123  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr) << FairLogger::endl;
124  }
125  attr = xml->GetNextAttr(attr);
126  }
127 
128  Int_t plane = -1;
129  Int_t ladderID = iladder;
130  if (iladder < getNLadders() / 2) {
131  plane = 0;
132  } else {
133  plane = 1;
134  // ladderID -= getNLadders()/2;
135  }
136 
137  UInt_t ladderUniqueID = mftGeom->getObjectID(Geometry::LadderType, mftGeom->getHalfID(GetUniqueID()),
138  mftGeom->getDiskID(GetUniqueID()), plane, ladderID);
139 
140  auto* ladder = new LadderSegmentation(ladderUniqueID);
141  ladder->setNSensors(nsensor);
142  ladder->setPosition(pos);
143  ladder->setRotationAngles(ang);
144 
145  ladder->createSensors(xml, node);
146 
147  new ((*mLadders)[iladder]) LadderSegmentation(*ladder);
148  delete ladder;
149 
150  // getLadder(iladder)->Print();
151  }
152 
153  // display all child nodes
154  XMLNodePointer_t child = xml->GetChild(node);
155  while (child != nullptr) {
156  createLadders(xml, child);
157  child = xml->GetNext(child);
158  }
159 }
