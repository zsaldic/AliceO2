109 void HalfSegmentation::createHalfDisks(TXMLEngine* xml, XMLNodePointer_t node)
110 {
111  // this function display all accessible information about xml node and its children
112  Int_t idisk;
113  Int_t nladder;
114  Double_t pos[3] = { 0., 0., 0. };
115  Double_t ang[3] = { 0., 0., 0. };
116 
117  Geometry* mftGeom = Geometry::instance();
118 
119  TString nodeName = xml->GetNodeName(node);
120  if (!nodeName.CompareTo("disk")) {
121  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
122  while (attr != nullptr) {
123  TString attrName = xml->GetAttrName(attr);
124  TString attrVal = xml->GetAttrValue(attr);
125  if (!attrName.CompareTo("idisk")) {
126  idisk = attrVal.Atoi();
127  if (idisk >= Constants::DisksNumber || idisk < 0) {
128  LOG(FATAL) << "Wrong disk number : " << idisk << FairLogger::endl;
129  }
130  } else if (!attrName.CompareTo("nladder")) {
131  nladder = attrVal.Atoi();
132  } else if (!attrName.CompareTo("xpos")) {
133  pos[0] = attrVal.Atof();
134  } else if (!attrName.CompareTo("ypos")) {
135  pos[1] = attrVal.Atof();
136  } else if (!attrName.CompareTo("zpos")) {
137  pos[2] = attrVal.Atof();
138  } else if (!attrName.CompareTo("phi")) {
139  ang[0] = attrVal.Atof();
140  } else if (!attrName.CompareTo("theta")) {
141  ang[1] = attrVal.Atof();
142  } else if (!attrName.CompareTo("psi")) {
143  ang[2] = attrVal.Atof();
144  } else {
145  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr) << FairLogger::endl;
146  }
147  attr = xml->GetNextAttr(attr);
148  }
149 
150  UInt_t diskUniqueID = mftGeom->getObjectID(Geometry::HalfDiskType, mftGeom->getHalfID(GetUniqueID()), idisk);
151 
152  auto* halfDisk = new HalfDiskSegmentation(diskUniqueID);
153  halfDisk->setPosition(pos);
154  halfDisk->setRotationAngles(ang);
155  halfDisk->setNLadders(nladder);
156  halfDisk->createLadders(xml, node);
157  if (halfDisk->getNLaddersBuild() != halfDisk->getNLadders()) {
158  LOG(FATAL) << "Number of ladder build " << halfDisk->getNLaddersBuild()
159  << " does not correspond to the number declared " << halfDisk->getNLadders() << " Check XML file"
160  << FairLogger::endl;
161  }
162  new ((*mHalfDisks)[idisk]) HalfDiskSegmentation(*halfDisk);
163  delete halfDisk;
164  }
165 
166  // display all child nodes
167  XMLNodePointer_t child = xml->GetChild(node);
168  while (child != nullptr) {
169  createHalfDisks(xml, child);
170  child = xml->GetNext(child);
171  }
172 }

177 void HalfSegmentation::findHalf(TXMLEngine* xml, XMLNodePointer_t node, XMLNodePointer_t& retnode)
178 {
179  // Find in the XML Geometry File the node corresponding to the Half-MFT being build
180  // Set Position and Orientation of the Half-MFT
181  Int_t isTop;
182  Int_t ndisk;
183  Double_t pos[3] = { 0., 0., 0. };
184  Double_t ang[3] = { 0., 0., 0. };
185 
186  TString nodeName = xml->GetNodeName(node);
187  if (!nodeName.CompareTo("half")) {
188  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
189  while (attr != nullptr) {
190  TString attrName = xml->GetAttrName(attr);
191  TString attrVal = xml->GetAttrValue(attr);
192  if (!attrName.CompareTo("top")) {
193  isTop = attrVal.Atoi();
194  if (isTop > 1 || isTop < 0) {
195  LOG(FATAL) << "Wrong Half MFT number : " << isTop << FairLogger::endl;
196  }
197  } else if (!attrName.CompareTo("ndisk")) {
198  ndisk = attrVal.Atoi();
199  if (ndisk > 5 || ndisk < 0) {
200  LOG(ERROR) << "Wrong number of disk : " << ndisk << FairLogger::endl;
201  }
202 
203  } else if (!attrName.CompareTo("xpos")) {
204  pos[0] = attrVal.Atof();
205  } else if (!attrName.CompareTo("ypos")) {
206  pos[1] = attrVal.Atof();
207  } else if (!attrName.CompareTo("zpos")) {
208  pos[2] = attrVal.Atof();
209  } else if (!attrName.CompareTo("phi")) {
210  ang[0] = attrVal.Atof();
211  } else if (!attrName.CompareTo("theta")) {
212  ang[1] = attrVal.Atof();
213  } else if (!attrName.CompareTo("psi")) {
214  ang[2] = attrVal.Atof();
215  } else {
216  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr) << FairLogger::endl;
217  }
218 
219  attr = xml->GetNextAttr(attr);
220  }
221 
222  Geometry* mftGeom = Geometry::instance();
223  if (isTop == mftGeom->getHalfID(GetUniqueID())) {
224  setPosition(pos);
225  setRotationAngles(ang);
226  retnode = node;
227  return;
228  }
229  }
230 
231  // display all child nodes
232  XMLNodePointer_t child = xml->GetChild(node);
233  while (child != nullptr) {
234  findHalf(xml, child, retnode);
235  child = xml->GetNext(child);
236  }
237 }
