277 
278  using type = T;

327  template <typename W>
328  typename std::enable_if<is_specialization<W, ROOTSerialized>::value == true, void>::type
329  snapshot(const Output& spec, W wrapper)
330  {
331  using T = typename W::wrapped_type;
332  static_assert(std::is_same<typename W::hint_type, const char>::value || //
333  std::is_same<typename W::hint_type, TClass>::value || //
334  std::is_void<typename W::hint_type>::value, //
335  "class hint must be of type TClass or const char");
336 
337  auto proxy = mContextRegistry->get<RootObjectContext>()->proxy();
338  FairMQMessagePtr payloadMessage(proxy.createMessage());
339  const TClass* cl = nullptr;
340  if (wrapper.getHint() == nullptr) {
341  // get TClass info by wrapped type
342  cl = TClass::GetClass(typeid(T));
343  } else if (std::is_same<typename W::hint_type, TClass>::value) {
344  // the class info has been passed directly
345  cl = reinterpret_cast<const TClass*>(wrapper.getHint());
346  } else if (std::is_same<typename W::hint_type, const char>::value) {
347  // get TClass info by optional name
348  cl = TClass::GetClass(reinterpret_cast<const char*>(wrapper.getHint()));
349  }
350  if (has_root_dictionary<T>::value == false && cl == nullptr) {
351  std::string msg("ROOT serialization not supported, dictionary not found for type ");
352  if (std::is_same<typename W::hint_type, const char>::value) {
353  msg += reinterpret_cast<const char*>(wrapper.getHint());
354  } else {
355  msg += typeid(T).name();
356  }
357  throw std::runtime_error(msg);
358  }
359  TMessageSerializer().Serialize(*payloadMessage, &wrapper(), cl);
360  addPartToContext(std::move(payloadMessage), spec, o2::header::gSerializationMethodROOT);
361  }
