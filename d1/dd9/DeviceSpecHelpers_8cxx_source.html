201  const std::vector<size_t>& outEdgeIndex,
202  const std::vector<DeviceConnectionEdge>& logicalEdges,
203  const std::vector<EdgeAction>& actions, const WorkflowSpec& workflow,
204  const std::vector<OutputSpec>& outputsMatchers,
205  const std::vector<ChannelConfigurationPolicy>& channelPolicies)
206 {
207  // The topology cannot be empty or not connected. If that is the case, than
208  // something before this went wrong.
209  // FIXME: is that really true???
210  assert(!workflow.empty());

212  // Edges are navigated in order for each device, so the device associaited to
213  // an edge is always the last one created.
214  auto deviceForEdge = [&actions, &workflow, &devices, &logicalEdges](size_t ei) {
215  auto& edge = logicalEdges[ei];
216  auto& action = actions[ei];
217 
218  if (action.requiresNewDevice == false) {
219  assert(devices.empty() == false);
220  return devices.size() - 1;
221  }
222  auto processor = workflow[edge.producer];
223  DeviceSpec device;
224  device.name = processor.name;
225  device.id = processor.name;
226  if (processor.maxInputTimeslices != 1) {
227  device.id = processor.name + "_t" + std::to_string(edge.producerTimeIndex);
228  }
229  device.algorithm = processor.algorithm;
230  device.options = processor.options;
231  device.rank = processor.rank;
232  device.nSlots = processor.nSlots;
233  device.inputTimesliceId = edge.timeIndex;
234  devices.push_back(device);
235  return devices.size() - 1;
236  };

238  auto channelFromDeviceEdgeAndPort = [&workflow, &channelPolicies](const DeviceSpec& device,
239  const DeviceConnectionEdge& edge, short port) {
240  OutputChannelSpec channel;
241  auto& consumer = workflow[edge.consumer];
242  std::string consumerDeviceId = consumer.name;
243  if (consumer.maxInputTimeslices != 1) {
244  consumerDeviceId += "_t" + std::to_string(edge.timeIndex);
245  }
246  channel.name = "from_" + device.id + "_to_" + consumerDeviceId;
247  channel.port = port;
248  for (auto& policy : channelPolicies) {
249  if (policy.match(device.id, consumerDeviceId)) {
250  policy.modifyOutput(channel);
251  break;
252  }
253  }
254  return std::move(channel);
255  };

257  auto connectionIdFromEdgeAndPort = [&connections](const DeviceConnectionEdge& edge, size_t port) {
258  DeviceConnectionId id{ edge.producer, edge.consumer, edge.timeIndex, edge.producerTimeIndex, port };
259  connections.push_back(id);
260  return connections.back();
261  };

263  auto isDifferentDestinationDeviceReferredBy = [&actions](size_t ei) { return actions[ei].requiresNewChannel; };
264 
265  // This creates a new channel for a given edge, if needed. Notice that we
266  // navigate edges in a per device fashion (creating those if they are not
267  // alredy there) and create a new channel only if it connects two new
268  // devices. Whether or not this is the case was previously computed
269  // in the action.requiresNewChannel field.
270  auto createChannelForDeviceEdge = [&devices, &logicalEdges, &resources, &channelFromDeviceEdgeAndPort,
271  &connectionIdFromEdgeAndPort, &outputsMatchers, &deviceIndex,
272  &workflow](size_t di, size_t ei) {
273  auto& device = devices[di];
274  auto& edge = logicalEdges[ei];
275 
276  deviceIndex.emplace_back(DeviceId{ edge.producer, edge.producerTimeIndex, di });
277 
278  OutputChannelSpec channel = channelFromDeviceEdgeAndPort(device, edge, resources.back().port);
279  const DeviceConnectionId& id = connectionIdFromEdgeAndPort(edge, resources.back().port);
280  resources.pop_back();
281 
282  device.outputChannels.push_back(channel);
283  return device.outputChannels.size() - 1;
284  };

286  // Notice how we need to behave in two different ways depending
287  // whether this is a real OutputRoute or if it's a forward from
288  // a previous consumer device.
289  // FIXME: where do I find the InputSpec for the forward?
290  auto appendOutputRouteToSourceDeviceChannel = [&outputsMatchers, &workflow, &devices, &logicalEdges](
291  size_t ei, size_t di, size_t ci) {
292  assert(ei < logicalEdges.size());
293  assert(di < devices.size());
294  assert(ci < devices[di].outputChannels.size());
295  auto& edge = logicalEdges[ei];
296  auto& device = devices[di];
297  assert(edge.consumer < workflow.size());
298  auto& consumer = workflow[edge.consumer];
299  auto& channel = devices[di].outputChannels[ci];
300  assert(edge.outputGlobalIndex < outputsMatchers.size());
301 
302  if (edge.isForward == false) {
303  OutputRoute route{
304  edge.timeIndex,
305  consumer.maxInputTimeslices,
306  outputsMatchers[edge.outputGlobalIndex],
307  channel.name
308  };
309  device.outputs.emplace_back(route);
310  } else {
311  ForwardRoute route{
312  edge.timeIndex,
313  consumer.maxInputTimeslices,
314  workflow[edge.consumer].inputs[edge.consumerInputIndex],
315  channel.name
316  };
317  device.forwards.emplace_back(route);
318  }
319  };

321  auto sortDeviceIndex = [&deviceIndex]() { std::sort(deviceIndex.begin(), deviceIndex.end()); };
322 
323  auto lastChannelFor = [&devices](size_t di) {
324  assert(di < devices.size());
325  assert(devices[di].outputChannels.empty() == false);
326  return devices[di].outputChannels.size() - 1;
327  };

329  //
330  // OUTER LOOP
331  //
332  // We need to create all the channels going out of a device, and associate
333  // routes to them for this reason
334  // we iterate over all the edges (which are per-datatype exchanged) and
335  // whenever we need to connect to a new device we create the channel. `device`
336  // here refers to the source device. This loop will therefore not create the
337  // devices which acts as sink, which are done in the preocessInEdgeActions
338  // function.
339  for (auto edge : outEdgeIndex) {
340  auto device = deviceForEdge(edge);
341  size_t channel = -1;
342  if (isDifferentDestinationDeviceReferredBy(edge)) {
343  channel = createChannelForDeviceEdge(device, edge);
344  } else {
345  channel = lastChannelFor(device);
346  }
347  appendOutputRouteToSourceDeviceChannel(edge, device, channel);
348  }
349  sortDeviceIndex();
350 }

352 void DeviceSpecHelpers::processInEdgeActions(std::vector<DeviceSpec>& devices,
353  std::vector<DeviceId>& deviceIndex,
354  std::vector<ComputingResource> &resources,
355  const std::vector<DeviceConnectionId>& connections,
356  const std::vector<size_t>& inEdgeIndex,
357  const std::vector<DeviceConnectionEdge>& logicalEdges,
358  const std::vector<EdgeAction>& actions, const WorkflowSpec& workflow,
359  std::vector<LogicalForwardInfo> const& availableForwardsInfo,
360  std::vector<ChannelConfigurationPolicy> const& channelPolicies)
361 {
362  auto const& constDeviceIndex = deviceIndex;
363 
364  auto findProducerForEdge = [&logicalEdges, &constDeviceIndex](size_t ei) {
365  auto& edge = logicalEdges[ei];
366 
367  DeviceId pid{ edge.producer, edge.producerTimeIndex, 0 };
368  auto deviceIt = std::lower_bound(constDeviceIndex.cbegin(), constDeviceIndex.cend(), pid);
369  // By construction producer should always be there
370  assert(deviceIt != constDeviceIndex.end());
371  assert(deviceIt->processorIndex == pid.processorIndex && deviceIt->timeslice == pid.timeslice);
372  return deviceIt->deviceIndex;
373  };
374 
375  // Notice that to start with, consumer exists only if they also are
376  // producers, so we need to create one if it does not exist. Given this is
377  // stateful, we keep an eye on what edge was last searched to make sure we
378  // are not screwing up.
379  //
380  // Notice this is not thread safe.
381  decltype(deviceIndex.begin()) lastConsumerSearch;
382  size_t lastConsumerSearchEdge;
383  auto hasConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge, &deviceIndex,
384  &logicalEdges](size_t ei) -> int {
385  auto& edge = logicalEdges[ei];
386  DeviceId cid{ edge.consumer, edge.timeIndex, 0 };
387  lastConsumerSearchEdge = ei; // This will invalidate the cache
388  lastConsumerSearch = std::lower_bound(deviceIndex.begin(), deviceIndex.end(), cid);
389  return lastConsumerSearch != deviceIndex.end() && cid.processorIndex == lastConsumerSearch->processorIndex &&
390  cid.timeslice == lastConsumerSearch->timeslice;
391  };
392 
393  // The passed argument is there just to check. We do know that the last searched
394  // is the one we want.
395  auto getConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge](size_t ei) {
396  assert(ei == lastConsumerSearchEdge);
397  return lastConsumerSearch->deviceIndex;
398  };
399 
400  auto createNewDeviceForEdge = [&workflow, &logicalEdges, &devices, &deviceIndex](size_t ei) {
401  auto& edge = logicalEdges[ei];
402  auto& processor = workflow[edge.consumer];
403  DeviceSpec device;
404  device.name = processor.name;
405  device.id = processor.name;
406  if (processor.maxInputTimeslices != 1) {
407  device.id += "_t" + std::to_string(edge.timeIndex);
408  }
409  device.algorithm = processor.algorithm;
410  device.options = processor.options;
411  device.rank = processor.rank;
412  device.nSlots = processor.nSlots;
413  device.inputTimesliceId = edge.timeIndex;
414  // FIXME: maybe I should use an std::map in the end
415  // but this is really not performance critical
416  auto id = DeviceId{ edge.consumer, edge.timeIndex, devices.size() };
417  devices.push_back(device);
418  deviceIndex.push_back(id);
419  std::sort(deviceIndex.begin(), deviceIndex.end());
420  return devices.size() - 1;
421  };
422 
423  // We search for a preexisting outgoing connection associated to this edge.
424  // This is to retrieve the port of the source.
425  // This has to exists, because we already created all the outgoing connections
426  // so it's just a matter of looking it up.
427  auto findMatchingOutgoingPortForEdge = [&logicalEdges, &connections](size_t ei) {
428  auto const& edge = logicalEdges[ei];
429  DeviceConnectionId connectionId{ edge.producer, edge.consumer, edge.timeIndex, edge.producerTimeIndex, 0 };
430 
431  auto it = std::lower_bound(connections.begin(), connections.end(), connectionId);
432 
433  assert(it != connections.end());
434  assert(it->producer == connectionId.producer);
435  assert(it->consumer == connectionId.consumer);
436  assert(it->timeIndex == connectionId.timeIndex);
437  assert(it->producerTimeIndex == connectionId.producerTimeIndex);
438  return it->port;
439  };

441  auto checkNoDuplicatesFor = [](std::vector<InputChannelSpec> const& channels, const std::string& name) {
442  for (auto const& channel : channels) {
443  if (channel.name == name) {
444  return false;
445  }
446  }

449  auto appendInputChannelForConsumerDevice = [&devices, &connections, &checkNoDuplicatesFor, &channelPolicies](
450  size_t pi, size_t ci, int16_t port) {
451  auto const& producerDevice = devices[pi];
452  auto& consumerDevice = devices[ci];
453  InputChannelSpec channel;
454  channel.name = "from_" + producerDevice.id + "_to_" + consumerDevice.id;
455  channel.port = port;
456  for (auto& policy : channelPolicies) {
457  if (policy.match(producerDevice.id, consumerDevice.id)) {
458  policy.modifyInput(channel);
459  break;
460  }
461  }
462  assert(checkNoDuplicatesFor(consumerDevice.inputChannels, channel.name));
463  consumerDevice.inputChannels.push_back(channel);
464  return consumerDevice.inputChannels.size() - 1;
465  };
466 
467  // I think this is trivial, since I think it should always be the last one,
468  // in case it's not actually the case, I should probably do an actual lookup
469  // here.
470  auto getChannelForEdge = [&devices](size_t pi, size_t ci) {
471  auto& consumerDevice = devices[ci];
472  return consumerDevice.inputChannels.size() - 1;
473  };
474 
475  // This is always called when adding a new channel, so we can simply refer
476  // to back. Notice also that this is the place where it makes sense to
477  // assign the forwarding, given that the forwarded stuff comes from some
478  // input.
479  auto appendInputRouteToDestDeviceChannel = [&devices, &logicalEdges, &workflow](size_t ei, size_t di, size_t ci) {
480  auto const& edge = logicalEdges[ei];
481  auto const& consumer = workflow[edge.consumer];
482  auto& consumerDevice = devices[di];
483 
484  InputRoute::CreationConfigurator creationConfigurator;
485  InputRoute::DanglingConfigurator danglingConfigurator;
486  InputRoute::ExpirationConfigurator expirationConfigurator;
487  auto const& inputSpec = consumer.inputs[edge.consumerInputIndex];
488  auto const& sourceChannel = consumerDevice.inputChannels[ci].name;
489 
490  switch (consumer.inputs[edge.consumerInputIndex].lifetime) {
491  case Lifetime::Timeframe:
492  creationConfigurator = ExpirationHandlerHelpers::dataDrivenConfigurator();
493  danglingConfigurator = ExpirationHandlerHelpers::danglingTimeframeConfigurator();
494  expirationConfigurator = ExpirationHandlerHelpers::expiringTimeframeConfigurator();
495  break;
496  case Lifetime::Condition:
497  creationConfigurator = ExpirationHandlerHelpers::dataDrivenConfigurator();
498  danglingConfigurator = ExpirationHandlerHelpers::danglingConditionConfigurator();
499  expirationConfigurator = ExpirationHandlerHelpers::expiringConditionConfigurator(inputSpec, sourceChannel);
500  break;
501  case Lifetime::QA:
502  creationConfigurator = ExpirationHandlerHelpers::dataDrivenConfigurator();
503  danglingConfigurator = ExpirationHandlerHelpers::danglingQAConfigurator();
504  expirationConfigurator = ExpirationHandlerHelpers::expiringQAConfigurator();
505  break;
506  case Lifetime::Timer:
507  creationConfigurator = ExpirationHandlerHelpers::timeDrivenConfigurator(inputSpec);
508  danglingConfigurator = ExpirationHandlerHelpers::danglingTimerConfigurator(inputSpec);
509  expirationConfigurator = ExpirationHandlerHelpers::expiringTimerConfigurator(inputSpec, sourceChannel);
510  break;
511  case Lifetime::Enumeration:
512  creationConfigurator = ExpirationHandlerHelpers::enumDrivenConfigurator(inputSpec);
513  danglingConfigurator = ExpirationHandlerHelpers::danglingEnumerationConfigurator(inputSpec);
514  expirationConfigurator = ExpirationHandlerHelpers::expiringEnumerationConfigurator(inputSpec, sourceChannel);
515  break;
516  case Lifetime::Transient:
517  creationConfigurator = ExpirationHandlerHelpers::dataDrivenConfigurator();
518  danglingConfigurator = ExpirationHandlerHelpers::danglingTransientConfigurator();
519  expirationConfigurator = ExpirationHandlerHelpers::expiringTransientConfigurator(inputSpec);
520  break;
521  }
522 
523  InputRoute route{
524  inputSpec,
525  sourceChannel,
526  edge.producerTimeIndex,
527  creationConfigurator,
528  danglingConfigurator,
529  expirationConfigurator
530  };
531 
532  consumerDevice.inputs.push_back(route);
533  };
534 
535  // Outer loop. A new device is needed for each
536  // of the sink data processors.
537  // New InputChannels need to refer to preexisting OutputChannels we create
538  // previously.
539  for (size_t edge : inEdgeIndex) {
540  auto& action = actions[edge];
541 
542  size_t consumerDevice;
543 
544  if (action.requiresNewDevice) {
545  if (hasConsumerForEdge(edge)) {
546  consumerDevice = getConsumerForEdge(edge);
547  } else {
548  consumerDevice = createNewDeviceForEdge(edge);
549  }
550  }
551  size_t producerDevice = findProducerForEdge(edge);
552 
553  size_t channel = -1;
554  if (action.requiresNewChannel) {
555  int16_t port = findMatchingOutgoingPortForEdge(edge);
556  channel = appendInputChannelForConsumerDevice(producerDevice, consumerDevice, port);
557  } else {
558  channel = getChannelForEdge(producerDevice, consumerDevice);
559  }
560  appendInputRouteToDestDeviceChannel(edge, consumerDevice, channel);
561  }
562 }

569  std::vector<CompletionPolicy> const& completionPolicies,

627 void DeviceSpecHelpers::prepareArguments(int argc, char** argv, bool defaultQuiet, bool defaultStopped,
628  const std::vector<DeviceSpec>& deviceSpecs,
629  const std::vector<ConfigParamSpec> &workflowOptions,
630  std::vector<DeviceExecution>& deviceExecutions,
631  std::vector<DeviceControl>& deviceControls)
632 {
633  assert(argc > 0); // we require to have the program name as the first argument
634  assert(deviceSpecs.size() == deviceExecutions.size());
635  assert(deviceControls.size() == deviceExecutions.size());
636  for (size_t si = 0; si < deviceSpecs.size(); ++si) {
637  auto& spec = deviceSpecs[si];
638  auto& control = deviceControls[si];
639  auto& execution = deviceExecutions[si];
640 
641  control.quiet = defaultQuiet;
642  control.stopped = defaultStopped;
643 
644  // We duplicate the list of options, filtering only those
645  // which are actually relevant for the given device. The additional
646  // four are to add
647  // * name of the executable
648  // * --framework-id <id> so that we can use the workflow
649  // executable also in other context where we do not fork, e.g. DDS.
650  // * final NULL required by execvp
651  //
652  // We do it here because we are still in the parent and we can therefore
653  // capture them to be displayed in the GUI or to populate the DDS configuration
654  // to dump
655 
656  // Set up options for the device running underneath
657  // FIXME: add some checksum in framework id. We could use this
658  // to avoid redeploys when only a portion of the workflow is changed.
659  // FIXME: this should probably be done in one go with char *, but I am lazy.
660  std::vector<std::string> tmpArgs = { argv[0], "--id", spec.id.c_str(), "--control", "static",
661  "--log-color", "false", "--color", "false" };
662 
663  // do the filtering of options:
664  // 1) forward options belonging to this specific DeviceSpec
665  // 2) global options defined in getForwardedDeviceOptions and workflow option are
666  // always forwarded and need to be handled separately
667  const char* name = spec.name.c_str();
668  bpo::options_description od; // option descriptions per process
669  bpo::options_description foDesc; // forwarded options for all processes
670  ConfigParamsHelper::prepareOptionsDescription(spec.options, od);
671  od.add_options()(name, bpo::value<std::string>());
672  ConfigParamsHelper::prepareOptionsDescription(workflowOptions, foDesc);
673  foDesc.add(getForwardedDeviceOptions());
674 
675  using FilterFunctionT = std::function<void(decltype(argc), decltype(argv), decltype(od))>;
676 
677  // the filter function will forward command line arguments based on the option
678  // definition passed to it. All options of the program option definition will be forwarded
679  // if found in the argument list. If not found they will be added with the default value
680  FilterFunctionT filterArgsFct = [&](int largc, char** largv, const bpo::options_description& odesc) {
681  // spec contains options
682  bpo::command_line_parser parser{ largc, largv };
683  parser.options(odesc).allow_unregistered();
684  bpo::parsed_options parsed_options = parser.run();
685 
686  bpo::variables_map varmap;
687  bpo::store(parsed_options, varmap);
688 
689  // options can be grouped per processor spec, the group is entered by
690  // the option created from the actual processor spec name
691  // if specified, the following string is interpreted as a sequence
692  // of arguments
693  if (varmap.count(name) > 0) {
694  // strangely enough, the first argument of the group argument string
695  // is marked as defaulted by the parser and is thus ignored. not fully
696  // understood but adding a dummy argument in front cures this
697  auto arguments = "--unused " + varmap[name].as<std::string>();
698  wordexp_t expansions;
699  wordexp(arguments.c_str(), &expansions, 0);
700  filterArgsFct(expansions.we_wordc, expansions.we_wordv, odesc);
701  wordfree(&expansions);
702  return;
703  }
704 
705  const char* child_driver_key = "child-driver";
706  if (varmap.count(child_driver_key) > 0) {
707  auto arguments = varmap[child_driver_key].as<std::string>();
708  wordexp_t expansions;
709  wordexp(arguments.c_str(), &expansions, 0);
710  tmpArgs.insert(tmpArgs.begin(), expansions.we_wordv, expansions.we_wordv + expansions.we_wordc);
711  }
712 
713  for (const auto varit : varmap) {
714  // find the option belonging to key, add if the option has been parsed
715  // and is not defaulted
716  const auto* description = odesc.find_nothrow(varit.first, false);
717  if (description && varmap.count(varit.first)) {
718  // check the semantics of the value
719  auto semantic = description->semantic();
720  const char* optarg = "";
721  if (semantic) {
722  // the value semantics allows different properties like
723  // multitoken, zero_token and composing
724  // currently only the simple case is supported
725  assert(semantic->min_tokens() <= 1);
726  //assert(semantic->max_tokens() && semantic->min_tokens());
727  if (semantic->min_tokens() > 0) {
728  tmpArgs.emplace_back("--");
729  tmpArgs.back() += varit.first;
730  // add the token
731  tmpArgs.emplace_back(varit.second.as<std::string>());
732  optarg = tmpArgs.back().c_str();
733  } else if (semantic->min_tokens() == 0 && varit.second.as<bool>()) {
734  tmpArgs.emplace_back("--");
735  tmpArgs.back() += varit.first;
736  }
737  }
738  control.options.insert(std::make_pair(varit.first, optarg));
739  }
740  }
741  };
742 
743  // filter global options and workflow options independent of option groups
744  filterArgsFct(argc, argv, foDesc);
745  // filter device options, and handle option groups
746  filterArgsFct(argc, argv, od);
747 
748  // Add the channel configuration
749  for (auto& channel : spec.outputChannels) {
750  tmpArgs.emplace_back(std::string("--channel-config"));
751  tmpArgs.emplace_back(outputChannel2String(channel));
752  }
753  for (auto& channel : spec.inputChannels) {
754  tmpArgs.emplace_back(std::string("--channel-config"));
755  tmpArgs.emplace_back(inputChannel2String(channel));
756  }
757 
758  // We create the final option list, depending on the channels
759  // which are present in a device.
760  for (auto& arg : tmpArgs) {
761  execution.args.emplace_back(strdup(arg.c_str()));
762  }
763  // execvp wants a NULL terminated list.
764  execution.args.push_back(nullptr);
765 
766  // FIXME: this should probably be reflected in the GUI
767  std::ostringstream str;
768  for (size_t ai = 0; ai < execution.args.size() - 1; ai++) {
769  assert(execution.args[ai]);
770  str << " " << execution.args[ai];
771  }
772  LOG(DEBUG) << "The following options are being forwarded to " << spec.id << ":" << str.str();
773  }
774 }
