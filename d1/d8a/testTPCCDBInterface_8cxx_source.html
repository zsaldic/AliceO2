71 void checkCalPadEqual(const CalPad& data, const CalPad& dataRead)
72 {
73  auto& mapper = Mapper::instance();
74  const auto numberOfPads = mapper.getPadsInSector() * 36;
75 
76  float sumROC = 0.f;
77 
78  int numberOfPadsROC = 0;
79 
80  for (auto const& arrays : boost::combine(data.getData(), dataRead.getData())) {
81  for (auto const& val : boost::combine(arrays.get<0>().getData(), arrays.get<1>().getData())) {
82  sumROC += (val.get<0>() - val.get<1>());
83  ++numberOfPadsROC;
84  }
85  }
86 
87  BOOST_CHECK_EQUAL(data.getName(), dataRead.getName());
88  BOOST_CHECK_CLOSE(sumROC, 0.f, 1.E-12);
89  BOOST_CHECK_EQUAL(numberOfPadsROC, numberOfPads);
90 }
