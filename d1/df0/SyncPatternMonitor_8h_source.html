103 void SyncPatternMonitor::checkWord(const short hw, const short pos) {
104  ++mCheckedWords;
105  if (hw == SYNC_PATTERN[mPosition]) ++mPosition;
106  else if (! (mPosition == SYNC_START+2 && hw == SYNC_PATTERN[mPosition-1])) mPosition = SYNC_START;
107  // Don't reset mPosition at the beginning to avoid missing of start of sync pattern in cases like
108  //
109  //
110  // ... A A A B B ...
111  //
112  // ^ ^ ^
113  // random A _| | |_ would trigger reset
114  // |
115  // real start
116 
117  if (mPosition == 32) patternFound(pos);
118 };
