44 // This is a prototype of a macro to test running the HLT O2 CA Tracking library on a root input file containg
45 // TClonesArray of clusters.
46 // It wraps the TPCCATracking class, forwwarding all parameters, which are passed as options.
47 int runCATrackingClusterNative(TString inputFile, TString outputFile, TString options = "")
48 {
49  if (inputFile.EqualTo("") || outputFile.EqualTo("")) {
50  printf("Filename missing\n");
51  return (1);
52  }
53  TPCCATracking tracker;
54 
55  if (tracker.initialize(options.Data())) {
56  printf("Error initializing tracker\n");
57  return (0);
58  }
59 
60  std::vector<ClusterNativeContainer> cont;
61  std::vector<MCLabelContainer> contMC;
62  bool doMC = true;
63 
64  TFile fin(inputFile);
65  for (int i = 0; i < Constants::MAXSECTOR; i++) {
66  for (int j = 0; j < Constants::MAXGLOBALPADROW; j++) {
67  TString contName = Form("clusters_sector_%d_row_%d", i, j);
68  TObject* tmp = fin.FindObjectAny(contName);
69  if (tmp == nullptr) {
70  printf("Error reading clusters %s\n", contName.Data());
71  } else {
72  cont.emplace_back(std::move(*reinterpret_cast<ClusterNativeContainer*>(tmp)));
73  tmp = fin.FindObjectAny(Form("clustersMCTruth_sector_%d_row_%d", i, j));
74 
75  if (tmp == nullptr) {
76  printf("Error, clustersMCTruth missing or clusters and clustersMCtruth out of sync! Disabling MC data\n");
77  doMC = false;
78  } else {
79  contMC.emplace_back(std::move(*reinterpret_cast<MCLabelContainer*>(tmp)));
80  }
81  }
82  }
83  }
84  fin.Close();
85 
86  std::unique_ptr<ClusterNativeAccessFullTPC> clusters =
87  ClusterNativeHelper::createClusterNativeIndex(cont, doMC ? &contMC : nullptr);
88 
89  vector<TrackTPC> tracks;
90  MCLabelContainer tracksMC;
91 
92  TFile fout(outputFile, "recreate");
93  TTree tout("events", "events");
94  tout.Branch("Tracks", &tracks);
95  tout.Branch("TracksMCTruth", &tracksMC);
96 
97  printf("Processing time frame\n");
98  if (tracker.runTracking(*clusters, &tracks, doMC ? &tracksMC : nullptr) == 0) {
99  printf("\tFound %d tracks\n", (int)tracks.size());
100  } else {
101  printf("\tError during tracking\n");
102  }
103 
104  float artificialVDrift = tracker.getPseudoVDrift();
105  float tfReferenceLength = tracker.getTFReferenceLength();
106 
107  // partial printout of 100 tracks
108  int step = tracks.size() / 100;
109  step = step < 1 ? 1 : step;
110  for (unsigned int i = 0; i < tracks.size(); i += step) {
111  // Loop over clusters
112  for (int j = tracks[i].getNClusterReferences() - 1; j >= 0; j--) {
113  // Get cluster references
114  uint8_t sector, row;
115  uint32_t clusterIndexInRow;
116  tracks[i].getClusterReference(j, sector, row, clusterIndexInRow);
117  const ClusterNative& cl = tracks[i].getCluster(j, *clusters, sector, row);
118  const ClusterNative& clLast = tracks[i].getCluster(0, *clusters);
119  // RS: TODO: account for possible A/C merged tracks
120  float sideFactor = tracks[i].hasASideClustersOnly() ? -1.f : 1.f;
121  printf(
122  "Track %d: Side %s Estimated timeVertex: %f, num clusters %d, innermost cluster: sector %d, row %d, "
123  "ClusterTime %f, TrackParam X %f Z %f --> T %f, LastClusterT: %f\n",
124  i, tracks[i].hasBothSidesClusters() ? "AC" : (tracks[i].hasASideClusters() ? "A" : "C"), tracks[i].getTime0(),
125  tracks[i].getNClusterReferences(), (int)sector, (int)row, cl.getTime(), tracks[i].getX(), tracks[i].getZ(),
126  tracks[i].getTime0() - sideFactor * tracks[i].getZ() / artificialVDrift, clLast.getTime());
127  break; // Reduce output in this example code
128  }
129  }
130 
131  tout.Fill();

135  tracker.deinitialize();
