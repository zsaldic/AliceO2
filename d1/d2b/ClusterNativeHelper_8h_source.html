100  // add clusters from a flattened buffer starting with an attribute, e.g. ClusterGroupAttribute followed
101  // by the array of ClusterNative, the number of clusters is determined from the size of the buffer
102  // FIXME: add mc labels
103  template <typename AttributeT>
104  static int addFlatBuffer(ClusterNativeAccessFullTPC& clusterIndex, unsigned char* buffer, size_t size)
105  {
106  if (buffer == nullptr || size < sizeof(AttributeT) || (size - sizeof(AttributeT)) % sizeof(ClusterNative) != 0) {
107  // this is not a valid message, incompatible size
108  return -1;
109  }
110  const auto& groupAttribute = *reinterpret_cast<AttributeT*>(buffer);
111  auto nofClusters = (size - sizeof(AttributeT)) / sizeof(ClusterNative);
112  auto ptrClusters = reinterpret_cast<ClusterNative*>(buffer + sizeof(groupAttribute));
113  clusterIndex.clusters[groupAttribute.sector][groupAttribute.globalPadRow] = ptrClusters;
114  clusterIndex.nClusters[groupAttribute.sector][groupAttribute.globalPadRow] = nofClusters;
115  return nofClusters;
116  }

137  template <typename DataArrayType, typename MCArrayType, typename CheckFct>
138  static int fillIndex(ClusterNativeAccessFullTPC& clusterIndex,
139  DataArrayType& inputs, MCArrayType& mcinputs,
140  CheckFct checkFct = [](auto&) { return true; })
141  {
142  static_assert(std::tuple_size<DataArrayType>::value == std::tuple_size<MCArrayType>::value);
143  memset(&clusterIndex, 0, sizeof(clusterIndex));
144  int result = 0;
145  for (size_t index = 0; index < NSectors; index++) {
146  if (!checkFct(index)) {
147  continue;
148  }
149  int locres = parseSector(inputs[index], mcinputs[index], clusterIndex);
150  if (result >= 0 && locres >= 0) {
151  result += locres;
152  } else if (result >= 0) {
153  result = locres;
154  }
155  }
156  return result;
157  }
