179 inline void Cluster::setClusterUsage(Int_t n)
180 {
181  // set cluster usage counter
182  mNxNzN &= ~(kMaskClUse << kOffsClUse);
183  mNxNzN |= (n & kMaskClUse) << kOffsClUse;
184  if (n < 2)
185  resetBit(kShared);
186  if (!n)
187  resetBit(kUsed);
188 }
