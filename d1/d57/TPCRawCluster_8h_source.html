115  int init(const char* filename) {
116  std::ifstream input(filename, std::ifstream::binary);
117  clear(0);
118  if (input) {
119  // get length of file:
120  input.seekg (0, input.end);
121  int length = input.tellg();
122  input.seekg (0, input.beg);
123 
124  // allocate memory:
125  mBuffer = new Buffer_t[length];
126  mBufferSize = length;
127 
128  // read data as a block:
129  input.read(reinterpret_cast<char*>(mBuffer),length);
130  if (!input.good()) {
131  clear(-1);
132  std::cerr << "failed to read " << length << " byte(s) from file " << filename << std::endl;
133  }
134 
135  input.close();
136  return init();
137  }
138  std::cerr << "failed to open file " << filename << std::endl;
139  return -1;
140  }
141 
142  int init(unsigned char* buffer, int size) {
143  if (!buffer || size <= 0) return -1;
144  clear(0);
145  mBuffer = new Buffer_t[size];
146  mBufferSize = size;
147  memcpy(mBuffer, buffer, size);
148  return init();

177 private:
178  int init() {
179  if (mBuffer == nullptr || mBufferSize == 0) return 0;
180  if (mBufferSize < sizeof(RawClusterData)) return -1;
181  RawClusterData& clusterData = *reinterpret_cast<RawClusterData*>(mBuffer);
182 
183  if (clusterData.fCount * sizeof(RawCluster) + sizeof(RawClusterData) > mBufferSize) {
184  std::cerr << "Format error, " << clusterData.fCount << " cluster(s) "
185  << "would require "
186  << (clusterData.fCount * sizeof(RawCluster) + sizeof(RawClusterData))
187  << " byte(s), but only " << mBufferSize << " available" << std::endl;
188  return clear(-1);
189  }
190 
191  mNClusters = clusterData.fCount;
192  mClusters = clusterData.fClusters;
193  mClustersEnd = mClusters + mNClusters;
194 
195  return mNClusters;
196  }
197
