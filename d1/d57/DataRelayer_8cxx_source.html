115 void DataRelayer::processDanglingInputs(std::vector<ExpirationHandler> const& expirationHandlers,
116  ServiceRegistry& services)
117 {
118  // Create any slot for the time based fields
119  for (size_t hi = 0; hi < expirationHandlers.size(); ++hi) {
120  expirationHandlers[hi].creator(mTimesliceIndex);
121  }
122  // Expire the records as needed.
123  for (size_t ti = 0; ti < mTimesliceIndex.size(); ++ti) {
124  TimesliceSlot slot{ ti };
125  if (mTimesliceIndex.isValid(slot) == false) {
126  continue;
127  }
128  assert(mDistinctRoutesIndex.empty() == false);
129  for (size_t ri = 0; ri < mDistinctRoutesIndex.size(); ++ri) {
130  auto& route = mInputRoutes[mDistinctRoutesIndex[ri]];
131  auto& expirator = expirationHandlers[mDistinctRoutesIndex[ri]];
132  auto timestamp = mTimesliceIndex.getTimesliceForSlot(slot);
133  auto& part = mCache[ti * mDistinctRoutesIndex.size() + ri];
134  if (part.header != nullptr) {
135  continue;
136  }
137  if (part.payload != nullptr) {
138  continue;
139  }
140  if (!expirator.checker) {
141  continue;
142  }
143  if (expirator.checker(timestamp.value) == false) {
144  continue;
145  }
146 
147  assert(ti * mDistinctRoutesIndex.size() + ri < mCache.size());
148  assert(expirator.handler);
149  expirator.handler(services, part, timestamp.value);
150  mTimesliceIndex.markAsDirty(slot, true);
151  assert(part.header != nullptr);
152  assert(part.payload != nullptr);
153  }
154  }
155 }

160 size_t matchToContext(void* data,
161  std::vector<DataDescriptorMatcher> const& matchers,
162  VariableContext& context)
163 {
164  for (size_t ri = 0, re = matchers.size(); ri < re; ++ri) {
165  auto& matcher = matchers[ri];
166 
167  if (matcher.match(reinterpret_cast<char const*>(data), context)) {
168  context.commit();
169  return ri;
170  }
171  context.discard();
172  }
173  return INVALID_INPUT;
174 }

195 DataRelayer::RelayChoice
196 DataRelayer::relay(std::unique_ptr<FairMQMessage> &&header,
197  std::unique_ptr<FairMQMessage> &&payload) {
198  // STATE HOLDING VARIABLES
199  // This is the class level state of the relaying. If we start supporting
200  // multithreading this will have to be made thread safe before we can invoke
201  // relay concurrently.
202  auto const& inputRoutes = mInputRoutes;
203  auto& index = mTimesliceIndex;
204 
205  auto& cache = mCache;
206  auto const& readonlyCache = mCache;
207  auto& metrics = mMetrics;
208  auto numInputTypes = mDistinctRoutesIndex.size();
209 
210  // IMPLEMENTATION DETAILS
211  //
212  // This returns the identifier for the given input. We use a separate
213  // function because while it's trivial now, the actual matchmaking will
214  // become more complicated when we will start supporting ranges.
215  auto getInputTimeslice = [& matchers = mInputMatchers,
216  &header,
217  &index ](VariableContext & context)
218  ->std::tuple<int, TimesliceId>
219  {
222  auto input = matchToContext(header->GetData(), matchers, context);
223 
224  if (input == INVALID_INPUT) {
225  return {
226  INVALID_INPUT,
227  TimesliceId{ TimesliceId::INVALID },
228  };
229  }
232  if (auto pval = std::get_if<uint64_t>(&context.get(0))) {
233  TimesliceId timeslice{ *pval };
234  return { input, timeslice };
235  }
236  // If we get here it means we need to push something out of the cache.
237  return {
238  INVALID_INPUT,
239  TimesliceId{ TimesliceId::INVALID },
240  };
241  };
242 
243  // We need to prune the cache from the old stuff, if any. Otherwise we
244  // simply store the payload in the cache and we mark relevant bit in the
245  // hence the first if.
246  auto pruneCache = [&cache, &numInputTypes, &index, &metrics](TimesliceSlot slot) {
247  assert(cache.empty() == false);
248  assert(index.size() * numInputTypes == cache.size());
249  // Prune old stuff from the cache, hopefully deleting it...
250  // We set the current slot to the timeslice value, so that old stuff
251  // will be ignored.
252  assert(numInputTypes * slot.index < cache.size());
253  for (size_t ai = slot.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
254  cache[ai].header.reset(nullptr);
255  cache[ai].payload.reset(nullptr);
256  metrics.send({ 0, sMetricsNames[ai] });
257  }
258  };
259 
260  // We need to check if the slot for the current input is already taken for
261  // the current timeslice.
262  // This should never happen, however given this is dependent on the input
263  // we want to protect again malicious / bad upstream source.
264  auto hasCacheInputAlreadyFor = [&cache, &index, &numInputTypes](TimesliceSlot slot, int input) {
265  PartRef& currentPart = cache[numInputTypes * slot.index + input];
266  return (currentPart.payload != nullptr) || (currentPart.header != nullptr);
267  };
268 
269  // Actually save the header / payload in the slot
270  auto saveInSlot = [&header, &payload, &cache, &numInputTypes, &metrics](TimesliceId timeslice, int input, TimesliceSlot slot) {
271  auto cacheIdx = numInputTypes * slot.index + input;
272  PartRef& currentPart = cache[cacheIdx];
273  metrics.send({ 1, sMetricsNames[cacheIdx] });
274  PartRef ref{std::move(header), std::move(payload)};
275  currentPart = std::move(ref);
276  assert(header.get() == nullptr && payload.get() == nullptr);
277  };

279  auto updateStatistics = [& stats = mStats](TimesliceIndex::ActionTaken action)
280  {
281  // Update statistics for what happened
282  switch (action) {
283  case TimesliceIndex::ActionTaken::DropObsolete:
284  stats.droppedIncomingMessages++;
285  break;
286  case TimesliceIndex::ActionTaken::DropInvalid:
287  stats.malformedInputs++;
288  stats.droppedIncomingMessages++;
289  break;
290  case TimesliceIndex::ActionTaken::ReplaceUnused:
291  stats.relayedMessages++;
292  break;
293  case TimesliceIndex::ActionTaken::ReplaceObsolete:
294  stats.droppedComputations++;
295  stats.relayedMessages++;
296  break;
297  }
298  };
299 
300  // OUTER LOOP
301  //
302  // This is the actual outer loop processing input as part of a given
303  // timeslice. All the other implementation details are hidden by the lambdas
304  auto input = INVALID_INPUT;
305  auto timeslice = TimesliceId{ TimesliceId::INVALID };
306  auto slot = TimesliceSlot{ TimesliceSlot::INVALID };
307 
308  // First look for matching slots which already have some
309  // partial match.
310  for (size_t ci = 0; ci < index.size(); ++ci) {
311  slot = TimesliceSlot{ ci };
312  if (index.isValid(slot) == false) {
313  continue;
314  }
315  std::tie(input, timeslice) = getInputTimeslice(index.getVariablesForSlot(slot));
316  if (input != INVALID_INPUT) {
317  break;
318  }
319  }
320 
321  // If we did not find anything, look for slots which
322  // are invalid.
323  if (input == INVALID_INPUT) {
324  for (size_t ci = 0; ci < index.size(); ++ci) {
325  slot = TimesliceSlot{ ci };
326  if (index.isValid(slot) == true) {
327  continue;
328  }
329  std::tie(input, timeslice) = getInputTimeslice(index.getVariablesForSlot(slot));
330  if (input != INVALID_INPUT) {
331  break;
332  }
333  }
334  }
335 
337  if (input != INVALID_INPUT && TimesliceId::isValid(timeslice) && TimesliceSlot::isValid(slot)) {
338  LOG(DEBUG) << "Received timeslice " << timeslice.value;
339  saveInSlot(timeslice, input, slot);
340  sendVariableContextMetrics(index.getVariablesForSlot(slot), slot, mMetrics, sVariablesMetricsNames);
341  index.markAsDirty(slot, true);
342  mStats.relayedMessages++;
343  return WillRelay;
344  }
345 
348  VariableContext pristineContext;
349  std::tie(input, timeslice) = getInputTimeslice(pristineContext);
350 
351  if (input == INVALID_INPUT) {
352  LOG(ERROR) << "Could not match incoming data to any input";
353  mStats.malformedInputs++;
354  mStats.droppedIncomingMessages++;
355  return WillNotRelay;
356  }
357 
358  if (TimesliceId::isValid(timeslice) == false) {
359  LOG(ERROR) << "Could not determine the timeslice for input";
360  mStats.malformedInputs++;
361  mStats.droppedIncomingMessages++;
362  return WillNotRelay;
363  }
364 
365  TimesliceIndex::ActionTaken action;
366  std::tie(action, slot) = index.replaceLRUWith(pristineContext);
367 
368  updateStatistics(action);
369 
370  if (action == TimesliceIndex::ActionTaken::DropObsolete) {
371  LOG(WARNING) << "Incoming data is already obsolete, not relaying.";
372  return WillNotRelay;
373  }
374 
375  if (action == TimesliceIndex::ActionTaken::DropInvalid) {
376  LOG(WARNING) << "Incoming data is invalid, not relaying.";
377  return WillNotRelay;
378  }
379 
380  // At this point the variables match the new input but the
381  // cache still holds the old data, so we prune it.
382  pruneCache(slot);
383  saveInSlot(timeslice, input, slot);
384  sendVariableContextMetrics(pristineContext, slot, mMetrics, sVariablesMetricsNames);
385  index.markAsDirty(slot, true);
386 
387  return WillRelay;
388 }

390 std::vector<DataRelayer::RecordAction>
391 DataRelayer::getReadyToProcess() {
392  // THE STATE
393  std::vector<RecordAction> completed;
394  completed.reserve(16);
395  const auto &cache = mCache;
396  const auto numInputTypes = mDistinctRoutesIndex.size();
397  //
398  // THE IMPLEMENTATION DETAILS
399  //
400  // We use this to bail out early from the check as soon as we find something
401  // which we know is not complete.
402  auto getPartialRecord = [&cache, &numInputTypes](int li) -> gsl::span<const PartRef> {
403  auto offset = li * numInputTypes;
404  assert(cache.size() >= offset + numInputTypes);
405  auto const start = cache.data() + offset;
406  auto const end = cache.data() + offset + numInputTypes;
407  return gsl::span<const PartRef>(start, end);
408  };
409 
410  // These two are trivial, but in principle the whole loop could be parallelised
411  // or vectorised so "completed" could be a thread local variable which needs
412  // merging at the end.
413  auto updateCompletionResults = [&completed](TimesliceSlot li, CompletionPolicy::CompletionOp op) {
414  completed.emplace_back(RecordAction{ li, op });
415  };
416 
417  auto completionResults = [&completed]() -> std::vector<RecordAction> {
418  return completed;
419  };
420 
421  // THE OUTER LOOP
422  //
423  // To determine if a line is complete, we iterate on all the arguments
424  // and check if they are ready. We do it this way, because in the end
425  // the number of inputs is going to be small and having a more complex
426  // structure will probably result in a larger footprint in any case.
427  // Also notice that ai == inputsNumber only when we reach the end of the
428  // iteration, that means we have found all the required bits.
429  //
430  // Notice that the only time numInputTypes is 0 is when we are a dummy
431  // device created as a source for timers / conditions.
432  if (numInputTypes == 0) {
433  return {};
434  }
435  size_t cacheLines = cache.size() / numInputTypes;
436  assert(cacheLines * numInputTypes == cache.size());
437 
438  for (size_t li = 0; li < cacheLines; ++li) {
439  TimesliceSlot slot{ li };
440  // We only check the cachelines which have been updated by an incoming
441  // message.
442  if (mTimesliceIndex.isDirty(slot) == false) {
443  continue;
444  }
445  auto partial = getPartialRecord(li);
446  auto action = mCompletionPolicy.callback(partial);
447  switch (action) {
448  case CompletionPolicy::CompletionOp::Consume:
449  case CompletionPolicy::CompletionOp::Process:
450  case CompletionPolicy::CompletionOp::Discard:
451  updateCompletionResults(slot, action);
452  break;
453  case CompletionPolicy::CompletionOp::Wait:
454  break;
455  }
456  // Given we have created an action for this cacheline, we need to wait for
457  // a new message before we look again into the given cacheline.
458  mTimesliceIndex.markAsDirty(slot, false);
459  }
460  return completionResults();
461 }

463 std::vector<std::unique_ptr<FairMQMessage>>
464  DataRelayer::getInputsForTimeslice(TimesliceSlot slot)
465 {
466  const auto numInputTypes = mDistinctRoutesIndex.size();
467  // State of the computation
468  std::vector<std::unique_ptr<FairMQMessage>> messages;
469  messages.reserve(numInputTypes*2);
470  auto& cache = mCache;
471  auto& index = mTimesliceIndex;
472  auto& metrics = mMetrics;
473 
474  // Nothing to see here, this is just to make the outer loop more understandable.
475  auto jumpToCacheEntryAssociatedWith = [](TimesliceSlot) {
476  return;
477  };
478 
479  // We move ownership so that the cache can be reused once the computation is
480  // finished. We mark the given cache slot invalid, so that it can be reused
481  // This means we can still handle old messages if there is still space in the
482  // cache where to put them.
483  auto moveHeaderPayloadToOutput = [&messages, &cache, &index, &numInputTypes, &metrics](TimesliceSlot s, size_t arg) {
484  auto cacheId = s.index * numInputTypes + arg;
485  metrics.send({ 2, sMetricsNames[cacheId] });
486  messages.emplace_back(std::move(cache[cacheId].header));
487  messages.emplace_back(std::move(cache[cacheId].payload));
488  index.markAsInvalid(s);
489  };
490 
491  // An invalid set of arguments is a set of arguments associated to an invalid
492  // timeslice, so I can simply do that. I keep the assertion there because in principle
493  // we should have dispatched the timeslice already!
494  // FIXME: what happens when we have enough timeslices to hit the invalid one?
495  auto invalidateCacheFor = [&numInputTypes, &index, &cache](TimesliceSlot s) {
496  for (size_t ai = s.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
497  assert(cache[ai].header.get() == nullptr);
498  assert(cache[ai].payload.get() == nullptr);
499  }
500  index.markAsInvalid(s);
501  };
502 
503  // Outer loop here.
504  jumpToCacheEntryAssociatedWith(slot);
505  for (size_t ai = 0, ae = numInputTypes; ai != ae; ++ai) {
506  moveHeaderPayloadToOutput(slot, ai);
507  }
508  invalidateCacheFor(slot);
509 
510  return std::move(messages);
511 }

542  sVariablesMetricsNames[i] = std::string("matcher_variables/") + std::to_string(i);
543  mMetrics.send({ std::string("null"), sVariablesMetricsNames[i % 16] });
544  }
