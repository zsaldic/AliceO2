54 void Dispatcher::run(ProcessingContext& ctx)
55 {
56  for (const auto& input : ctx.inputs()) {
57  if (input.header != nullptr && input.spec != nullptr) {
58 
59  for (auto& policy : mPolicies) {
60  // todo: consider getting the outputSpec in match to improve performance
61  // todo: consider matching (and deciding) in completion policy to save some time
62  if (policy->match(*input.spec) && policy->decide(input)) {
63 
64  if (!policy->getFairMQOutputChannel().empty()) {
65  sendFairMQ(ctx.services().get<RawDeviceService>().device(), input, policy->getFairMQOutputChannelName());
66  } else {
67  send(ctx.outputs(), input, policy->prepareOutput(*input.spec));
68  }
69  }
70  }
71  }
72  }

75 void Dispatcher::send(DataAllocator& dataAllocator, const DataRef& inputData, const Output& output) const
76 {
77  //todo: support other serialization methods
78  const auto* inputHeader = header::get<header::DataHeader*>(inputData.header);
79  if (inputHeader->payloadSerializationMethod == header::gSerializationMethodInvalid) {
80  LOG(WARNING) << "DataSampling::dispatcherCallback: input of origin'" << inputHeader->dataOrigin.str
81  << "', description '" << inputHeader->dataDescription.str
82  << "' has gSerializationMethodInvalid.";
83  } else if (inputHeader->payloadSerializationMethod == header::gSerializationMethodROOT) {
84  dataAllocator.adopt(output, DataRefUtils::as<TObject>(inputData).release());
85  } else { // POD
86  // todo: do it non-copy, when API is available
87  auto outputMessage = dataAllocator.newChunk(output, inputHeader->payloadSize);
88  memcpy(outputMessage.data, inputData.payload, inputHeader->payloadSize);
89  }
90 }

93 void Dispatcher::sendFairMQ(FairMQDevice* device, const DataRef& inputData, const std::string& fairMQChannel) const
94 {
95  const auto* dh = header::get<header::DataHeader*>(inputData.header);
96  assert(dh);
97  const auto* dph = header::get<DataProcessingHeader*>(inputData.header);
98  assert(dph);
99 
100  header::DataHeader dhout{ dh->dataDescription, dh->dataOrigin, dh->subSpecification, dh->payloadSize };
101  dhout.payloadSerializationMethod = dh->payloadSerializationMethod;
102  DataProcessingHeader dphout{ dph->startTime, dph->duration };
103  o2::header::Stack headerStack{ dhout, dphout };
104 
105  auto channelAlloc = o2::pmr::getTransportAllocator(device->Transport());
106  FairMQMessagePtr msgHeaderStack = o2::pmr::getMessage(std::move(headerStack), channelAlloc);
107 
108  char* payloadCopy = new char[dh->payloadSize];
109  memcpy(payloadCopy, inputData.payload, dh->payloadSize);
110  auto cleanupFcn = [](void* data, void*) { delete[] reinterpret_cast<char*>(data); };
111  FairMQMessagePtr msgPayload(device->NewMessage(payloadCopy, dh->payloadSize, cleanupFcn, payloadCopy));
112 
113  FairMQParts message;
114  message.AddPart(move(msgHeaderStack));
115  message.AddPart(move(msgPayload));
116 
117  int64_t bytesSent = device->Send(message, fairMQChannel);
118 }

120 void Dispatcher::registerPath(const std::pair<InputSpec, OutputSpec>& path)
121 {
122  //todo: take care of inputs inclusive in others, when subSpec matchers are supported
123  auto cmp = [a = path.first](const InputSpec b)
124  {
125  return a.matcher == b.matcher && a.lifetime == b.lifetime;
126  };
127 
128  if (std::find_if(inputs.begin(), inputs.end(), cmp) == inputs.end()) {
129  inputs.push_back(path.first);
130  LOG(DEBUG) << "Registering input " << DataSpecUtils::describe(path.first);
131  } else {
132  LOG(DEBUG) << "Input " << DataSpecUtils::describe(path.first)
133  << " already registered";
134  }
135 
136  outputs.push_back(path.second);
137 }
