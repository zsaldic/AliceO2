331  // this merges several entries from the TBranch brname from the origin TTree
332  // into a single entry in a target TTree / same branch
333  // (assuming T is typically a vector; merging is simply done by appending)
334  // make this function a (static helper)
335  template <typename T>
336  void mergeAndAdjustHits(std::string const& brname, TTree& origin, TTree& target,
337  std::map<int, std::vector<int>> const& entrygroups,
338  std::map<int, std::vector<int>> const& trackoffsets)
339  {
340  auto originbr = origin.GetBranch(brname.c_str());
341  if (originbr) {
342  auto targetdata = new T;
343  T* incomingdata = nullptr;
344  originbr->SetAddress(&incomingdata);
345 
346  T* filladdress;
347  for (auto& event_entries_pair : entrygroups) {
348  const auto& entries = event_entries_pair.second;
349  int offset = 0;
350  const auto& offsetvector = trackoffsets.find(event_entries_pair.first)->second;
351  int entrycounter = 0;
352  if (entries.size() == 1) {
353  originbr->GetEntry(entries[0]);
354  filladdress = incomingdata;
355  } else {
356  for (auto& e : entries) {
357  filladdress = targetdata;
358  originbr->GetEntry(e);
359  if (incomingdata) {
360  if (offset != 0) {
361  // fix the trackIDs for this data
362  for (auto& hit : *incomingdata) {
363  const auto oldID = hit.GetTrackID();
364  hit.SetTrackID(oldID + offset);
365  }
366  }
367  // this could be further generalized by using a policy for T
368  std::copy(incomingdata->begin(), incomingdata->end(), std::back_inserter(*targetdata));
369  // adjust offset
370  offset += offsetvector[entrycounter];
371  delete incomingdata;
372  incomingdata = nullptr;
373  }
374  entrycounter++;
375  }
376  }
377  // fill target for this event
378  auto targetbr = o2::Base::getOrMakeBranch(target, brname.c_str(), &filladdress);
379  targetbr->SetAddress(&filladdress);
380  targetbr->Fill();
381  targetbr->ResetAddress();
382  targetdata->clear();
383  if (incomingdata) {
384  delete incomingdata;
385  incomingdata = nullptr;
386  }
387  }
388  delete targetdata;
389  }
390  }

392  void mergeHitEntries(TTree& origin, TTree& target, std::map<int, std::vector<int>> const& entries,
393  std::map<int, std::vector<int>> const& trackoffsets) final
394  {
395  // loop over hit containers / different branches
396  // adjust trackID in hits on the go
397  int probe = 0;
398  using Hit_t = decltype(static_cast<Det*>(this)->Det::getHits(probe));
399  std::string name = static_cast<Det*>(this)->getHitBranchNames(probe++);
400  while (name.size() > 0) {
401  mergeAndAdjustHits<typename std::remove_pointer<Hit_t>::type>(name, origin, target, entries, trackoffsets);
