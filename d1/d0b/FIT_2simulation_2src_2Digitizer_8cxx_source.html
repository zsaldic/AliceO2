27 void Digitizer::process(const std::vector<HitType>* hits, Digit* digit)
28 {
29  //parameters constants TO DO: move to class
30 
31  constexpr Float_t C_side_cable_cmps = 2.8; //ns
32  constexpr Float_t A_side_cable_cmps = 11.; //ns
33  constexpr Float_t signal_width = 5.; // time gate for signal, ns
34  constexpr Float_t nPe_in_mip = 250.; // n ph. e. in one mip
35 
36  Int_t nlbl = 0; //number of MCtrues
37 
38  digit->setTime(mEventTime);
39  digit->setBC(mBC);
40  digit->setOrbit(mOrbit);
41 
42  //Calculating signal time, amplitude in mean_time +- time_gate --------------
43  Double_t cfd[300] = {};
44  Float_t amp[300] = {};
45  Int_t ch_signal_nPe[300] = {};
46  Double_t ch_signal_MIP[300] = {};
47  Double_t ch_signal_time[300] = {};
48  for (auto& hit : *hits) {
49  Int_t hit_ch = hit.GetDetectorID();
50  Double_t hit_time = hit.GetTime();
51  Bool_t is_A_side = (hit_ch <= 4 * Geometry::NCellsA);
52  Float_t time_compensate = is_A_side ? A_side_cable_cmps : C_side_cable_cmps;
53 
54  Bool_t is_hit_in_signal_gate = (hit_time > time_compensate - signal_width * .5) &&
55  (hit_time < time_compensate + signal_width * .5);
56 
57  Double_t hit_time_corr = hit_time - time_compensate /* + mBC_clk_center + mEventTime*/;
58 
59  if (/*is_time_in_gate &&*/ is_hit_in_signal_gate) {
60  ch_signal_nPe[hit_ch]++;
61  ch_signal_time[hit_ch] += hit_time_corr;
62  }
63 
64  //charge particles in MCLabel
65  if (hit.GetEnergyLoss() > 0) {
66  o2::fit::MCLabel label(hit.GetTrackID(), mEventID, mSrcID, hit_ch);
67  int lblCurrent;
68  if (mMCLabels) {
69  lblCurrent = mMCLabels->getIndexedSize(); // this is the size of mHeaderArray;
70  mMCLabels->addElement(lblCurrent, label);
71  nlbl++;
72  }
73  }
74  }
75 
76  // sum different sources
77  std::vector<ChannelData> mChDgDataArr;
78  for (const auto& d : digit->getChDgData()) {
79  Int_t mcp = d.ChId;
80  cfd[mcp] = d.CFDTime;
81  amp[mcp] = d.QTCAmpl;
82  }
83 
84  for (Int_t ch_iter = 0; ch_iter < mMCPs; ch_iter++) {
85  if (ch_signal_nPe[ch_iter] != 0) {
86  ch_signal_MIP[ch_iter] = amp[ch_iter] + ch_signal_nPe[ch_iter] / nPe_in_mip;
87  if (cfd[ch_iter] > 0) {
88  cfd[ch_iter] = cfd[ch_iter] - mBC_clk_center - mEventTime;
89  ch_signal_time[ch_iter] = ((cfd[ch_iter] + ch_signal_time[ch_iter] / (float)ch_signal_nPe[ch_iter]) / 2.) + mBC_clk_center + mEventTime;
90  } else
91  ch_signal_time[ch_iter] = (ch_signal_time[ch_iter] / (float)ch_signal_nPe[ch_iter]) + mBC_clk_center + mEventTime;
92 
93  if (ch_signal_MIP[ch_iter] > mCFD_trsh_mip) {
94  LOG(DEBUG) << ch_iter << " : "
95  << " : " << ch_signal_time[ch_iter] - mBC_clk_center - mEventTime << " : "
96  << ch_signal_MIP[ch_iter] << " : " << mEventTime << " cfd " << cfd[ch_iter] << FairLogger::endl;
97  mChDgDataArr.emplace_back(ChannelData{ ch_iter, ch_signal_time[ch_iter], ch_signal_MIP[ch_iter] });
98  } else {
99  ch_signal_MIP[ch_iter] = 0;
100  ch_signal_time[ch_iter] = 0;
101  }
102  }
103  }
104 
105  digit->setChDgData(std::move(mChDgDataArr));
106 }

109 void Digitizer::smearCFDtime(Digit* digit)
110 {
111  //smeared CFD time for 50ps
112  std::vector<ChannelData> mChDgDataArr;
113  for (const auto& d : digit->getChDgData()) {
114  Int_t mcp = d.ChId;
115  Double_t cfd = d.CFDTime - mBC_clk_center - mEventTime;
116  Float_t amp = d.QTCAmpl;
117  if (amp > mCFD_trsh_mip) {
118  Double_t smeared_time = gRandom->Gaus(cfd, 0.050) + mBC_clk_center + mEventTime;
119  mChDgDataArr.emplace_back(ChannelData{ mcp, smeared_time, amp });
120  }
121  }
122  digit->setChDgData(std::move(mChDgDataArr));
123 }

126 void Digitizer::setTriggers(Digit* digit)
127 {
128  constexpr Double_t BC_clk_center = 12.5; // clk center
129  constexpr Double_t trg_central_trh = 100.; // mip
130  constexpr Double_t trg_semicentral_trh = 50.; // mip
131  constexpr Double_t trg_vertex_min = -3.; //ns
132  constexpr Double_t trg_vertex_max = 3.; //ns
133 
134  // Calculating triggers -----------------------------------------------------
135  Int_t n_hit_A = 0, n_hit_C = 0;
136  Float_t mean_time_A = 0.;
137  Float_t mean_time_C = 0.;
138  Float_t summ_ampl_A = 0.;
139  Float_t summ_ampl_C = 0.;
140  Float_t vertex_time;
141 
142  Double_t cfd[300] = {};
143  Float_t amp[300] = {};
144  for (const auto& d : digit->getChDgData()) {
145  Int_t mcp = d.ChId;
146  cfd[mcp] = d.CFDTime - mBC_clk_center - mEventTime;
147  amp[mcp] = d.QTCAmpl;
148  if (amp[mcp] < mCFD_trsh_mip)
149  continue;
150  if (cfd[mcp] < -mTime_trg_gate / 2. || cfd[mcp] > mTime_trg_gate / 2.)
151  continue;
152 
153  Bool_t is_A_side = (mcp <= 4 * Geometry::NCellsA);
154  if (is_A_side) {
155  n_hit_A++;
156  summ_ampl_A += amp[mcp];
157  mean_time_A += cfd[mcp];
158  } else {
159  n_hit_C++;
160  summ_ampl_C += amp[mcp];
161  mean_time_C += cfd[mcp];
162  }
163  }
164 
165  Bool_t is_A = n_hit_A > 0;
166  Bool_t is_C = n_hit_C > 0;
167  Bool_t is_Central = summ_ampl_A + summ_ampl_C >= trg_central_trh;
168  Bool_t is_SemiCentral = summ_ampl_A + summ_ampl_C >= trg_semicentral_trh;
169 
170  mean_time_A = is_A ? mean_time_A / n_hit_A : 0;
171  mean_time_C = is_C ? mean_time_C / n_hit_C : 0;
172  vertex_time = (mean_time_A + mean_time_C) * .5;
173  Bool_t is_Vertex = (vertex_time > trg_vertex_min) && (vertex_time < trg_vertex_max);
174 
175  //filling digit
176  digit->setTriggers(is_A, is_C, is_Central, is_SemiCentral, is_Vertex);
177 
178  // Debug output -------------------------------------------------------------
179  LOG(DEBUG) << "\n\nTest digizing data ===================" << FairLogger::endl;
180 
181  LOG(INFO) << "Event ID: " << mEventID << " Event Time " << mEventTime << FairLogger::endl;
182  LOG(INFO) << "N hit A: " << n_hit_A << " N hit C: " << n_hit_C << " summ ampl A: " << summ_ampl_A
183  << " summ ampl C: " << summ_ampl_C << " mean time A: " << mean_time_A
184  << " mean time C: " << mean_time_C << FairLogger::endl;
185 
186  LOG(INFO) << "IS A " << is_A << " IS C " << is_C << " is Central " << is_Central
187  << " is SemiCentral " << is_SemiCentral << " is Vertex " << is_Vertex << FairLogger::endl;
188 
189  LOG(DEBUG) << "======================================\n\n"
190  << FairLogger::endl;
191  // --------------------------------------------------------------------------
192 }
