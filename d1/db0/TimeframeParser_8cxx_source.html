61 void streamTimeframe(std::istream &stream,
62  std::function<void(FairMQParts &parts, char *buffer, size_t size)> onAddPart,
63  std::function<void(FairMQParts &parts)> onSend) {
64  FairMQParts parts;
65  StreamingState state;
66  assert(state.state == PARSE_BEGIN_STREAM);
67  while(true) {
68  switch(state.state) {
69  case PARSE_BEGIN_STREAM:
70  LOG(INFO) << "In PARSE_BEGIN_STREAM\n";
71  state.state = PARSE_BEGIN_TIMEFRAME;
72  break;
73  case PARSE_BEGIN_TIMEFRAME:
74  LOG(INFO) << "In PARSE_BEGIN_TIMEFRAME\n";
75  state.state = PARSE_BEGIN_PAIR;
76  break;
77  case PARSE_BEGIN_PAIR:
78  LOG(INFO) << "In PARSE_BEGIN_PAIR\n";
79  state.state = PARSE_DATA_HEADER;
80  state.hasDataHeader = false;
81  state.payloadBuffer = nullptr;
82  state.headerBuffer = nullptr;
83  break;
84  case PARSE_DATA_HEADER:
85  LOG(INFO) << "In PARSE_DATA_HEADER\n";
86  if (state.hasDataHeader) {
87  throw std::runtime_error("DataHeader already present.");
88  } else if (state.payloadBuffer) {
89  throw std::runtime_error("Unexpected payload.");
90  }
91  LOG(INFO) << "Reading dataheader of " << sizeof(state.dh) << " bytes\n";
92  stream.read(reinterpret_cast<char *>(&state.dh), sizeof(state.dh));
93  // If we have a TIMEFRAMEINDEX part and we find the eof, we are done.
94  if (stream.eof()) {
95  throw std::runtime_error("Premature end of stream");
96  }
97 
98  // Otherwise we move to the state which is responsible for parsing the
99  // kind of header.
100  state.state = PARSE_CONCRETE_HEADER;
101  break;
102  case PARSE_CONCRETE_HEADER:
103  LOG(INFO) << "In PARSE_CONCRETE_HEADER\n";
104  if (state.headerBuffer)
105  {
106  throw std::runtime_error("File has two consecutive headers");
107  }
108  if (state.dh.headerSize < sizeof(DataHeader))
109  {
110  std::ostringstream str;
111  str << "Bad header size. Should be greater then "
112  << sizeof(DataHeader)
113  << ". Found " << state.dh.headerSize << "\n";
114  throw std::runtime_error(str.str());
115  }
116  // We get the full header size and read the rest of the header
117  state.headerBuffer = malloc(state.dh.headerSize);
118  memcpy(state.headerBuffer, &state.dh, sizeof(state.dh));
119  LOG(INFO) << "Reading rest of the header of " << state.dh.headerSize - sizeof(state.dh) << " bytes\n";
120  stream.read(reinterpret_cast<char*>(state.headerBuffer)+ sizeof(state.dh),
121  state.dh.headerSize - sizeof(state.dh));
122  // Handle the case the file was truncated.
123  if (stream.eof())
124  {
125  throw std::runtime_error("Unexpected end of file");
126  }
127  onAddPart(parts, reinterpret_cast<char *>(state.headerBuffer), state.dh.headerSize);
128  // Move to parse the payload
129  state.state = PARSE_PAYLOAD;
130  break;
131  case PARSE_PAYLOAD:
132  LOG(INFO) << "In PARSE_PAYLOAD\n";
133  if(state.payloadBuffer)
134  {
135  throw std::runtime_error("File has two consecutive payloads");
136  }
137  state.payloadBuffer = new char[state.dh.payloadSize];
138  LOG(INFO) << "Reading payload of " << state.dh.payloadSize << " bytes\n";
139  stream.read(reinterpret_cast<char *>(state.payloadBuffer), state.dh.payloadSize);
140  if (stream.eof())
141  {
142  throw std::runtime_error("Unexpected end of file");
143  }
144  onAddPart(parts, reinterpret_cast<char *>(state.payloadBuffer), state.dh.payloadSize);
145  state.state = PARSE_END_PAIR;
146  break;
147  case PARSE_END_PAIR:
148  LOG(INFO) << "In PARSE_END_PAIR\n";
149  state.state = state.dh == DataDescription("TIMEFRAMEINDEX") ? PARSE_END_TIMEFRAME : PARSE_BEGIN_PAIR;
150  break;
151  case PARSE_END_TIMEFRAME:
152  LOG(INFO) << "In PARSE_END_TIMEFRAME\n";
153  onSend(parts);
154  // Check if we have more. If not, we can declare success.
155  stream.peek();
156  if (stream.eof()) {
157  state.state = PARSE_END_STREAM;
158  } else {
159  state.state = PARSE_BEGIN_TIMEFRAME;
160  }
161  break;
162  case PARSE_END_STREAM:
163  return;
164  break;
165  default:
166  break;
167  }
168  }

171 void streamTimeframe(std::ostream &stream, FairMQParts &parts) {
172  if (parts.Size() < 2)
173  {
174  throw std::runtime_error("Expecting at least 2 parts\n");
175  }
176 
177  auto indexHeader = o2::header::get<DataHeader*>(parts.At(parts.Size() - 2)->GetData());
178  // FIXME: Provide iterator pair API for the index
179  // Index should really be something which provides an
180  // iterator pair API so that we can sort / find / lower_bound
181  // easily. Right now we simply use it a C-style array.
182  auto index = reinterpret_cast<IndexElement*>(parts.At(parts.Size() - 1)->GetData());
183 
184  LOG(INFO) << "This time frame has " << parts.Size() << " parts.\n";
185  auto indexEntries = indexHeader->payloadSize / sizeof(IndexElement);
186  if (indexHeader->dataDescription != DataDescription("TIMEFRAMEINDEX")) {
187  throw std::runtime_error("Could not find a valid index header\n");
188  }
189  LOG(INFO) << indexHeader->dataDescription.str << "\n";
190  LOG(INFO) << "This time frame has " << indexEntries << "entries in the index.\n";
191  if ((indexEntries * 2 + 2) != (parts.Size())) {
192  std::stringstream err;
193  err << "Mismatched index and received parts. Expected "
194  << (parts.Size() - 2 * 2) << " found " << indexEntries ;
195  throw std::runtime_error(err.str());
196  }
197 
198  LOG(INFO) << "Everything is fine with received timeframe\n";
199  for (size_t i = 0; i < parts.Size(); ++i)
200  {
201  stream.write(reinterpret_cast<const char *>(parts.At(i)->GetData()),
202  parts.At(i)->GetSize());
203  }
204 }
