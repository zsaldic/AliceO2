65  float getR()
66  {
67  if (mR == RadiusOutOfRange) {
68  mR = TMath::Sqrt(getX() * getX() + getY() * getY() + getZ() * getZ());
69  }
70  return mR;
71  } // Cluster Radius
72  float getPhi()
73  {
74  if (mPhi == PhiOutOfRange) {
75  mPhi = TMath::ATan2(getY(), getX());
76  }
77  return mPhi;
78  } // Cluster Phi
79  int getSector()
80  {
81  if (mSector == -1) {
82  mSector = (TMath::ATan2(-getY(), -getX()) + TMath::Pi()) * TMath::RadToDeg() * 0.05;
83  }
84  return mSector;
85  } // Cluster Sector
