38  unsigned int ClusterTopology::hashFunction(const void* key, int len)
39  {
40  //
41  // Developed from https://github.com/rurban/smhasher , function MurMur2
42  //
43  // 'm' and 'r' are mixing constants generated offline.
44  const unsigned int m = 0x5bd1e995;
45  const int r = 24;
46  // Initialize the hash
47  unsigned int h = len ^ 0xdeadbeef;
48  // Mix 4 bytes at a time into the hash
49  const unsigned char* data = (const unsigned char*)key;
50  // int recIndex=0;
51  while (len >= 4) {
52  unsigned int k = *(unsigned int*)data;
53  k *= m;
54  k ^= k >> r;
55  k *= m;
56  h *= m;
57  h ^= k;
58  data += 4;
59  len -= 4;
60  }
61  // Handle the last few bytes of the input array
62  switch (len) {
63  case 3:
64  h ^= data[2] << 16;
65  case 2:
66  h ^= data[1] << 8;
67  case 1:
68  h ^= data[0];
69  h *= m;
70  };
71  // Do a few final mixes of the hash to ensure the last few
72  // bytes are well-incorporated.
73  h ^= h >> 13;
74  h *= m;
75  h ^= h >> 15;
76  return h;
77  }

79  unsigned long ClusterTopology::getCompleteHash(int nRow, int nCol,
80  const unsigned char patt[Cluster::kMaxPatternBytes])
81  {
82  unsigned char extended_pattern[ClusterPattern::kExtendedPatternBytes] = { 0 };
83  extended_pattern[0] = (unsigned char)nRow;
84  extended_pattern[1] = (unsigned char)nCol;
85  int nBits = nRow * nCol;
86  int nBytes = nBits / 8;
87  if (nBits % 8 != 0)
88  nBytes++;
89  memcpy(&extended_pattern[2], patt, nBytes);
90 
91  unsigned long partialHash = (unsigned long)hashFunction(extended_pattern, nBytes);
92  // The first four bytes are directly taken from partialHash
93  unsigned long completeHash = partialHash << 32;
94  // The last four bytes of the hash are the first 32 pixels of the topology.
95  // The bits reserved for the pattern that are not used are set to 0.
96  if (nBytes == 1) {
97  completeHash += ((((unsigned long)extended_pattern[2]) << 24));
98  } else if (nBytes == 2) {
99  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16));
100  } else if (nBytes == 3) {
101  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
102  (((unsigned long)extended_pattern[4]) << 8));
103  } else if (nBytes >= 4) {
104  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
105  (((unsigned long)extended_pattern[4]) << 8) + ((unsigned long)extended_pattern[5]));
106  } else {
107  LOG(ERROR) << "No fired pixels in small topology" << FairLogger::endl;
108  throw std::runtime_error("No fired pixels in small topology");
109  }
110  return completeHash;
111  }
