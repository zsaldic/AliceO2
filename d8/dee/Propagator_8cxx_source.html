50 bool Propagator::PropagateToXBxByBz(o2::track::TrackParCov& track, float xToGo, float mass, float maxSnp, float maxStep,
51  int matCorr, o2::track::TrackLTIntegral* tofInfo, int signCorr)
52 {
53  //----------------------------------------------------------------
54  //
55  // Propagates the track to the plane X=xk (cm)
56  // taking into account all the three components of the magnetic field
57  // and correcting for the crossed material.
58  //
59  // mass - mass used in propagation - used for energy loss correction (if <0 then q=2)
60  // maxStep - maximal step for propagation
61  // tofInfo - optional container for track length and PID-dependent TOF integration
62  //----------------------------------------------------------------
63  const float Epsilon = 0.00001;
64  auto dx = xToGo - track.getX();
65  int dir = dx > 0.f ? 1 : -1;
66  if (!signCorr) {
67  signCorr = -dir; // sign of eloss correction is not imposed
68  }
69 
70  std::array<float, 3> b;
71  while (std::abs(dx) > Epsilon) {
72  auto step = std::min(std::abs(dx), maxStep);
73  if (dir < 0) {
74  step = -step;
75  }
76  auto x = track.getX() + step;
77  auto xyz0 = track.getXYZGlo();
78  mField->Field(xyz0, b.data());
79 
80  if (!track.propagateTo(x, b)) {
81  return false;
82  }
83  if (maxSnp > 0 && std::abs(track.getSnp()) >= maxSnp) {
84  return false;
85  }
86  if (matCorr) {
87  auto xyz1 = track.getXYZGlo();
88  auto mb = GeometryManager::MeanMaterialBudget(xyz0, xyz1);
89  if (!track.correctForMaterial(mb.meanX2X0, ((signCorr < 0) ? -mb.length : mb.length) * mb.meanRho, mass)) {
90  return false;
91  }
92 
93  if (tofInfo) {
94  tofInfo->addStep(mb.length, track); // fill L,ToF info using already calculated step length
95  tofInfo->addX2X0(mb.meanX2X0);
96  }
97  } else if (tofInfo) { // if tofInfo filling was requested w/o material correction, we need to calculate the step lenght
98  auto xyz1 = track.getXYZGlo();
99  Vector3D<float> stepV(xyz1.X() - xyz0.X(), xyz1.Y() - xyz0.Y(), xyz1.Z() - xyz0.Z());
100  tofInfo->addStep(stepV.R(), track);
101  }
102  dx = xToGo - track.getX();
103  }
104  return true;
105 }

165 bool Propagator::propagateToDCA(const Point3D<float>& vtx, o2::track::TrackParCov& track, float bZ,
166  float mass, float maxStep, int matCorr,
167  o2::track::TrackLTIntegral* tofInfo, int signCorr, float maxD)
168 {
169  // propagate track to DCA to the vertex
170  float sn, cs, alp = track.getAlpha();
171  o2::utils::sincosf(alp, sn, cs);
172  float x = track.getX(), y = track.getY(), snp = track.getSnp(), csp = std::sqrt((1.f - snp) * (1.f + snp));
173  float xv = vtx.X() * cs + vtx.Y() * sn, yv = -vtx.X() * sn + vtx.Y() * cs, zv = vtx.Z();
174  x -= xv;
175  y -= yv;
176  //Estimate the impact parameter neglecting the track curvature
177  Double_t d = std::abs(x * snp - y * csp);
178  if (d > maxD) {
179  return false;
180  }
181  float crv = track.getCurvature(bZ);
182  float tgfv = -(crv * x - snp) / (crv * y + csp);
183  sn = tgfv / std::sqrt(1.f + tgfv * tgfv);
184  cs = std::sqrt((1. - sn) * (1. + sn));
185  cs = (std::abs(tgfv) > o2::constants::math::Almost0) ? sn / tgfv : o2::constants::math::Almost1;
186 
187  x = xv * cs + yv * sn;
188  yv = -xv * sn + yv * cs;
189  xv = x;
190 
191  o2::track::TrackParCov tmpT(track); // operate on the copy to recover after the failure
192  alp += std::asin(sn);
193  if (!tmpT.rotate(alp) || !propagateToX(tmpT, xv, bZ, mass, 0.85, maxStep, matCorr, tofInfo, signCorr)) {
194  LOG(ERROR) << "failed to propagate to alpha=" << alp << " X=" << xv << " for vertex "
195  << vtx.X() << ' ' << vtx.Y() << ' ' << vtx.Z() << " | Track is: ";
196  tmpT.print();
197  return false;
198  }
199  track = tmpT;
200  return true;
201 }
