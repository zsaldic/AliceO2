73 void ClustererTask::run(const std::string inpName, const std::string outName, bool entryPerROF)
74 {
75  // standalone execution
76  setSelfManagedMode(true);
77  Init(); // create reader, clusterer
78 
79  std::unique_ptr<TFile> outFile(TFile::Open(outName.data(), "recreate"));
80  if (!outFile || outFile->IsZombie()) {
81  LOG(FATAL) << "Failed to open output file " << outName;
82  }
83  std::unique_ptr<TTree> outTree = std::make_unique<TTree>("o2sim", "ITS Clusters");
84  std::unique_ptr<TTree> outTreeROF = std::make_unique<TTree>("ITSClustersROF", "ROF records tree");
85 
86  if (mClusterer.getWantFullClusters()) {
87  mFullClusPtr = &mFullClus;
88  outTree->Branch("ITSCluster", &mFullClusPtr);
89  LOG(INFO) << Class()->GetName() << " output of full clusters is requested";
90  } else {
91  LOG(INFO) << Class()->GetName() << " output of full clusters is not requested";
92  }
93 
94  if (mClusterer.getWantCompactClusters()) {
95  mCompClusPtr = &mCompClus;
96  outTree->Branch("ITSClusterComp", &mCompClusPtr);
97  LOG(INFO) << Class()->GetName() << " output of compact clusters is requested";
98  } else {
99  LOG(INFO) << Class()->GetName() << " output of compact clusters is not requested";
100  }
101 
102  mROFRecVecPtr = &mROFRecVec;
103  outTreeROF->Branch("ITSClustersROF", mROFRecVecPtr);
104 
105  if (entryPerROF) {
106  mClusterer.setOutputTree(outTree.get()); // this will force flushing at every ROF
107  }
108 
109  if (mRawDataMode) {
110  mReaderRaw->openInput(inpName);
111  mClusterer.process(*mReaderRaw.get(), mFullClusPtr, mCompClusPtr, nullptr, mROFRecVecPtr);
112  } else {
113  mReaderMC->openInput(inpName, o2::detectors::DetID("ITS"));
114 
115  if (mUseMCTruth && !(mClusterer.getWantFullClusters() || mClusterer.getWantCompactClusters())) {
116  mUseMCTruth = false;
117  LOG(WARNING) << "ITS clusters storage is not requested, suppressing MCTruth storage";
118  }
119 
120  if (mUseMCTruth && mReaderMC->getDigitsMCTruth()) {
121  // digit labels are provided directly to clusterer
122  mClsLabelsPtr = &mClsLabels;
123  outTree->Branch("ITSClusterMCTruth", &mClsLabelsPtr);
124  } else {
125  mUseMCTruth = false;
126  }
127  LOG(INFO) << Class()->GetName() << " | MCTruth: " << (mUseMCTruth ? "ON" : "OFF");
128 
129  // loop over entries of the input tree
130  while (mReaderMC->readNextEntry()) {
131  mClusterer.process(*mReaderMC.get(), mFullClusPtr, mCompClusPtr, mClsLabelsPtr, mROFRecVecPtr);
132  }
133  }
134 
135  if (!mRawDataMode && mReaderMC->getMC2ROFRecords()) {
136  std::unique_ptr<TTree> outTreeMC2ROF = std::make_unique<TTree>("ITSClustersMC2ROF", "MC->ROF records tree");
137  auto mc2rof = *mReaderMC->getMC2ROFRecords(); // clone
138  auto* mc2rofPtr = &mc2rof;
139  outTreeMC2ROF->Branch("ITSClustersMC2ROF", &mc2rofPtr);
140  outTreeMC2ROF->Fill();
141  outTreeMC2ROF->Write();
142  }
143 
144  if (!entryPerROF) { // Clustered was not managing the output, do it here
145  outTree->Fill(); // in this mode all ROF will go to single entry of the tree
146  }
147  outTreeROF->Fill(); // ROF records are stored as a single vector
148 
149  outTree->Write();
150  outTreeROF->Write();
151 
152  outTree.reset(); // tree should be destroyed before the file is closed
153  outTreeROF.reset();
154 
155  outFile->Close();
156 
157  mClusterer.clear();
158 }
