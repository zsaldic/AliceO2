31  ImVec2 const& sizeHint,

39  {
40  ImVec2 size = ImVec2(sizeHint.x, std::min(sizeHint.y, 16.f * getNumItems(0) + 2));

42  ImU32 BACKGROUND_COLOR = ImColor(20, 20, 20, 255);

46  ImVec2 winPos = ImGui::GetCursorScreenPos();

49 
50  ImGui::InvisibleButton("sensible area", ImVec2(size.x, size.y));
51  if (ImGui::IsItemHovered()) {
52  auto pos = ImGui::GetMousePos() - winPos;
53  auto slot = std::lround(std::trunc(pos.x / size.x * records));
54  auto row = std::lround(std::trunc(pos.y / size.y));
55  describeCell(row, slot);
56  }
57 
58  drawList->AddRectFilled(
59  ImVec2(0., 0.) + winPos,
60  ImVec2{ size.x, size.y } + winPos,
61  BACKGROUND_COLOR);
62  drawList->AddRect(
63  ImVec2(0., 0.) + winPos,
64  ImVec2{ size.x - 1, size.y } + winPos,
65  BORDER_COLOR);
66  float padding = 1;
67  for (size_t ri = 0, re = getNumRecords(); ri < re; ri++) {
68  auto record = getRecord(ri);
69  ImVec2 xOffset{ (ri * boxSizeX) + padding, 0 };
70  ImVec2 xSize{ boxSizeX - 2 * padding, 0 };
71  auto boxSizeY = std::min(size.y / getNumItems(record), MAX_BOX_Y_SIZE);
72  for (size_t mi = 0, me = getNumItems(record); mi < me; mi++) {
73  ImVec2 yOffSet{ 0, (mi * boxSizeY) + padding };
74  ImVec2 ySize{ 0, boxSizeY - 2 * padding };
75 
76  drawList->AddRectFilled(
77  xOffset + yOffSet + winPos,
78  xOffset + xSize + yOffSet + ySize + winPos,
79  getColor(getValue(getItem(record, mi))));
80  }
81  }
82 
83  ImGui::SetCursorScreenPos(winPos + size);
84  }

87 void displayDataRelayer(DeviceMetricsInfo const& metrics,
88  DeviceInfo const& info,
89  ImVec2 const &size)
90 {
91  auto& viewIndex = info.dataRelayerViewIndex;
92  auto& variablesIndex = info.variablesViewIndex;
93  auto& queriesIndex = info.queriesViewIndex;
94 
95  auto getNumRecords = [&viewIndex]() -> size_t {
96  if (viewIndex.isComplete()) {
97  return viewIndex.w;
98  }
99  return 0;
100  };
101  auto getRecord = [&metrics](size_t i) -> int {
102  return i;
103  };
104  auto getNumItems = [&viewIndex](int record) -> int {
105  if (viewIndex.isComplete()) {
106  return viewIndex.h;
107  }
108  return 0;
109  };
110  auto getItem = [&metrics, &viewIndex](int const& record, size_t i) -> int const& {
111  // Calculate the index in the viewIndex.
112  auto idx = record * viewIndex.h + i;
113  assert(viewIndex.indexes.size() > idx);
114  MetricInfo const& metricInfo = metrics.metrics[viewIndex.indexes[idx]];
115  assert(metrics.intMetrics.size() > metricInfo.storeIdx);
116  auto& data = metrics.intMetrics[metricInfo.storeIdx];
117  return data[(metricInfo.pos - 1) % data.size()];
118  };
119  auto getValue = [](int const& item) -> int { return item; };
120  auto getColor = [](int value) {
121  const ImU32 SLOT_EMPTY = ImColor(70, 70, 70, 255);
122  const ImU32 SLOT_FULL = ImColor(0xf9, 0xcd, 0xad, 255);
123  const ImU32 SLOT_DISPATCHED = ImColor(0xc8, 0xc8, 0xa9, 255);
124  const ImU32 SLOT_DONE = ImColor(0x83, 0xaf, 0, 255);
125  const ImU32 SLOT_ERROR = ImColor(0xfe, 0x43, 0x65, 255);
126  switch (value) {
127  case 0:
128  return SLOT_EMPTY;
129  case 1:
130  return SLOT_FULL;
131  case 2:
132  return SLOT_DISPATCHED;
133  case 3:
134  return SLOT_DONE;
135  }
136  return SLOT_ERROR;
137  };
138  auto describeCell = [&metrics, &variablesIndex, &queriesIndex](int input, int slot) -> void {
139  ImGui::BeginTooltip();
140  for (size_t vi = 0; vi < variablesIndex.w; ++vi) {
141  auto idx = (slot * variablesIndex.w) + vi;
142  assert(idx < variablesIndex.indexes.size());
143  MetricInfo const& metricInfo = metrics.metrics[variablesIndex.indexes[idx]];
144  assert(metricInfo.storeIdx < metrics.stringMetrics.size());
145  //assert(metricInfo.type == MetricType::String);
146  auto& data = metrics.stringMetrics[metricInfo.storeIdx];
147  ImGui::Text("$%zu: %s", vi, data[(metricInfo.pos - 1) % data.size()].data);
148  }
149  ImGui::EndTooltip();
150  };
151 
152  if (getNumRecords()) {
153  HeatMapHelper::draw<int, int>("DataRelayer",
154  size,
155  getNumRecords,
156  getRecord,
157  getNumItems,
158  getItem,
159  getValue,
160  getColor,
161  describeCell);
162  }
163 }
