160 void o2::header::hexDump (const char* desc, const void* voidaddr, size_t len, size_t max)
161 {
162  size_t i;
163  unsigned char buff[17]; // stores the ASCII data
164  memset(&buff[0],'\0',17);
165  const byte* addr = reinterpret_cast<const byte*>(voidaddr);
166 
167  // Output description if given.
168  if (desc != nullptr)
169  printf ("%s, ", desc);
170  printf("%zu bytes:", len);
171  if (max>0 && len>max) {
172  len = max; //limit the output if requested
173  printf(" output limited to %zu bytes\n", len);
174  } else {
175  printf("\n");
176  }
177 
178  // In case of null pointer addr
179  if (addr==nullptr) {printf(" nullptr, size: %zu\n", len); return;}
180 
181  // Process every byte in the data.
182  for (i = 0; i < len; i++) {
183  // Multiple of 16 means new line (with line offset).
184  if ((i % 16) == 0) {
185  // Just don't print ASCII for the zeroth line.
186  if (i != 0)
187  printf (" %s\n", buff);
188 
189  // Output the offset.
190  //printf (" %04x ", i);
191  printf (" %p ", &addr[i]);
192  }
193 
194  // Now the hex code for the specific character.
195  printf (" %02x", addr[i]);
196 
197  // And store a printable ASCII character for later.
198  if ((addr[i] < 0x20) || (addr[i] > 0x7e))
199  buff[i % 16] = '.';
200  else
201  buff[i % 16] = addr[i];
202  buff[(i % 16) + 1] = '\0';
203  fflush(stdout);
204  }
205 
206  // Pad out last line if not exactly 16 characters.
207  while ((i % 16) != 0) {
208  printf (" ");
209  fflush(stdout);
210  i++;
211  }
212 
213  // And print the final ASCII bit.
214  printf (" %s\n", buff);
215  fflush(stdout);
216 }
