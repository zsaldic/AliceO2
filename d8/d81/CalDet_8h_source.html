95 template <class T>
96 inline const T CalDet<T>::getValue(const int sector, const int globalPadInSector) const
97 {
98  // This shold be a temporary speedup, a proper restructuring of Mapper and CalDet/CalArray is needed.
99  // The default granularity for the moment should be ROC, for the assumptions below this should be assured
100  assert(mPadSubset == PadSubset::ROC);
101  int roc = sector;
102  int padInROC = globalPadInSector;
103  const int padsInIROC = Mapper::getPadsInIROC();
104  if (globalPadInSector >= padsInIROC) {
105  roc += Mapper::getNumberOfIROCs();
106  padInROC -= padsInIROC;
107  }
108  return mData[roc].getValue(padInROC);
109 }

143 template <class T>
144 inline const T CalDet<T>::getValue(const CRU cru, const size_t row, const size_t pad) const
145 {
146  // TODO: might need speedup and beautification
147  static const Mapper& mapper = Mapper::instance();
148  const auto& info = mapper.getPadRegionInfo(cru.region());
149 
150  // bind row and pad to the maximum rows and pads in the requested region
151  const size_t nRows = info.getNumberOfPadRows();
152  const size_t mappedRow = row % nRows;
153  const size_t nPads = info.getPadsInRowRegion(mappedRow);
154  const size_t mappedPad = pad % nPads;
155 
156  switch (mPadSubset) {
157  case PadSubset::ROC: {
158  const ROC roc = cru.roc();
159  const size_t irocOffset = (cru.rocType() == RocType::IROC) ? 0 : mapper.getNumberOfRowsROC(0);
160  const size_t rowROC = mappedRow + info.getGlobalRowOffset() - irocOffset;
161  const size_t channel = mapper.getPadNumberInROC(PadROCPos(roc, rowROC, mappedPad));
162  // printf("roc %2d, row %2zu, pad %3zu, channel: %3zu\n", roc.getRoc(), rowROC, mappedPad, channel);
163  return mData[roc].getValue(channel);
164  break;
165  }
166  case PadSubset::Partition: {
167  break;
168  }
169  case PadSubset::Region: {
170  return mData[cru].getValue(mappedRow, mappedPad);
171  break;
172  }
173  }
174  return T{};
175 }

286 template <class T>
287 void CalDet<T>::initData()
288 {
289  const auto& mapper = Mapper::instance();
290 
291  // ---| Define number of sub pad regions |------------------------------------
292  size_t size = 0;
293  std::string frmt;
294  switch (mPadSubset) {
295  case PadSubset::ROC: {
296  size = ROC::MaxROC;
297  frmt = "%1%_ROC_%2$02d";
298  break;
299  }
300  case PadSubset::Partition: {
301  size = Sector::MAXSECTOR * mapper.getNumberOfPartitions();
302  frmt = "%1%_Partition_%2$02d";
303  break;
304  }
305  case PadSubset::Region: {
306  size = Sector::MAXSECTOR * mapper.getNumberOfPadRegions();
307  frmt = "%1%_Region_%2$02d";
308  break;
309  }
310  }
311 
312  for (size_t i = 0; i < size; ++i) {
313  mData.push_back(CalType(mPadSubset, i));
314  mData.back().setName(boost::str(format(frmt) % mName % i));
315  }
316 }

319 }
