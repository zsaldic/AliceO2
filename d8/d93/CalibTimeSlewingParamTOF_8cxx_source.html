55 float CalibTimeSlewingParamTOF::evalTimeSlewing(int channel, float tot) const
56 {
57  int sector = channel / NCHANNELXSECTOR;
58  channel = channel % NCHANNELXSECTOR;
59 
60  if (sector >= NSECTORS)
61  return 0.; // something went wrong!
62 
63  int n = mChannelStart[sector][channel];
64  if (n < 0)
65  return 0.;
66 
67  int nstop = (*(mTimeSlewing[sector])).size();
68  if (channel < NCHANNELXSECTOR - 1)
69  nstop = mChannelStart[sector][channel + 1];
70 
71  if (n >= nstop)
72  return 0.; // something went wrong!
73 
74  while (n < nstop && tot < (*(mTimeSlewing[sector]))[n].first)
75  n++;
76  n--;
77 
78  if (n < 0) { // tot is lower than the first available value
79  return 0;
80  }
81 
82  if (n == nstop - 1)
83  return (*(mTimeSlewing[sector]))[n].second; // use the last value stored for that channel
84 
85  float w1 = tot - (*(mTimeSlewing[sector]))[n].first;
86  float w2 = (*(mTimeSlewing[sector]))[n + 1].first - tot;
87 
88  return ((*(mTimeSlewing[sector]))[n].second * w2 + (*(mTimeSlewing[sector]))[n + 1].second * w1) / ((*(mTimeSlewing[sector]))[n + 1].first - (*(mTimeSlewing[sector]))[n].first);
89 }

92 void CalibTimeSlewingParamTOF::addTimeSlewingInfo(int channel, float tot, float time)
93 {
94  // WE ARE ASSUMING THAT:
95  // channels have to be filled in increasing order (within the sector)
96  // tots have to be filled in increasing order (within the channel)
97  int sector = channel / NCHANNELXSECTOR;
98  channel = channel % NCHANNELXSECTOR;
99 
100  // printf("DBG: addTimeSlewinginfo sec=%i\n",sector);
101 
102  if (sector >= NSECTORS)
103  return; // something went wrong!
104 
105  int currentch = channel;
106  while (mChannelStart[sector][currentch] == -1 && currentch > -1) {
107  // printf("DBG: fill channel %i\n",currentch);
108  // set also all the previous ones which were not filled
109  mChannelStart[sector][currentch] = (*(mTimeSlewing[sector])).size();
110  currentch--;
111  }
112  // printf("DBG: emplace back (%f,%f)\n",tot,time);
113  (*(mTimeSlewing[sector])).emplace_back(tot, time);
