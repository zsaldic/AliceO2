70 int TPCCATracking::runTracking(const ClusterNativeAccessFullTPC& clusters, std::vector<TrackTPC>* outputTracks,
71  MCLabelContainer* outputTracksMCTruth)
72 {
73  const static ParameterDetector& detParam = ParameterDetector::defaultInstance();
74  const static ParameterGas& gasParam = ParameterGas::defaultInstance();
75  const static ParameterElectronics& elParam = ParameterElectronics::defaultInstance();
76  float vzbin = (elParam.getZBinWidth() * gasParam.getVdrift());
77  float vzbinInv = 1.f / vzbin;
78  Mapper& mapper = Mapper::instance();
79 
80  const AliHLTTPCGMMergedTrack* tracks;
81  int nTracks;
82  const AliHLTTPCGMMergedTrackHit* trackClusters;
83  int retVal = mTrackingCAO2Interface->RunTracking(&clusters, tracks, nTracks, trackClusters);
84  if (retVal == 0) {
85  std::vector<std::pair<int, float>> trackSort(nTracks);
86  int tmp = 0, tmp2 = 0;
87  for (char cside = 0; cside < 2; cside++) {
88  for (int i = 0; i < nTracks; i++) {
89  if (tracks[i].OK() && tracks[i].CSide() == cside)
90  trackSort[tmp++] = { i, (cside == 0 ? 1.f : -1.f) * tracks[i].GetParam().GetZOffset() };
91  }
92  std::sort(trackSort.data() + tmp2, trackSort.data() + tmp,
93  [](const auto& a, const auto& b) { return (a.second > b.second); });
94  tmp2 = tmp;
95  if (cside == 0)
96  mNTracksASide = tmp;
97  }
98  nTracks = tmp;
99 
100  outputTracks->resize(nTracks);
101 
102  for (int iTmp = 0; iTmp < nTracks; iTmp++) {
103  auto& oTrack = (*outputTracks)[iTmp];
104  const int i = trackSort[iTmp].first;
105  float time0 = 0.f, tFwd = 0.f, tBwd = 0.f;
106 
107  float zHigh = 0, zLow = 0;
108  if (mTrackingCAO2Interface->GetParamContinuous()) {
109  float zoffset = tracks[i].CSide() ? -tracks[i].GetParam().GetZOffset() : tracks[i].GetParam().GetZOffset();
110  time0 = sContinuousTFReferenceLength - zoffset * vzbinInv;
111 
112  if (tracks[i].CCE()) {
113  bool lastSide = trackClusters[tracks[i].FirstClusterRef()].fSlice < Sector::MAXSECTOR / 2;
114  float delta = 0.f;
115  for (int iCl = 1; iCl < tracks[i].NClusters(); iCl++) {
116  if (lastSide ^ (trackClusters[tracks[i].FirstClusterRef() + iCl].fSlice < Sector::MAXSECTOR / 2)) {
117  auto& hltcl1 = trackClusters[tracks[i].FirstClusterRef() + iCl];
118  auto& hltcl2 = trackClusters[tracks[i].FirstClusterRef() + iCl - 1];
119  auto& cl1 = clusters.clusters[hltcl1.fSlice][hltcl1.fRow][hltcl1.fNum];
120  auto& cl2 = clusters.clusters[hltcl2.fSlice][hltcl2.fRow][hltcl2.fNum];
121  delta = fabs(cl1.getTime() - cl2.getTime()) * 0.5f;
122  break;
123  }
124  }
125  tFwd = tBwd = delta;
126  } else {
127  // estimate max/min time increments which still keep track in the physical limits of the TPC
128  zHigh = trackClusters[tracks[i].FirstClusterRef()].fZ - tracks[i].GetParam().GetZOffset(); // high R cluster
129  zLow = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fZ -
130  tracks[i].GetParam().GetZOffset(); // low R cluster
131 
132  bool sideHighA = trackClusters[tracks[i].FirstClusterRef()].fSlice < Sector::MAXSECTOR / 2;
133  bool sideLowA =
134  trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fSlice < Sector::MAXSECTOR / 2;
135 
136  // calculate time bracket
137  float zLowAbs = zLow < 0.f ? -zLow : zLow;
138  float zHighAbs = zHigh < 0.f ? -zHigh : zHigh;
139  //
140  // tFwd = (Lmax - max(|zLow|,|zAbs|))/vzbin = drift time from cluster current Z till endcap
141  // tBwd = min(|zLow|,|zAbs|))/vzbin = drift time from CE till cluster current Z
142  //
143  if (zLowAbs < zHighAbs) {
144  tFwd = (detParam.getTPClength() - zHighAbs) * vzbinInv;
145  tBwd = zLowAbs * vzbinInv;
146  } else {
147  tFwd = (detParam.getTPClength() - zLowAbs) * vzbinInv;
148  tBwd = zHighAbs * vzbinInv;
149  }
150  }
151  }
152 
153  oTrack =
154  TrackTPC(tracks[i].GetParam().GetX(), tracks[i].GetAlpha(),
155  { tracks[i].GetParam().GetY(), tracks[i].GetParam().GetZ(), tracks[i].GetParam().GetSinPhi(),
156  tracks[i].GetParam().GetDzDs(), tracks[i].GetParam().GetQPt() },
157  { tracks[i].GetParam().GetCov(0), tracks[i].GetParam().GetCov(1), tracks[i].GetParam().GetCov(2),
158  tracks[i].GetParam().GetCov(3), tracks[i].GetParam().GetCov(4), tracks[i].GetParam().GetCov(5),
159  tracks[i].GetParam().GetCov(6), tracks[i].GetParam().GetCov(7), tracks[i].GetParam().GetCov(8),
160  tracks[i].GetParam().GetCov(9), tracks[i].GetParam().GetCov(10), tracks[i].GetParam().GetCov(11),
161  tracks[i].GetParam().GetCov(12), tracks[i].GetParam().GetCov(13), tracks[i].GetParam().GetCov(14) });
162  oTrack.setTime0(time0);
163  oTrack.setDeltaTBwd(tBwd);
164  oTrack.setDeltaTFwd(tFwd);
165  if (tracks[i].CCE()) {
166  oTrack.setHasCSideClusters();
167  oTrack.setHasASideClusters();
168  } else if (tracks[i].CSide()) {
169  oTrack.setHasCSideClusters();
170  } else {
171  oTrack.setHasASideClusters();
172  }
173 
174  oTrack.setChi2(tracks[i].GetParam().GetChi2());
175  auto& outerPar = tracks[i].OuterParam();
176  oTrack.setOuterParam(o2::track::TrackParCov(
177  outerPar.fX, outerPar.fAlpha,
178  { outerPar.fP[0], outerPar.fP[1], outerPar.fP[2], outerPar.fP[3], outerPar.fP[4] },
179  { outerPar.fC[0], outerPar.fC[1], outerPar.fC[2], outerPar.fC[3], outerPar.fC[4], outerPar.fC[5],
180  outerPar.fC[6], outerPar.fC[7], outerPar.fC[8], outerPar.fC[9], outerPar.fC[10], outerPar.fC[11],
181  outerPar.fC[12], outerPar.fC[13], outerPar.fC[14] }));
182  oTrack.resetClusterReferences(tracks[i].NClusters());
183  std::vector<std::pair<MCCompLabel, unsigned int>> labels;
184  for (int j = 0; j < tracks[i].NClusters(); j++) {
185  int clusterId = trackClusters[tracks[i].FirstClusterRef() + j].fNum;
186  Sector sector = trackClusters[tracks[i].FirstClusterRef() + j].fSlice;
187  int globalRow = trackClusters[tracks[i].FirstClusterRef() + j].fRow;
188  const ClusterNative& cl = clusters.clusters[sector][globalRow][clusterId];
189  int regionNumber = 0;
190  while (globalRow > mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber))
191  regionNumber++;
192  CRU cru(sector, regionNumber);
193  oTrack.addCluster(Cluster(cru, globalRow - mapper.getGlobalRowOffsetRegion(regionNumber), cl.qTot, cl.qMax,
194  cl.getPad(), cl.getSigmaPad(), cl.getTime(), cl.getSigmaTime()));
195  oTrack.setClusterReference(j, sector, globalRow, clusterId);
196  if (outputTracksMCTruth) {
197  for (const auto& element : clusters.clustersMCTruth[sector][globalRow]->getLabels(clusterId)) {
198  bool found = false;
199  for (int l = 0; l < labels.size(); l++) {
200  if (labels[l].first == element) {
201  labels[l].second++;
202  found = true;
203  break;
204  }
205  }
206  if (!found)
207  labels.emplace_back(element, 1);
208  }
209  }
210  }
211  if (outputTracksMCTruth) {
212  if (labels.size() == 0) {
213  outputTracksMCTruth->addElement(iTmp, MCCompLabel()); //default constructor creates NotSet label
214  } else {
215  int bestLabelNum = 0, bestLabelCount = 0;
216  for (int j = 0; j < labels.size(); j++) {
217  if (labels[j].second > bestLabelCount) {
218  bestLabelNum = j;
219  bestLabelCount = labels[j].second;
220  }
221  }
222  MCCompLabel& bestLabel = labels[bestLabelNum].first;
223  if (bestLabelCount < (1.f - sTrackMCMaxFake) * tracks[i].NClusters())
224  bestLabel.set(-bestLabel.getTrackID(), bestLabel.getEventID(), bestLabel.getSourceID());
225  outputTracksMCTruth->addElement(iTmp, bestLabel);
226  }
227  }
228  int lastSector = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fNum >> 24;
229  }
230  }
231  mTrackingCAO2Interface->Cleanup();
232  return (retVal);
233 }
