149 void render3D()
150 {
151  int cur_width, cur_height;
152  glfwGetFramebufferSize(g3dCtx.window, &cur_width, &cur_height);
153 
154  int lastDrawState = -1;
155  int lastUniform = -1;
156 
157  // Cleaning happens somewhere else
158  sg_begin_default_pass(&g3dCtx.passes.back(), cur_width, cur_height);
159 
160  for (auto& command : g3dCtx.commands) {
162  if (command.drawStateIdx != lastDrawState) {
163  sg_draw_state const& state = g3dCtx.drawStates[command.drawStateIdx];
164  lastDrawState = command.drawStateIdx;
165  sg_apply_draw_state(&state);
166  }
168  if (command.uniformIdx != lastUniform) {
169  vs_params_t const& params = g3dCtx.params[command.uniformIdx];
170  sg_apply_uniform_block(SG_SHADERSTAGE_VS, 0, &params, sizeof(params));
171  lastUniform = command.uniformIdx;
172  }
174  sg_draw(command.start, command.end, command.instances);
175  }
176  sg_end_pass();
177  sg_commit();
178 }
