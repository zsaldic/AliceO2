66  size_t finalise(char **out, MergeableId &id) {
67  *out = nullptr;
68  if (mCheckIfComplete(id, mPartsMap) == false) {
69  return 0;
70  }
71  // If we are here, it means we can send the messages that belong
72  // to some predefined class of equivalence, identified by the MERGEABLE_ID,
73  // to the receiver. This is done by the following process:
74  //
75  // - Extract what we actually want to send (this might be data embedded inside the message itself)
76  // - Calculate the aggregate size of all the payloads.
77  // - Copy all the parts into a final payload
78  // - Create the header part
79  // - Create the payload part
80  // - Send
81  std::vector<std::pair<char *, size_t>> parts;
82 
83  size_t sum = 0;
84  auto range = mPartsMap.equal_range(id);
85  for (auto hi = range.first, he = range.second; hi != he; ++hi) {
86  std::unique_ptr<FairMQMessage> &payload = hi->second;
87  std::pair<char *, size_t> part;
88  part.second = mExtractPayload(&part.first, reinterpret_cast<char *>(payload->GetData()), payload->GetSize());
89  parts.push_back(part);
90  sum += part.second;
91  }
92 
93  auto *payload = new char[sum]();
94  size_t offset = 0;
95  for (auto &part : parts) {
96  // Right now this does a copy. In principle this could be done with some sort of
97  // vectorized I/O
98  memcpy(payload + offset, part.first, part.second);
99  offset += part.second;
100  }
101 
102  mPartsMap.erase(id);
103  *out = payload;
104  return sum;
105  }
