67 void Detector::ConstructGeometry()

69  LOG(DEBUG) << "Creating FIT geometry\n";
70  CreateMaterials();
71 
72  Float_t zdetA = 333;
73  Float_t zdetC = 82;
74 
75  Int_t idrotm[999];
76  Double_t x, y, z;
77  Float_t pstartC[3] = { 20., 20, 5 };
78  Float_t pstartA[3] = { 20, 20, 5 };
79  Float_t pinstart[3] = { 2.95, 2.95, 4.34 };
80  Float_t pmcp[3] = { 2.949, 2.949, 2.8 }; // MCP
81 
82  int nCellsA = Geometry::NCellsA;
83  int nCellsC = Geometry::NCellsC;
84 
85  Geometry geometry;
86  TVector3 centerMCP = geometry.centerMCP(2);
87  Matrix(idrotm[901], 90., 0., 90., 90., 180., 0.);
88 
89  // C side Concave Geometry
90 
91  Double_t crad = 82.; // define concave c-side radius here
92 
93  Double_t dP = pmcp[0]; // side length of mcp divided by 2
94 
95  // uniform angle between detector faces==
96  Double_t btta = 2 * TMath::ATan(dP / crad);
97 
98  // get noncompensated translation data
99  Double_t grdin[6] = { -3, -2, -1, 1, 2, 3 };
100  Double_t gridpoints[6];
101  for (Int_t i = 0; i < 6; i++) {
102  gridpoints[i] = crad * TMath::Sin((1 - 1 / (2 * TMath::Abs(grdin[i]))) * grdin[i] * btta);
103  }
104 
105  Double_t xi[Geometry::NCellsC] = { gridpoints[1], gridpoints[2], gridpoints[3], gridpoints[4], gridpoints[0],
106  gridpoints[1], gridpoints[2], gridpoints[3], gridpoints[4], gridpoints[5],
107  gridpoints[0], gridpoints[1], gridpoints[4], gridpoints[5], gridpoints[0],
108  gridpoints[1], gridpoints[4], gridpoints[5], gridpoints[0], gridpoints[1],
109  gridpoints[2], gridpoints[3], gridpoints[4], gridpoints[5], gridpoints[1],
110  gridpoints[2], gridpoints[3], gridpoints[4] };
111  Double_t yi[Geometry::NCellsC] = { gridpoints[5], gridpoints[5], gridpoints[5], gridpoints[5], gridpoints[4],
112  gridpoints[4], gridpoints[4], gridpoints[4], gridpoints[4], gridpoints[4],
113  gridpoints[3], gridpoints[3], gridpoints[3], gridpoints[3], gridpoints[2],
114  gridpoints[2], gridpoints[2], gridpoints[2], gridpoints[1], gridpoints[1],
115  gridpoints[1], gridpoints[1], gridpoints[1], gridpoints[1], gridpoints[0],
116  gridpoints[0], gridpoints[0], gridpoints[0] };
117  Double_t zi[Geometry::NCellsC];
118  for (Int_t i = 0; i < Geometry::NCellsC; i++) {
119  zi[i] = TMath::Sqrt(TMath::Power(crad, 2) - TMath::Power(xi[i], 2) - TMath::Power(yi[i], 2));
120  }
121 
122  // get rotation data
123  Double_t ac[Geometry::NCellsC], bc[Geometry::NCellsC], gc[Geometry::NCellsC];
124  for (Int_t i = 0; i < Geometry::NCellsC; i++) {
125  ac[i] = TMath::ATan(yi[i] / xi[i]) - TMath::Pi() / 2 + 2 * TMath::Pi();
126  if (xi[i] < 0) {
127  bc[i] = TMath::ACos(zi[i] / crad);
128  } else {
129  bc[i] = -1 * TMath::ACos(zi[i] / crad);
130  }
131  }
132  Double_t xc2[Geometry::NCellsC], yc2[Geometry::NCellsC], zc2[Geometry::NCellsC];
133 
134  // compensation based on node position within individual detector geometries
135  // determine compensated radius
136  Double_t rcomp = crad + pstartC[2] / 2.0; //
137  for (Int_t i = 0; i < Geometry::NCellsC; i++) {
138  // Get compensated translation data
139  xc2[i] = rcomp * TMath::Cos(ac[i] + TMath::Pi() / 2) * TMath::Sin(-1 * bc[i]);
140  yc2[i] = rcomp * TMath::Sin(ac[i] + TMath::Pi() / 2) * TMath::Sin(-1 * bc[i]);
141  zc2[i] = rcomp * TMath::Cos(bc[i]);
142 
143  // Convert angles to degrees
144  ac[i] *= 180 / TMath::Pi();
145  bc[i] *= 180 / TMath::Pi();
146  gc[i] = -1 * ac[i];
147  }
148  // A Side
149 
150  Float_t xa[Geometry::NCellsA] = { -11.8, -5.9, 0, 5.9, 11.8, -11.8, -5.9, 0, 5.9, 11.8, -12.8, -6.9,
151  6.9, 12.8, -11.8, -5.9, 0, 5.9, 11.8, -11.8, -5.9, 0, 5.9, 11.8 };
152 
153  Float_t ya[Geometry::NCellsA] = { 11.9, 11.9, 12.9, 11.9, 11.9, 6.0, 6.0, 7.0, 6.0, 6.0, -0.1, -0.1,
154  0.1, 0.1, -6.0, -6.0, -7.0, -6.0, -6.0, -11.9, -11.9, -12.9, -11.9, -11.9 };
155 
156  TGeoVolumeAssembly* stlinA = new TGeoVolumeAssembly("0STL"); // A side mother
157  TGeoVolumeAssembly* stlinC = new TGeoVolumeAssembly("0STR"); // C side mother
158 
159  // FIT interior
160  TVirtualMC::GetMC()->Gsvolu("0INS", "BOX", getMediumID(kAir), pinstart, 3);
161  TGeoVolume* ins = gGeoManager->GetVolume("0INS");
162  //
163  TGeoTranslation* tr[Geometry::NCellsA + Geometry::NCellsC];
164  TString nameTr;
165 
166  // A side Translations
167  for (Int_t itr = 0; itr < Geometry::NCellsA; itr++) {
168  nameTr = Form("0TR%i", itr + 1);
169  z = -pstartA[2] + pinstart[2];
170  tr[itr] = new TGeoTranslation(nameTr.Data(), xa[itr], ya[itr], z);
171  tr[itr]->RegisterYourself();
172  stlinA->AddNode(ins, itr, tr[itr]);
173  }
174 
175  TGeoRotation* rot[Geometry::NCellsC];
176  TString nameRot;
177 
178  TGeoCombiTrans* com[Geometry::NCellsC];
179  TString nameCom;
180 
181  // C Side Transformations
182  for (Int_t itr = Geometry::NCellsA; itr < Geometry::NCellsA + Geometry::NCellsC; itr++) {
183  nameTr = Form("0TR%i", itr + 1);
184  nameRot = Form("0Rot%i", itr + 1);
185  int ic = itr - Geometry::NCellsA;
186  // nameCom = Form("0Com%i",itr+1);

188  rot[ic]->RegisterYourself();
189 
190  tr[itr] = new TGeoTranslation(nameTr.Data(), xc2[ic], yc2[ic], (zc2[ic] - 80.));
191  tr[itr]->RegisterYourself();
192 
193  // com[itr-Geometry::NCellsA] = new TGeoCombiTrans(tr[itr],rot[itr-Geometry::NCellsA]);
194  com[ic] = new TGeoCombiTrans(xc2[ic], yc2[ic], (zc2[ic] - 80), rot[ic]);
195  TGeoHMatrix hm = *com[ic];
196  TGeoHMatrix* ph = new TGeoHMatrix(hm);
197  stlinC->AddNode(ins, itr, ph);
198  }
199

201  alice->AddNode(stlinA, 1, new TGeoTranslation(0, 0, zdetA));

204  alice->AddNode(stlinC, 1, new TGeoCombiTrans(0., 0., -zdetC, rotC));

218 void Detector::SetOneMCP(TGeoVolume* ins)
219 {
220  Double_t x, y, z;
221  Float_t pinstart[3] = { 2.95, 2.95, 4.34 };
222  Float_t pmcp[3] = { 2.949, 2.949, 2.8 }; // MCP
223  Float_t ptop[3] = { 1.324, 1.324, 1. }; // cherenkov radiator
224  Float_t ptopref[3] = { 1.3241, 1.3241, 1. }; // cherenkov radiator wrapped with reflection
225  Float_t preg[3] = { 1.324, 1.324, 0.005 }; // photcathode
226  Double_t prfv[3] = { 0.0002, 1.323,
227  1. }; // vertical refracting layer bettwen radiators and bettwen radiator and not optical Air
228  Double_t prfh[3] = { 1.323, 0.0002, 1. }; // horizontal refracting layer bettwen radiators a
229  Double_t pal[3] = { 2.648, 2.648, 0.25 }; // 5mm Al top on th eeach radiator
230  // Entry window (glass)
231  TVirtualMC::GetMC()->Gsvolu("0TOP", "BOX", getMediumID(kOpGlass), ptop, 3); // glass radiator
232  TGeoVolume* top = gGeoManager->GetVolume("0TOP");
233  TVirtualMC::GetMC()->Gsvolu("0TRE", "BOX", getMediumID(kAir), ptopref, 3); // air: wrapped radiator
234  TGeoVolume* topref = gGeoManager->GetVolume("0TRE");
235  TVirtualMC::GetMC()->Gsvolu("0REG", "BOX", getMediumID(kOpGlassCathode), preg, 3);
236  TGeoVolume* cat = gGeoManager->GetVolume("0REG");
237  TVirtualMC::GetMC()->Gsvolu("0MCP", "BOX", getMediumID(kGlass), pmcp, 3); // glass
238  TGeoVolume* mcp = gGeoManager->GetVolume("0MCP");
239  TVirtualMC::GetMC()->Gsvolu("0RFV", "BOX", getMediumID(kOpAir), prfv, 3); // optical Air vertical
240  TGeoVolume* rfv = gGeoManager->GetVolume("0RFV");
241  TVirtualMC::GetMC()->Gsvolu("0RFH", "BOX", getMediumID(kOpAir), prfh, 3); // optical Air horizontal
242  TGeoVolume* rfh = gGeoManager->GetVolume("0RFH");
243 
244  TVirtualMC::GetMC()->Gsvolu("0PAL", "BOX", getMediumID(kAl), pal, 3); // 5mmAl top on the radiator
245  TGeoVolume* altop = gGeoManager->GetVolume("0PAL");
246 
247  // wrapped radiator + reflectiong layers
248  Int_t ntops = 0, nrfvs = 0, nrfhs = 0;
249  Float_t xin = 0, yin = 0, xinv = 0, yinv = 0, xinh = 0, yinh = 0;
250  x = y = z = 0;
251  topref->AddNode(top, 1, new TGeoTranslation(0, 0, 0));
252  xinv = -ptop[0] - prfv[0];
253  topref->AddNode(rfv, 1, new TGeoTranslation(xinv, 0, 0));
254  xinv = ptop[0] + prfv[0];
255  topref->AddNode(rfv, 2, new TGeoTranslation(xinv, 0, 0));
256  yinv = -ptop[1] - prfh[1];
257  topref->AddNode(rfh, 1, new TGeoTranslation(0, yinv, 0));
258  yinv = ptop[1] + prfh[1];
259  topref->AddNode(rfh, 2, new TGeoTranslation(0, yinv, 0));
260 
261  // container for radiator, cathod
262  for (Int_t ix = 0; ix < 2; ix++) {
263  xin = -pinstart[0] + 0.3 + (ix + 0.5) * 2 * ptopref[0];
264  for (Int_t iy = 0; iy < 2; iy++) {
265  z = -pinstart[2] + 2 * pal[2] + ptopref[2];
266  yin = -pinstart[1] + 0.3 + (iy + 0.5) * 2 * ptopref[1];
267  ntops++;
268  ins->AddNode(topref, ntops, new TGeoTranslation(xin, yin, z));
269  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + preg[2];
270  ins->AddNode(cat, ntops, new TGeoTranslation(xin, yin, z));
271  }
272  }
273  // Al top
274  z = -pinstart[2] + pal[2];
275  ins->AddNode(altop, 1, new TGeoTranslation(0, 0, z));
276 
277  // MCP
278  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + 2 * preg[2] + pmcp[2];
279  // z=-pinstart[2] + 2*ptopref[2] + preg[2];
280  ins->AddNode(mcp, 1, new TGeoTranslation(0, 0, z));
281 }

283 Bool_t Detector::ProcessHits(FairVolume* v)
284 {
285  Int_t quadrant, mcp;
286  if (fMC->IsTrackEntering()) {
287  float x, y, z;
288  fMC->TrackPosition(x, y, z);
289  fMC->CurrentVolID(quadrant);
290  fMC->CurrentVolOffID(1, mcp);
291  float time = fMC->TrackTime() * 1.0e9; //time from seconds to ns
292  int trackID = fMC->GetStack()->GetCurrentTrackNumber();
293  int detID = 4 * mcp + quadrant - 1;
294  float etot = fMC->Etot();
295  int iPart = fMC->TrackPid();
296  float enDep = fMC->Edep();
297  if (fMC->TrackCharge()) { //charge particles for MCtrue
298  AddHit(x, y, z, time, 10, trackID, detID);
299  }
300  if (iPart == 50000050) // If particles is photon then ...
301  {
302  if (RegisterPhotoE(etot)) {
303  AddHit(x, y, z, time, enDep, trackID, detID);
304  }
305  }
306 
307  return kTRUE;
308  }
309  return kFALSE;
310 }

446 Int_t Detector::ReadOptProperties(const std::string filePath)
447 {
448  std::ifstream infile;
449  infile.open(filePath.c_str());
450 
451  // Check if file is opened correctly
452  if (infile.fail() == true) {
453  // AliFatal(Form("Error opening ascii file: %s", filePath.c_str()));
454  return -1;
455  }
456 
457  std::string comment; // dummy, used just to read 4 first lines and move the cursor to the 5th, otherwise unused
458  if (!getline(infile, comment)) { // first comment line
459  // AliFatal(Form("Error opening ascii file (it is probably a folder!): %s", filePath.c_str()));
460  return -2;
461  }
462  getline(infile, comment); // 2nd comment line
463 
464  // Get number of elements required for the array
465  Int_t nLines;
466  infile >> nLines;
467  if (nLines < 0 || nLines > 1e4) {
468  // AliFatal(Form("Input arraySize out of range 0..1e4: %i. Check input file: %s", kNbins, filePath.c_str()));
469  return -4;
470  }
471 
472  getline(infile, comment); // finish 3rd line after the nEntries are read
473  getline(infile, comment); // 4th comment line
474 
475  // read the main body of the file (table of values: energy, absorption length and refractive index)
476  Int_t iLine = 0;
