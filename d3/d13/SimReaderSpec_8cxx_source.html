36 DataProcessorSpec getSimReaderSpec(int fanoutsize, const std::vector<int>& tpcsectors,
37  std::shared_ptr<std::vector<int>> tpcsubchannels)
38 {
39  // this container will contain the TPC sector assignment per subchannel per invocation
40  // it will allow that we snapshot/send exactly one sector assignment per algorithm invocation
41  // to ensure that they all have different timeslice ids
42  auto tpcsectormessages = std::make_shared<std::vector<std::vector<int>>>();
43  tpcsectormessages->resize(tpcsubchannels->size());
44  int tpcchannelcounter = 0;
45  uint64_t activeSectors = 0;
46  for (const auto& tpcsector : tpcsectors) {
47  activeSectors |= (uint64_t)0x1 << tpcsector;
48  auto actualchannel = (*tpcsubchannels.get())[tpcchannelcounter % tpcsubchannels->size()];
49  LOG(DEBUG) << " WILL ASSIGN SECTOR " << tpcsector << " to subchannel " << actualchannel;
50  tpcsectormessages->operator[](actualchannel).emplace_back(tpcsector);
51  tpcchannelcounter++;
52  }
53 
54  // this is the number of invocations of the algorithm needed for the TPC
55  size_t tpcinvocations = 0;
56  for (int i = 0; i < tpcsubchannels->size(); ++i) {
57  tpcinvocations = std::max(tpcinvocations, tpcsectormessages->operator[](i).size());
58  }
59  // in principle each channel needs to be invoked exactly the same number of times (sigh)
60  // so I am adding some kind of "NOP" sectors in case needed
61  for (int i = 0; i < tpcsubchannels->size(); ++i) {
62  auto size = tpcsectormessages->operator[](i).size();
63  if (size < tpcinvocations) {
64  for (int k = 0; k < tpcinvocations - size; ++k) {
65  tpcsectormessages->operator[](i).emplace_back(-2); // -2 is NOP
66  LOG(INFO) << "ADDING NOP TO CHANNEL " << i << "\n";
67  }
68  assert(tpcsectormessages->operator[](i).size() == tpcinvocations);
69  }
70  }
71  // at this moment all tpc digitizers should receive the exact same number of messages/invocations
72 
73  auto doit = [fanoutsize, tpcsectormessages, tpcinvocations, tpcsubchannels, activeSectors](ProcessingContext& pc) {
74  auto& mgr = steer::HitProcessingManager::instance();
75  auto eventrecords = mgr.getRunContext().getEventRecords();
76  const auto& context = mgr.getRunContext();
77 
78  // counter to make sure we are sending the data only once
79  static int counter = 0;
80 
81  static bool finished = false;
82  static bool tpc_end_messagesent = false;
83  if (finished) {
84  if (!tpc_end_messagesent) {
85  // we need to send this in a different time slice
86  // send message telling tpc workers that they can terminate
87  // do this only one
88  for (const auto& channel : *tpcsubchannels.get()) {
89  // -1 is marker for end of work
90  o2::TPC::TPCSectorHeader header{ -1 };
91  header.activeSectors = activeSectors;
92  pc.outputs().snapshot(
93  OutputRef{ "collisioncontext", static_cast<SubSpecificationType>(channel), { header } },
94  context);
95  }
96  tpc_end_messagesent = true;
97  }
98  // now mark the reader as ready to finish
99  pc.services().get<ControlService>().readyToQuit(false);
100  return;
101  }
102 
103  for (int tpcchannel = 0; tpcchannel < tpcsubchannels->size(); ++tpcchannel) {
104  auto& sectors = tpcsectormessages->operator[](tpcchannel);
105  if (counter < sectors.size()) {
106  auto sector = sectors[counter];
107  // send the sectorassign as header with the collision context data
108  o2::TPC::TPCSectorHeader header{ sector };
109  header.activeSectors = activeSectors;
110  pc.outputs().snapshot(
111  OutputRef{ "collisioncontext", static_cast<SubSpecificationType>(tpcchannel), { header } },
112  context);
113  }
114  }
115 
116  // everything not done previously treat here (this is to be seen how since other things than TPC will have
117  // a different number of invocations)
118  for (int subchannel = 0; subchannel < fanoutsize; ++subchannel) {
119  // TODO: this is temporary ... we should find a more clever+ faster + scalable mechanism
120  if (std::find(tpcsubchannels->begin(), tpcsubchannels->end(), subchannel) != tpcsubchannels->end()) {
121  continue;
122  }
123  LOG(INFO) << "SENDING SOMETHING TO OTHERS";
124  pc.outputs().snapshot(
125  OutputRef{ "collisioncontext", static_cast<SubSpecificationType>(subchannel) },
126  context);
127  }
128  counter++;
129  if (tpcinvocations == 0 || counter == tpcinvocations) {
130  finished = true;
131  }
132  };
133 
134  // init function return a lambda taking a ProcessingContext
135  auto initIt = [doit](InitContext& ctx) {
136  // initialize fundamental objects
137  auto& mgr = steer::HitProcessingManager::instance();
138  mgr.addInputFile(ctx.options().get<std::string>("simFile").c_str());
139  if (ctx.options().get<std::string>("simFileS").size() > 0) {
140  mgr.addInputSignalFile(ctx.options().get<std::string>("simFileS").c_str());
141  }
142 
143  // do we start from an existing context
144  auto incontextstring = ctx.options().get<std::string>("incontext");
145  LOG(INFO) << "INCONTEXTSTRING " << incontextstring;
146  if (incontextstring.size() > 0) {
147  auto success = mgr.setupRunFromExistingContext(incontextstring.c_str());
148  if (!success) {
149  LOG(FATAL) << "Could not read collision context from " << incontextstring;
150  }
151  } else {
152  // number of collisions asked?
153  auto col = ctx.options().get<int>("ncollisions");
154  if (col != 0) {
155  mgr.setupRun(col);
156  } else {
157  mgr.setupRun();
158  }
159  LOG(INFO) << "Initializing Spec ... have " << mgr.getRunContext().getEventRecords().size() << " times ";
160  LOG(INFO) << "Serializing Context for later reuse";
161  mgr.writeRunContext(ctx.options().get<std::string>("outcontext").c_str());
162  }
163 
164  return doit;
165  };

168  for (int subchannel = 0; subchannel < fanoutsize; ++subchannel) {
169  outputs.emplace_back(
