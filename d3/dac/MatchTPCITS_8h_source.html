529 inline void MatchTPCITS::removeTPCfromITS(int tpcMatchID, int itsMatchID)
530 {
532  auto& itsMatch = mMatchesITS[itsMatchID];
533  if (isValidatedITS(itsMatch))
534  return;
535  int topID = MinusOne, next = itsMatch.first;
536  while (next > MinusOne) {
537  auto& rcITS = mMatchRecordsITS[next];
538  if (rcITS.matchID == tpcMatchID) {
539  if (topID < 0) {
540  itsMatch.first = rcITS.nextRecID;
541  } else {
542  mMatchRecordsITS[topID].nextRecID = rcITS.nextRecID;
543  }
544  return;
545  }
546  topID = next;
547  next = rcITS.nextRecID;
548  }
549 }
