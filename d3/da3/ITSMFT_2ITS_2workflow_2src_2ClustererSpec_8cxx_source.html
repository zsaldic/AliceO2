36 void ClustererDPL::init(InitContext& ic)
37 {
38  o2::Base::GeometryManager::loadGeometry(); // for generating full clusters
39  o2::ITS::GeometryTGeo* geom = o2::ITS::GeometryTGeo::Instance();
40  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L));
41 
42  mClusterer = std::make_unique<o2::ITSMFT::Clusterer>();
43  mClusterer->setGeometry(geom);
44  mClusterer->setNChips(o2::ITSMFT::ChipMappingITS::getNChips());
45 
46  auto filenameGRP = ic.options().get<std::string>("grp-file");
47  const auto grp = o2::parameters::GRPObject::loadFrom(filenameGRP.c_str());
48 
49  if (grp) {
50  mClusterer->setContinuousReadOut(grp->isDetContinuousReadOut("ITS"));
51  } else {
52  LOG(ERROR) << "Cannot retrieve GRP from the " << filenameGRP.c_str() << " file !";
53  mState = 0;
54  return;
55  }
56 
57  auto filename = ic.options().get<std::string>("its-dictionary-file");
58  mFile = std::make_unique<std::ifstream>(filename.c_str(), std::ios::in | std::ios::binary);
59  if (mFile->good()) {
60  mClusterer->loadDictionary(filename);
61  LOG(INFO) << "ITSClusterer running with a provided dictionary: " << filename.c_str();
62  mState = 1;
63  } else {
64  LOG(ERROR) << "Cannot open the " << filename.c_str() << " file !";
65  mState = 0;
66  return;
67  }
68 
69  mClusterer->print();
70 }

72 void ClustererDPL::run(ProcessingContext& pc)
73 {
74  if (mState > 1)
75  return;
76 
77  auto digits = pc.inputs().get<const std::vector<o2::ITSMFT::Digit>>("digits");
78  auto labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
79  auto rofs = pc.inputs().get<const std::vector<o2::ITSMFT::ROFRecord>>("ROframes");
80  auto mc2rofs = pc.inputs().get<const std::vector<o2::ITSMFT::MC2ROFRecord>>("MC2ROframes");
81 
82  LOG(INFO) << "ITSClusterer pulled " << digits.size() << " digits, "
83  << labels->getIndexedSize() << " MC label objects, in "
84  << rofs.size() << " RO frames and "
85  << mc2rofs.size() << " MC events";
86 
87  o2::ITSMFT::DigitPixelReader reader;
88  reader.setDigits(&digits);
89  reader.setROFRecords(&rofs);
90  reader.setMC2ROFRecords(&mc2rofs);
91  reader.setDigitsMCTruth(labels.get());
92  reader.init();
93 
94  std::vector<o2::ITSMFT::CompClusterExt> compClusters;
95  std::vector<o2::ITSMFT::Cluster> clusters;
96  o2::dataformats::MCTruthContainer<o2::MCCompLabel> clusterLabels;
97  std::vector<o2::ITSMFT::ROFRecord> clusterROframes; // To be filled in future
98  std::vector<o2::ITSMFT::MC2ROFRecord>& clusterMC2ROframes = mc2rofs; // Simply, replicate it from digits ?
99 
100  mClusterer->process(reader, &clusters, &compClusters, &clusterLabels, &clusterROframes);
101  // TODO: in principle, after masking "overflow" pixels the MC2ROFRecord maxROF supposed to change, nominally to minROF
102  // -> consider recalculationg maxROF
103 
104  LOG(INFO) << "ITSClusterer pushed " << clusters.size() << " clusters, in "
105  << clusterROframes.size() << " RO frames and "
106  << clusterMC2ROframes.size() << " MC events";
107 
108  pc.outputs().snapshot(Output{ "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe }, compClusters);
109  pc.outputs().snapshot(Output{ "ITS", "CLUSTERS", 0, Lifetime::Timeframe }, clusters);
110  pc.outputs().snapshot(Output{ "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe }, clusterLabels);
111  pc.outputs().snapshot(Output{ "ITS", "ITSClusterROF", 0, Lifetime::Timeframe }, clusterROframes);
112  pc.outputs().snapshot(Output{ "ITS", "ITSClusterMC2ROF", 0, Lifetime::Timeframe }, clusterMC2ROframes);
113 
114  mState = 2;
115  //pc.services().get<ControlService>().readyToQuit(true);
116 }
