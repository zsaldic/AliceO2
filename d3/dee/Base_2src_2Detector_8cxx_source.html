118 void Detector::initFieldTrackingParams(int& integration, float& maxfield)
119 {
120  // set reasonable default values
121  integration = 2;
122  maxfield = 10;
123  auto vmc = TVirtualMC::GetMC();
124  if (vmc) {
125  auto field = vmc->GetMagField();
126  // see if we can query the o2 field
127  if (auto o2field = dynamic_cast<o2::field::MagneticField*>(field)) {
128  integration = o2field->Integral(); // default integration method?
129  maxfield = o2field->Max();
130  return;
131  }
132  }
133  LOG(INFO) << "No magnetic field found; using default tracking values " << integration << " " << maxfield
134  << " to initialize media\n";
135 }
