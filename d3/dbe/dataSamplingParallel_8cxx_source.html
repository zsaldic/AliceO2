138 void someDataProducerAlgorithm(ProcessingContext& ctx)
139 {
140  size_t index = ctx.services().get<ParallelContext>().index1D();
141  sleep(1);
142  // Creates a new message of size collectionChunkSize which
143  // has "TPC" as data origin and "CLUSTERS" as data description.
144  auto tpcClusters = ctx.outputs().make<FakeCluster>(Output{ "TPC", "CLUSTERS", index }, collectionChunkSize);
145  int i = 0;
146 
147  for (auto& cluster : tpcClusters) {
148  assert(i < collectionChunkSize);
149  cluster.x = index;
150  cluster.y = i;
151  cluster.z = i;
152  cluster.q = rand() % 1000;
153  i++;
154  }
155 }

157 void someProcessingStageAlgorithm(ProcessingContext& ctx)
158 {
159  size_t index = ctx.services().get<ParallelContext>().index1D();
160 
161  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
162  auto processedTpcClusters =
163  ctx.outputs().make<FakeCluster>(Output{ "TPC", "CLUSTERS_P", index }, collectionChunkSize);
164 
165  int i = 0;
166  for (auto& cluster : processedTpcClusters) {
167  assert(i < collectionChunkSize);
168  cluster.x = -inputDataTpc[i].x;
169  cluster.y = 2 * inputDataTpc[i].y;
170  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
171  cluster.q = inputDataTpc[i].q;
172  i++;
173  }
