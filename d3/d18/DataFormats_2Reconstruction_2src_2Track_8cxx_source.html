21 TrackPar::TrackPar(const array<float, 3>& xyz, const array<float, 3>& pxpypz, int charge, bool sectorAlpha)
22  : mX{ 0.f }, mAlpha{ 0.f }, mP{ 0.f }
23 {
24  // construct track param from kinematics
25 
26  // Alpha of the frame is defined as:
27  // sectorAlpha == false : -> angle of pt direction
28  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
29  // angle of pt direction for r==0
30  //
31  //
32  constexpr float kSafe = 1e-5f;
33  float radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
34  float alp = 0;
35  if (sectorAlpha || radPos2 < 1) {
36  alp = atan2f(pxpypz[1], pxpypz[0]);
37  } else {
38  alp = atan2f(xyz[1], xyz[0]);
39  }
40  if (sectorAlpha) {
41  alp = utils::Angle2Alpha(alp);
42  }
43  //
44  float sn, cs;
45  utils::sincosf(alp, sn, cs);
46  // protection: avoid alpha being too close to 0 or +-pi/2
47  if (fabs(sn) < 2 * kSafe) {
48  if (alp > 0) {
49  alp += alp < PIHalf ? 2 * kSafe : -2 * kSafe;
50  } else {
51  alp += alp > -PIHalf ? -2 * kSafe : 2 * kSafe;
52  }
53  utils::sincosf(alp, sn, cs);
54  } else if (fabs(cs) < 2 * kSafe) {
55  if (alp > 0) {
56  alp += alp > PIHalf ? 2 * kSafe : -2 * kSafe;
57  } else {
58  alp += alp > -PIHalf ? 2 * kSafe : -2 * kSafe;
59  }
60  utils::sincosf(alp, sn, cs);
61  }
62  // get the vertex of origin and the momentum
63  array<float, 3> ver{ xyz[0], xyz[1], xyz[2] };
64  array<float, 3> mom{ pxpypz[0], pxpypz[1], pxpypz[2] };
65  //
66  // Rotate to the local coordinate system
67  utils::RotateZ(ver, -alp);
68  utils::RotateZ(mom, -alp);
69  //
70  float ptI = 1.f / sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
71  mX = ver[0];
72  mAlpha = alp;
73  mP[kY] = ver[1];
74  mP[kZ] = ver[2];
75  mP[kSnp] = mom[1] * ptI;
76  mP[kTgl] = mom[2] * ptI;
77  mP[kQ2Pt] = ptI * charge;
78  //
79  if (fabs(1 - getSnp()) < kSafe) {
80  mP[kSnp] = 1. - kSafe; // Protection
81  } else if (fabs(-1 - getSnp()) < kSafe) {
82  mP[kSnp] = -1. + kSafe; // Protection
83  }
84  //
85 }

88 bool TrackPar::getPxPyPzGlo(array<float, 3>& pxyz) const
89 {
90  // track momentum
91  if (fabs(getQ2Pt()) < Almost0 || fabs(getSnp()) > Almost1) {
92  return false;
93  }
94  float cs, sn, pt = fabs(1.f / getQ2Pt());
95  float r = sqrtf((1.f - getSnp()) * (1.f + getSnp()));
96  utils::sincosf(getAlpha(), sn, cs);
97  pxyz[0] = pt * (r * cs - getSnp() * sn);
98  pxyz[1] = pt * (getSnp() * cs + r * sn);
99  pxyz[2] = pt * getTgl();
100  return true;
101 }

104 bool TrackPar::getPosDirGlo(array<float, 9>& posdirp) const
105 {
106  // fill vector with lab x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha
107  float ptI = fabs(getQ2Pt());
108  float snp = getSnp();
109  if (ptI < Almost0 || fabs(snp) > Almost1) {
110  return false;
111  }
112  float &sn = posdirp[7], &cs = posdirp[8];
113  float csp = sqrtf((1.f - snp) * (1.f + snp));
114  float cstht = sqrtf(1.f + getTgl() * getTgl());
115  float csthti = 1.f / cstht;
116  utils::sincosf(getAlpha(), sn, cs);
117  posdirp[0] = getX() * cs - getY() * sn;
118  posdirp[1] = getX() * sn + getY() * cs;
119  posdirp[2] = getZ();
120  posdirp[3] = (csp * cs - snp * sn) * csthti; // px/p
121  posdirp[4] = (snp * cs + csp * sn) * csthti; // py/p
122  posdirp[5] = getTgl() * csthti; // pz/p
123  posdirp[6] = cstht / ptI; // p
124  return true;
125 }

128 bool TrackPar::rotateParam(float alpha)
129 {
130  // rotate to alpha frame
131  if (fabs(getSnp()) > Almost1) {
132  printf("Precondition is not satisfied: |sin(phi)|>1 ! %f\n", getSnp());
133  return false;
134  }
135  //
136  utils::BringToPMPi(alpha);
137  //
138  float ca = 0, sa = 0;
139  utils::sincosf(alpha - getAlpha(), sa, ca);
140  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp)); // Improve precision
141  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
142  // direction in local frame is along the X axis
143  if ((csp * ca + snp * sa) < 0) {
144  //printf("Rotation failed: local cos(phi) would become %.2f\n", csp * ca + snp * sa);
145  return false;
146  }
147  //
148  float tmp = snp * ca - csp * sa;
149  if (fabs(tmp) > Almost1) {
150  printf("Rotation failed: new snp %.2f\n", tmp);
151  return false;
152  }
153  float xold = getX(), yold = getY();
154  mAlpha = alpha;
155  mX = xold * ca + yold * sa;
156  mP[kY] = -xold * sa + yold * ca;
157  mP[kSnp] = tmp;
158  return true;
159 }

162 bool TrackPar::propagateParamTo(float xk, const array<float, 3>& b)
163 {
164  //----------------------------------------------------------------
165  // Extrapolate this track params (w/o cov matrix) to the plane X=xk in the field b[].
166  //
167  // X [cm] is in the "tracking coordinate system" of this track.
168  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
169  //----------------------------------------------------------------
170 
171  float dx = xk - getX();
172  if (fabs(dx) < Almost0) {
173  return true;
174  }
175  // Do not propagate tracks outside the ALICE detector
176  if (fabs(dx) > 1e5 || fabs(getY()) > 1e5 || fabs(getZ()) > 1e5) {
177  printf("Anomalous track, target X:%f\n", xk);
178  // print();
179  return false;
180  }
181  float crv = (fabs(b[2]) < Almost0) ? 0.f : getCurvature(b[2]);
182  float x2r = crv * dx;
183  float f1 = getSnp(), f2 = f1 + x2r;
184  if (fabs(f1) > Almost1 || fabs(f2) > Almost1) {
185  return false;
186  }
187  if (fabs(getQ2Pt()) < Almost0) {
188  return false;
189  }
190  float r1 = sqrtf((1.f - f1) * (1.f + f1));
191  if (fabs(r1) < Almost0) {
192  return false;
193  }
194  float r2 = sqrtf((1.f - f2) * (1.f + f2));
195  if (fabs(r2) < Almost0) {
196  return false;
197  }
198  float dy2dx = (f1 + f2) / (r1 + r2);
199  float step = (fabs(x2r) < 0.05f) ? dx * fabs(r2 + f2 * dy2dx) // chord
200  : 2.f * asinf(0.5f * dx * sqrtf(1.f + dy2dx * dy2dx) * crv) / crv; // arc
201  step *= sqrtf(1.f + getTgl() * getTgl());
202  //
203  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
204  array<float, 9> vecLab{ 0.f };
205  if (!getPosDirGlo(vecLab)) {
206  return false;
207  }
208 
209  // rotate to the system where Bx=By=0.
210  float bxy2 = b[0] * b[0] + b[1] * b[1];
211  float bt = sqrtf(bxy2);
212  float cosphi = 1.f, sinphi = 0.f;
213  if (bt > Almost0) {
214  cosphi = b[0] / bt;
215  sinphi = b[1] / bt;
216  }
217  float bb = sqrtf(bxy2 + b[2] * b[2]);
218  float costet = 1., sintet = 0.;
219  if (bb > Almost0) {
220  costet = b[2] / bb;
221  sintet = bt / bb;
222  }
223  array<float, 7> vect{ costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
224  -sinphi * vecLab[0] + cosphi * vecLab[1],
225  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
226  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
227  -sinphi * vecLab[3] + cosphi * vecLab[4],
228  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
229  vecLab[6] };
230 
231  // Do the helix step
232  float sgn = getSign();
233  g3helx3(sgn * bb, step, vect);
234 
235  // rotate back to the Global System
236  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
237  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
238  vecLab[2] = -sintet * vect[0] + costet * vect[2];
239 
240  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
241  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
242  vecLab[5] = -sintet * vect[3] + costet * vect[5];
243 
244  // rotate back to the Tracking System
245  float sinalp = -vecLab[7], cosalp = vecLab[8];
246  float t = cosalp * vecLab[0] - sinalp * vecLab[1];
247  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
248  vecLab[0] = t;
249  t = cosalp * vecLab[3] - sinalp * vecLab[4];
250  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
251  vecLab[3] = t;
252 
253  // Do the final correcting step to the target plane (linear approximation)
254  float x = vecLab[0], y = vecLab[1], z = vecLab[2];
255  if (fabs(dx) > Almost0) {
256  if (fabs(vecLab[3]) < Almost0) {
257  return false;
258  }
259  dx = xk - vecLab[0];
260  x += dx;
261  y += vecLab[4] / vecLab[3] * dx;
262  z += vecLab[5] / vecLab[3] * dx;
263  }
264 
265  // Calculate the track parameters
266  t = 1.f / sqrtf(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
267  mX = x;
268  mP[kY] = y;
269  mP[kZ] = z;
270  mP[kSnp] = vecLab[4] * t;
271  mP[kTgl] = vecLab[5] * t;
272  mP[kQ2Pt] = sgn * t / vecLab[6];
273 
274  return true;
275 }

278 bool TrackPar::propagateParamTo(float xk, float b)
279 {
280  //----------------------------------------------------------------
281  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
282  // Only parameters are propagated, not the matrix. To be used for small
283  // distances only (<mm, i.e. misalignment)
284  //----------------------------------------------------------------
285  float dx = xk - getX();
286  if (fabs(dx) < Almost0) {
287  return true;
288  }
289  float crv = (fabs(b) < Almost0) ? 0.f : getCurvature(b);
290  float x2r = crv * dx;
291  float f1 = getSnp(), f2 = f1 + x2r;
292  if ((fabs(f1) > Almost1) || (fabs(f2) > Almost1) || (fabs(getQ2Pt()) < Almost0)) {
293  return false;
294  }
295  float r1 = sqrtf((1.f - f1) * (1.f + f1));
296  if (fabs(r1) < Almost0) {
297  return false;
298  }
299  float r2 = sqrtf((1.f - f2) * (1.f + f2));
300  if (fabs(r2) < Almost0) {
301  return false;
302  }
303  mX = xk;
304  double dy2dx = (f1 + f2) / (r1 + r2);
305  mP[kY] += dx * dy2dx;
306  mP[kSnp] += x2r;
307  if (fabs(x2r) < 0.05f) {
308  mP[kZ] += dx * (r2 + f2 * dy2dx) * getTgl();
309  } else {
310  // for small dx/R the linear apporximation of the arc by the segment is OK,
311  // but at large dx/R the error is very large and leads to incorrect Z propagation
312  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
313  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
314  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
315  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
316  // track1 += rot/crv*track3;
317  //
318  float rot = asinf(r1 * f2 - r2 * f1); // more economic version from Yura.
319  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles
320  if (f2 > 0.f) {
321  rot = PI - rot; //
322  } else {
323  rot = -PI - rot;
324  }
325  }
326  mP[kZ] += getTgl() / crv * rot;
327  }
328  return true;
329 }

472  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one

475  //

478  if (f2 > 0.f) {
479  rot = PI - rot; //
480  } else {
481  rot = -PI - rot;
482  }
483  }
484  dP[kZ] = getTgl() / crv * rot;
485  }
486 
487  updateParams(dP); // apply corrections

506  double b02 = f24 * c40;
507  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
508  double b12 = f24 * c41;
509  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
510  double b22 = f24 * c42;
511  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
512  double b42 = f24 * c44;
513  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
514  double b32 = f24 * c43;
515 
516  // a = f*b = f*C*ft
517  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
518  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
519  double a22 = f24 * b42;
520 
521  // F*C*Ft = C + (b + bt + a)
522  c00 += b00 + b00 + a00;
523  c10 += b10 + b01 + a01;
524  c20 += b20 + b02 + a02;
525  c30 += b30;
526  c40 += b40;
527  c11 += b11 + b11 + a11;
528  c21 += b21 + b12 + a12;
529  c31 += b31;
530  c41 += b41;
531  c22 += b22 + b22 + a22;
532  c32 += b32;
533  c42 += b42;
534 
535  checkCovariance();
536 
537  return true;

541 bool TrackParCov::rotate(float alpha)
542 {
543  // rotate to alpha frame
544  if (fabs(getSnp()) > Almost1) {
545  printf("Precondition is not satisfied: |sin(phi)|>1 ! %f\n", getSnp());
546  return false;
547  }
548  //
549  utils::BringToPMPi(alpha);
550  //
551  float ca = 0, sa = 0;
552  utils::sincosf(alpha - getAlpha(), sa, ca);
553  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp)); // Improve precision
554  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
555  // direction in local frame is along the X axis
556  if ((csp * ca + snp * sa) < 0) {
557  //printf("Rotation failed: local cos(phi) would become %.2f\n", csp * ca + snp * sa);
558  return false;
559  }
560  //
561 
562  float updSnp = snp * ca - csp * sa;
563  if (fabs(updSnp) > Almost1) {
564  printf("Rotation failed: new snp %.2f\n", updSnp);
565  return false;
566  }
567  float xold = getX(), yold = getY();
568  setAlpha(alpha);
569  setX(xold * ca + yold * sa);
570  setY(-xold * sa + yold * ca);
571  setSnp(updSnp);
572 
573  if (fabs(csp) < Almost0) {
574  printf("Too small cosine value %f\n", csp);
575  csp = Almost0;
576  }
577 
578  float rr = (ca + snp / csp * sa);
579 
580  mC[kSigY2] *= (ca * ca);
581  mC[kSigZY] *= ca;
582  mC[kSigSnpY] *= ca * rr;
583  mC[kSigSnpZ] *= rr;
584  mC[kSigSnp2] *= rr * rr;
585  mC[kSigTglY] *= ca;
586  mC[kSigTglSnp] *= rr;
587  mC[kSigQ2PtY] *= ca;
588  mC[kSigQ2PtSnp] *= rr;
589 
590  checkCovariance();
591  return true;
592 }

595 TrackParCov::TrackParCov(const array<float, 3>& xyz, const array<float, 3>& pxpypz,
596  const array<float, kLabCovMatSize>& cv, int charge, bool sectorAlpha)
597 {
598  // construct track param and covariance from kinematics and lab errors
599 
600  // Alpha of the frame is defined as:
601  // sectorAlpha == false : -> angle of pt direction
602  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
603  // angle of pt direction for r==0
604  //
605  //
606  constexpr float kSafe = 1e-5f;
607  float radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
608  float alp = 0;
609  if (sectorAlpha || radPos2 < 1) {
610  alp = atan2f(pxpypz[1], pxpypz[0]);
611  } else {
612  alp = atan2f(xyz[1], xyz[0]);
613  }
614  if (sectorAlpha) {
615  alp = utils::Angle2Alpha(alp);
616  }
617  //
618  float sn, cs;
619  utils::sincosf(alp, sn, cs);
620  // protection: avoid alpha being too close to 0 or +-pi/2
621  if (fabs(sn) < 2.f * kSafe) {
622  if (alp > 0) {
623  alp += alp < PIHalf ? 2.f * kSafe : -2.f * kSafe;
624  } else {
625  alp += alp > -PIHalf ? -2.f * kSafe : 2.f * kSafe;
626  }
627  utils::sincosf(alp, sn, cs);
628  } else if (fabs(cs) < 2.f * kSafe) {
629  if (alp > 0) {
630  alp += alp > PIHalf ? 2.f * kSafe : -2.f * kSafe;
631  } else {
632  alp += alp > -PIHalf ? 2.f * kSafe : -2.f * kSafe;
633  }
634  utils::sincosf(alp, sn, cs);
635  }
636  // get the vertex of origin and the momentum
637  array<float, 3> ver{ xyz[0], xyz[1], xyz[2] };
638  array<float, 3> mom{ pxpypz[0], pxpypz[1], pxpypz[2] };
639  //
640  // Rotate to the local coordinate system
641  utils::RotateZ(ver, -alp);
642  utils::RotateZ(mom, -alp);
643  //
644  float pt = sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
645  float ptI = 1.f / pt;
646  setX(ver[0]);
647  setAlpha(alp);
648  setY(ver[1]);
649  setZ(ver[2]);
650  setSnp(mom[1] * ptI); // cos(phi)
651  setTgl(mom[2] * ptI); // tg(lambda)
652  setQ2Pt(ptI * charge);
653  //
654  if (fabs(1.f - getSnp()) < kSafe) {
655  setSnp(1.f - kSafe); // Protection
656  } else if (fabs(-1.f - getSnp()) < kSafe) {
657  setSnp(-1.f + kSafe); // Protection
658  }
659  //
660  // Covariance matrix (formulas to be simplified)
661  float r = mom[0] * ptI; // cos(phi)
662  float cv34 = sqrtf(cv[3] * cv[3] + cv[4] * cv[4]);
663  //
664  int special = 0;
665  float sgcheck = r * sn + getSnp() * cs;
666  if (fabs(sgcheck) > 1 - kSafe) { // special case: lab phi is +-pi/2
667  special = 1;
668  sgcheck = sgcheck < 0 ? -1.f : 1.f;
669  } else if (fabs(sgcheck) < kSafe) {
670  sgcheck = cs < 0 ? -1.0f : 1.0f;
671  special = 2; // special case: lab phi is 0
672  }
673  //
674  mC[kSigY2] = cv[0] + cv[2];
675  mC[kSigZY] = (-cv[3] * sn) < 0 ? -cv34 : cv34;
676  mC[kSigZ2] = cv[5];
677  //
678  float ptI2 = ptI * ptI;
679  float tgl2 = getTgl() * getTgl();
680  if (special == 1) {
681  mC[kSigSnpY] = cv[6] * ptI;
682  mC[kSigSnpZ] = -sgcheck * cv[8] * r * ptI;
683  mC[kSigSnp2] = fabs(cv[9] * r * r * ptI2);
684  mC[kSigTglY] = (cv[10] * getTgl() - sgcheck * cv[15]) * ptI / r;
685  mC[kSigTglZ] = (cv[17] - sgcheck * cv[12] * getTgl()) * ptI;
686  mC[kSigTglSnp] = (-sgcheck * cv[18] + cv[13] * getTgl()) * r * ptI2;
687  mC[kSigTgl2] = fabs(cv[20] - 2 * sgcheck * cv[19] * mC[4] + cv[14] * tgl2) * ptI2;
688  mC[kSigQ2PtY] = cv[10] * ptI2 / r * charge;
689  mC[kSigQ2PtZ] = -sgcheck * cv[12] * ptI2 * charge;
690  mC[kSigQ2PtSnp] = cv[13] * r * ptI * ptI2 * charge;
691  mC[kSigQ2PtTgl] = (-sgcheck * cv[19] + cv[14] * getTgl()) * r * ptI2 * ptI;
692  mC[kSigQ2Pt2] = fabs(cv[14] * ptI2 * ptI2);
693  } else if (special == 2) {
694  mC[kSigSnpY] = -cv[10] * ptI * cs / sn;
695  mC[kSigSnpZ] = cv[12] * cs * ptI;
696  mC[kSigSnp2] = fabs(cv[14] * cs * cs * ptI2);
697  mC[kSigTglY] = (sgcheck * cv[6] * getTgl() - cv[15]) * ptI / sn;
698  mC[kSigTglZ] = (cv[17] - sgcheck * cv[8] * getTgl()) * ptI;
699  mC[kSigTglSnp] = (cv[19] - sgcheck * cv[13] * getTgl()) * cs * ptI2;
700  mC[kSigTgl2] = fabs(cv[20] - 2 * sgcheck * cv[18] * getTgl() + cv[9] * tgl2) * ptI2;
701  mC[kSigQ2PtY] = sgcheck * cv[6] * ptI2 / sn * charge;
702  mC[kSigQ2PtZ] = -sgcheck * cv[8] * ptI2 * charge;
703  mC[kSigQ2PtSnp] = -sgcheck * cv[13] * cs * ptI * ptI2 * charge;
704  mC[kSigQ2PtTgl] = (-sgcheck * cv[18] + cv[9] * getTgl()) * ptI2 * ptI * charge;
705  mC[kSigQ2Pt2] = fabs(cv[9] * ptI2 * ptI2);
706  } else {
707  double m00 = -sn; // m10=cs;
708  double m23 = -pt * (sn + getSnp() * cs / r), m43 = -pt * pt * (r * cs - getSnp() * sn);
709  double m24 = pt * (cs - getSnp() * sn / r), m44 = -pt * pt * (r * sn + getSnp() * cs);
710  double m35 = pt, m45 = -pt * pt * getTgl();
711  //
712  m43 *= charge;
713  m44 *= charge;
714  m45 *= charge;
715  //
716  double a1 = cv[13] - cv[9] * (m23 * m44 + m43 * m24) / m23 / m43;
717  double a2 = m23 * m24 - m23 * (m23 * m44 + m43 * m24) / m43;
718  double a3 = m43 * m44 - m43 * (m23 * m44 + m43 * m24) / m23;
719  double a4 = cv[14] + 2. * cv[9];
720  double a5 = m24 * m24 - 2. * m24 * m44 * m23 / m43;
721  double a6 = m44 * m44 - 2. * m24 * m44 * m43 / m23;
722  //
723  mC[kSigSnpY] = (cv[10] * m43 - cv[6] * m44) / (m24 * m43 - m23 * m44) / m00;
724  mC[kSigQ2PtY] = (cv[6] / m00 - mC[kSigSnpY] * m23) / m43;
725  mC[kSigTglY] = (cv[15] / m00 - mC[kSigQ2PtY] * m45) / m35;
726  mC[kSigSnpZ] = (cv[12] * m43 - cv[8] * m44) / (m24 * m43 - m23 * m44);
727  mC[kSigQ2PtZ] = (cv[8] - mC[kSigSnpZ] * m23) / m43;
728  mC[kSigTglZ] = cv[17] / m35 - mC[kSigQ2PtZ] * m45 / m35;
729  mC[kSigSnp2] = fabs((a4 * a3 - a6 * a1) / (a5 * a3 - a6 * a2));
730  mC[kSigQ2Pt2] = fabs((a1 - a2 * mC[kSigSnp2]) / a3);
731  mC[kSigQ2PtSnp] = (cv[9] - mC[kSigSnp2] * m23 * m23 - mC[kSigQ2Pt2] * m43 * m43) / m23 / m43;
732  double b1 = cv[18] - mC[kSigQ2PtSnp] * m23 * m45 - mC[kSigQ2Pt2] * m43 * m45;
733  double b2 = m23 * m35;
734  double b3 = m43 * m35;
735  double b4 = cv[19] - mC[kSigQ2PtSnp] * m24 * m45 - mC[kSigQ2Pt2] * m44 * m45;
736  double b5 = m24 * m35;
737  double b6 = m44 * m35;
738  mC[kSigTglSnp] = (b4 - b6 * b1 / b3) / (b5 - b6 * b2 / b3);
739  mC[kSigQ2PtTgl] = b1 / b3 - b2 * mC[kSigTglSnp] / b3;
740  mC[kSigTgl2] = fabs((cv[20] - mC[kSigQ2Pt2] * (m45 * m45) - mC[kSigQ2PtTgl] * 2. * m35 * m45) / (m35 * m35));
741  }
742  checkCovariance();
743 }

745 //____________________________________________________________
746 bool TrackParCov::propagateTo(float xk, const array<float, 3>& b)
747 {
748  //----------------------------------------------------------------
749  // Extrapolate this track to the plane X=xk in the field b[].
750  //
751  // X [cm] is in the "tracking coordinate system" of this track.
752  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
753  //----------------------------------------------------------------
754 
755  float dx = xk - getX();
756  if (fabs(dx) < Almost0) {
757  return true;
758  }
759  // Do not propagate tracks outside the ALICE detector
760  if (fabs(dx) > 1e5 || fabs(getY()) > 1e5 || fabs(getZ()) > 1e5) {
761  printf("Anomalous track, target X:%f\n", xk);
762  // print();
763  return false;
764  }
765  float crv = (fabs(b[2]) < Almost0) ? 0.f : getCurvature(b[2]);
766  float x2r = crv * dx;
767  float f1 = getSnp(), f2 = f1 + x2r;
768  if ((fabs(f1) > Almost1) || (fabs(f2) > Almost1) || (fabs(getQ2Pt()) < Almost0)) {
769  return false;
770  }
771  float r1 = sqrtf((1.f - f1) * (1.f + f1));
772  if (fabs(r1) < Almost0) {
773  return false;
774  }
775  float r2 = sqrtf((1.f - f2) * (1.f + f2));
776  if (fabs(r2) < Almost0) {
777  return false;
778  }
779 
780  float dy2dx = (f1 + f2) / (r1 + r2);
781  float step = (fabs(x2r) < 0.05f) ? dx * fabs(r2 + f2 * dy2dx) // chord
782  : 2.f * asinf(0.5f * dx * sqrtf(1.f + dy2dx * dy2dx) * crv) / crv; // arc
783  step *= sqrtf(1.f + getTgl() * getTgl());
784  //
785  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
786  array<float, 9> vecLab{ 0.f };
787  if (!getPosDirGlo(vecLab)) {
788  return false;
789  }
790  //
791  // matrix transformed with Bz component only
792  float &c00 = mC[kSigY2], &c10 = mC[kSigZY], &c11 = mC[kSigZ2], &c20 = mC[kSigSnpY], &c21 = mC[kSigSnpZ],
793  &c22 = mC[kSigSnp2], &c30 = mC[kSigTglY], &c31 = mC[kSigTglZ], &c32 = mC[kSigTglSnp], &c33 = mC[kSigTgl2],
794  &c40 = mC[kSigQ2PtY], &c41 = mC[kSigQ2PtZ], &c42 = mC[kSigQ2PtSnp], &c43 = mC[kSigQ2PtTgl],
795  &c44 = mC[kSigQ2Pt2];
796  // evaluate matrix in double prec.
797  double rinv = 1. / r1;
798  double r3inv = rinv * rinv * rinv;
799  double f24 = dx * b[2] * B2C; // x2r/track[kQ2Pt];
800  double f02 = dx * r3inv;
801  double f04 = 0.5 * f24 * f02;
802  double f12 = f02 * getTgl() * f1;
803  double f14 = 0.5 * f24 * f12; // 0.5*f24*f02*getTgl()*f1;
804  double f13 = dx * rinv;
805 
806  // b = C*ft
807  double b00 = f02 * c20 + f04 * c40, b01 = f12 * c20 + f14 * c40 + f13 * c30;
808  double b02 = f24 * c40;
809  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
810  double b12 = f24 * c41;
811  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
812  double b22 = f24 * c42;
813  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
814  double b42 = f24 * c44;
815  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
816  double b32 = f24 * c43;
817 
818  // a = f*b = f*C*ft
819  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
820  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
821  double a22 = f24 * b42;
822 
823  // F*C*Ft = C + (b + bt + a)
824  c00 += b00 + b00 + a00;
825  c10 += b10 + b01 + a01;
826  c20 += b20 + b02 + a02;
827  c30 += b30;
828  c40 += b40;
829  c11 += b11 + b11 + a11;
830  c21 += b21 + b12 + a12;
831  c31 += b31;
832  c41 += b41;
833  c22 += b22 + b22 + a22;
834  c32 += b32;
835  c42 += b42;
836 
837  checkCovariance();
838 
839  // Rotate to the system where Bx=By=0.
840  float bxy2 = b[0] * b[0] + b[1] * b[1];
841  float bt = sqrtf(bxy2);
842  float cosphi = 1.f, sinphi = 0.f;
843  if (bt > Almost0) {
844  cosphi = b[0] / bt;
845  sinphi = b[1] / bt;
846  }
847  float bb = sqrtf(bxy2 + b[2] * b[2]);
848  float costet = 1., sintet = 0.;
849  if (bb > Almost0) {
850  costet = b[2] / bb;
851  sintet = bt / bb;
852  }
853  array<float, 7> vect{ costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
854  -sinphi * vecLab[0] + cosphi * vecLab[1],
855  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
856  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
857  -sinphi * vecLab[3] + cosphi * vecLab[4],
858  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
859  vecLab[6] };
860 
861  // Do the helix step
862  float sgn = getSign();
863  g3helx3(sgn * bb, step, vect);
864 
865  // Rotate back to the Global System
866  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
867  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
868  vecLab[2] = -sintet * vect[0] + costet * vect[2];
869 
870  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
871  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
872  vecLab[5] = -sintet * vect[3] + costet * vect[5];
873 
874  // Rotate back to the Tracking System
875  float sinalp = -vecLab[7], cosalp = vecLab[8];
876  float t = cosalp * vecLab[0] - sinalp * vecLab[1];
877  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
878  vecLab[0] = t;
879  t = cosalp * vecLab[3] - sinalp * vecLab[4];
880  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
881  vecLab[3] = t;
882 
883  // Do the final correcting step to the target plane (linear approximation)
884  float x = vecLab[0], y = vecLab[1], z = vecLab[2];
885  if (fabs(dx) > Almost0) {
886  if (fabs(vecLab[3]) < Almost0) {
887  return false;
888  }
889  dx = xk - vecLab[0];
890  x += dx;
891  y += vecLab[4] / vecLab[3] * dx;
892  z += vecLab[5] / vecLab[3] * dx;
893  }
894 
895  // Calculate the track parameters
896  t = 1.f / sqrtf(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
897  setX(x);
898  setY(y);
899  setZ(z);
900  setSnp(vecLab[4] * t);
901  setTgl(vecLab[5] * t);
902  setQ2Pt(sgn * t / vecLab[6]);
903 
904  return true;
905 }

907 //______________________________________________
908 void TrackParCov::checkCovariance()
909 {
910  // This function forces the diagonal elements of the covariance matrix to be positive.
911  // In case the diagonal element is bigger than the maximal allowed value, it is set to
912  // the limit and the off-diagonal elements that correspond to it are set to zero.
913 
914  mC[kSigY2] = fabs(mC[kSigY2]);
915  if (mC[kSigY2] > kCY2max) {
916  float scl = sqrtf(kCY2max / mC[kSigY2]);
917  mC[kSigY2] = kCY2max;
918  mC[kSigZY] *= scl;
919  mC[kSigSnpY] *= scl;
920  mC[kSigTglY] *= scl;
921  mC[kSigQ2PtY] *= scl;
922  }
923  mC[kSigZ2] = fabs(mC[kSigZ2]);
924  if (mC[kSigZ2] > kCZ2max) {
925  float scl = sqrtf(kCZ2max / mC[kSigZ2]);
926  mC[kSigZ2] = kCZ2max;
927  mC[kSigZY] *= scl;
928  mC[kSigSnpZ] *= scl;
929  mC[kSigTglZ] *= scl;
930  mC[kSigQ2PtZ] *= scl;
931  }
932  mC[kSigSnp2] = fabs(mC[kSigSnp2]);
933  if (mC[kSigSnp2] > kCSnp2max) {
934  float scl = sqrtf(kCSnp2max / mC[kSigSnp2]);
935  mC[kSigSnp2] = kCSnp2max;
936  mC[kSigSnpY] *= scl;
937  mC[kSigSnpZ] *= scl;
938  mC[kSigTglSnp] *= scl;
939  mC[kSigQ2PtSnp] *= scl;
940  }

1071 {

1073

1082  return false;
1083  }
1084

1088  matC0(kZ, kY) = getSigmaZY();

1107  for (int i = kNParams; i--;) {

1125  mC[kSigTglZ] -= matK(kTgl, kZ);

1131  mC[kSigQ2PtTgl] -= matK(kQ2Pt, kTgl);

1133

1221 bool TrackParCov::correctForMaterial(float x2x0, float xrho, float mass, bool anglecorr, float dedx)
1222 {
1223  //------------------------------------------------------------------
1224  // This function corrects the track parameters for the crossed material.
1225  // "x2x0" - X/X0, the thickness in units of the radiation length.
1226  // "xrho" - is the product length*density (g/cm^2).
1227  // It should be passed as negative when propagating tracks
1228  // from the intreaction point to the outside of the central barrel.
1229  // "mass" - the mass of this particle (GeV/c^2). Negative mass means charge=2 particle
1230  // "dedx" - mean enery loss (GeV/(g/cm^2), if <=kCalcdEdxAuto : calculate on the fly
1231  // "anglecorr" - switch for the angular correction
1232  //------------------------------------------------------------------
1233  constexpr float kMSConst2 = 0.0136f * 0.0136f;
1234  constexpr float kMaxELossFrac = 0.3f; // max allowed fractional eloss
1235  constexpr float kMinP = 0.01f; // kill below this momentum
1236 
1237  float& fC22 = mC[kSigSnp2];
1238  float& fC33 = mC[kSigTgl2];
1239  float& fC43 = mC[kSigQ2PtTgl];
1240  float& fC44 = mC[kSigQ2Pt2];
1241  //
1242  float csp2 = (1.f - getSnp()) * (1.f + getSnp()); // cos(phi)^2
1243  float cst2I = (1.f + getTgl() * getTgl()); // 1/cos(lambda)^2
1244  // Apply angle correction, if requested
1245  if (anglecorr) {
1246  float angle = sqrtf(cst2I / (csp2));
1247  x2x0 *= angle;
1248  xrho *= angle;
1249  }
1250 
1251  float p = getP();
1252  if (mass < 0) {
1253  p += p; // q=2 particle
1254  }
1255  float p2 = p * p, mass2 = mass * mass;
1256  float e2 = p2 + mass2;
1257  float beta2 = p2 / e2;
1258 
1259  // Calculating the multiple scattering corrections******************
1260  float cC22(0.f), cC33(0.f), cC43(0.f), cC44(0.f);
1261  if (x2x0 != 0.f) {
1262  float theta2 = kMSConst2 / (beta2 * p2) * fabs(x2x0);
1263  if (mass < 0) {
1264  theta2 *= 4.f; // q=2 particle
1265  }
1266  if (theta2 > PI * PI) {
1267  return false;
1268  }
1269  float fp34 = getTgl() * getQ2Pt();
1270  float t2c2I = theta2 * cst2I;
1271  cC22 = t2c2I * csp2;
1272  cC33 = t2c2I * cst2I;
1273  cC43 = t2c2I * fp34;
1274  cC44 = theta2 * fp34 * fp34;
1275  // optimes this
1276  // cC22 = theta2*((1.-getSnp())*(1.+getSnp()))*(1. + getTgl()*getTgl());
1277  // cC33 = theta2*(1. + getTgl()*getTgl())*(1. + getTgl()*getTgl());
1278  // cC43 = theta2*getTgl()*getQ2Pt()*(1. + getTgl()*getTgl());
1279  // cC44 = theta2*getTgl()*getQ2Pt()*getTgl()*getQ2Pt();
1280  }
1281 
1282  // Calculating the energy loss corrections************************
1283  float cP4 = 1.f;
1284  if ((xrho != 0.f) && (beta2 < 1.f)) {
1285  if (dedx < kCalcdEdxAuto + Almost1) { // request to calculate dedx on the fly
1286  dedx = BetheBlochSolid(p / fabs(mass));
1287  if (mass < 0) {
1288  dedx *= 4.f; // z=2 particle
1289  }
1290  }
1291 
1292  float dE = dedx * xrho;
1293  float e = sqrtf(e2);
1294  if (fabs(dE) > kMaxELossFrac * e) {
1295  return false; // 30% energy loss is too much!
1296  }
1297  float eupd = e + dE;
1298  float pupd2 = eupd * eupd - mass2;
1299  if (pupd2 < kMinP * kMinP) {
1300  return false;
1301  }
1302  cP4 = p / sqrtf(pupd2);
1303  //
1304  // Approximate energy loss fluctuation (M.Ivanov)
1305  constexpr float knst = 0.07f; // To be tuned.
1306  float sigmadE = knst * sqrtf(fabs(dE)) * e / p2 * getQ2Pt();
1307  cC44 += sigmadE * sigmadE;
1308  }
1309 
1310  // Applying the corrections*****************************
1311  fC22 += cC22;
1312  fC33 += cC33;
1313  fC43 += cC43;
1314  fC44 += cC44;
1315  setQ2Pt(getQ2Pt() * cP4);
1316 
1317  checkCovariance();
1318 
1319  return true;
1320 }

1344 void o2::track::g3helx3(float qfield, float step, array<float, 7>& vect)

1347  * *
1348  * GEANT3 tracking routine in a constant field oriented *
1349  * along axis 3 *

1351  * helix step method *
1352  * *
1353  * Authors R.Brun, M.Hansroul ********* *
1354  * Rewritten V.Perevoztchikov *
1355  * *
1356  * Rewritten in C++ by I.Belikov *
1357  * *
1358  * qfield (kG) - particle charge times magnetic field *
1359  * step (cm) - step length along the helix *
1360  * vect[7](cm,GeV/c) - input/output x, y, z, px/p, py/p ,pz/p, p *
1361  * *
1362  ******************************************************************/

1370

1372  if (fabs(tet) > 0.03f) {
1373  sint = sinf(tet);
1374  sintt = sint / tet;
1375  tsint = (tet - sint) / tet;
1376  float t = sinf(0.5f * tet);
1377  cos1t = 2 * t * t / tet;
1378  } else {
1379  tsint = tet * tet / 6.f;
1380  sintt = (1.f - tet * kOvSqSix) * (1.f + tet * kOvSqSix); // 1.- tsint;
1381  sint = tet * sintt;

1383  }

1386  float f2 = step * cos1t;

1388  float f4 = -tet * cos1t;
1389  float f5 = sint;
1390

1392  vect[iy] += f1 * cosy + f2 * cosx;

1401 {
1402  //
1403  // This is the parameterization of the Bethe-Bloch formula inspired by Geant.
1404  //
1405  // bg - beta*gamma
1406  // rho - density [g/cm^3]
1407  // kp1 - density effect first junction point
1408  // kp2 - density effect second junction point
1409  // meanI - mean excitation energy [GeV]
1410  // meanZA - mean Z/A
1411  //
1412  // The default values for the kp* parameters are for silicon.
1413  // The returned value is in [GeV/(g/cm^2)].
1414  //
1415  constexpr float mK = 0.307075e-3f; // [GeV*cm^2/g]
1416  constexpr float me = 0.511e-3f; // [GeV/c^2]
1417  kp1 *= 2.303f;
1418  kp2 *= 2.303f;
1419  float bg2 = bg * bg;
1420  float maxT = 2.f * me * bg2; // neglecting the electron mass
1421 
1422  //*** Density effect
1423  float d2 = 0.;
1424  const float x = log(bg);
1425  const float lhwI = log(28.816 * 1e-9 * sqrtf(rho * meanZA) / meanI);
1426  if (x > kp2) {
1427  d2 = lhwI + x - 0.5;
1428  } else if (x > kp1) {
1429  double r = (kp2 - x) / (kp2 - kp1);
1430  d2 = lhwI + x - 0.5 + (0.5 - lhwI - kp1) * r * r * r;
1431  }
1432  return mK * meanZA * (1 + bg2) / bg2 * (0.5 * log(2 * me * bg2 * maxT / (meanI * meanI)) - bg2 / (1 + bg2) - d2);
1433 }
