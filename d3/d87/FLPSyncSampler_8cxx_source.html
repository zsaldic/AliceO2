67 bool FLPSyncSampler::ConditionalRun()
68 {
69  FairMQMessagePtr msg(NewSimpleMessage(mTimeFrameId));
70 
71  if (fChannels.at(mOutChannelName).at(0).Send(msg) >= 0) {
72  mTimeframeRTT[mTimeFrameId].start = steady_clock::now();
73 
74  if (++mTimeFrameId == UINT16_MAX - 1) {
75  mTimeFrameId = 0;
76  }
77  }
78 
79  // rate limiting
80  --mEventCounter;
81  while (mEventCounter == 0) {
82  this_thread::sleep_for(milliseconds(1));
83  }
84 
85  if (mMaxEvents > 0 && mTimeFrameId >= mMaxEvents) {
86  LOG(INFO) << "Reached configured maximum number of events (" << mMaxEvents << "). Exiting Run().";
87  return false;
88  }
89 
90  return true;
91 }

113  string name = s.str();

133  }
