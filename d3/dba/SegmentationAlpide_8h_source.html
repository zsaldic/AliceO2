101 inline void SegmentationAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
102 {
103  // convert to row/col w/o over/underflow check
104  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt top edge of Active matrix
105  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt left edge of Active matrix
106  iRow = int(xRow / PitchRow);
107  iCol = int(zCol / PitchCol);
108  if (xRow<0) iRow -= 1;
109  if (zCol<0) iCol -= 1;
110 }

113 inline bool SegmentationAlpide::localToDetector(float xRow, float zCol, int& iRow, int& iCol)
114 {
115  // convert to row/col
116  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt left edge of Active matrix
117  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
118  if (xRow<0 || xRow>=ActiveMatrixSizeRows || zCol<0 || zCol>=ActiveMatrixSizeCols) {
119  iRow = iCol = -1;
120  return false;
121  }
122  iRow = int(xRow / PitchRow);
123  iCol = int(zCol / PitchCol);
124  return true;
125 }
