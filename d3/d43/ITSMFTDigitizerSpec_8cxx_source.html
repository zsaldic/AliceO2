54  mSimChains.emplace_back(new TChain("o2sim"));

93  void run(framework::ProcessingContext& pc)
94  {
95  if (mFinished) {
96  return;
97  }
98  std::string detStr = mID.getName();
99  // read collision context from input
100  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
101  auto& timesview = context->getEventRecords();
102  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES" << FairLogger::endl;
103  // if there is nothing to do ... return
104  if (timesview.size() == 0) {
105  return;
106  }
107  TStopwatch timer;
108  timer.Start();
109  LOG(INFO) << " CALLING ITS DIGITIZATION " << FairLogger::endl;
110 
111  mDigitizer.setDigits(&mDigits);
112  mDigitizer.setROFRecords(&mROFRecords);
113  mDigitizer.setMCLabels(&mLabels);
114 
115  // attach optional QED digits branch
116  setupQEDChain();
117 
118  auto& eventParts = context->getEventParts();
119  // loop over all composite collisions given from context (aka loop over all the interaction records)
120  for (int collID = 0; collID < timesview.size(); ++collID) {
121  auto eventTime = timesview[collID].timeNS;
122 
123  if (mQEDChain.GetEntries()) { // QED must be processed before other inputs since done in small time steps
124  processQED(eventTime);
125  }
126 
127  mDigitizer.setEventTime(eventTime);
128  mDigitizer.resetEventROFrames(); // to estimate min/max ROF for this collID
129  // for each collision, loop over the constituents event and source IDs
130  // (background signal merging is basically taking place here)
131  for (auto& part : eventParts[collID]) {
132 
133  // get the hits for this event and this source
134  mHits.clear();
135  retrieveHits(part.sourceID, part.entryID);
136 
137  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID
138  << " found " << mHits.size() << " hits " << FairLogger::endl;
139 
140  mDigitizer.process(&mHits, part.entryID, part.sourceID); // call actual digitization procedure
141  }
142  mMC2ROFRecordsAccum.emplace_back(collID, -1, mDigitizer.getEventROFrameMin(), mDigitizer.getEventROFrameMax());
143  accumulate();
144  }
145  // finish digitization ... stream any remaining digits/labels
146  if (mQEDChain.GetEntries()) { // fill last slots from QED input
147  processQED(mDigitizer.getEndTimeOfROFMax());
148  }
149 
150  mDigitizer.fillOutputContainer();
151  accumulate();
152 
153  // here we have all digits and labels and we can send them to consumer (aka snapshot it onto output)
154  pc.outputs().snapshot(Output{ mOrigin, "DIGITS", 0, Lifetime::Timeframe }, mDigitsAccum);
155  pc.outputs().snapshot(Output{ mOrigin, "DIGITSROF", 0, Lifetime::Timeframe }, mROFRecordsAccum);
156  pc.outputs().snapshot(Output{ mOrigin, "DIGITSMC2ROF", 0, Lifetime::Timeframe }, mMC2ROFRecordsAccum);
157  pc.outputs().snapshot(Output{ mOrigin, "DIGITSMCTR", 0, Lifetime::Timeframe }, mLabelsAccum);
158  LOG(INFO) << mID.getName() << ": Sending ROMode= " << mROMode << " to GRPUpdater";
159  pc.outputs().snapshot(Output{ mOrigin, "ROMode", 0, Lifetime::Timeframe }, mROMode);
160 
161  timer.Stop();
162  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s" << FairLogger::endl;
163 
164  // we should be only called once; tell DPL that this process is ready to exit
165  pc.services().get<ControlService>().readyToQuit(false);
166 
167  mFinished = true;
168  }

173  void processQED(double tMax)
174  {
175  auto tQEDNext = mLastQEDTimeNS + mQEDEntryTimeBinNS; // timeslice to retrieve
176  std::string detStr = mID.getName();
177  auto br = mQEDChain.GetBranch((detStr + "Hit").c_str());
178  while (tQEDNext < tMax) {
179  mLastQEDTimeNS = tQEDNext; // time used for current QED slot
180  tQEDNext += mQEDEntryTimeBinNS; // prepare time for next QED slot
181  if (++mLastQEDEntry >= mQEDChain.GetEntries()) {
182  mLastQEDEntry = 0; // wrapp if needed
183  }
184  br->GetEntry(mLastQEDEntry);
185  mDigitizer.setEventTime(mLastQEDTimeNS);
186  mDigitizer.process(&mHits, mLastQEDEntry, QEDSourceID);
187  //
188  }
189  }

191  void setupQEDChain()
192  {
193  if (!mQEDChain.GetEntries()) {
194  return;
195  }
196  std::string detStr = mID.getName();
197  auto qedBranch = mQEDChain.GetBranch((detStr + "Hit").c_str());
198  assert(qedBranch != nullptr);
199  assert(mQEDEntryTimeBinNS >= 1.0);
200  assert(QEDSourceID < o2::MCCompLabel::maxSourceID());
201  mLastQEDTimeNS = -mQEDEntryTimeBinNS / 2; // time will be assigned to the middle of the bin
202  qedBranch->SetAddress(&mHitsP);
203  LOG(INFO) << "Attaching QED ITS hits as sourceID=" << int(QEDSourceID) << ", entry integrates "
204  << mQEDEntryTimeBinNS << " ns" << FairLogger::endl;
205  }

208  void retrieveHits(int sourceID, int entryID)
209  {
210  std::string detStr = mID.getName();
211  auto br = mSimChains[sourceID]->GetBranch((detStr + "Hit").c_str());
212  if (!br) {

216  br->SetAddress(&mHitsP);
217  br->GetEntry(entryID);

220  void accumulate()
221  {
222  // accumulate result of single event processing, called after processing every event supplied
223  // AND after the final flushing via digitizer::fillOutputContainer
224  if (!mDigits.size()) {
225  return; // no digits were flushed, nothing to accumulate
226  }
227  static int fixMC2ROF = 0; // 1st entry in mc2rofRecordsAccum to be fixed for ROFRecordID
228  auto ndigAcc = mDigitsAccum.size();
229  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(mDigitsAccum));
230 
231  // fix ROFrecords references on ROF entries
232  auto nROFRecsOld = mROFRecordsAccum.size();
233 
234  for (int i = 0; i < mROFRecords.size(); i++) {
235  auto& rof = mROFRecords[i];
236  rof.getROFEntry().shiftIndex(ndigAcc);
237  rof.print();
238 
239  if (mFixMC2ROF < mMC2ROFRecordsAccum.size()) { // fix ROFRecord entry in MC2ROF records
240  for (int m2rid = mFixMC2ROF; m2rid < mMC2ROFRecordsAccum.size(); m2rid++) {
241  // need to register the ROFRecors entry for MC event starting from this entry
242  auto& mc2rof = mMC2ROFRecordsAccum[m2rid];
243  if (rof.getROFrame() == mc2rof.minROF) {
244  mFixMC2ROF++;
245  mc2rof.rofRecordID = nROFRecsOld + i;
246  mc2rof.print();
247  }
248  }
249  }
250  }
251  std::copy(mROFRecords.begin(), mROFRecords.end(), std::back_inserter(mROFRecordsAccum));
252  mLabelsAccum.mergeAtBack(mLabels);
253  LOG(INFO) << "Added " << mDigits.size() << " digits " << FairLogger::endl;
254  // clean containers from already accumulated stuff
255  mLabels.clear();
256  mDigits.clear();
257  mROFRecords.clear();
258  }
