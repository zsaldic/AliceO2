94 inline void IndexTableParameters::ComputeInverseBinSizes()
95 {
96  InversePhiBinSize = PhiBins / Constants::Math::TwoPi;
97  for (int iL = 0; iL < Constants::ITS::LayersNumber; ++iL) {
