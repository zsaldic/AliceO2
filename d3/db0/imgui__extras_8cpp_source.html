47 static float binner(BinFunction binFunction,
48  const void* const* datas,
49  size_t data_idx,
50  size_t values_count,
51  float binStart, float binEnd,
52  size_t domain_min, size_t domain_max, float scale,
53  float (*getterY)(const void* data, int idx),
54  size_t (*getterX)(const void* data, int idx),
55  size_t& cacheXIdx,
56  size_t& count)
57 {
58  float value;
59  switch (binFunction) {
60  case BinFunction::CEIL:
61  case BinFunction::MAX:
62  value = -FLT_MAX;
63  break;
64  case BinFunction::FLOOR:
65  case BinFunction::MIN:
66  value = FLT_MAX;
67  break;
68  default:
69  value = 0;
70  }
71  // Accumulate the matching datapoints.
72  while (cacheXIdx < values_count) {
73  size_t bin = getterX(datas[data_idx], cacheXIdx);
74  if ((bin < domain_min)) {
75  cacheXIdx++;
76  continue;
77  }
78  if ((bin > domain_max)) {
79  return value;
80  }
81  // We are withing (domain_min, domain_max), what we have have below
82  // works.
83  float binPos = ImSaturate(float(bin - domain_min) * scale);
84  if ((binPos < binStart)) {
85  cacheXIdx++;
86  continue;
87  }
88  if ((binPos > binEnd)) {
89  return value;
90  }
91  count++;
92  float binValue = getterY(datas[data_idx], cacheXIdx);
93  cacheXIdx++;
94  switch (binFunction) {
95  case BinFunction::CEIL:
96  value = std::max(value, ceilf(binValue));
97  break;
98  case BinFunction::MAX:
99  value = std::max(value, binValue);
100  break;
101  case BinFunction::FLOOR:
102  value = std::min(value, floorf(binValue));
103  break;
104  case BinFunction::MIN:
105  value = std::min(value, binValue);
106  break;
107  case BinFunction::COUNT:
108  value = count;
109  break;
110  case BinFunction::AVG:
111  // NOTE: this will not actually work with too many values because of floating point
112  value = (value * (count - 1) / count) + (binValue / count);
113  break;
114  }
115  }
116  return value;
117 }
