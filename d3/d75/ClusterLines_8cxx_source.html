35 bool Line::areParallel(const Line& firstLine, const Line& secondLine, const float precision)
36 {
37  float crossProdX{ firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2] -
38  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1] };
39  float module{ std::abs(firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2]) +
40  std::abs(firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1]) };
41  if (std::abs(crossProdX) > precision * module)
42  return false;
43 
44  float crossProdY{ -firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2] +
45  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0] };
46  module = std::abs(firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2]) +
47  std::abs(firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0]);
48  if (std::abs(crossProdY) > precision * module)
49  return false;
50 
51  float crossProdZ = firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1] -
52  firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0];
53  module = std::abs(firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1]) +
54  std::abs(firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0]);
55  if (std::abs(crossProdZ) > precision * module)
56  return false;
57 
58  return true;

61 float Line::getDCA(const Line& firstLine, const Line& secondLine, const float precision)
62 {
63  std::array<float, 3> normalVector;
64  normalVector[0] = firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2] -
65  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1];
66  normalVector[1] = -firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2] +
67  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0];
68  normalVector[2] = firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1] -
69  firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0];
70 
71  float norm{ 0.f }, distance{ 0.f };
72  for (int i{ 0 }; i < 3; ++i) {
73  norm += normalVector[i] * normalVector[i];
74  distance += (secondLine.originPoint[i] - firstLine.originPoint[i]) * normalVector[i];
75  }
76  if (norm > precision) {
77  return std::abs(distance / std::sqrt(norm));
78  } else {
79  return getDistanceFromPoint(firstLine, secondLine.originPoint);
80  }
81 }

83 float Line::getDistanceFromPoint(const Line& line, const std::array<float, 3> point)
84 {
85  float DCASquared{ 0 };
86  float cdelta{ 0 };
87  for (int i{ 0 }; i < 3; ++i)
88  cdelta -= line.cosinesDirector[i] * (line.originPoint[i] - point[i]);
89  for (int i{ 0 }; i < 3; ++i) {
90  DCASquared += (line.originPoint[i] - point[i] + line.cosinesDirector[i] * cdelta) *
91  (line.originPoint[i] - point[i] + line.cosinesDirector[i] * cdelta);
92  }
93  return std::sqrt(DCASquared);
94 }

97 {
98  std::array<float, 6> components{ 0., 0., 0., 0., 0., 0. };
99  float cdelta{ 0. };
100  for (int i{ 0 }; i < 3; ++i)
101  cdelta -= line.cosinesDirector[i] * (line.originPoint[i] - point[i]);
102 
103  components[0] = line.originPoint[0] - point[0] + line.cosinesDirector[0] * cdelta;
104  components[3] = line.originPoint[1] - point[1] + line.cosinesDirector[1] * cdelta;
105  components[5] = line.originPoint[2] - point[2] + line.cosinesDirector[2] * cdelta;
106  components[1] = std::sqrt(components[0] * components[0] + components[3] * components[3]);
107  components[2] = std::sqrt(components[0] * components[0] + components[5] * components[5]);
108  components[4] = std::sqrt(components[3] * components[3] + components[5] * components[5]);
