51  void setMatrix(const T& mat, int sensID)
52  {
53  // assign matrix for given sensor. The cache must be booked in advance

55  LOG(FATAL) << "SensID " << sensID << " exceeds cache size of " << mCache.size();

167  LOG(FATAL) << "SendID " << sensID << " exceeds indirect cache size of " << mIndirectSize;
168  }
169  if (mIndirection[sensID] >= 0) {
