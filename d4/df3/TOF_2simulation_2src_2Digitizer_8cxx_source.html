81 Int_t Digitizer::processHit(const HitType &hit,Double_t event_time)
82 {
83  Float_t pos[3] = { hit.GetX(), hit.GetY(), hit.GetZ() };
84  Float_t deltapos[3];
85  Int_t detInd[5];
86  Int_t detIndOtherPad[5];
87 
88  Geo::getPadDxDyDz(pos, detInd, deltapos); // Get DetId and residuals
89 
90  detIndOtherPad[0] = detInd[0], detIndOtherPad[1] = detInd[1],
91  detIndOtherPad[2] = detInd[2]; // same sector, plate, strip
92 
93  Int_t otherraw = 0;
94  if (detInd[3] == 0)
95  otherraw = 1;
96 
97  Int_t iZshift = otherraw ? 1 : -1;
98 
99  Int_t channel = Geo::getIndex(detInd);
100 
101  Float_t charge = getCharge(hit.GetEnergyLoss());
102  // NOTE: FROM NOW ON THE TIME IS IN PS ... AND NOT IN NS
103  Float_t time = getShowerTimeSmeared((event_time + hit.GetTime()) * 1E3, charge);
104 
105  Float_t xLocal = deltapos[0];
106  Float_t zLocal = deltapos[2];
107 
108  // extract trackID
109  auto trackID = hit.GetTrackID();
110 
111  // PadId - Pad Identifier
112  // E | F --> PadId = 5 | 6
113  // A | B --> PadId = 1 | 2
114  // C | D --> PadId = 3 | 4
115 
116  Int_t ndigits = 0;//Number of digits added
117 
118  UInt_t istrip = channel / Geo::NPADS;
119 
120  // check the fired PAD 1 (A)
121  if (isFired(xLocal, zLocal, charge)){
122  ndigits++;
123  addDigit(channel, istrip, time, xLocal, zLocal, charge, 0, 0, detInd[3], trackID);
124  }
125 
126  // check PAD 2
127  detIndOtherPad[3] = otherraw;
128  detIndOtherPad[4] = detInd[4];
129  channel = Geo::getIndex(detIndOtherPad);
130  xLocal = deltapos[0]; // recompute local coordinates
131  if (otherraw)
132  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
133  else
134  zLocal = deltapos[2] + Geo::ZPAD;
135  if (isFired(xLocal, zLocal, charge)) {
136  ndigits++;
137  addDigit(channel, istrip, time, xLocal, zLocal, charge, 0, iZshift, detInd[3], trackID);
138  }
139 
140  // check PAD 3
141  detIndOtherPad[3] = detInd[3];
142  detIndOtherPad[4] = detInd[4] - 1;
143  if (detIndOtherPad[4] >= 0) {
144  channel = Geo::getIndex(detIndOtherPad);
145  xLocal = deltapos[0] + Geo::XPAD; // recompute local coordinates
146  zLocal = deltapos[2]; // recompute local coordinates
147  if (isFired(xLocal, zLocal, charge)) {
148  ndigits++;
149  addDigit(channel, istrip, time, xLocal, zLocal, charge, -1, 0, detInd[3], trackID);
150  }
151  }
152 
153  // check PAD 5
154  detIndOtherPad[3] = detInd[3];
155  detIndOtherPad[4] = detInd[4] + 1;
156  if (detIndOtherPad[4] < Geo::NPADX) {
157  channel = Geo::getIndex(detIndOtherPad);
158  xLocal = deltapos[0] - Geo::XPAD; // recompute local coordinates
159  zLocal = deltapos[2]; // recompute local coordinates
160  if (isFired(xLocal, zLocal, charge)) {
161  ndigits++;
162  addDigit(channel, istrip, time, xLocal, zLocal, charge, 1, 0, detInd[3], trackID);
163  }
164  }
165 
166  // check PAD 4
167  detIndOtherPad[3] = otherraw;
168  detIndOtherPad[4] = detInd[4] - 1;
169  if (detIndOtherPad[4] >= 0) {
170  channel = Geo::getIndex(detIndOtherPad);
171  xLocal = deltapos[0] + Geo::XPAD; // recompute local coordinates
172  if (otherraw)
173  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
174  else
175  zLocal = deltapos[2] + Geo::ZPAD;
176  if (isFired(xLocal, zLocal, charge)) {
177  ndigits++;
178  addDigit(channel, istrip, time, xLocal, zLocal, charge, -1, iZshift, detInd[3], trackID);
179  }
180  }
181 
182  // check PAD 6
183  detIndOtherPad[3] = otherraw;
184  detIndOtherPad[4] = detInd[4] + 1;
185  if (detIndOtherPad[4] < Geo::NPADX) {
186  channel = Geo::getIndex(detIndOtherPad);
187  xLocal = deltapos[0] - Geo::XPAD; // recompute local coordinates
188  if (otherraw)
189  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
190  else
191  zLocal = deltapos[2] + Geo::ZPAD;
192  if (isFired(xLocal, zLocal, charge)) {
193  ndigits++;
194  addDigit(channel, istrip, time, xLocal, zLocal, charge, 1, iZshift, detInd[3], trackID);
195  }
196  }
197  return ndigits;
198 
199 }

202 void Digitizer::addDigit(Int_t channel, UInt_t istrip, Float_t time, Float_t x, Float_t z, Float_t charge, Int_t iX, Int_t iZ,
203  Int_t padZfired, Int_t trackID)
204 {
205  // TOF digit requires: channel, time and time-over-threshold
206 
207  time = getDigitTimeSmeared(time, x, z, charge); // add time smearing
208 
209  charge *= getFractionOfCharge(x, z);
210  Float_t tot = 12; // time-over-threshold
211 
212  Float_t xborder = Geo::XPAD * 0.5 - TMath::Abs(x);
213  Float_t zborder = Geo::ZPAD * 0.5 - TMath::Abs(z);
214  Float_t border = TMath::Min(xborder, zborder);
215 
216  Float_t timewalkX = x * mTimeWalkeSlope;
217  Float_t timewalkZ = (z - (padZfired - 0.5) * Geo::ZPAD) * mTimeWalkeSlope;
218 
219  if (border < 0) { // keep the effect onlu if hit out of pad
220  border *= -1;
221  Float_t extraTimeSmear = border * mTimeSlope;
222  time += gRandom->Gaus(mTimeDelay, extraTimeSmear);
223  } else {
224  border = 1 - border;
225  // if(border > 0) printf("deltat =%f\n",mTimeDelay*border*border*border);
226  // else printf("deltat=0\n");
227  // getchar();
228  if (border > 0)
229  time += mTimeDelay * border * border * border;
230  }
231  time += TMath::Sqrt(timewalkX * timewalkX + timewalkZ * timewalkZ) - mTimeDelayCorr - mTimeWalkeSlope * 2;
232 
233  Int_t nbc = Int_t(time * Geo::BC_TIME_INPS_INV); // time elapsed in number of bunch crossing
234  //Digit newdigit(time, channel, (time - Geo::BC_TIME_INPS * nbc) * Geo::NTDCBIN_PER_PS, tot * Geo::NTOTBIN_PER_NS, nbc);
235 
236  auto tdc = (time - Geo::BC_TIME_INPS * nbc) * Geo::NTDCBIN_PER_PS;
237 
238  int lblCurrent = 0;
239 
240  bool iscurrent = true; // if we are in the current readout window
241  Int_t isnext = -1;
242 
243  if (mContinuous) {
244  isnext = Int_t(time * 1E-3 * Geo::READOUTWINDOW_INV) - mReadoutWindowCurrent; // to be replaced with uncalibrated time
245 
246  if (isnext < 0 || isnext >= MAXWINDOWS - 1) {
247 
248  lblCurrent = mFutureIevent.size(); // this is the size of mHeaderArray;
249  mFutureIevent.push_back(mEventID);
250  mFutureIsource.push_back(mSrcID);
251  mFutureItrackID.push_back(trackID);
252 
253  // fill temporary digits array
254  mFutureDigits.emplace_back(channel, tdc, tot * Geo::NTOTBIN_PER_NS, nbc, lblCurrent);
255 
256  return; // don't fill if doesn't match any available readout window
257  }
258 
259  if (isnext)
260  iscurrent = false;
261  }
262 
263  // printf("add TOF digit c=%i n=%i\n",iscurrent,isnext);
264 
265  std::vector<Strip>* strips;
266  o2::dataformats::MCTruthContainer<o2::tof::MCLabel>* mcTruthContainer;
267 
268  if (iscurrent) {
269  strips = mStripsCurrent;
270  mcTruthContainer = mMCTruthContainerCurrent;
271  } else {
272  strips = mStripsNext[isnext - 1];
273  mcTruthContainer = mMCTruthContainerNext[isnext - 1];
274  }
275 
276  fillDigitsInStrip(strips, mcTruthContainer, channel, tdc, tot, nbc, istrip, trackID, mEventID, mSrcID);
277 }

329 Bool_t Digitizer::isFired(Float_t x, Float_t z, Float_t charge)
330 {
331  if (TMath::Abs(x) > Geo::XPAD * 0.5 + 0.3)
332  return kFALSE;
333  if (TMath::Abs(z) > Geo::ZPAD * 0.5 + 0.3)
334  return kFALSE;
335 
336  Float_t effX = getEffX(x);
337  Float_t effZ = getEffZ(z);
338 
339  Float_t efficiency = TMath::Min(effX, effZ);
340 
341  if (gRandom->Rndm() > efficiency)
342  return kFALSE;
343 
344  return kTRUE;
345 }

348 Float_t Digitizer::getEffX(Float_t x)
349 {
350  Float_t xborder = Geo::XPAD * 0.5 - TMath::Abs(x);
351 
352  if (xborder > 0) {
353  if (xborder > mBound1)
354  return mEffCenter;
355  else if (xborder > mBound2)
356  return mEffBoundary1 + (mEffCenter - mEffBoundary1) * (xborder - mBound2) / (mBound1 - mBound2);
357  else
358  return mEffBoundary2 + (mEffBoundary1 - mEffBoundary2) * xborder / mBound2;
359  } else {
360  xborder *= -1;
361  if (xborder > mBound4)
362  return 0;
363  else if (xborder > mBound3)
364  return mEffBoundary3 - mEffBoundary3 * (xborder - mBound3) / (mBound4 - mBound3);
365  else
366  return mEffBoundary2 + (mEffBoundary3 - mEffBoundary2) * xborder / mBound3;
367  }
368 
369  return 0;
370 }

373 Float_t Digitizer::getEffZ(Float_t z)
374 {
375  Float_t zborder = Geo::ZPAD * 0.5 - TMath::Abs(z);
376 
377  if (zborder > 0) {
378  if (zborder > mBound1)
379  return mEffCenter;
380  else if (zborder > mBound2)
381  return mEffBoundary1 + (mEffCenter - mEffBoundary1) * (zborder - mBound2) / (mBound1 - mBound2);
382  else
383  return mEffBoundary2 + (mEffBoundary1 - mEffBoundary2) * zborder / mBound2;
384  } else {
385  zborder *= -1;
386  if (zborder > mBound4)
387  return 0;
388  else if (zborder > mBound3)
389  return mEffBoundary3 - mEffBoundary3 * (zborder - mBound3) / (mBound4 - mBound3);
390  else
391  return mEffBoundary2 + (mEffBoundary3 - mEffBoundary2) * zborder / mBound3;
392  }
393 
394  return 0;

401 void Digitizer::initParameters()
402 {
403  // boundary references for interpolation of efficiency and resolution
404  mBound1 = 0.4; // distance from border when efficiency starts to decrese
405  mBound2 = 0.15; // second step in the fired pad
406  mBound3 = 0.55; // distance from border (not fired pad)
407  mBound4 = 0.9; // distance from border (not fired pad) when efficiency vanishes
408 
409  // resolution parameters
410  mTOFresolution = 60; // TOF global resolution in ps
411  mShowerResolution = 50; // smearing correlated for all digits of the same hit
412  if (mTOFresolution > mShowerResolution)
413  mDigitResolution = TMath::Sqrt(mTOFresolution * mTOFresolution -
414  mShowerResolution * mShowerResolution); // independent smearing for each digit
415  else {
416  mShowerResolution = mTOFresolution;
417  mDigitResolution = 0;
418  }
419 
420  mTimeSlope = 100; // ps/cm extra smearing if hit out of pad propto the distance from the border
421  mTimeDelay = 70; // time delay if hit out of pad
422  mTimeWalkeSlope = 40; // ps/cm
423 
424  if (mMode == 0) {
425  mTimeSlope = 0;
426  mTimeDelay = 0;
427  mTimeWalkeSlope = 0;
428  }
429 
430  mTimeDelayCorr = mTimeDelay / 3.5;
431  if (mShowerResolution > mTimeDelayCorr)
432  mShowerResolution = TMath::Sqrt(mShowerResolution * mShowerResolution - mTimeDelayCorr * mTimeDelayCorr);
433  else
434  mShowerResolution = 0;
435 
436  if (mShowerResolution > mTimeWalkeSlope * 0.8)
437  mShowerResolution = TMath::Sqrt(mShowerResolution * mShowerResolution - mTimeWalkeSlope * mTimeWalkeSlope * 0.64);
438  else
439  mShowerResolution = 0;
440 
441  // efficiency parameters
442  mEffCenter = 0.995; // efficiency in the center of the fired pad
443  mEffBoundary1 = 0.94; // efficiency in mBound2
444  mEffBoundary2 = 0.833; // efficiency in the pad border
445  mEffBoundary3 = 0.1; // efficiency in mBound3
446 }

754 void Digitizer::checkIfReuseFutureDigits()
755 {
756  // check if digits stored very far in future match the new readout windows currently available
757  int idigit = 0;
758  for (auto& digit : mFutureDigits) {
759  double timestamp = digit.getBC() * 25 + digit.getTDC() * Geo::TDCBIN * 1E-3; // in ns
760  int isnext = Int_t(timestamp * Geo::READOUTWINDOW_INV) - (mReadoutWindowCurrent + 1); // to be replaced with uncalibrated time
761  if (isnext < 0) // we jump too ahead in future, digit will be not stored
762  LOG(INFO) << "Digit lost because we jump too ahead in future. Current RO window=" << isnext << "\n";
763  if (isnext < MAXWINDOWS - 1) { // move from digit buffer array to the proper window
764  if (isnext >= 0) {
765  std::vector<Strip>* strips = mStripsCurrent;
766  o2::dataformats::MCTruthContainer<o2::tof::MCLabel>* mcTruthContainer = mMCTruthContainerCurrent;
767 
768  if (isnext) {
769  strips = mStripsNext[isnext - 1];
770  mcTruthContainer = mMCTruthContainerNext[isnext - 1];
771  }
772 
773  int trackID = mFutureItrackID[digit.getLabel()];
774  int sourceID = mFutureIsource[digit.getLabel()];
775  int eventID = mFutureIevent[digit.getLabel()];
776  fillDigitsInStrip(strips, mcTruthContainer, digit.getChannel(), digit.getTDC(), digit.getTOT(), digit.getBC(), digit.getChannel() / Geo::NPADS, trackID, eventID, sourceID);
777  }
778 
779  // remove the element from the buffers
780  mFutureItrackID.erase(mFutureItrackID.begin() + digit.getLabel());
781  mFutureIsource.erase(mFutureIsource.begin() + digit.getLabel());
782  mFutureIevent.erase(mFutureIevent.begin() + digit.getLabel());
783 
784  int labelremoved = digit.getLabel();
785  // adjust labels
786  for (auto& digit2 : mFutureDigits) {
787  if (digit2.getLabel() > labelremoved)
788  digit2.setLabel(digit2.getLabel() - 1);
789  }
790  // remove also digit from buffer
791  mFutureDigits.erase(mFutureDigits.begin() + idigit);
792  } else {
793  idigit++; // increment when moving to the next only if the current is not removed from the buffer
794  }
795  } // close future digit loop
796 }
