68 Int_t CalibPulser::updateROC(const Int_t roc, const Int_t row, const Int_t pad,
69  const Int_t timeBin, Float_t signal)
70 {
71  // ===| range checks |========================================================
72  if (signal<mADCMin || signal>mADCMax) return 0;
73  if (timeBin<mFirstTimeBin || timeBin>mLastTimeBin) return 0;
74 
75  // ===| correct the signal |==================================================
76  // TODO before or after signal range check?
77  if (mPedestal) {
78  signal -= mPedestal->getValue(ROC(roc), row, pad);
79  }
80 
81  // ===| temporary calibration data |==========================================
82  const PadROCPos padROCPos(roc, row, pad);
83  VectorType& adcData = mPulserData[padROCPos];
84 
85  if ( !adcData.size() ) {
86  // accept first and last time bin, so difference +1
87  adcData.resize(mLastTimeBin-mFirstTimeBin+1);
88  //printf("new pad pos\n");
89  }
90 
91  adcData[timeBin-mFirstTimeBin] = signal;
92  //printf("%2d, %3d, %3d, %3d: %.2f\n", roc, row, pad, timeBin, signal);
93  return 1;
94 }

140 CalibPulser::PulserData CalibPulser::processPadData(const PadROCPos& padROCPos, const CalibPulser::VectorType& adcData)
141 {
142  // data to return
143  PulserData data;
144 
145  const auto vectorSize = adcData.size();
146  const auto maxElement = std::max_element(std::begin(adcData), std::end(adcData));
147  const auto maxPosition = std::distance(std::begin(adcData), maxElement);
148 
149  double weightedSum=0.;
150  double weightedSum2=0.;
151  double chargeSum=0.;
152  for (int t = maxPosition-mPeakIntMinus; t<=maxPosition+mPeakIntPlus; ++t) {
153  const auto signal = adcData[t];
154  // check time bounds
155  if (t<0 || t>=vectorSize ) continue;
156  weightedSum += signal * (t+0.5); // +0.5 to get the center of the time bin
157  weightedSum2 += signal * (t+0.5)*(t+0.5);
158  chargeSum += signal;
159  }
160 
161  if (chargeSum > mMinimumQtot) {
162  weightedSum /= chargeSum;
163  weightedSum2 = std::sqrt(std::abs(weightedSum2/chargeSum - weightedSum*weightedSum));
164  // L1 phase correction?
165  data.mT0 = weightedSum;
166  data.mWidth = weightedSum2;
167  data.mQtot = chargeSum;
168  }
169 
170  return data;
171 }

188 void CalibPulser::analyse()
189 {
190  for (ROC roc; !roc.looped(); ++roc) {
191  auto histT0 = mT0Histograms.at(roc).get();
192  auto histWidth = mWidthHistograms.at(roc).get();
193  auto histQtot = mQtotHistograms.at(roc).get();
194 
195  if (!histT0 || !histWidth || !histQtot) continue;
196 
197  // array pointer
198  const auto arrT0 = histT0 ->GetArray();
199  const auto arrWidth = histWidth->GetArray();
200  const auto arrQtot = histQtot ->GetArray();
201 
202  const auto numberOfPads = mMapper.getNumberOfPads(roc);
203  for (auto iChannel=0; iChannel<numberOfPads; ++iChannel) {
204  const int offsetT0 = (mNbinsT0 +2)*(iChannel+1)+1;
205  const int offsetQtot = (mNbinsQtot +2)*(iChannel+1)+1;
206  const int offsetWidth= (mNbinsWidth+2)*(iChannel+1)+1;
207 
208  StatisticsData dataT0 = getStatisticsData(arrT0 + offsetT0, mNbinsT0, mXminT0, mXmaxT0);
209  StatisticsData dataWidth = getStatisticsData(arrWidth + offsetWidth, mNbinsWidth, mXminWidth, mXmaxWidth);
210  StatisticsData dataQtot = getStatisticsData(arrQtot + offsetQtot, mNbinsQtot, mXminQtot, mXmaxQtot);
211 
212  mT0. getCalArray(roc).setValue(iChannel, dataT0.mCOG);
213  mWidth.getCalArray(roc).setValue(iChannel, dataWidth.mCOG);
214  mQtot. getCalArray(roc).setValue(iChannel, dataQtot.mCOG);
215  }
216  }
217 }
