38  int nRowClusters[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
39  int containerRowCluster[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
40  Mapper& mapper = Mapper::instance();
41  int numberOfOutputContainers = 0;
42  for (int loop = 0;loop < 2;loop++)
43  {
44  for (int i = 0;i < inputClusters.size();i++)
45  {
46  if (outMCLabels && inputClusters[i].second > 1)
47  {
48  LOG(ERROR) << "Decoding of ClusterHardware to ClusterNative with MC labels is yet only support for single 8kb pages of ClusterHardwareContainer\n";

50  }
51  for (int j = 0;j < inputClusters[i].second;j++)
52  {
53  const char* tmpPtr = reinterpret_cast<const char*> (inputClusters[i].first);
54  tmpPtr += j * 8192; //TODO: FIXME: Compute correct offset based on the size of the actual packet in the RDH
55  const ClusterHardwareContainer& cont = *(reinterpret_cast<const ClusterHardwareContainer*> (tmpPtr));
56  const CRU cru(cont.CRU);
57  const Sector sector = cru.sector();
58  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
59  const int rowOffset = region.getGlobalRowOffset();

61  for (int k = 0;k < cont.numberOfClusters;k++)
62  {
63  const int padRowGlobal = rowOffset + cont.clusters[k].getRow();
64  int& nCls = nRowClusters[sector][padRowGlobal];
65  if (loop == 1)
66  {
67  //Fill cluster in the respective output buffer
68  const ClusterHardware& cIn = cont.clusters[k];
69  ClusterNative& cOut = outputClusters[containerRowCluster[sector][padRowGlobal]].clusters[nCls];
70  float pad = cIn.getPad();
71  cOut.setPad(pad);
72  cOut.setTimeFlags(cIn.getTimeLocal() + cont.timeBinOffset, cIn.getFlags());
73  cOut.setSigmaPad(std::sqrt(cIn.getSigmaPad2()));
74  cOut.setSigmaTime(std::sqrt(cIn.getSigmaTime2()));
75  cOut.qMax = cIn.getQMax();

78  if (outMCLabels)
79  {
80  MCLabelContainer& mcOut = (*outMCLabels)[containerRowCluster[sector][padRowGlobal]];
81  for (const auto& element : (*inMCLabels)[i].getLabels(k)) {
82  mcOut.addElement(nCls, element);
83  }
84  }
85  }
86  else
87  {
88  //Count how many output buffers we need (and how large they are below)
89  if (nCls == 0) numberOfOutputContainers++;
90  }
91  nCls++;
92  }
93  }
94  }
95  if (loop == 1)
96  {

98  for (int i = 0;i < outputClusters.size();i++)
99  {

102  } else {

104  std::sort(cl.data(), cl.data() + cl.size(), ClusterNativeContainer::sortComparison);
105  }
106  }
107  }
108  else
109  {
110  //Now we know the size of all output buffers, allocate them
111  outputClusters.resize(numberOfOutputContainers);
112  if (outMCLabels) outMCLabels->resize(numberOfOutputContainers);
113  numberOfOutputContainers = 0;
114  for (int i = 0;i < Constants::MAXSECTOR;i++)
115  {
116  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++)
117  {
118  if (nRowClusters[i][j] == 0) continue;
119  outputClusters[numberOfOutputContainers].clusters.resize(nRowClusters[i][j]);
120  outputClusters[numberOfOutputContainers].sector = i;
121  outputClusters[numberOfOutputContainers].globalPadRow = j;
122  containerRowCluster[i][j] = numberOfOutputContainers++;
123  mIntegrator->initRow(i, j);
124  }
125  }
126  memset(nRowClusters, 0, sizeof(nRowClusters));
127  }
128  }
129  return(0);
