54  bool decide(const o2::framework::DataRef& dataRef) override
55  {
56  const auto* dpHeader = get<DataProcessingHeader*>(dataRef.header);
57  assert(dpHeader);
58 
59  int64_t diff = dpHeader->startTime - mCurrentTimesliceID;
60  if (diff == -1) {
61  return mLastDecision;
62  } else if (diff < -1) {
63  mGenerator.backstep(static_cast<uint64_t>(-diff));
64  } else if (diff > 0) {
65  mGenerator.advance(static_cast<uint64_t>(diff));
66  }
67 
68  mLastDecision = mGenerator() < mThreshold;
69  mCurrentTimesliceID = dpHeader->startTime + 1;
70  return mLastDecision;
71  }
72 
73  private:
74  uint32_t mThreshold;
75  pcg32_fast mGenerator;

77  DataProcessingHeader::StartTime mCurrentTimesliceID;
