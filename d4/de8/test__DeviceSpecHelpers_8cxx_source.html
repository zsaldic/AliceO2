34 // search for an option in the device execution
35 bool search(DeviceExecution const& execution, std::string const& option, std::string const& argument)
36 {
37  bool foundOption = false;
38  for (auto const& execArg : execution.args) {
39  if (execArg == nullptr) {
40  break;
41  }
42  if (!foundOption) {
43  foundOption = option == execArg;
44  } else if (argument == execArg) {
45  return true;
46  } else {
47  // the required argument to the option is not found
48  foundOption = false;
49  }
50  }
51  return false;
52 }
