24 // Basic cluster class with X,Y,Z position detector ID information + user fields
25 // The position is ALWAYS stored in tracking frame and is misaligned (in opposite
26 // to AliRoot). The errors are defined in *ideal* tracking frame
27 // DetectorID should correspond to continuous (no jumps between detector layers
28 // planes etc.) internal sensor ID within detector
29 // Detector specific clusters should be composed by including it as data member
30 template <typename T>
31 class BaseCluster
32 {
33  private:
34  Point3D<T> mPos; // cartesian position
35  T mSigmaY2; // error in Y direction (usually rphi)
36  T mSigmaZ2; // error in Z direction (usually Z)
37  T mSigmaYZ; // non-diagonal term of error matrix
38  std::uint16_t mSensorID = 0; // the sensor id
39  std::int8_t mCount = 0; // user field reserved for counting
40  std::uint8_t mBits = 0; // user field reserved for bit flags
