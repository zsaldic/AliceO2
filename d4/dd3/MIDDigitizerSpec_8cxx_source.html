86  void run(framework::ProcessingContext& pc)
87  {
88  static bool finished = false;
89  if (finished) {
90  return;
91  }
92  LOG(INFO) << "Doing MID digitization";
93 
94  // read collision context from input
95  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
96  auto& irecords = context->getEventRecords();
97 
98  for (auto& record : irecords) {
99  LOG(INFO) << "MID TIME RECEIVED " << record.timeNS;
100  }
101 
102  auto& eventParts = context->getEventParts();
103  std::vector<o2::mid::ColumnDataMC> digitsAccum; // accumulator for digits
104  o2::dataformats::MCTruthContainer<o2::mid::MCLabel> labelsAccum;
105 
106  // loop over all composite collisions given from context
107  // (aka loop over all the interaction records)
108  for (int collID = 0; collID < irecords.size(); ++collID) {
109  mDigitizer->setEventTime(irecords[collID].timeNS);
110 
111  // for each collision, loop over the constituents event and source IDs
112  // (background signal merging is basically taking place here)
113  for (auto& part : eventParts[collID]) {
114  mDigitizer->setEventID(part.entryID);
115  mDigitizer->setSrcID(part.sourceID);
116 
117  // get the hits for this event and this source
118  std::vector<o2::mid::Hit> hits;
119  retrieveHits(mSimChains, "MIDHit", part.sourceID, part.entryID, &hits);
120  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found MID " << hits.size() << " hits ";
121 
122  std::vector<o2::mid::ColumnDataMC> digits; // digits which get filled
123  o2::dataformats::MCTruthContainer<o2::mid::MCLabel> labels;
124  mDigitizer->process(hits, digits, labels);
125  std::copy(digits.begin(), digits.end(), std::back_inserter(digitsAccum));
126  labelsAccum.mergeAtBack(labels);
127  }
128  }
129 
130  LOG(INFO) << "MID: Sending " << digitsAccum.size() << " digits";
131  pc.outputs().snapshot(Output{ "MID", "DIGITS", 0, Lifetime::Timeframe }, digitsAccum);
132  pc.outputs().snapshot(Output{ "MID", "DIGITSMC", 0, Lifetime::Timeframe }, labelsAccum);
133 
134  LOG(INFO) << "MID: Sending ROMode= " << mROMode << " to GRPUpdater";
135  pc.outputs().snapshot(Output{ "MID", "ROMode", 0, Lifetime::Timeframe }, mROMode);
136 
137  // we should be only called once; tell DPL that this process is ready to exit
138  pc.services().get<ControlService>().readyToQuit(false);
139  finished = true;
140  }
141 
142  private:
143  std::unique_ptr<Digitizer> mDigitizer;
144  std::vector<TChain*> mSimChains;
145  // RS: at the moment using hardcoded flag for continuos readout
146  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
147 };
