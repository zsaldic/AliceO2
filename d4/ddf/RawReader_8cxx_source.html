59 bool RawReader::addInputFile(std::string infile) {
60  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
61  boost::char_separator<char> sep{":"};
62  tokenizer tok{infile,sep};
63  if (std::distance(tok.begin(), tok.end()) < 3) {
64  LOG(ERROR) << "Not enough arguments" << FairLogger::endl;
65  return false;
66  }
67 
68  tokenizer::iterator it1= tok.begin();
69  std::string path = boost::lexical_cast<std::string>(*it1);
70  int run = std::stoi(path.substr(path.find("run")+3,6));
71 
72  int region, link;
73  try {
74  ++it1;
75  region = boost::lexical_cast<int>(*it1);
76  } catch (boost::bad_lexical_cast) {
77  LOG(ERROR) << "Please enter a region for " << path << FairLogger::endl;
78  return false;
79  }
80 
81  try {
82  ++it1;
83  link = boost::lexical_cast<int>(*it1);
84  } catch (boost::bad_lexical_cast) {
85  LOG(ERROR) << "Please enter a link number for " << path << FairLogger::endl;
86  return false;
87  }
88 
89  int sampaVersion = -1;
90  if (++it1 == tok.end()) {
91  LOG(DEBUG) << "No SAMPA version was entered, assuming latest." << FairLogger::endl;
92  } else {
93  try {
94  sampaVersion = boost::lexical_cast<int>(*it1);
95  } catch (boost::bad_lexical_cast) {
96  LOG(ERROR) << "Please enter a valid SAMPA version for " << path << FairLogger::endl;
97  return false;
98  }
99  }
100 
101  return addInputFile(region, link, sampaVersion, path, run);
102 }

104 bool RawReader::addInputFile(int region, int link, int sampaVersion, std::string path, int run)
105 {
106  if (mRun == -1)
107  mRun = run;
108  if (mRegion == -1)
109  mRegion = region;
110  if (mLink == -1)
111  mLink = link;
112  if (mSampaVersion == -1)
113  mSampaVersion = sampaVersion;
114 
115  if (run != mRun) {
116  LOG(DEBUG) << "Run of RawReader is " << mRun << " and not " << run << FairLogger::endl;
117  return false;
118  }
119 
120  if (region != mRegion) {
121  LOG(DEBUG) << "Region of RawReader is " << mRegion << " and not " << region << FairLogger::endl;
122  return false;
123  }
124 
125  if (link != mLink) {
126  LOG(DEBUG) << "Link of RawReader is " << mLink << " and not " << link << FairLogger::endl;
127  return false;
128  }
129 
130  if (sampaVersion != mSampaVersion) {
131  LOG(DEBUG) << "SAMPA Version of RawReader is " << mSampaVersion << " and not " << sampaVersion << FairLogger::endl;
132  return false;
133  }
134 
135  std::ifstream file(path);
136  if (!file.is_open()) {
137  LOG(ERROR) << "Can't read file " << path << FairLogger::endl;
138  return false;
139  }
140 
141  Header h;
142  int pos = 0;
143  file.seekg(0,file.end);
144  int length = file.tellg();
145  file.seekg(0,file.beg);
146 
147  while (pos < length) {
148  file.read((char*)&h, sizeof(h));
149  if (h.reserved_01 != 0x0F || h.reserved_2() != 0x3fec2fec1fec0fec) {
150  LOG(ERROR) << "Header does not look consistent" << FairLogger::endl;
151  }
152  EventInfo eD;
153  eD.path = path;
154  eD.posInFile = file.tellg();
155  eD.header = h;
156  if (h.headerVersion == 1) {
157 // auto ins = mEvents.insert(std::make_pair(h.eventCount(),std::shared_ptr<std::vector<EventInfo>>(new std::vector<EventInfo>)));
158  auto ins = mEvents.insert(std::make_pair(h.eventCount(),std::make_shared<std::vector<EventInfo>>()));
159  ins.first->second->push_back(eD);
160 
161  if (h.eventCount() == 0) {
162  LOG(DEBUG) << "Processing event 0 to find sync pattern for event synchronization" << FairLogger::endl;
163  loadEvent(0); // processing first event to find the SYNC pattern (mode 1) or start of data (mode 2)
164  mLastEvent=-1;
165  }
166 
167  mEventSynchronizer->addEventOffset(
168  h.eventCount(),
169  (h.timeStamp() - mTimestampOfFirstData[0]));
170 
171  file.seekg(pos+(h.nWords*4));
172  pos = file.tellg();
173  } else {
174  LOG(ERROR) << "Header version " << (int)h.headerVersion << " not implemented." << FairLogger::endl;
175  return false;
176  }
177  }
178 
179  return true;

182 bool RawReader::loadEvent(int64_t event) {
183  const Mapper& mapper = Mapper::instance();
184  const PartitionInfo& partInfo = mapper.getPartitionInfo(mRegion/2);
185  LOG(DEBUG) << "Loading event " << event << " for Link " << (mRegion*partInfo.getNumberOfFECs())+mLink << FairLogger::endl;
186 
187  if (mLastEvent == -1 && event != getFirstEvent() ) {
188  LOG(DEBUG) << "First event was not loaded (maybe important for decoding RAW GBT frames) loading first event." << FairLogger::endl;
189  mSyncPos.fill(-1);
190  mTimestampOfFirstData.fill(0);
191  loadEvent(getFirstEvent());
192  LOG(DEBUG) << "Continue with event " << event << FairLogger::endl;
193  }
194  mData.clear();
195 
196  auto ev = mEvents.find(event);
197 
198  if (ev == mEvents.end()) return false;
199  mLastEvent = event;
200 
201  for (auto &eventInfo : *(ev->second)) {
202  switch (eventInfo.header.dataType) {
203  case 1: // RAW GBT frames
204  {
205  LOG(DEBUG) << "Data of readout mode 1 (RAW GBT frames)" << FairLogger::endl;
206  if (!decodeRawGBTFrames(eventInfo)) return false;
207  break;
208  }
209  case 2: // Decoded data
210  {
211  LOG(DEBUG) << "Data of readout mode 2 (decoded data)" << FairLogger::endl;
212  if (!decodePreprocessedData(eventInfo)) return false;
213  break;
214  }
215  case 3: // both, RAW GBT frames and decoded data
216  {
217  if (mUseRawInMode3) {
218  LOG(DEBUG) << "Data of readout mode 3 (decoding RAW GBT frames)" << FairLogger::endl;
219  if (!decodeRawGBTFrames(eventInfo)) return false;
220  } else {
221  LOG(DEBUG) << "Data of readout mode 3 (using decoded data)" << FairLogger::endl;
222  if (!decodePreprocessedData(eventInfo)) return false;
223  }
224  break;
225  }
226  default:
227  {
228  LOG(DEBUG) << "Readout mode not known" << FairLogger::endl;
229  break;
230  }
231  }
232  }
233 
234  mDataIterator = mData.begin();
235  LOG(DEBUG) << FairLogger::endl;
236  LOG(DEBUG) << FairLogger::endl;
237  return true;
238 }

240 bool RawReader::decodePreprocessedData(EventInfo eventInfo) {
241  std::ifstream file(eventInfo.path);
242  if (!file.is_open()) {
243  LOG(ERROR) << "Can't read file " << eventInfo.path << FairLogger::endl;
244  return false;
245  }
246 
247  int nWords = eventInfo.header.nWords-8;
248  uint32_t words[nWords];
249  LOG(DEBUG) << "reading " << nWords << " words from position " << eventInfo.posInFile << " in file " << eventInfo.path << FairLogger::endl;
250  file.seekg(eventInfo.posInFile);
251  file.read((char*)&words, nWords*sizeof(words[0]));
252 
253  const Mapper& mapper = Mapper::instance();
254 
255  std::array<char,5> sampas;
256  std::array<char,5> sampaChannelStart;
257  for (char i=0; i<5; ++i) {
258  sampas[i] = (i == 4) ? 2 : (mRegion%2) ? i/2+3 : i/2;
259  sampaChannelStart[i] = (i == 4) ? // 5th half SAMPA corresponds to SAMPA2
260  ((mRegion%2) ? 16 : 0) : // every even CRU receives channel 0-15 from SAMPA 2, the odd ones channel 16-31
261  ((i%2) ? 16 : 0); // every even half SAMPA containes channel 0-15, the odd ones channel 16-31
262  }
263 
264 
265  std::array<uint32_t,5> ids;
266  std::array<bool,5> writeValue;
267  writeValue.fill(false);
268  std::array<std::array<uint16_t,16>,5> adcValues;
269 
270 
271  int indexStep = (eventInfo.header.dataType == 3) ? 8 : 4;
272  int offset = (eventInfo.header.dataType == 3) ? 4: 0;
273 
274  long i_start = 0;
275  if (eventInfo.header.eventCount() > 0) {
276  LOG(DEBUG) << "Using offset [" << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) << "] "
277  << "instead of [" << eventInfo.header.timeStamp() << "] "
278  << " (diff = " << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() << ") "
279  << FairLogger::endl;
280 
281  i_start = mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() -1;
282  i_start *= indexStep;
283  }
284 
285  LOG(DEBUG) << "Start index for Event " << eventInfo.header.eventCount() << " is " << i_start << FairLogger::endl;
286 
287 
288  for (long i=i_start; i<nWords; i=i+indexStep) {
289  ids[4] = (words[i+offset] >> 4) & 0xF;
290  ids[3] = (words[i+offset] >> 8) & 0xF;
291  ids[2] = (words[i+offset] >> 12) & 0xF;
292  ids[1] = (words[i+offset] >> 16) & 0xF;
293  ids[0] = (words[i+offset] >> 20) & 0xF;
294 
295  adcValues[4][((ids[4] & 0x7)*2)+1] = (((ids[4]>>3)&0x1) == 0) ? 0 : words[i+offset+3] & 0x3FF;
296  adcValues[4][((ids[4] & 0x7)*2) ] = (((ids[4]>>3)&0x1) == 0) ? 0 : (words[i+offset+3] >> 10) & 0x3FF;
297  adcValues[3][((ids[3] & 0x7)*2)+1] = (((ids[3]>>3)&0x1) == 0) ? 0 : (words[i+offset+3] >> 20) & 0x3FF;
298  adcValues[3][((ids[3] & 0x7)*2) ] = (((ids[3]>>3)&0x1) == 0) ? 0 : ((words[i+offset+2] & 0xFF) << 2) | ((words[i+offset+3] >> 30) & 0x3);
299  adcValues[2][((ids[2] & 0x7)*2)+1] = (((ids[2]>>3)&0x1) == 0) ? 0 : (words[i+offset+2] >> 8) & 0x3FF;
300  adcValues[2][((ids[2] & 0x7)*2) ] = (((ids[2]>>3)&0x1) == 0) ? 0 : (words[i+offset+2] >> 18) & 0x3FF;
301  adcValues[1][((ids[1] & 0x7)*2)+1] = (((ids[1]>>3)&0x1) == 0) ? 0 : ((words[i+offset+1] & 0x3F) << 4) | ((words[i+offset+2] >> 28) & 0xF);
302  adcValues[1][((ids[1] & 0x7)*2) ] = (((ids[1]>>3)&0x1) == 0) ? 0 : (words[i+offset+1] >> 6) & 0x3FF;
303  adcValues[0][((ids[0] & 0x7)*2)+1] = (((ids[0]>>3)&0x1) == 0) ? 0 : (words[i+offset+1] >> 16) & 0x3FF;
304  adcValues[0][((ids[0] & 0x7)*2) ] = (((ids[0]>>3)&0x1) == 0) ? 0 : ((words[i+offset+0] & 0xF) << 6) | ((words[i+offset+1] >> 26) & 0x3F);
305 
306  for (char j=0; j<5; ++j) {
307  if (ids[j] == 0x8) {
308  writeValue[j] = true;
309  // header TS is one before first word -> +1
310  if (mTimestampOfFirstData[j] == 0) mTimestampOfFirstData[j] = eventInfo.header.timeStamp() + 1 + i/indexStep;
311  }
312  }
313 
314  for (char j=0; j<5; ++j) {
315  if (writeValue[j] & (ids[j] == 0xF)) {
316  for (int k=0; k<16; ++k) {
317  const PadPos padPos = mapper.padPosRegion(
318  mRegion, mLink,sampas[j],k+sampaChannelStart[j]);
319 
320  if (mApplyChannelMask && // channel mask should be applied
321  (mChannelMask != nullptr) && // channel mask is available
322  !mChannelMask->getValue(CRU(mRegion),padPos.getPad(),padPos.getRow())) { // channel mask
323  continue;
324  }
325 
326 // auto ins = mData.insert(std::make_pair(padPos, std::shared_ptr<std::vector<uint16_t>>(new std::vector<uint16_t>)));
327  auto ins = mData.insert(std::make_pair(padPos, std::make_shared<std::vector<uint16_t>>()));
328  ins.first->second->push_back(adcValues[j][k]);
329  }
330  }
331  }
332 
333  }
334  return true;
335 }

337 bool RawReader::decodeRawGBTFrames(EventInfo eventInfo) {
338  std::ifstream file(eventInfo.path);
339  if (!file.is_open()) {
340  LOG(ERROR) << "Can't read file " << eventInfo.path << FairLogger::endl;
341  return false;
342  }
343 
344  int nWords = eventInfo.header.nWords-8;
345  uint32_t words[nWords];
346  LOG(DEBUG) << "reading " << nWords << " words from position " << eventInfo.posInFile << " in file " << eventInfo.path << FairLogger::endl;
347  LOG(DEBUG) << "Header time stamp is " << eventInfo.header.timeStamp() << FairLogger::endl;
348  file.seekg(eventInfo.posInFile);
349  file.read((char*)&words, nWords*sizeof(words[0]));

351  const Mapper& mapper = Mapper::instance();
352 
353  std::array<char,5> sampas;
354  std::array<char,5> sampaChannelStart;

357  sampaChannelStart[i] = (i == 4) ? // 5th half SAMPA corresponds to SAMPA2
358  ((mRegion%2) ? 16 : 0) : // every even CRU receives channel 0-15 from SAMPA 2, the odd ones channel 16-31
359  ((i%2) ? 16 : 0); // every even half SAMPA containes channel 0-15, the odd ones channel 16-31
360  }
361 
362  std::array<SyncPatternMonitor,5> syncMon{
363  SyncPatternMonitor((mRegion%2) ? 3 : 0,0),
364  SyncPatternMonitor((mRegion%2) ? 3 : 0,1),
365  SyncPatternMonitor((mRegion%2) ? 4 : 1,0),
366  SyncPatternMonitor((mRegion%2) ? 4 : 1,1),
367  SyncPatternMonitor(2,(mRegion%2) ? 1 : 0)};
368  std::array<AdcClockMonitor,3> adcClockMon{
369  AdcClockMonitor((mRegion%2) ? 3 : 0),
370  AdcClockMonitor((mRegion%2) ? 4 : 1),
371  AdcClockMonitor(2)};
372  std::array<bool,3> adcClockFound{false, false, false};
373 
374  std::array<short,5> lastSyncPos;
375  std::array<std::list<uint16_t>,5> adcValues;
376  mAdcError->clear();
377  uint64_t timebin = 0;
378 
379  int indexStep = (eventInfo.header.dataType == 3) ? 8 : 4;
380 
381  long i_start = 0;
382  if (eventInfo.header.eventCount() > 0) {
383  LOG(DEBUG) << "Using offset [" << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) << "] "
384  << "instead of [" << eventInfo.header.timeStamp() << "] "
385  << " (diff = " << mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() << ") "
386  << FairLogger::endl;
387 
388  i_start = mTimestampOfFirstData[0] + mEventSynchronizer->getEventOffset(eventInfo.header.eventCount()) - eventInfo.header.timeStamp() -1;
389  i_start *= indexStep;
390  }

392  LOG(DEBUG) << "Start index for Event " << eventInfo.header.eventCount() << " is " << i_start << FairLogger::endl;
393 
394  GBTFrame frame;
395  if (eventInfo.header.eventCount() > 0) frame.setData(words[i_start-indexStep],words[i_start-indexStep+1],words[i_start-indexStep+2],words[i_start-indexStep+3]);
396  GBTFrame lastFrame;
397 
398  for (long i=i_start; i<nWords; i=i+indexStep) {
399 
400  lastSyncPos = mSyncPos;
401  lastFrame = frame;
402  frame.setData(words[i],words[i+1],words[i+2],words[i+3]);
403 
404  if (syncMon[0].addSequence(frame.getHalfWord(0,0,0), frame.getHalfWord(0,1,0), frame.getHalfWord(0,2,0), frame.getHalfWord(0,3,0))) mSyncPos[0] = syncMon[0].getPosition();
405  if (syncMon[1].addSequence(frame.getHalfWord(0,0,1), frame.getHalfWord(0,1,1), frame.getHalfWord(0,2,1), frame.getHalfWord(0,3,1))) mSyncPos[1] = syncMon[1].getPosition();
406  if (syncMon[2].addSequence(frame.getHalfWord(1,0,0), frame.getHalfWord(1,1,0), frame.getHalfWord(1,2,0), frame.getHalfWord(1,3,0))) mSyncPos[2] = syncMon[2].getPosition();
407  if (syncMon[3].addSequence(frame.getHalfWord(1,0,1), frame.getHalfWord(1,1,1), frame.getHalfWord(1,2,1), frame.getHalfWord(1,3,1))) mSyncPos[3] = syncMon[3].getPosition();
408  if (syncMon[4].addSequence(frame.getHalfWord(2,0), frame.getHalfWord(2,1), frame.getHalfWord(2,2), frame.getHalfWord(2,3))) mSyncPos[4] = syncMon[4].getPosition();
409 
410  if (mCheckAdcClock) {
411 
412  bool adcCheckErr0 = adcClockMon[0].addSequence(frame.getAdcClock(0));
413  bool adcCheckErr1 = adcClockMon[1].addSequence(frame.getAdcClock(1));
414  bool adcCheckErr2 = adcClockMon[2].addSequence(frame.getAdcClock(2));
415 
416  if (mSyncPos[0] >= 0) adcClockFound[0] = adcClockFound[0] | !adcCheckErr0;
417  if (mSyncPos[2] >= 0) adcClockFound[1] = adcClockFound[1] | !adcCheckErr1;
418  if (mSyncPos[4] >= 0) adcClockFound[2] = adcClockFound[2] | !adcCheckErr2;
419  if(adcClockFound[0] & adcCheckErr0) {
420  adcClockFound[0] = false;
421  LOG(DEBUG) << "ADC clock error of SAMPA " << ((mRegion%2) ? 3 : 0) << " in frame [" << i/indexStep << "]" << FairLogger::endl;
422  mAdcError->emplace_back(std::make_tuple((mRegion%2) ? 3 : 0,i,mSyncPos[0]));
423  };
424  if(adcClockFound[1] & adcCheckErr1) {
425  adcClockFound[1] = false;
426  LOG(DEBUG) << "ADC clock error of SAMPA " << ((mRegion%2) ? 4 : 1) << " in frame [" << i/indexStep << "]" << FairLogger::endl;
427  mAdcError->emplace_back(std::make_tuple((mRegion%2) ? 4 : 1,i,mSyncPos[2]));
428  };
429  if(adcClockFound[2] & adcCheckErr2) {
430  adcClockFound[2] = false;
431  LOG(DEBUG) << "ADC clock error of SAMPA " << 2 << " in frame [" << i/indexStep << "]" << FairLogger::endl;
432  mAdcError->emplace_back(std::make_tuple(2,i,mSyncPos[4]));
433  };
434  }
435 
436  short value1;
437  short value2;
438  for (short iHalfSampa = 0; iHalfSampa < 5; ++iHalfSampa) {
439  if (mSyncPos[iHalfSampa] < 0 ) {
440  LOG(DEBUG1) << "Sync pattern for half SAMPA " << iHalfSampa << " not yet found" << FairLogger::endl;
441  continue;
442  }
443  if (lastSyncPos[iHalfSampa] < 0 ) {
444  LOG(DEBUG1) << "Sync pattern for half SAMPA " << iHalfSampa << " not yet found" << FairLogger::endl;
445  LOG(DEBUG1) << lastFrame << FairLogger::endl;
446  continue;
447  }
448  if (mTimestampOfFirstData[iHalfSampa] == 0) {
449  if(iHalfSampa == 0) LOG(DEBUG) << "Setting timestamp of first data to " << eventInfo.header.timeStamp() + 1 + i/indexStep << " for half SAMPA " << iHalfSampa << " (" << 1 + i/indexStep << ")" << FairLogger::endl;
450  mTimestampOfFirstData[iHalfSampa] = eventInfo.header.timeStamp() + 1 + i/indexStep;
451  }
452 
453  switch(mSyncPos[iHalfSampa]) {
454  case 0:
455  value1 = (frame.getHalfWord(iHalfSampa/2,1,iHalfSampa%2) << 5) | frame.getHalfWord(iHalfSampa/2,0,iHalfSampa%2);
456  value2 = (frame.getHalfWord(iHalfSampa/2,3,iHalfSampa%2) << 5) | frame.getHalfWord(iHalfSampa/2,2,iHalfSampa%2);
457  break;
458 
459  case 1:
460  value1 = (lastFrame.getHalfWord(iHalfSampa/2,2,iHalfSampa%2) << 5) | lastFrame.getHalfWord(iHalfSampa/2,1,iHalfSampa%2);
461  value2 = (frame.getHalfWord(iHalfSampa/2,0,iHalfSampa%2) << 5) | lastFrame.getHalfWord(iHalfSampa/2,3,iHalfSampa%2);
462  break;
463 
464  case 2:
465  value1 = (lastFrame.getHalfWord(iHalfSampa/2,3,iHalfSampa%2) << 5) | lastFrame.getHalfWord(iHalfSampa/2,2,iHalfSampa%2);
466  value2 = (frame.getHalfWord(iHalfSampa/2,1,iHalfSampa%2) << 5) | frame.getHalfWord(iHalfSampa/2,0,iHalfSampa%2);
467  break;
468 
469  case 3:
470  value1 = (frame.getHalfWord(iHalfSampa/2,0,iHalfSampa%2) << 5) | lastFrame.getHalfWord(iHalfSampa/2,3,iHalfSampa%2);
471  value2 = (frame.getHalfWord(iHalfSampa/2,2,iHalfSampa%2) << 5) | frame.getHalfWord(iHalfSampa/2,1,iHalfSampa%2);
472  break;
473 
474  default:
475  return false;
