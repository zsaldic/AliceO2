35 Point3D<double> HitFinder::getIntersectInDefaultPlane(const Track& track, int chamber) const
36 {
38  double defaultZ = Constants::sDefaultChamberZ[chamber];
39  double linePar = ((track.getPosition().z() - defaultZ) * mTanTheta - track.getPosition().y()) /
40  (track.getDirection().y() - track.getDirection().z() * mTanTheta);
41  Point3D<double> point;
42  point.SetX(track.getPosition().x() + linePar * track.getDirection().x());
43  point.SetY(track.getPosition().y() + linePar * track.getDirection().y());
44  point.SetZ(track.getPosition().z() + linePar * track.getDirection().z());
45  return point;
46 }

79 std::vector<int> HitFinder::getFiredDE(const Track& track, int chamber) const
80 {
85  std::vector<int> deIdList;
86  Point3D<double> defPos = getIntersectInDefaultPlane(track, chamber);
87  double xPos = defPos.x();
88  double xErr = std::sqrt(track.getCovarianceParameter(Track::CovarianceParamIndex::VarX));
89  double yPos = defPos.y() / mCosTheta;
90  double yErr = std::sqrt(track.getCovarianceParameter(Track::CovarianceParamIndex::VarY));
91  double positions[3] = { yPos, yPos - yErr, yPos + yErr };
92  int centerRpc = -1;
93  for (int ipos = 0; ipos < 3; ++ipos) {
94  int rpc = guessRPC(positions[ipos], chamber);
95  if (rpc < 0 || rpc > 8) {
96  continue;
97  }
98  if (ipos == 0) {
99  centerRpc = rpc;
100  } else if (rpc == centerRpc) {
101  continue;
102  }
103  if (xPos > 0) {
104  deIdList.push_back(Constants::getDEId(true, chamber, rpc));
105  if (xPos - xErr < 0) {
106  deIdList.push_back(Constants::getDEId(false, chamber, rpc));
107  }
108  } else {
109  deIdList.push_back(Constants::getDEId(false, chamber, rpc));
110  if (xPos + xErr > 0) {
111  deIdList.push_back(Constants::getDEId(true, chamber, rpc));
112  }
113  }
114  }
115 
116  return deIdList;
117 }
