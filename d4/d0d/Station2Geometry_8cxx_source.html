125 TGeoVolume* createSegment(int iSeg)
126 {
127 
131 
132  // materials
133  const auto kGasMed = assertMedium(Medium::Gas);
134  const auto kCathodeMed = assertMedium(Medium::Copper);
135  const auto kInsuMed = assertMedium(Medium::FR4);
136  const auto kRohacellMed = assertMedium(Medium::Rohacell);
137  const auto kMEBMed = assertMedium(Medium::Epoxy); // changed w.r.t AliMUONSt2GeometryBuilderv2 after investigation
138  const auto kEERBMed = assertMedium(Medium::Copper);
139 
140  const int kNofLayers = 6;
141 
142  const std::string kLayerName[kNofLayers] = { "gas", "cathode", "insulator", "rohacell", "MEB", "EERB" };
143  const std::array<const TGeoMedium*, kNofLayers> kLayerMedium = { kGasMed, kCathodeMed, kInsuMed, kRohacellMed, kMEBMed, kEERBMed };
144  const double kLayerWidth[kNofLayers] = { kGasWidth, kCathodeWidth, kInsuWidth, kRohaWidth, kMEBWidth, kEERBWidth };
145 
146  auto segment = new TGeoVolumeAssembly(Form("Segment %d", iSeg));
147 
148  // volume dimensions
149  double halfLength = kSegHalfLength[iSeg];
150  double halfHeight = kSegHalfHeight[iSeg];
151  double halfWidth = 0.001; //random value just to initialize the variable
152 
153  double z = 0.; // increment this variable when adding a new layer
154 
155  switch (iSeg) {
156  case 1: // polygon
157 
158  // parameters
159  double par[10];
160  par[0] = kStartAngle; // initial angle
161  par[1] = kStopAngle; // increment in angle starting from initial angle
162  par[2] = kNofEdges; // number of sides
163  par[3] = kNofPlanes; // number of planes
164  par[4] = -halfWidth; // z-position of the first plane
165  par[5] = kSecondInnerRadius; // inner radius first plane
166  par[6] = kOuterRadius; // outer radious first plane
167  par[7] = halfWidth; // z-position of the second plane
168  par[8] = par[5]; // inner radius of second plane
169  par[9] = par[6]; // outer radious of second plane
170 
171  // create and place the layers in the segment
172 
173  // start with gas
174  halfWidth = kLayerWidth[0] / 2.;
175 
176  par[4] = -halfWidth;
177  par[7] = halfWidth;
178  segment->AddNode(new TGeoVolume(Form("Segment %d %s", iSeg, kLayerName[0].data()), new TGeoPgon(par), kLayerMedium[0]), 1);
179 
180  z = halfWidth;
181 
182  for (int j = 1; j < kNofLayers; j++) {
183 
184  halfWidth = kLayerWidth[j] / 2.;
185 
186  par[4] = -halfWidth;
187  par[7] = halfWidth;
188  auto layer = new TGeoVolume(Form("Segment %d %s", iSeg, kLayerName[j].data()), new TGeoPgon(par), kLayerMedium[j]);
189 
190  z += halfWidth;
191  segment->AddNode(layer, 1, new TGeoTranslation(0., 0., z));
192  segment->AddNode(layer, 2, new TGeoTranslation(0., 0., -z));
193  z += halfWidth;
194 
195  } // end of the layer loop
196  break;
197 
198  default: //box
199  // create and place the layers in the segment
200 
201  // start with gas
202  halfWidth = kLayerWidth[0] / 2.;
203 
204  segment->AddNode(new TGeoVolume(Form("Segment %d %s", iSeg, kLayerName[0].data()),
205  new TGeoBBox(halfLength, halfHeight, halfWidth), kLayerMedium[0]),
206  1);
207 
208  z = halfWidth;
209 
210  for (int j = 1; j < kNofLayers; j++) {
211 
212  halfWidth = kLayerWidth[j] / 2.;
213 
214  auto layer = new TGeoVolume(Form("Segment %d %s", iSeg, kLayerName[j].data()),
215  new TGeoBBox(halfLength, halfHeight, halfWidth), kLayerMedium[j]);
216 
217  z += halfWidth;
218  segment->AddNode(layer, 1, new TGeoTranslation(0., 0., z));
219  segment->AddNode(layer, 2, new TGeoTranslation(0., 0., -z));
220  z += halfWidth;
221 
222  } // end of the layer loop
223 
224  break;
225  } // end of the switch
226 
227  return segment;
228 }

376 {
377 
379  auto quadrant = new TGeoVolumeAssembly("Station 2 quadrant");
380 
382  const double kSegXPos[kNofSeg] = { kSegHalfLength[0] + kSecondInnerRadius, 0., -kSegHalfLength[2] };
383  const double kSegYPos[kNofSeg] = { -kSegHalfHeight[0], 0., kSegHalfHeight[2] + kSecondInnerRadius };
384 
385  for (int i = 0; i < kNofSeg; i++)
386  quadrant->AddNode(createSegment(i), 0, new TGeoTranslation(kSegXPos[i], kSegYPos[i], 0.));
387 
389  createFrames();
390 
391  // positions
392  const double kFrameShift = 3.7;
393  const double kFrameXPos[kNofFrames] = { kFrameLength[0] / 2. + kFirstInnerRadius,
394  kFrameLength[1] / 2. + kOuterRadius,
395  0.,
396  -kFrameLength[3] / 2.,
397  kFrameLength[4] / 2. - kFrameShift,
398  -kFrameLength[5] / 2.,
399  0.,
400  kFrameLength[7] / 2. + kFirstInnerRadius };
401 
402  const double kFrameYPos[kNofFrames] = { kFrameHeight[0] / 2. - kFrameShift,
403  -kFrameHeight[1] / 2.,
404  0.,
405  kFrameHeight[3] / 2. + kOuterRadius,
406  kFrameHeight[4] / 2. + kFirstInnerRadius,
407  kFrameHeight[5] / 2. + kFirstInnerRadius,
408  0.,
409  -kFrameHeight[7] / 2. };
410 
411  for (int i = 1; i <= kNofFrames; i++)
412  quadrant->AddNode(gGeoManager->GetVolume(Form("Frame %d", i)), 1,
413  new TGeoTranslation(kFrameXPos[i - 1], kFrameYPos[i - 1], 0.));
414 
415  return quadrant;
416 }

419 void createStation2Geometry(TGeoVolume& topVolume)
420 {
422 
423  // create a quadrant
424  auto quadrant = createQuadrant();
425 
426  const int kNQuad = 4;
427 
428  // rotation matrices to place the quadrants in the half-chambers
429  auto rot0 = new TGeoRotation();
430  auto rot1 = new TGeoRotation("reflXZ", 90., 180., 90., 90., 180., 0.);
431  auto rot2 = new TGeoRotation("reflXY", 90., 180., 90., 270., 0., 0.);
432  auto rot3 = new TGeoRotation("reflYZ", 90., 0., 90., -90., 180., 0.);
433  std::array<TGeoRotation*, kNQuad> rot = { rot0, rot1, rot2, rot3 };
434 
435  // Build the two chambers
436  double z = kQuadZpos;
437  int detElemID = 0;
438  const int kFirstChamberNumber = 3;
439 
440  for (int ich = kFirstChamberNumber; ich < kFirstChamberNumber + 2; ich++) {
441 
442  // create two half-chambers (new compared to AliRoot !)
443  auto in = new TGeoVolumeAssembly(Form("SC0%dI", ich));
444  auto out = new TGeoVolumeAssembly(Form("SC0%dO", ich));
445 
446  // Place the quadrant in the half-chambers
447  for (int i = 0; i < kNQuad; i++) {
448  // alternate the z position
449  z *= -1.;
450 
451  // compute the detection element ID
452  detElemID = 100 * ich + i;
453 
454  if (i == 0 || i == 3) {
455  in->AddNode(quadrant, detElemID, new TGeoCombiTrans(0., 0., z, rot[i]));
456  } else
457  out->AddNode(quadrant, detElemID, new TGeoCombiTrans(0., 0., z, rot[i]));
458  }
459 
460  // place the half-chambers in the top volume
461  topVolume.AddNode(in, 2 * (ich - 1), new TGeoTranslation(0., 0., kChamberZpos[ich - kFirstChamberNumber]));
462  topVolume.AddNode(out, 2 * ich - 1, new TGeoTranslation(0., 0., kChamberZpos[ich - kFirstChamberNumber]));
463 
464  } // end of the chamber loop
465 }
