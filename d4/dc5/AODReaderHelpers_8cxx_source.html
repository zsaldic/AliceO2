41  // FIXME: handle return code
42  arrow::Status Close() override { return arrow::Status::OK(); }
43  bool closed() const override { return false; }

45  arrow::Status Tell(int64_t* position) const override
46  {
47  *position = mPos;
48  return arrow::Status::OK();
49  }
50 
51  arrow::Status Read(int64_t nbytes, int64_t* bytes_read, void* out) override
52  {
53  auto count = fread(out, nbytes, 1, mStream);
54  if (ferror(mStream) == 0) {
55  *bytes_read = nbytes;
56  mPos += nbytes;
57  LOG(INFO) << mPos;
58  } else {
59  *bytes_read = 0;
60  }
61  return arrow::Status::OK();
62  }
63 
64  arrow::Status Read(int64_t nbytes, std::shared_ptr<arrow::Buffer>* out) override
65  {
66  std::shared_ptr<arrow::ResizableBuffer> buffer;
67  ARROW_RETURN_NOT_OK(AllocateResizableBuffer(nbytes, &buffer));
68  int64_t bytes_read;
69  ARROW_RETURN_NOT_OK(Read(nbytes, &bytes_read, buffer->mutable_data()));
70  ARROW_RETURN_NOT_OK(buffer->Resize(bytes_read, false));
71  buffer->ZeroPadding();
72  *out = buffer;
73  return arrow::Status::OK();
74  }
75 
76  private:
77  FILE* mStream = nullptr;
78  int64_t mPos = 0;
79 };

83 AlgorithmSpec AODReaderHelpers::run2ESDConverterCallback()
84 {
85  auto callback = AlgorithmSpec{ adaptStateful([](ConfigParamRegistry const& options,
86  ControlService& control,
87  DeviceSpec const& spec) {
88  std::vector<std::string> filenames;
89  auto filename = options.get<std::string>("esd-file");
90 
91  if (filename.empty()) {
92  LOG(error) << "Option --esd-file did not provide a filename";
93  control.readyToQuit(true);
94  return adaptStateless([](RawDeviceService& service) {
95  service.device()->WaitFor(std::chrono::milliseconds(1000));
96  });
97  }
98 
99  // If option starts with a @, we consider the file as text which contains a list of
100  // files.
101  if (filename.size() && filename[0] == '@') {
102  try {
103  filename.erase(0, 1);
104  std::ifstream filelist(filename);
105  while (std::getline(filelist, filename)) {
106  filenames.push_back(filename);
107  }
108  } catch (...) {
109  LOG(error) << "Unable to process file list: " << filename;
110  }
111  } else {
112  filenames.push_back(filename);
113  }
114 
115  bool readTracks = false;
116  bool readTracksCov = false;
117  bool readTracksExtra = false;
118  bool readCalo = false;
119  bool readMuon = false;
120  bool readVZ = false;
121 
122  // FIXME: bruteforce but effective.
123  for (auto& route : spec.outputs) {
124  if (route.matcher.origin != header::DataOrigin{ "RN2" }) {
125  continue;
126  }
127  auto description = route.matcher.description;
128  if (description == header::DataDescription{ "TRACKPAR" }) {
129  readTracks = true;
130  } else if (description == header::DataDescription{ "TRACKPARCOV" }) {
131  readTracksCov = true;
132  } else if (description == header::DataDescription{ "TRACKEXTRA" }) {
133  readTracksExtra = true;
134  } else if (description == header::DataDescription{ "CALO" }) {
135  readCalo = true;
136  } else if (description == header::DataDescription{ "MUON" }) {
137  readMuon = true;
138  } else if (description == header::DataDescription{ "VZERO" }) {
139  readVZ = true;
140  } else {
141  throw std::runtime_error(std::string("Unknown AOD type: ") + route.matcher.description.str);
142  }
143  }
144 
145  auto counter = std::make_shared<int>(0);
146  return adaptStateless([readTracks,
147  readTracksCov,
148  readTracksExtra,
149  readCalo,
150  readMuon,
151  readVZ,
152  counter,
153  filenames](DataAllocator& outputs, ControlService& ctrl, RawDeviceService& service) {
154  if (*counter >= filenames.size()) {
155  LOG(info) << "All input files processed";
156  ctrl.readyToQuit(false);
157  service.device()->WaitFor(std::chrono::milliseconds(1000));
158  return;
159  }
160  auto f = filenames[*counter];
161  setenv("O2RUN2CONVERTER", "run2ESD2Run3AOD", 0);
162  auto command = std::string(getenv("O2RUN2CONVERTER"));
163  FILE* pipe = popen((command + " " + f).c_str(), "r");
164  if (pipe == nullptr) {
165  LOG(ERROR) << "Unable to run converter: " << (command + " " + f).c_str() << f;
166  ctrl.readyToQuit(true);
167  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
168  return;
169  }
170  *counter += 1;
171  auto input = std::make_shared<FileStream>(pipe);
172 
174  while ((feof(pipe) == false) && (ferror(pipe) == 0)) {
175  std::shared_ptr<arrow::RecordBatchReader> reader;
176  auto readerStatus = arrow::ipc::RecordBatchStreamReader::Open(input.get(), &reader);
177  if (readerStatus.ok() == false) {
178  break;
179  }
180 
181  while (true) {
182  std::shared_ptr<arrow::RecordBatch> batch;
183  auto status = reader->ReadNext(&batch);
184  if (batch.get() == nullptr) {
185  LOG(INFO) << "End of batches reached";
186  break;
187  }
188  std::unordered_map<std::string, std::string> meta;
189  batch->schema()->metadata()->ToUnorderedMap(&meta);
190  LOG(INFO) << "description:" << meta["description"];
191  std::shared_ptr<arrow::ipc::RecordBatchWriter> writer;
192  if (meta["description"] == "TRACKPAR" && readTracks) {
193  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{ "RN2", "TRACKPAR" }, batch->schema());
194  } else if (meta["description"] == "TRACKPARCOV" && readTracksCov) {
195  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{ "RN2", "TRACKPARCOV" }, batch->schema());
196  } else if (meta["description"] == "TRACKEXTRA" && readTracksExtra) {
197  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{ "RN2", "TRACKEXTRA" }, batch->schema());
198  } else if (meta["description"] == "CALO" && readCalo) {
199  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{ "RN2", "CALO" }, batch->schema());
200  } else if (meta["description"] == "MUON" && readMuon) {
201  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{ "RN2", "MUON" }, batch->schema());
202  } else if (meta["description"] == "VZEOR" && readVZ) {
203  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{ "RN2", "VZERO" }, batch->schema());
204  } else {
205  continue;
206  }
207  auto writeStatus = writer->WriteRecordBatch(*batch);
208  if (writeStatus.ok() == false) {
209  throw std::runtime_error("Error while writing record");
210  }
211  }
212  }
213  pclose(pipe);
214  });
215  }) };
216 
217  return callback;
218 }
