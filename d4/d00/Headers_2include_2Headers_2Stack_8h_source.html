120  template <typename T, typename... Args>
121  static o2::byte* inject(o2::byte* here, T&& h, Args&&... args) noexcept
122  {
123  auto alsohere = inject(here, h);
124  // the type might be a stack itself, loop through headers and set the flag in the last one
125  if (h.size() > 0) {
126  BaseHeader* next = BaseHeader::get(here);
127  while (next->flagsNextHeader) {
128  next = next->next();
129  }
130  next->flagsNextHeader = hasNonEmptyArg(args...);
131  }
132  return inject(alsohere, args...);
133  }
