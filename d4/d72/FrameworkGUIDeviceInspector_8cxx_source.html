60 void optionsTable(const DeviceSpec& spec, const DeviceControl& control)
61 {
62  if (spec.options.empty()) {
63  return;
64  }
65  if (ImGui::CollapsingHeader("Options:", ImGuiTreeNodeFlags_DefaultOpen)) {
66  ImGui::Columns(2);
67  auto labels = { "Name", "Value" };
68  for (auto& label : labels) {
69  ImGui::TextUnformatted(label);
70  ImGui::NextColumn();
71  }
72  for (auto& option : spec.options) {
73  ImGui::TextUnformatted(option.name.c_str());
74  ImGui::NextColumn();
75  auto currentValueIt = control.options.find(option.name);
76 
77  // Did not find the option
78  if (currentValueIt == control.options.end()) {
79  switch (option.type) {
80  case VariantType::String:
81  ImGui::Text(R"("%s" (default))", option.defaultValue.get<const char*>());
82  break;
83  case VariantType::Int:
84  ImGui::Text("%d (default)", option.defaultValue.get<int>());
85  break;
86  case VariantType::Float:
87  ImGui::Text("%f (default)", option.defaultValue.get<float>());
88  break;
89  case VariantType::Double:
90  ImGui::Text("%f (default)", option.defaultValue.get<double>());
91  break;
92  case VariantType::Empty:
93  ImGui::TextUnformatted(""); // no default value
94  default:
95  ImGui::TextUnformatted("unknown");
96  }
97  } else {
98  ImGui::TextUnformatted(currentValueIt->second.c_str());
99  }
100  ImGui::NextColumn();
101  }
102  }
103  ImGui::Columns(1);
104 }
