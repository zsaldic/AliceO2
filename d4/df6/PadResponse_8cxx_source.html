42 bool PadResponse::importPRF(std::string file, std::unique_ptr<TGraph2D> & grPRF) const
43 {
44  std::string inputDir;
45  const char* aliceO2env=std::getenv("O2_ROOT");
46  if (aliceO2env) inputDir=aliceO2env;
47  inputDir+="/share/Detectors/TPC/files/";
48 
49  float x, y, normalizedPadResponse;
50  int i=0;
51  std::ifstream prfFile(inputDir+file, std::ifstream::in);
52  if(!prfFile) {
53  LOG(FATAL) << "TPC::PadResponse - Input file '" << inputDir+file << "' does not exist! No PRF loaded!" << FairLogger::endl;
54  return false;
55  }
56  for (std::string line; std::getline(prfFile, line); ) {
57  std::istringstream is(line);
58  while(is >> x >> y >> normalizedPadResponse) {
59  grPRF->SetPoint(i++, x, y, normalizedPadResponse);
60  }
61  }
62  return true;
63 }

74 
76 
77  const int gemStack = int(cru.gemStack());
78  const float offsetX = std::abs(posEle.X() - padCentre.X())*10.f;
79  const float offsetY = std::abs(posEle.Y() - padCentre.Y())*10.f;
80  float normalizedPadResponse = 0;
81  if(gemStack == 0) {
82  normalizedPadResponse = mIROC->Interpolate(offsetX, offsetY);

84  else if(gemStack == 1 || gemStack == 2) {
85  normalizedPadResponse = mOROC12->Interpolate(offsetX, offsetY);
