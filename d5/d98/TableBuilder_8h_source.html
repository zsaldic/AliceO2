102  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;

104  auto status = builder->Append();
105  auto valueBuilder = reinterpret_cast<ValueBuilderType*>(builder->value_builder());
106  status &= valueBuilder->AppendValues(&*ip.first, std::distance(ip.first, ip.second));

227 class TableBuilder
228 {
229  template <typename... ARGS>
230  void validate(std::vector<std::string> const& columnNames)
231  {
232  constexpr int nColumns = sizeof...(ARGS);
233  if (nColumns != columnNames.size()) {
234  throw std::runtime_error("Mismatching number of column types and names");
235  }
236  if (mBuilders != nullptr) {
237  throw std::runtime_error("TableBuilder::persist can only be invoked once per instance");
238  }
239  }
240 
241  template <typename... ARGS>
242  auto makeBuilders(std::vector<std::string> const& columnNames, size_t nRows)
243  {
244  using BuildersTuple = typename std::tuple<std::unique_ptr<typename BuilderTraits<ARGS>::BuilderType>...>;
245  mSchema = std::make_shared<arrow::Schema>(TableBuilderHelpers::makeFields<ARGS...>(columnNames));
246 
247  BuildersTuple* builders = new BuildersTuple(BuilderMaker<ARGS>::make(mMemoryPool)...);
248  if (nRows != -1) {
249  auto seq = std::make_index_sequence<sizeof...(ARGS)>{};
250  TableBuilderHelpers::reserveAll(*builders, nRows, seq);
251  }
252  mBuilders = builders; // We store the builders
253  }
254 
255  template <typename... ARGS>
256  auto makeFinalizer()
257  {
258  using BuildersTuple = typename std::tuple<std::unique_ptr<typename BuilderTraits<ARGS>::BuilderType>...>;
259  mFinalizer = [ schema = mSchema, &arrays = mArrays, builders = (BuildersTuple*)mBuilders ]()->std::shared_ptr<arrow::Table>
260  {
261  auto status = TableBuilderHelpers::finalize(arrays, *builders, std::make_index_sequence<sizeof...(ARGS)>{});
262  if (status == false) {
263  throw std::runtime_error("Unable to finalize");
264  }
265  return arrow::Table::Make(schema, arrays);
266  };
267  }

278  template <typename... ARGS>
279  auto persist(std::vector<std::string> const& columnNames)
280  {
281  using BuildersTuple = typename std::tuple<std::unique_ptr<typename BuilderTraits<ARGS>::BuilderType>...>;
282  constexpr int nColumns = sizeof...(ARGS);
283  validate<ARGS...>(columnNames);
284  mArrays.resize(nColumns);
285  makeBuilders<ARGS...>(columnNames, 1000);
286  makeFinalizer<ARGS...>();
287 
288  // Callback used to fill the builders
289  return [builders = (BuildersTuple*)mBuilders](unsigned int slot, typename BuilderMaker<ARGS>::FillType... args)->void
290  {
291  auto status = TableBuilderHelpers::append(*builders, std::index_sequence_for<ARGS...>{}, std::forward_as_tuple(args...));
292  if (status == false) {
293  throw std::runtime_error("Unable to append");
294  }
295  };
