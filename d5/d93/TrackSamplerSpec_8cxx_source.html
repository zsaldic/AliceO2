62  void run(framework::ProcessingContext& pc)
63  {
65 
66  int event(0);
67  mInputFile.read(reinterpret_cast<char*>(&event), SSizeOfInt);
68  if (mInputFile.fail()) {
69  return; // probably reached eof
70  }
71 
72  int size(0);
73  mInputFile.read(reinterpret_cast<char*>(&size), SSizeOfInt);
74  if (size == 0) {
75  LOG(INFO) << "event " << event + 1 << " is empty";
76  return; // skip empty event
77  }
78 
79  // create the output message
80  auto msgOut = pc.outputs().make<char>(Output{ "MCH", "TRACKS", 0, Lifetime::Timeframe }, SHeaderSize + size);
81  if (msgOut.size() != SHeaderSize + size) {
82  throw length_error("incorrect message payload");
83  }
84 
85  auto bufferPtr = msgOut.data();
86 
87  // fill header info
88  memcpy(bufferPtr, &event, SSizeOfInt);
89  bufferPtr += SSizeOfInt;
90  memcpy(bufferPtr, &size, SSizeOfInt);
91  bufferPtr += SSizeOfInt;
92 
93  // fill tracks and clusters info
94  mInputFile.read(bufferPtr, size);
95  }
96 
97  private:
98  static constexpr int SSizeOfInt = sizeof(int);
99  static constexpr int SHeaderSize = 2 * SSizeOfInt;
100 
101  std::ifstream mInputFile{};
102 };
