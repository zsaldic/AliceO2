26 void ClusterNativeHelper::convert(const char* fromFile, const char* toFile, const char* toTreeName)
27 {
28  Reader reader;
29  TreeWriter writer;
30  reader.init(fromFile);
31  writer.init(toFile, toTreeName);
32  size_t nEntries = reader.getTreeSize();
33  ClusterNativeAccessFullTPC clusterIndex;
34  int result = 0;
35  int nClusters = 0;
36  for (size_t entry = 0; entry < nEntries; ++entry) {
37  LOG(INFO) << "converting entry " << entry;
38  reader.read(entry);
39  result = reader.fillIndex(clusterIndex);
40  if (result >= 0) {
41  LOG(INFO) << "added " << result << " clusters to index";
42  } else {
43  LOG(ERROR) << "filling of clusters index failed with " << result;
44  }
45  result = writer.fillFrom(clusterIndex);
46  if (result >= 0) {
47  LOG(INFO) << "wrote " << result << " clusters to tree";
48  nClusters += result;
49  } else {
50  LOG(ERROR) << "filling of tree failed with " << result;
51  }
52  }
53  LOG(INFO) << "... done, converted " << nClusters << " clusters";
54  writer.close();
55 }

83 void ClusterNativeHelper::Reader::init(const char* filename, const char* treename)
84 {
85  if (treename != nullptr && treename[0] != 0) {
86  mTreeName = treename;
87  }
88  mFile.reset(TFile::Open(filename));
89  if (!mFile) {
90  return;
91  }
92  mTree = reinterpret_cast<TTree*>(mFile->GetObjectUnchecked(mTreeName.c_str()));
93  if (!mTree) {
94  LOG(ERROR) << "can not find tree " << mTreeName << " in file " << filename;
95  return;
96  }
97  size_t nofDataBranches = 0;
98  size_t nofMCBranches = 0;
99  for (size_t sector = 0; sector < NSectors; ++sector) {
100  auto branchname = mDataBranchName + "_" + std::to_string(sector);
101  TBranch* branch = mTree->GetBranch(branchname.c_str());
102  if (branch) {
103  TBranch* sizebranch = mTree->GetBranch((branchname + "Size").c_str());
104  if (sizebranch) {
105  branch->SetAddress(&mSectorRaw[sector]);
106  sizebranch->SetAddress(&mSectorRawSize[sector]);
107  ++nofDataBranches;
108  } else {
109  LOG(ERROR) << "can not find corresponding 'Size' branch for data branch " << branchname << ", skipping it";
110  }
111  }
112  branchname = mMCBranchName + " " + std::to_string(sector);
113  branch = mTree->GetBranch(branchname.c_str());
114  if (branch) {
115  branch->SetAddress(&mSectorMC);
116  ++nofMCBranches;
117  }
118  }
119  LOG(INFO) << "reading " << nofDataBranches << " data branch(es) and " << nofMCBranches << " mc branch(es)";
120 }

142 int ClusterNativeHelper::Reader::fillIndex(ClusterNativeAccessFullTPC& clusterIndex)
143 {
144  for (size_t index = 0; index < mSectorRaw.size(); ++index) {
145  if (mSectorRaw[index] && mSectorRaw[index]->size() != mSectorRawSize[index]) {
146  LOG(ERROR) << "inconsistent raw size for sector " << index << ": " << mSectorRaw[index]->size() << " v.s. " << mSectorRawSize[index];
147  mSectorRaw[index]->clear();
148  }
149  }
150  int result = fillIndex(clusterIndex, mSectorRaw, mSectorMC, [](auto&) { return true; });
151  return result;

154 int ClusterNativeHelper::Reader::parseSector(const char* buffer, size_t size, std::vector<MCLabelContainer>& mcinput, ClusterNativeAccessFullTPC& clusterIndex)
155 {
156  if (!buffer || size == 0) {
157  return 0;
158  }
159  auto mcIterator = mcinput.begin();
160  using ClusterGroupParser = o2::algorithm::ForwardParser<o2::TPC::ClusterGroupHeader>;
161  ClusterGroupParser parser;
162  size_t numberOfClusters = 0;
163  parser.parse(buffer, size,
164  [](const typename ClusterGroupParser::HeaderType& h) {
165  // check the header, but in this case there is no validity check
166  return true;
167  },
168  [](const typename ClusterGroupParser::HeaderType& h) {
169  // get the size of the frame including payload
170  // and header and trailer size, e.g. payload size
171  // from a header member
172  return h.nClusters * sizeof(ClusterNative) + ClusterGroupParser::totalOffset;
173  },
174  [&](typename ClusterGroupParser::FrameInfo& frame) {
175  int sector = frame.header->sector;
176  int padrow = frame.header->globalPadRow;
177  int nClusters = frame.header->nClusters;
178  clusterIndex.clusters[sector][padrow] = reinterpret_cast<const ClusterNative*>(frame.payload);
179  clusterIndex.nClusters[sector][padrow] = nClusters;
180  numberOfClusters += nClusters;
181  if (mcIterator != mcinput.end()) {
182  clusterIndex.clustersMCTruth[sector][padrow] = &(*mcIterator);
183  ++mcIterator;
184  }
185 
186  return true;
187  });
188  return numberOfClusters;
189 }

213 int ClusterNativeHelper::TreeWriter::fillFrom(ClusterNativeAccessFullTPC const& clusterIndex)
214 {
215  if (!mTree) {
216  return -1;
217  }
218  int result = 0;
219  for (size_t sector = 0; sector < o2::TPC::Constants::MAXSECTOR; ++sector) {
220  for (size_t padrow = 0; padrow < o2::TPC::Constants::MAXGLOBALPADROW; ++padrow) {
221  int locres = fillFrom(sector, padrow, clusterIndex.clusters[sector][padrow], clusterIndex.nClusters[sector][padrow]);
222  if (result >= 0 && locres >= 0) {
223  result += locres;
224  } else if (result >= 0) {
225  result = locres;
226  }
227  }
228  }
229  return result;
230 }

233  o2::dataformats::MCTruthContainer<o2::MCCompLabel>*)
234 {

239  if (clusters != nullptr && nClusters > 0) {
