57  auto initFunction = [](InitContext& ic) {
58  // FIXME: the clusterer needs to be initialized with the sector number, so we need one
59  // per sector. Taking a closer look to the HwClusterer, the sector number is only used
60  // for calculating the CRU id. This could be achieved by passing the current sector as
61  // parameter to the clusterer processing function.
62  auto processAttributes = std::make_shared<ProcessAttributes>();
63 
64  auto processSectorFunction = [processAttributes](ProcessingContext& pc, std::string inputKey, std::string labelKey) -> bool {
65  auto& clusterArray = processAttributes->clusterArray;
66  auto& mctruthArray = processAttributes->mctruthArray;
67  auto& clusterers = processAttributes->clusterers;
68  auto& verbosity = processAttributes->verbosity;
69  auto dataref = pc.inputs().get(inputKey);
70  auto const* sectorHeader = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(dataref);
71  if (sectorHeader == nullptr) {
72  LOG(ERROR) << "sector header missing on header stack";
73  return false;
74  }
75  auto const* dataHeader = DataRefUtils::getHeader<o2::header::DataHeader*>(dataref);
76  o2::header::DataHeader::SubSpecificationType fanSpec = dataHeader->subSpecification;
77 
78  const auto sector = sectorHeader->sector;
79  if (sector < 0) {
80  // forward the control information
81  // FIXME define and use flags in TPCSectorHeader
82  o2::TPC::TPCSectorHeader header{ sector };
83  pc.outputs().snapshot(Output{ gDataOriginTPC, "CLUSTERHW", fanSpec, Lifetime::Timeframe, { header } }, fanSpec);
84  if (!labelKey.empty()) {
85  pc.outputs().snapshot(Output{ gDataOriginTPC, "CLUSTERHWMCLBL", fanSpec, Lifetime::Timeframe, { header } }, fanSpec);
86  }
87  return (sectorHeader->sector == -1);
88  }
89  std::unique_ptr<const MCLabelContainer> inMCLabels;
90  if (!labelKey.empty()) {
91  inMCLabels = std::move(pc.inputs().get<const MCLabelContainer*>(labelKey.c_str()));
92  }
93  auto inDigits = pc.inputs().get<const std::vector<o2::TPC::Digit>>(inputKey.c_str());
94  if (verbosity > 0 && inMCLabels) {
95  LOG(INFO) << "received " << inDigits.size() << " digits, "
96  << inMCLabels->getIndexedSize() << " MC label objects";
97  }
98  if (!clusterers[sector]) {
99  // create the clusterer for this sector, take the same target arrays for all clusterers
100  // as they are not invoked in parallel
101  // the cost of creating the clusterer should be small so we do it in the processing
102  clusterers[sector] = std::make_shared<o2::TPC::HwClusterer>(&clusterArray, sector, &mctruthArray);
103  }
104  auto& clusterer = clusterers[sector];
105 
106  if (verbosity > 0) {
107  LOG(INFO) << "processing " << inDigits.size() << " digit object(s) of sector " << sectorHeader->sector;
108  }
109  // process the digits and MC labels, the bool parameter controls whether to clear all
110  // internal data or not. Have to clear it inside the process method as not only the containers
111  // are cleared but also the cluster counter. Clearing the containers externally leaves the
112  // cluster counter unchanged and leads to an inconsistency between cluster container and
113  // MC label container (the latter just grows with every call).
114  clusterer->process(inDigits, inMCLabels.get(), true /* clear output containers and cluster counter */);
115  const std::vector<o2::TPC::Digit> emptyDigits;
116  clusterer->finishProcess(emptyDigits, nullptr, false); // keep here the false, otherwise the clusters are lost of they are not stored in the meantime
117  if (verbosity > 0) {
118  LOG(INFO) << "clusterer produced "
119  << std::accumulate(clusterArray.begin(), clusterArray.end(), size_t(0), [](size_t l, auto const& r) { return l + r.getContainer()->numberOfClusters; })
120  << " cluster(s)";
121  if (!labelKey.empty()) {
122  LOG(INFO) << "clusterer produced " << mctruthArray.getIndexedSize() << " MC label object(s)";
123  }
124  }
125  // FIXME: that should be a case for pmr, want to send the content of the vector as a binary
126  // block by using move semantics
127  auto outputPages = pc.outputs().make<ClusterHardwareContainer8kb>(Output{ gDataOriginTPC, "CLUSTERHW", fanSpec, Lifetime::Timeframe, { *sectorHeader } }, clusterArray.size());
128  std::copy(clusterArray.begin(), clusterArray.end(), outputPages.begin());
129  if (!labelKey.empty()) {
130  pc.outputs().snapshot(Output{ gDataOriginTPC, "CLUSTERHWMCLBL", fanSpec, Lifetime::Timeframe, { *sectorHeader } }, mctruthArray);
131  }
132  return false;
133  };

135  auto processingFct = [processAttributes, processSectorFunction](ProcessingContext& pc) {
136  if (processAttributes->finished) {
137  return;
138  }
139 
140  struct SectorInputDesc {
141  std::string inputKey = "";
142  std::string labelKey = "";
143  };
144  std::map<o2::header::DataHeader::SubSpecificationType, SectorInputDesc> inputs;
145  for (auto const& inputRef : pc.inputs()) {
146  auto const* dataHeader = DataRefUtils::getHeader<o2::header::DataHeader*>(inputRef);
147  assert(dataHeader);
148  if (dataHeader->dataOrigin == gDataOriginTPC && dataHeader->dataDescription == o2::header::DataDescription("DIGITS")) {
149  inputs[dataHeader->subSpecification].inputKey = inputRef.spec->binding;
150  } else if (dataHeader->dataOrigin == gDataOriginTPC && dataHeader->dataDescription == o2::header::DataDescription("DIGITSMCTR")) {
151  inputs[dataHeader->subSpecification].labelKey = inputRef.spec->binding;
152  }
153  }
154  bool finished = true;
155  for (auto const& input : inputs) {
156  if (!processSectorFunction(pc, input.second.inputKey, input.second.labelKey)) {
157  finished = false;
158  }
159  }
160  if (finished) {
161  // got EOD on all inputs
162  processAttributes->finished = true;
163  pc.services().get<ControlService>().readyToQuit(false);
164  }
165  };
166  return processingFct;
167  };

169  auto createInputSpecs = [](bool makeMcInput, bool makeTriggersInput = false) {
170  std::vector<InputSpec> inputSpecs{
171  InputSpec{ "digits", gDataOriginTPC, "DIGITS", 0, Lifetime::Timeframe },
172  };
173  if (makeMcInput) {
174  constexpr o2::header::DataDescription datadesc("DIGITSMCTR");
175  inputSpecs.emplace_back("mclabels", gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
176  }
177  if (makeTriggersInput) {
178  // this is an additional output by the TPC digitizer, need to check if that
179  // has to go into the clusterer as well
180  constexpr o2::header::DataDescription datadesc("DIGTRIGGERS");
181  inputSpecs.emplace_back("digtriggers", gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
182  }
183  return std::move(inputSpecs);
184  };
185 
186  auto createOutputSpecs = [](bool makeMcOutput) {
187  std::vector<OutputSpec> outputSpecs{
188  OutputSpec{ { "clusters" }, gDataOriginTPC, "CLUSTERHW", 0, Lifetime::Timeframe },
189  };
190  if (makeMcOutput) {
191  OutputLabel label{ "clusterlbl" };
192  // FIXME: define common data type specifiers
193  constexpr o2::header::DataDescription datadesc("CLUSTERHWMCLBL");
194  outputSpecs.emplace_back(label, gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
195  }
196  return std::move(outputSpecs);
197  };
198 
199  return DataProcessorSpec{ processorName,
200  { createInputSpecs(sendMC, haveDigTriggers) },
201  { createOutputSpecs(sendMC) },
202  AlgorithmSpec(initFunction) };
