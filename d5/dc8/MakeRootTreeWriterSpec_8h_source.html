265  // the init function is returned to the DPL in order to init the process
266  auto initFct = [processAttributes, TerminationPolicyMap = TerminationPolicyMap](InitContext& ic) {
267  auto& branchNameOptions = processAttributes->branchNameOptions;
268  auto filename = ic.options().get<std::string>("outfile");
269  auto treename = ic.options().get<std::string>("treename");
270  processAttributes->nEvents = ic.options().get<int>("nevents");
271  if (processAttributes->nEvents > 0 && processAttributes->activeInputs.size() != processAttributes->nofBranches) {
272  LOG(WARNING) << "the n inputs serve in total m branches with n != m, this means that there will be data for\n"
273  << "different branches on the same input. Be aware that the --nevents option might lead to incomplete\n"
274  << "data in the output file as the number of processed input sets is counted";
275  }
276  try {
277  processAttributes->terminationPolicy = TerminationPolicyMap.at(ic.options().get<std::string>("terminate"));
278  } catch (std::out_of_range&) {
279  throw std::invalid_argument(std::string("invalid termination policy: ") + ic.options().get<std::string>("terminate"));
280  }
281  if (filename.empty() || treename.empty()) {
282  throw std::invalid_argument("output file name and tree name are mandatory options");
283  }
284  for (size_t branchIndex = 0; branchIndex < branchNameOptions.size(); branchIndex++) {
285  // pair of key (first) - value (second)
286  if (branchNameOptions[branchIndex].first.empty()) {
287  continue;
288  }
289  auto branchName = ic.options().get<std::string>(branchNameOptions[branchIndex].first.c_str());
290  processAttributes->writer->setBranchName(branchIndex, branchName.c_str());
291  }
292  processAttributes->writer->init(filename.c_str(), treename.c_str());
293 
294  // the callback to be set as hook at stop of processing for the framework
295  auto finishWriting = [processAttributes]() {
296  processAttributes->writer->close();
297  };
298  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);

300  auto processingFct = [processAttributes](ProcessingContext& pc) {
301  auto& writer = processAttributes->writer;
302  auto& terminationPolicy = processAttributes->terminationPolicy;
303  auto& terminationCondition = processAttributes->terminationCondition;
304  auto& activeInputs = processAttributes->activeInputs;
305  auto& counter = processAttributes->counter;
306  auto& nEvents = processAttributes->nEvents;
307  if (writer->isClosed()) {
308  return;
309  }
310 
311  // if the termination condition contains function of type CheckProcessing, this is checked
312  // before the processing, currently implemented logic:
313  // - processing is skipped if this is indicated at least one input
314  // - treated as 'ready' if this is indicated at least one input
315  auto checkProcessing = [&terminationCondition, &activeInputs](auto const& inputs) {
316  bool doProcessing = true;
317  if (std::holds_alternative<TerminationCondition::CheckProcessing>(terminationCondition.check)) {
318  auto& check = std::get<TerminationCondition::CheckProcessing>(terminationCondition.check);
319  for (auto const& ref : inputs) {
320  bool ready = false;
321  auto iter = activeInputs.find(ref.spec->binding);
322  if (check(ref, ready) == TerminationCondition::Action::SkipProcessing) {
323  // this condition tells us not to process the data any further
324  doProcessing = false;
325  }
326  if (iter != activeInputs.end() && ready) {
327  // this input is ready, remove from active inputs
328  activeInputs.erase(iter);
329  }
330  }
331  }
332  return doProcessing;
333  };
334  // if the termination condition contains function of type CheckReady, this is checked
335  // after the processing, treated as 'ready' if indicated by at least one input
336  auto checkReady = [&terminationCondition, &activeInputs](auto const& inputs) {
337  if (std::holds_alternative<TerminationCondition::CheckReady>(terminationCondition.check)) {
338  auto& check = std::get<TerminationCondition::CheckReady>(terminationCondition.check);
339  for (auto const& ref : inputs) {
340  auto iter = activeInputs.find(ref.spec->binding);
341  if (iter != activeInputs.end() && check(ref)) {
342  // this input is ready, remove from active inputs
343  activeInputs.erase(iter);
344  }
345  }
346  }
347  return activeInputs.size() == 0;
348  };
349 
350  if (checkProcessing(pc.inputs())) {
351  (*writer)(pc.inputs());
352  counter = counter + 1;
353  }
354 
355  if ((nEvents >= 0 && counter == nEvents) || checkReady(pc.inputs())) {
356  writer->close();
357  pc.services().get<ControlService>().readyToQuit(terminationPolicy == TerminationPolicy::Workflow);
358  }
359  };
360 
361  return processingFct;
362  };
