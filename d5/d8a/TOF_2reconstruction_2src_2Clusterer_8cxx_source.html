40 void Clusterer::processStrip(std::vector<Cluster>& clusters, MCLabelContainer const* digitMCTruth)
41 {
42  // method to clusterize the current strip
43 
44  Int_t detId[5];
45  Int_t chan, chan2, chan3;
46  Int_t strip1, strip2;
47  Int_t iphi, iphi2, iphi3;
48  Int_t ieta, ieta2, ieta3; // it is the number of padz-row increasing along the various strips
49 
50  for (int idig = 0; idig < mStripData.digits.size(); idig++) {
51  // LOG(DEBUG) << "Checking digit " << idig;
52  Digit* dig = &mStripData.digits[idig];
53  if (dig->isUsedInCluster())
54  continue; // the digit was already used to build a cluster
55 
56  mNumberOfContributingDigits = 0;
57  dig->getPhiAndEtaIndex(iphi, ieta);
58  if (mStripData.digits.size() > 1)
59  LOG(DEBUG) << "idig = " << idig;
60 
61  // first we make a cluster out of the digit
62  int noc = clusters.size();
63  // LOG(DEBUG) << "noc = " << noc << "\n";
64  clusters.emplace_back();
65  Cluster& c = clusters[noc];
66  addContributingDigit(dig);
67  float timeDig = dig->getTDC() * Geo::TDCBIN;
68 
69  for (int idigNext = idig + 1; idigNext < mStripData.digits.size(); idigNext++) {
70  Digit* digNext = &mStripData.digits[idigNext];
71  if (digNext->isUsedInCluster())
72  continue; // the digit was already used to build a cluster
73  // check if the TOF time are close enough to be merged; if not, it means that nothing else will contribute to the cluster (since digits are ordered in time)
74  float timeDigNext = digNext->getTDC() * Geo::TDCBIN; // we assume it calibrated (for now); in ps
75  LOG(DEBUG) << "Time difference = " << timeDigNext - timeDig;
76  if (timeDigNext - timeDig > 500 /*in ps*/)
77  break;
78  digNext->getPhiAndEtaIndex(iphi2, ieta2);
79 
80  // check if the fired pad are close in space
81  LOG(DEBUG) << "phi difference = " << iphi - iphi2;
82  LOG(DEBUG) << "eta difference = " << ieta - ieta2;
83  if ((TMath::Abs(iphi - iphi2) > 1) || (TMath::Abs(ieta - ieta2) > 1))
84  continue;
85 
86  // if we are here, the digit contributes to the cluster
87  addContributingDigit(digNext);
88 
89  } // loop on the second digit
90 
91  buildCluster(c, digitMCTruth);
92 
93  } // loop on the first digit
94 }
