30 ChipPixelData* DigitPixelReader::getNextChipData(std::vector<ChipPixelData>& chipDataVec)
31 {
32  // decode data of single chip to corresponding slot of chipDataVec
33  if (!mLastDigit) { // new ROF record should be started
34  if (mIdDig >= mDigits->size()) { // nothing left
35  return nullptr;
36  }
37  mLastDigit = &((*mDigits)[mIdDig++]);
38  }
39  auto chipID = mLastDigit->getChipIndex();
40  return getNextChipData(chipDataVec[chipID]) ? &chipDataVec[chipID] : nullptr;

44 bool DigitPixelReader::getNextChipData(ChipPixelData& chipData)
45 {
46  // decode data of single chip to chipData
47  if (!mLastDigit) { // new ROF record should be started
48  if (mIdDig >= mDigits->size()) { // nothing left
49  return false;
50  }
51  mLastDigit = &((*mDigits)[mIdDig++]);
52  }
53  // get corresponding ROF record
54  int lim = -1;
55  if (mIdROF >= 0) {
56  const auto& rofRec = (*mROFRecVec)[mIdROF];
57  lim = rofRec.getROFEntry().getIndex() + rofRec.getNROFEntries();
58  }
59  while (mIdDig > lim) {
60  const auto& rofRec = (*mROFRecVec)[++mIdROF];
61  lim = rofRec.getROFEntry().getIndex() + rofRec.getNROFEntries();
62  mInteractionRecord = rofRec.getBCData(); // update interaction record
63  }
64  chipData.clear();
65  chipData.setStartID(mIdDig - 1); // for the MC references
66  chipData.setChipID(mLastDigit->getChipIndex());
67  chipData.setROFrame(mLastDigit->getROFrame());
68  chipData.setInteractionRecord(mInteractionRecord);
69  chipData.setTrigger(mTrigger);
70  chipData.getData().emplace_back(mLastDigit);
71  mLastDigit = nullptr;
72 
73  for (; mIdDig < lim;) {
74  mLastDigit = &((*mDigits)[mIdDig++]);
75  if (mLastDigit->getChipIndex() != chipData.getChipID()) { // new chip starts
76  return true;
77  }
78  chipData.getData().emplace_back(mLastDigit);
79  mLastDigit = nullptr; // reset pointer of already used digit
80  }
81  return true;
82 }

85 void DigitPixelReader::openInput(const std::string inpName, o2::detectors::DetID det)
86 {
87  // open input file, load digits, MC labels
88  assert(det.getID() == o2::detectors::DetID::ITS || det.getID() == o2::detectors::DetID::MFT);
89 
90  clear();
91  std::string detName = det.getName();
92 
93  if (!(mInputTree = o2::utils::RootChain::load("o2sim", inpName))) {
94  LOG(FATAL) << "Failed to load Digits tree from " << inpName << FairLogger::endl;
95  }
96  mInputTree->SetBranchAddress((detName + "Digit").c_str(), &mDigitsSelf);
97  if (!mDigitsSelf) {
98  LOG(FATAL) << "Failed to find " << (detName + "Digit").c_str() << " branch in the " << mInputTree->GetName()
99  << " from file " << inpName << FairLogger::endl;
100  }
101  setDigits(mDigitsSelf);
102 
103  if (!(mInputTreeROF = o2::utils::RootChain::load((detName + "DigitROF").c_str(), inpName))) {
104  LOG(FATAL) << "Failed to load ROF records tree from " << inpName << FairLogger::endl;
105  }
106  mInputTreeROF->SetBranchAddress((detName + "DigitROF").c_str(), &mROFRecVecSelf);
107  if (!mROFRecVecSelf) {
108  LOG(FATAL) << "Failed to find " << (detName + "DigitROF").c_str() << " branch in the " << mInputTree->GetName()
109  << " from file " << inpName << FairLogger::endl;
110  }
111  setROFRecords(mROFRecVecSelf);
112 
113  if (!(mInputTreeMC2ROF = o2::utils::RootChain::load((detName + "DigitMC2ROF").c_str(), inpName))) {
114  LOG(FATAL) << "Failed to load MC2ROF records tree from " << inpName << FairLogger::endl;
115  }
116  mInputTreeMC2ROF->SetBranchAddress((detName + "DigitMC2ROF").c_str(), &mMC2ROFRecVecSelf);
117  if (!mMC2ROFRecVecSelf) {
118  LOG(FATAL) << "Failed to find " << (detName + "DigitMC2ROF").c_str() << " branch in the " << mInputTree->GetName()
119  << " from file " << inpName << FairLogger::endl;
120  }
121  setMC2ROFRecords(mMC2ROFRecVecSelf);
122 
123  mInputTree->SetBranchAddress((detName + "DigitMCTruth").data(), &mDigitsMCTruthSelf);
124  setDigitsMCTruth(mDigitsMCTruthSelf);
125 }

128 bool DigitPixelReader::readNextEntry()
129 {
130  // load next entry from the self-managed input
131  auto nev = mInputTree->GetEntries();
132  if (mInputTreeROF->GetEntries() != nev || nev != 1) {
133  LOG(FATAL) << "In the self-managed mode the Digits and ROFRecords trees must have 1 entry only";
134  }
135  auto evID = mInputTree->GetReadEntry();
136  if (evID < -1)
137  evID = -1;
138  if (++evID < nev) {
139  init();
140  mInputTree->GetEntry(evID);
141  mInputTreeROF->GetEntry(evID);
142  if (evID == 0) {
143  mInputTreeMC2ROF->GetEntry(0); // onle one entry is expected
144  }
145  return true;
146  } else {
147  return false;
148  }
149 }
