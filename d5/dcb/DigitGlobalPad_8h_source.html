122 inline bool DigitGlobalPad::compareMClabels(const MCCompLabel& label1, const MCCompLabel& label2) const
123 {
124  // we compare directly on the bare label (in which eventID, labelID etc. are encoded)
125  // this avoids any logical operator on the label; optimization motivated from an Intel VTune analysis
126  // (note that this is also faster than using the operator= of MCCompLabel)
127  return label1.getRawValue() == label2.getRawValue();
128 }

130 template <DigitzationMode MODE>
131 inline void DigitGlobalPad::fillOutputContainer(std::vector<Digit>& output,
132  dataformats::MCTruthContainer<MCCompLabel>& mcTruth,
133  const CRU& cru, TimeBin timeBin,
134  GlobalPadNumber globalPad,
135  o2::dataformats::LabelContainer<std::pair<MCCompLabel, int>, false>& labels,
136  float commonMode)
137 {
138  const static Mapper& mapper = Mapper::instance();
139  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
140  const PadPos pad = mapper.padPos(globalPad);
141  static std::vector<std::pair<MCCompLabel, int>> labelCollector; // static workspace container for sorting
142 
145  const float totalADC = mChargePad - commonMode; // common mode is subtracted here in order to properly apply noise,
146  // pedestals and saturation of the SAMPA
147 
148  float noise, pedestal;
149  const float mADC = sampaProcessing.makeSignal<MODE>(totalADC, cru.sector(), globalPad, pedestal, noise);
150 
152  if (mADC > 0 && mChargePad > 0) {
153  auto labelview = labels.getLabels(mID);
154 
156  const auto digiPos = output.size();
157  output.emplace_back(cru, mADC, pad.getRow(), pad.getPad(), timeBin);
158 
159  labelCollector.clear();
160  for (auto& mcLabel : labelview) {
161  labelCollector.push_back(mcLabel);
162  }
163  if (labelview.size() > 1) {
165  using P = std::pair<MCCompLabel, int>;
166  std::sort(labelCollector.begin(), labelCollector.end(), [](const P& a, const P& b) { return a.second > b.second; });
167  }
168  for (auto& mcLabel : labelCollector) {
169  mcTruth.addElement(digiPos, mcLabel.first);
170  }
171  }
172 }
