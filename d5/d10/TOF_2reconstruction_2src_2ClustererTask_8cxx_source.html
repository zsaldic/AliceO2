49 InitStatus ClustererTask::Init()
50 {
51  FairRootManager* mgr = FairRootManager::Instance();
52  if (!mgr) {
53  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
54  return kERROR;
55  }
56 
57  const std::vector<o2::tof::Digit>* arr = mgr->InitObjectAs<const std::vector<o2::tof::Digit>*>("TOFDigit");
58  if (!arr) {
59  LOG(ERROR) << "TOF digits not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
60  return kERROR;
61  }
62  mReader.setDigitArray(arr);
63 
64  if (mClsLabels) { // here we take the array of labels used for the digits
65  mDigitMCTruth =
66  mgr->InitObjectAs<const dataformats::MCTruthContainer<MCCompLabel>*>("TOFDigitMCTruth");
67  if (!mDigitMCTruth) {
68  LOG(ERROR) << "TOF MC Truth not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
69  return kERROR;
70  }
71  }
72 
73  // Register output container
74  mgr->RegisterAny("TOFCluster", mClustersArray, kTRUE);
75 
76  // Register new MC Truth container --> here we will now associate to the clusters all labels that belonged to all digits that formed that cluster
77  if (mClsLabels)
78  mgr->RegisterAny("TOFClusterMCTruth", mClsLabels, kTRUE);
79 
80  mClusterer.setMCTruthContainer(mClsLabels);
81 
82  return kSUCCESS;
83 }
