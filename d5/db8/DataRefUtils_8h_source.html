33 struct DataRefUtils {
34  // SFINAE makes this available only for the case we are using
35  // trivially copyable type, this is to distinguish it from the
36  // alternative below, which works for TObject (which are serialised).
37  template <typename T>
38  static typename std::enable_if<is_messageable<T>::value == true, gsl::span<T>>::type
39  as(DataRef const& ref)
40  {
41  using DataHeader = o2::header::DataHeader;
42  auto header = o2::header::get<const DataHeader*>(ref.header);
43  if (header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
44  throw std::runtime_error("Attempt to extract a POD from a wrong message kind");
45  }
46  if ((header->payloadSize % sizeof(T)) != 0) {
47  throw std::runtime_error("Cannot extract POD from message as size do not match");
48  }
49  //FIXME: provide a const collection
50  return gsl::span<T>(reinterpret_cast<T *>(const_cast<char *>(ref.payload)), header->payloadSize/sizeof(T));
51  }
52 
53  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
54  // with a somewhat uniform API.
55  // Classes using the ROOT ClassDef/ClassImp macros can be detected automatically
56  // and the serialization method can be deduced. If the class is also
57  // messageable, gSerializationMethodNone is used by default. This substitution
58  // does not apply for such types, the serialization method needs to be specified
59  // explicitely using type wrapper @a ROOTSerialized.
60  template <typename T>
61  static typename std::enable_if<has_root_dictionary<T>::value == true &&
62  is_messageable<T>::value == false,
63  std::unique_ptr<T>>::type
64  as(DataRef const& ref)
65  {
66  using DataHeader = o2::header::DataHeader;
67  auto header = o2::header::get<const DataHeader*>(ref.header);
68  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
69  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
70  }
71 
72  o2::framework::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
73  auto* storedClass = ftm.GetClass();
74  auto* requestedClass = TClass::GetClass(typeid(T));
75  // should always have the class description if has_root_dictionary is true
76  assert(requestedClass != nullptr);
77 
78  auto* object = ftm.ReadObjectAny(storedClass);
79  if (object == nullptr) {
80  std::ostringstream ss;
81  ss << "Failed to read object with name "
82  << (storedClass != nullptr ? storedClass->GetName() : "<unknown>")
83  << " from message using ROOT serialization.";
84  throw std::runtime_error(ss.str());
85  }
86 
87  std::unique_ptr<T> result;
88  if (std::is_base_of<TObject, T>::value) { // compile time switch
89  // if the type to be extracted inherits from TObject, the class descriptions
90  // do not need to match exactly, only the dynamic_cast has to work
91  // FIXME: could probably try to extend this to arbitrary types T being a base
92  // to the stored type, but this would require to cast the extracted object to
93  // the actual type. This requires this information to be available as a type
94  // in the ROOT dictionary, check if this is the case or if TClass::DynamicCast
95  // can be used for the test. Right now, this case was and is not covered and
96  // not yet checked in the unit test either.
97  auto* r = dynamic_cast<T*>(static_cast<TObject*>(object));
98  if (r) {
99  result.reset(r);
100  }
101  } else if (storedClass == requestedClass) {
102  result.reset(static_cast<T*>(object));
103  }
104  if (result == nullptr) {
105  // did not manage to cast the pointer to result
106  // delete object via the class info if available, not the case for all types,
107  // e.g. for standard containers of ROOT objects apparently this is not always
108  // the case
109  auto* delfunc = storedClass->GetDelete();
110  if (delfunc) {
111  (*delfunc)(object);
112  }
113 
114  std::ostringstream ss;
115  ss << "Attempting to extract a "
116  << (requestedClass != nullptr ? requestedClass->GetName() : "<unknown>")
117  << " but a "
118  << (storedClass != nullptr ? storedClass->GetName() : "<unknown>")
119  << " is actually stored which cannot be casted to the requested one.";
120  throw std::runtime_error(ss.str());
121  }
122 
123  return std::move(result);
124  }

126  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
127  // with a somewhat uniform API. ROOT serialization method is enforced by using
128  // type wrapper @a ROOTSerialized
129  template <typename W>
130  static typename std::enable_if<is_specialization<W, ROOTSerialized>::value == true,
131  std::unique_ptr<typename W::wrapped_type>>::type
132  as(DataRef const& ref)
133  {
134  using T = typename W::wrapped_type;
135  using DataHeader = o2::header::DataHeader;
136  auto header = o2::header::get<const DataHeader*>(ref.header);
137  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
138  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
139  }
140  auto* cl = TClass::GetClass(typeid(T));
141  if (has_root_dictionary<T>::value == false && cl == nullptr) {
142  throw std::runtime_error("ROOT serialization not supported, dictionary not found for data type");
143  }
144 
145  o2::framework::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
146  std::unique_ptr<T> result(static_cast<T*>(ftm.ReadObjectAny(cl)));
147  if (result.get() == nullptr) {
148  std::ostringstream ss;
149  ss << "Unable to extract class ";
150  if (cl == nullptr) {
151  ss << "<name not available>";
152  } else {
153  ss << cl->GetName();
154  }
155  throw std::runtime_error(ss.str());
156  }
157  return std::move(result);
158  }
