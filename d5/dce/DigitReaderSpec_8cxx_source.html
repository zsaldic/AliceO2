44 void DigitReader::run(ProcessingContext& pc)
45 {
46  if (mState != 1) {
47  return;
48  }
49 
50  std::unique_ptr<TTree> treeDig((TTree*)mFile->Get("o2sim"));
51  std::unique_ptr<TTree> treeROF((TTree*)mFile->Get("ITSDigitROF"));
52  std::unique_ptr<TTree> treeMC2ROF((TTree*)mFile->Get("ITSDigitMC2ROF"));
53 
54  if (treeDig && treeROF && treeMC2ROF) {
55 
56  std::vector<o2::ITSMFT::Digit> allDigits;
57  std::vector<o2::ITSMFT::Digit> digits, *pdigits = &digits;
58 
59  treeDig->SetBranchAddress("ITSDigit", &pdigits);
60  o2::dataformats::MCTruthContainer<o2::MCCompLabel> allLabels;
61  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labels, *plabels = &labels;
62  treeDig->SetBranchAddress("ITSDigitMCTruth", &plabels);
63 
64  std::vector<ROFRecord> rofs, *profs = &rofs;
65  treeROF->SetBranchAddress("ITSDigitROF", &profs);
66  treeROF->GetEntry(0);
67 
68  std::vector<MC2ROFRecord> mc2rofs, *pmc2rofs = &mc2rofs;
69  treeMC2ROF->SetBranchAddress("ITSDigitMC2ROF", &pmc2rofs);
70  treeMC2ROF->GetEntry(0);
71 
72  int ne = treeDig->GetEntries();
73  for (int e = 0; e < ne; e++) { // RS normally we should not have multiple entries here, if it happens, the references will be wrong
74  treeDig->GetEntry(e);
75  std::copy(digits.begin(), digits.end(), std::back_inserter(allDigits));
76  allLabels.mergeAtBack(labels);
77  }
78  LOG(INFO) << "ITSDigitReader pushed " << allDigits.size() << " digits, in "
79  << profs->size() << " RO frames and "
80  << pmc2rofs->size() << " MC events";
81  pc.outputs().snapshot(Output{ "ITS", "DIGITS", 0, Lifetime::Timeframe }, allDigits);
82  pc.outputs().snapshot(Output{ "ITS", "DIGITSMCTR", 0, Lifetime::Timeframe }, allLabels);
83  pc.outputs().snapshot(Output{ "ITS", "ITSDigitROF", 0, Lifetime::Timeframe }, *profs);
84  pc.outputs().snapshot(Output{ "ITS", "ITSDigitMC2ROF", 0, Lifetime::Timeframe }, *pmc2rofs);
85  } else {
86  LOG(ERROR) << "Cannot read the ITS digits !";
87  return;
88  }
89  mState = 2;
90  //pc.services().get<ControlService>().readyToQuit(true);
91 }
