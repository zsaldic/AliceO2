47 DataProcessorSpec getTOFClusterWriterSpec()
48 {
49  auto initFunction = [](InitContext& ic) {
50  // get the option from the init context
51  auto filename = ic.options().get<std::string>("tof-cluster-outfile");
52  auto treename = ic.options().get<std::string>("treename");
53 
54  auto outputfile = std::make_shared<TFile>(filename.c_str(), "RECREATE");
55  auto outputtree = std::make_shared<TTree>(treename.c_str(), treename.c_str());
56 
57  // container for incoming data
58  auto digits = std::make_shared<std::vector<o2::tof::Cluster>>();
59 
60  // the callback to be set as hook at stop of processing for the framework
61  auto finishWriting = [outputfile, outputtree]() {
62  outputtree->SetEntries(1);
63  outputtree->Write();
64  outputfile->Close();
65  };
66  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
67 
68  // setup the processing function
69  // using by-copy capture of the worker instance shared pointer
70  // the shared pointer makes sure to clean up the instance when the processing
71  // function gets out of scope
72  auto processingFct = [outputfile, outputtree, digits](ProcessingContext& pc) {
73  static bool finished = false;
74  if (finished) {
75  // avoid being executed again when marked as finished;
76  return;
77  }
78 
79  // retrieve the digits from the input
80  auto indata = pc.inputs().get<std::vector<o2::tof::Cluster>>("tofclusters");
81  LOG(INFO) << "RECEIVED CLUSTERS SIZE " << indata.size();
82  *digits.get() = std::move(indata);
83 
84  // connect this to a particular branch
85  auto br = getOrMakeBranch(*outputtree.get(), "TOFCluster", digits.get());
86  br->Fill();
87 
88  // retrieve labels from the input
89  auto labeldata = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("tofclusterlabels");
90  LOG(INFO) << "TOF GOT " << labeldata->getNElements() << " LABELS ";
91  auto labeldataraw = labeldata.get();
92  // connect this to a particular branch
93  auto labelbr = getOrMakeBranch(*outputtree.get(), "TOFClusterMCTruth", &labeldataraw);
94  labelbr->Fill();
95 
96  finished = true;
97  pc.services().get<ControlService>().readyToQuit(false);
98  };
99 
100  // return the actual processing function as a lambda function using variables
101  // of the init function
102  return processingFct;
103  };
104 
105  return DataProcessorSpec{
106  "TOFClusterWriter",
107  Inputs{ InputSpec{ "tofclusters", "TOF", "CLUSTERS", 0, Lifetime::Timeframe },
108  InputSpec{ "tofclusterlabels", "TOF", "CLUSTERSMCTR", 0, Lifetime::Timeframe } },
109  {}, // no output
110  AlgorithmSpec(initFunction),
111  Options{
112  { "tof-cluster-outfile", VariantType::String, "tofclusters.root", { "Name of the input file" } },
113  { "treename", VariantType::String, "o2sim", { "Name of top-level TTree" } },
114  }
115  };
116 }
