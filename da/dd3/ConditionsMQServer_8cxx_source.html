90 void ConditionsMQServer::Run()
91 {
92  std::unique_ptr<FairMQPoller> poller(
93  fTransportFactory->CreatePoller(fChannels, { "data-put", "data-get", "broker-get" }));
94 
95  while (CheckCurrentState(RUNNING)) {
96 
97  poller->Poll(-1);
98 
99  if (poller->CheckInput("data-get", 0)) {
100  std::unique_ptr<FairMQMessage> input(fTransportFactory->CreateMessage());
101 
102  if (Receive(input, "data-get") > 0) {
103  std::string serialString(static_cast<char*>(input->GetData()), input->GetSize());
104 
105  //LOG(DEBUG) << "Received a GET client message: " << serialString;
106 
107  std::string dataSource;
108  ParseDataSource(dataSource, serialString);
109 
110  if (dataSource == "OCDB") {
111  // Retrieve the key from the serialized message
112  std::string key;
113  Deserialize(serialString, key);
114 
115  getFromOCDB(key);
116  } else if (dataSource == "Riak") {
117  // No need to de-serialize, just forward message to the broker
118  fChannels.at("broker-get").at(0).Send(input);
119  }
120  }
121  }
122 
123  if (poller->CheckInput("data-put", 0)) {
124  std::unique_ptr<FairMQMessage> input(fTransportFactory->CreateMessage());
125 
126  if (Receive(input, "data-put") > 0) {
127  std::string serialString(static_cast<char*>(input->GetData()), input->GetSize());
128 
129  // LOG(DEBUG) << "Received a PUT client message: " << serialString;
130  LOG(DEBUG) << "Message size: " << input->GetSize();
131 
132  std::string dataSource;
133  ParseDataSource(dataSource, serialString);
134 
135  if (dataSource == "OCDB") {
136  LOG(ERROR) << "The GET operation is not supported for the OCDB data source yet";
137  } else if (dataSource == "Riak") {
138  fChannels.at("broker-put").at(0).Send(input);
139  }
140  }
141  }
142 
143  if (poller->CheckInput("broker-get", 0)) {
144  std::unique_ptr<FairMQMessage> input(fTransportFactory->CreateMessage());
145 
146  if (Receive(input, "broker-get") > 0) {
147  LOG(DEBUG) << "Received object from broker with a size of: " << input->GetSize();
148 
149  fChannels.at("data-get").at(0).Send(input);
150  }
151  }
152  }
153 }

155 // Query OCDB for the condition
156 void ConditionsMQServer::getFromOCDB(std::string key)
157 {
158  // Change key from i.e. "/DET/Calib/Histo/Run2008_2008_v1_s0" to (DET/Calib/Histo, 2008)
159  // FIXME: This will have to be changed in the future by adapting IdPath and getObject accordingly
160  std::size_t pos = key.rfind("/");
161  std::string identifier = key.substr(0, pos);
162  key.erase(0, pos + 4);
163  std::size_t pos2 = key.find("_");
164  int runId = atoi(key.substr(0, pos2).c_str());
165 
166  Condition* aCondition = nullptr;
167 
168  mCdbManager->setRun(runId);
169  aCondition = mCdbManager->getCondition(IdPath(identifier), runId);
170 
171  if (aCondition) {
172  LOG(DEBUG) << "Sending following parameter to the client:";
173  aCondition->printConditionMetaData();
174  TMessage* tmsg = new TMessage(kMESS_OBJECT);
175  tmsg->WriteObject(aCondition);
176 
177  std::unique_ptr<FairMQMessage> message(
178  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
179 
180  fChannels.at("data-get").at(0).Send(message);
181  } else {
182  LOG(ERROR) << R"(Could not get a condition for ")" << key << R"(" and run )" << runId << "!";
183  }
184 }
