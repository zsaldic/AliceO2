438 inline float TrackResiduals::getDXI(int ix) const
439 {
440  if (mUniformBins[VoxX]) {
441  return mDXI;
442  } else {
443  if (ix < param::NRowsPerROC[0]) {
444  // we are in the IROC
445  return 1.f / param::RowDX[0];
446  } else if (ix > param::NRowsAccumulated[param::NROCTypes - 1]) {
447  // we are in the last OROC
448  return 1.f / param::RowDX[param::NROCTypes - 1];
449  }
450 #ifdef TPC_RUN2
451  else {
452  // we are in OROC1
453  return 1.f / param::RowDX[1];
454  }
455 #else
456  else if (ix < param::NRowsAccumulated[2]) {
457  // OROC1
458  return 1.f / param::RowDX[1];
459  } else {
460  // OROC2
461  return 1.f / param::RowDX[2];
462  }
463 #endif
464  }
465 }

494 //_____________________________________________________
495 inline int TrackResiduals::getY2XBin(float y2x, int ix) const
496 {
497  int bp = (y2x + mMaxY2X[ix]) * mDY2XI[ix];

499  bp = 0;

502 }

504 //_____________________________________________________
505 inline int TrackResiduals::getZ2XBin(float z2x) const
506 {
507  int bz = z2x * mDZI;
508  if (bz < 0) {
509  // accounting for clusters which were moved to the wrong side
510  bz = 0;
511  }
512  return (bz < mNZ2XBins) ? bz : mNZ2XBins - 1;
