40 MultiView::MultiView()
41 {
42  // set scene names and descriptions
43  mSceneNames[Scene3dGeom] = "3D Geometry Scene";
44  mSceneNames[SceneRPhiGeom] = "R-Phi Geometry Scene";
45  mSceneNames[SceneZrhoGeom] = "Rho-Z Geometry Scene";
46  mSceneNames[Scene3dEvent] = "3D Event Scene";
47  mSceneNames[SceneRphiEvent] = "R-Phi Event Scene";
48  mSceneNames[SceneZrhoEvent] = "Rho-Z Event Scene";
49 
50  mSceneDescriptions[Scene3dGeom] = "Scene holding 3D geometry.";
51  mSceneDescriptions[SceneRPhiGeom] = "Scene holding projected geometry for the R-Phi view.";
52  mSceneDescriptions[SceneZrhoGeom] = "Scene holding projected geometry for the Rho-Z view.";
53  mSceneDescriptions[Scene3dEvent] = "Scene holding 3D event.";
54  mSceneDescriptions[SceneRphiEvent] = "Scene holding projected event for the R-Phi view.";
55  mSceneDescriptions[SceneZrhoEvent] = "Scene holding projected event for the Rho-Z view.";
56 
57  // spawn scenes
58  for(int i=0;i<NumberOfScenes;++i){
59  mScenes[i] = gEve->SpawnNewScene(mSceneNames[i].c_str(), mSceneDescriptions[i].c_str());
60  }
61 
62  // Projection managers
63  mProjections[ProjectionRphi] = new TEveProjectionManager();
64  mProjections[ProjectionZrho] = new TEveProjectionManager();
65 
66  mProjections[ProjectionRphi]->SetProjection(TEveProjection::kPT_RPhi);
67  mProjections[ProjectionZrho]->SetProjection(TEveProjection::kPT_RhoZ);
68 
69  gEve->AddToListTree(static_cast<TEveElement*>(mProjections[ProjectionRphi]),false);
70  gEve->AddToListTree(static_cast<TEveElement*>(mProjections[ProjectionZrho]),false);
71 
72  // add axes
73  TEnv settings;
74  ConfigurationManager::getInstance().getConfig(settings);
75  const bool showAxes = settings.GetValue("axes.show", false);
76  if(showAxes){
77  for(int i=0;i<NumberOfProjections;++i){
78  TEveProjectionAxes axes(mProjections[static_cast<EProjections>(i)]);
79  axes.SetMainColor(kWhite);
80  axes.SetTitle("R-Phi");
81  axes.SetTitleSize(0.05);
82  axes.SetTitleFont(102);
83  axes.SetLabelSize(0.025);
84  axes.SetLabelFont(102);
85  mScenes[getSceneOfProjection(static_cast<EProjections>(i))]->AddElement(&axes);
86  }
87  }
88 
89  setupMultiview();
90  sInstance = this;
91 }

101 void MultiView::setupMultiview()

105  TEveWindowPack *pack = slot->MakePack();
106 
107  pack->SetElementName("Multi View");
108  pack->SetHorizontal();
109  pack->SetShowTitleBar(kFALSE);
110 
111  pack->NewSlotWithWeight(2)->MakeCurrent(); // new slot is created from pack
112  mViews[View3d] = gEve->SpawnNewViewer("3D View", "");
113  mViews[View3d]->AddScene(mScenes[Scene3dGeom]);
114  mViews[View3d]->AddScene(mScenes[Scene3dEvent]);
115 
116  pack = pack->NewSlot()->MakePack();

118  pack->NewSlot()->MakeCurrent();
119  mViews[ViewRphi] = gEve->SpawnNewViewer("R-Phi View", "");

122  mViews[ViewRphi]->AddScene(mScenes[SceneRphiEvent]);

149 void MultiView::registerGeometry(TEveGeoShape *geom, bool threeD, bool rPhi, bool zRho)
150 {
151  if(!geom){
152  cout<<"MultiView::registerGeometry -- geometry is NULL!"<<endl;
153  return;
154  }
155  mGeomVector.push_back(geom);
156 
157  TEveProjectionManager *projection;
158 
159  if(threeD){
160  gEve->AddElement(geom,getScene(Scene3dGeom));
161  }
162  if(rPhi){
163  projection = getProjection(ProjectionRphi);
164  projection->SetCurrentDepth(-10);
165  projection->ImportElements(geom, getScene(SceneRPhiGeom));
166  projection->SetCurrentDepth(0);
167  }
168  if(zRho){
169  projection = getProjection(ProjectionZrho);
170  projection->SetCurrentDepth(-10);
171  projection->ImportElements(geom, getScene(SceneZrhoGeom));
172  projection->SetCurrentDepth(0);
173  }
174 }
