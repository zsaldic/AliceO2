184 void BoxClusterer::findLocalMaxima(const Int_t iCRU)
185 {
190 
191  R__ASSERT(mAllBins);
192 
193  Int_t nLocalMaxima = 0;
194  // loop over rows
195  for (Int_t iRow = 0; iRow < mRowsMax; iRow++) {
196 
197  Float_t* allBins = mAllBins[iRow];
198  Int_t* sigBins = mAllSigBins[iRow];
199  const Int_t nSigBins = mAllNSigBins[iRow];
200 
201  // loop over all signals
202  for (Int_t iSig = 0; iSig < nSigBins; iSig++) {
203 
204  Int_t bin = sigBins[iSig];
205  // Array of charged centered at the current signal
206  Float_t *qArray = &allBins[bin];
207  Float_t qMax = qArray[0];
208 
209  // First check that the charge is bigger than the threshold
210  if ( qMax < mMinQMax )
211  continue;
212 
213  // Require at least one neighboring time bin with signal
214  if ( qArray[-1] + qArray[1] <= 0 ) continue;
215  // Require at least one neighboring pad with signal
216  const Int_t maxTimeBin = mTimeBinsMax+4; // Used to step between neighboring
217  if ( mRequireNeighbouringPad
218  && (qArray[-maxTimeBin]+qArray[maxTimeBin]<=0) ) continue;
219  //
220  // Check that this is a local maximum
221  // Note that the checking is done so that if 2 charges has the same
222  // qMax then only 1 cluster is generated
223  // (that is why there is BOTH > and >=)
224  //
225  if (qArray[-maxTimeBin] >= qMax) continue;
226  if (qArray[+maxTimeBin] > qMax) continue;
227  if (qArray[-1 ] >= qMax) continue;
228  if (qArray[+1 ] > qMax) continue;
229  if (qArray[-maxTimeBin-1] >= qMax) continue;
230  if (qArray[+maxTimeBin+1] > qMax) continue;
231  if (qArray[+maxTimeBin-1] >= qMax) continue;
232  if (qArray[-maxTimeBin+1] > qMax) continue;
233 
234 // Short_t tb; Short_t pa;
235 // GetPadAndTimeBin(bin,pa,tb);
236 // if ((iCRU == 179 && iRow == 2 && pa == 104 && tb == 171) || (iCRU == 256 && iRow == 10 && pa == 27 && tb == 181) ) {
237 // std::cout << qArray[+maxTimeBin-1] << " " << qArray[+maxTimeBin] << " " << qArray[+maxTimeBin+1] << std::endl;
238 // std::cout << qArray[-1] << " " << qArray[0] << " " << qArray[+1] << std::endl;
239 // std::cout << qArray[-maxTimeBin-1] << " " << qArray[-maxTimeBin] << " " << qArray[-maxTimeBin+1] << std::endl;
240 // }
241  // We accept the local maximum as a cluster and calculates its
242  // parameters
243  ++nLocalMaxima;
244 
245  //
246  // Calculate the total charge as the sum over the region:
247  //
248  // o o o o o
249  // o i i i o
250  // o i C i o
251  // o i i i o
252  // o o o o o
253  //
254  // with qmax at the center C.
255  //
256  // The inner charge (i) we always add, but we only add the outer
257  // charge (o) if the neighboring inner bin (i) has a signal.
258  //
259  Short_t minP = 0, maxP = 0, minT = 0, maxT = 0;
260  Double_t meanP = 0; // mean pad position
261  Double_t meanT = 0; // mean time position
262  Double_t sigmaP = 0; // sigma pad position
263  Double_t sigmaT = 0; // sigma time position
264  Float_t qTot = qMax; // total charge
265  for(Short_t dTime = -1; dTime<=1; dTime++) { // delta time
266  for(Short_t dPad = -1; dPad<=1; dPad++) { // delta pad
267 
268  if( dPad==0 && dTime==0 ) // central pad
269  continue;
270 
271  Float_t charge = getQ(qArray, dPad, dTime, minT, maxT, minP, maxP);
272  updateCluster(charge, dPad, dTime, qTot,
273  meanP, sigmaP, meanT, sigmaT);
274 
275  if( !mRequirePositiveCharge || charge>0 ) {
276  // see if the next neighbor is also above threshold
277 
278  if(dPad*dTime==0) { // we are above/below or to the sides
279  // (dPad, dTime) = (+-1, 0), or (0, +-1)
280  // so we only have 1 neighbor
281  charge = getQ(qArray, 2 * dPad, 2 * dTime, minT, maxT, minP, maxP);
282  updateCluster(charge, 2 * dPad, 2 * dTime, qTot,
283  meanP, sigmaP, meanT, sigmaT);
284  } else { // we are in a diagonal corner so we have 3 neighbors
285  // (dPad, dTime) = (+-1, +-1), or (+-1, -+1)
286  charge = getQ(qArray, dPad, 2 * dTime, minT, maxT, minP, maxP);
287  updateCluster(charge, dPad, 2 * dTime, qTot,
288  meanP, sigmaP, meanT, sigmaT);
289  charge = getQ(qArray, 2 * dPad, dTime, minT, maxT, minP, maxP);
290  updateCluster(charge, 2 * dPad, dTime, qTot,
291  meanP, sigmaP, meanT, sigmaT);
292  charge = getQ(qArray, 2 * dPad, 2 * dTime, minT, maxT, minP, maxP);
293  updateCluster(charge, 2 * dPad, 2 * dTime, qTot,
294  meanP, sigmaP, meanT, sigmaT);
295  }
296  }
297  }
298  }
299 
300  // calculate cluster parameters
301  if(qTot > 0) {
302  meanP /= qTot;
303  meanT /= qTot;
304  sigmaP /= qTot;
305  sigmaT /= qTot;
306  sigmaP = TMath::Sqrt(sigmaP - meanP*meanP);
307  sigmaT = TMath::Sqrt(sigmaT - meanT*meanT);
308  Short_t pad, timebin;
309  getPadAndTimeBin(bin, pad, timebin);
310  meanP += pad;
311  meanT += timebin;
312  Short_t nPad = maxP-minP+1;
313  Short_t nTimeBins = maxT-minT+1;
314  Short_t size = 10*nPad+nTimeBins;
315  Cluster* cluster = ClusterContainer::addCluster<Cluster>(mClusterArray, iCRU, iRow, qTot, qMax, meanP, meanT, sigmaP, sigmaT);
316 
317  // if ((iCRU == 179)) {// && iRow == 5)){// && (int)meanP == 103 && (int)meanT == 170) ||
319  // std::cout << "BoxCluster - ";
320  // cluster->Print(std::cout);
321  // std::cout << " " << std::endl;
322  // for(Short_t dTime = -2; dTime<=2; dTime++) { // delta time
323  // for(Short_t dPad = -2; dPad<=2; dPad++) { // delta pad
324  // Float_t charge = GetQ(qArray, dPad, dTime, minT, maxT, minP, maxP);
325  // std::cout << "\t" << charge;
326  // }
327  // std::cout << std::endl;
328  // }
329  // std::cout << std::endl;
330  // }
332  }
333  } // end loop over signals
334  } // end loop over rows
335 }

371 Int_t BoxClusterer::update(const Int_t iCRU,
372  const Int_t iRow,
373  const Int_t iPad,
374  const Int_t iTimeBin,
375  Float_t signal)
376 {
378 
379  // -ne could have a list of active chambers
380  // if (!fActiveChambers[iSector]) return 0;
381 
382  // Stop processing if input is out of range
383  R__ASSERT(iRow>=0 && iRow<mRowsMax);
384  R__ASSERT(iPad>=0 && iPad<(mPadsMax+4));
385  R__ASSERT(iTimeBin>=0 && iTimeBin<(mTimeBinsMax+4));
386 
387  if (mRequirePositiveCharge && (signal <= 0)){
388  return 0; // signal was not accepted
389  }
390 
391  // ===| get pedestal |========================================================
392  if (mPedestalObject) {
393  signal -= mPedestalObject->getValue(CRU(iCRU), iRow, iPad);
394  }
395 
396  // Fill signal in array. Add 2 to pad and time to make sure that the 2D
397  // array even for (0, 0) has a valid 5x5 matrix.
398  Int_t bin = (iPad+2)*(mTimeBinsMax+4) + (iTimeBin+2);
399 
400  mAllBins[iRow][bin] = signal;
401  mAllSigBins[iRow][mAllNSigBins[iRow]] = bin;
402  mAllNSigBins[iRow]++;
403 
404  return 1; // signal was accepted
405 }
