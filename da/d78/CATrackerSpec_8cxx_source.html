48 DataProcessorSpec getCATrackerSpec(bool processMC, std::vector<int> const& inputIds)
49 {
50  constexpr static size_t NSectors = o2::TPC::Sector::MAXSECTOR;
51  using ClusterGroupParser = o2::algorithm::ForwardParser<o2::TPC::ClusterGroupHeader>;
52  struct ProcessAttributes {
53  // the input comes in individual calls and we need to buffer until
54  // data set is complete, have to think about a DPL feature to take
55  // ownership of an input
56  std::array<std::vector<char>, NSectors> bufferedInputs;
57  std::array<std::vector<MCLabelContainer>, NSectors> mcInputs;
58  std::bitset<NSectors> validInputs = 0;
59  std::bitset<NSectors> validMcInputs = 0;
60  std::unique_ptr<ClusterGroupParser> parser;
61  std::unique_ptr<o2::TPC::TPCCATracking> tracker;
62  int verbosity = 1;
63  std::vector<int> inputIds;
64  bool readyToQuit = false;
65  };

67  auto initFunction = [processMC, inputIds](InitContext& ic) {
68  auto options = ic.options().get<std::string>("tracker-options");
69 
70  auto processAttributes = std::make_shared<ProcessAttributes>();
71  {
72  processAttributes->inputIds = inputIds;
73  auto& parser = processAttributes->parser;
74  auto& tracker = processAttributes->tracker;
75  parser = std::make_unique<ClusterGroupParser>();
76  tracker = std::make_unique<o2::TPC::TPCCATracking>();
77  if (tracker->initialize(options.c_str()) != 0) {
78  throw std::invalid_argument("TPCCATracking initialization failed");
79  }
80  processAttributes->validInputs.reset();
81  processAttributes->validMcInputs.reset();
82  }
83 
84  auto processingFct = [processAttributes, processMC](ProcessingContext& pc) {
85  if (processAttributes->readyToQuit) {
86  return;
87  }
88  auto& parser = processAttributes->parser;
89  auto& tracker = processAttributes->tracker;
90  uint64_t activeSectors = 0;
91  auto& verbosity = processAttributes->verbosity;
92 
93  // FIXME cleanup almost duplicated code
94  auto& validMcInputs = processAttributes->validMcInputs;
95  auto& mcInputs = processAttributes->mcInputs;
96  if (processMC) {
97  // we can later extend this to multiple inputs
98  for (auto const& inputId : processAttributes->inputIds) {
99  std::string inputLabel = "mclblin" + std::to_string(inputId);
100  auto ref = pc.inputs().get(inputLabel);
101  auto const* sectorHeader = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(ref);
102  if (sectorHeader == nullptr) {
103  // FIXME: think about error policy
104  LOG(ERROR) << "sector header missing on header stack";
105  return;
106  }
107  const int& sector = sectorHeader->sector;
108  if (sector < 0) {
109  continue;
110  }
111  if (validMcInputs.test(sector)) {
112  // have already data for this sector, this should not happen in the current
113  // sequential implementation, for parallel path merged at the tracker stage
114  // multiple buffers need to be handled
115  throw std::runtime_error("can only have one data set per sector");
116  }
117  mcInputs[sector] = std::move(pc.inputs().get<std::vector<MCLabelContainer>>(inputLabel.c_str()));
118  validMcInputs.set(sector);
119  activeSectors |= sectorHeader->activeSectors;
120  if (verbosity > 1) {
121  LOG(INFO) << "received " << *(ref.spec) << " MC label containers"
122  << " for sector " << sector //
123  << std::endl //
124  << " mc input status: " << validMcInputs //
125  << std::endl //
126  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
127  }
128  }
129  }
130 
131  auto& validInputs = processAttributes->validInputs;
132  int operation = 0;
133  std::map<int, DataRef> datarefs;
134  for (auto const& inputId : processAttributes->inputIds) {
135  std::string inputLabel = "input" + std::to_string(inputId);
136  auto ref = pc.inputs().get(inputLabel);
137  auto const* sectorHeader = DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(ref);
138  if (sectorHeader == nullptr) {
139  // FIXME: think about error policy
140  LOG(ERROR) << "sector header missing on header stack";
141  return;
142  }
143  const int& sector = sectorHeader->sector;
144  // check the current operation, this is used to either signal eod or noop
145  // FIXME: the noop is not needed any more once the lane configuration with one
146  // channel per sector is used
147  if (sector < 0) {
148  if (operation < 0 && operation != sector) {
149  // we expect the same operation on all inputs
150  LOG(ERROR) << "inconsistent lane operation, got " << sector << ", expecting " << operation;
151  } else if (operation == 0) {
152  // store the operation
153  operation = sector;
154  }
155  continue;
156  }
157  if (validInputs.test(sector)) {
158  // have already data for this sector, this should not happen in the current
159  // sequential implementation, for parallel path merged at the tracker stage
160  // multiple buffers need to be handled
161  throw std::runtime_error("can only have one data set per sector");
162  }
163  activeSectors |= sectorHeader->activeSectors;
164  validInputs.set(sector);
165  datarefs[sector] = ref;
166  }
167 
168  if (operation == -1) {
169  // EOD is transmitted in the sectorHeader with sector number equal to -1
170  o2::TPC::TPCSectorHeader sh{ -1 };
171  sh.activeSectors = activeSectors;
172  pc.outputs().snapshot(OutputRef{ "output", 0, { sh } }, -1);
173  if (processMC) {
174  pc.outputs().snapshot(OutputRef{ "mclblout", 0, { sh } }, -1);
175  }
176  pc.services().get<ControlService>().readyToQuit(false);
177  processAttributes->readyToQuit = true;
178  return;
179  }
180  auto printInputLog = [&verbosity, &validInputs, &activeSectors](auto& r, const char* comment, auto& s) {
181  if (verbosity > 1) {
182  LOG(INFO) << comment << " " << *(r.spec) << ", size " << DataRefUtils::getPayloadSize(r) //
183  << " for sector " << s //
184  << std::endl //
185  << " input status: " << validInputs //
186  << std::endl //
187  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
188  }
189  };
190  auto& bufferedInputs = processAttributes->bufferedInputs;
191  if (activeSectors == 0 || (activeSectors & validInputs.to_ulong()) != activeSectors ||
192  (processMC && (activeSectors & validMcInputs.to_ulong()) != activeSectors)) {
193  // not all sectors available, we have to buffer the inputs
194  for (auto const& refentry : datarefs) {
195  auto& sector = refentry.first;
196  auto& ref = refentry.second;
197  auto payploadSize = DataRefUtils::getPayloadSize(ref);
198  bufferedInputs[sector].resize(payploadSize);
199  std::copy(ref.payload, ref.payload + payploadSize, bufferedInputs[sector].begin());
200  printInputLog(ref, "buffering", sector);
201  }
202 
203  // not needed to send something, DPL will simply drop this timeslice, whenever the
204  // data for all sectors is available, the output is sent in that time slice
205  return;
206  }
207  assert(processMC == false || validMcInputs == validInputs);
208  std::array<gsl::span<const char>, NSectors> inputs;
209  auto inputStatus = validInputs;
210  for (auto const& refentry : datarefs) {
211  auto& sector = refentry.first;
212  auto& ref = refentry.second;
213  inputs[sector] = gsl::span(ref.payload, DataRefUtils::getPayloadSize(ref));
214  inputStatus.reset(sector);
215  printInputLog(ref, "received", sector);
216  }
217  if (inputStatus.any()) {
218  // some of the inputs have been buffered
219  for (size_t sector = 0; sector < inputStatus.size(); ++sector) {
220  if (inputStatus.test(sector)) {
221  inputs[sector] = gsl::span(&bufferedInputs[sector].front(), bufferedInputs[sector].size());
222  }
223  }
224  }
225  if (verbosity > 0) {
226  if (inputStatus.any()) {
227  LOG(INFO) << "using buffered data for " << inputStatus.count() << " sector(s)";
228  }
229  // make human readable information from the bitfield
230  std::string bitInfo;
231  auto nActiveBits = validInputs.count();
232  if (((uint64_t)0x1 << nActiveBits) == validInputs.to_ulong() + 1) {
233  // sectors 0 to some upper bound are active
234  bitInfo = "0-" + std::to_string(nActiveBits - 1);
235  } else {
236  int rangeStart = -1;
237  int rangeEnd = -1;
238  for (size_t sector = 0; sector < validInputs.size(); sector++) {
239  if (validInputs.test(sector)) {
240  if (rangeStart < 0) {
241  if (rangeEnd >= 0) {
242  bitInfo += ",";
243  }
244  bitInfo += std::to_string(sector);
245  if (nActiveBits == 1) {
246  break;
247  }
248  rangeStart = sector;
249  }
250  rangeEnd = sector;
251  } else {
252  if (rangeStart >= 0 && rangeEnd > rangeStart) {
253  bitInfo += "-" + std::to_string(rangeEnd);
254  }
255  rangeStart = -1;
256  }
257  }
258  if (rangeStart >= 0 && rangeEnd > rangeStart) {
259  bitInfo += "-" + std::to_string(rangeEnd);
260  }
261  }
262  LOG(INFO) << "running tracking for sector(s) " << bitInfo;
263  }
264  ClusterNativeAccessFullTPC clusterIndex;
265  memset(&clusterIndex, 0, sizeof(clusterIndex));
266  ClusterNativeHelper::Reader::fillIndex(clusterIndex, inputs, mcInputs, [&validInputs](auto& index) { return validInputs.test(index); });
267 
268  std::vector<TrackTPC> tracks;
269  MCLabelContainer tracksMCTruth;
270  int retVal = tracker->runTracking(clusterIndex, &tracks, (processMC ? &tracksMCTruth : nullptr));
271  if (retVal != 0) {
272  // FIXME: error policy
273  LOG(ERROR) << "tracker returned error code " << retVal;
274  }
275  LOG(INFO) << "found " << tracks.size() << " track(s)";
276  pc.outputs().snapshot(OutputRef{ "output" }, tracks);
277  if (processMC) {
278  LOG(INFO) << "sending " << tracksMCTruth.getIndexedSize() << " track label(s)";
279  pc.outputs().snapshot(OutputRef{ "mclblout" }, tracksMCTruth);
280  }
281 
282  validInputs.reset();
283  if (processMC) {
284  validMcInputs.reset();
285  for (auto& mcInput : mcInputs) {
286  mcInput.clear();
287  }
288  }
289  };
290 
291  return processingFct;
292  };
293 
294  // FIXME: find out how to handle merge inputs in a simple and intuitive way
295  // changing the binding name of the input in order to identify inputs by unique labels
296  // in the processing. Think about how the processing can be made agnostic of input size,
297  // e.g. by providing a span of inputs under a certain label
298  auto createInputSpecs = [inputIds](bool makeMcInput) {
299  Inputs inputs = { InputSpec{ "input", gDataOriginTPC, "CLUSTERNATIVE", 0, Lifetime::Timeframe } };
300  if (makeMcInput) {
301  inputs.emplace_back(InputSpec{ "mclblin", gDataOriginTPC, "CLNATIVEMCLBL", 0, Lifetime::Timeframe });
302  }
303 
304  return std::move(mergeInputs(inputs, inputIds.size(),
305  [inputIds](InputSpec& input, size_t index) {
306  // using unique input names for the moment but want to find
307  // an input-multiplicity-agnostic way of processing
308  input.binding += std::to_string(inputIds[index]);
309  DataSpecUtils::updateMatchingSubspec(input, inputIds[index]);
310  }));
311  };
312 
313  auto createOutputSpecs = [](bool makeMcOutput) {
314  std::vector<OutputSpec> outputSpecs{
315  OutputSpec{ { "output" }, gDataOriginTPC, "TRACKS", 0, Lifetime::Timeframe },
316  };
317  if (makeMcOutput) {
318  OutputLabel label{ "mclblout" };
319  constexpr o2::header::DataDescription datadesc("TRACKMCLBL");
320  outputSpecs.emplace_back(label, gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
321  }
322  return std::move(outputSpecs);
323  };
324 
325  return DataProcessorSpec{ "tpc-tracker", // process id
326  { createInputSpecs(processMC) },
327  { createOutputSpecs(processMC) },
328  AlgorithmSpec(initFunction),
329  Options{
330  { "tracker-options", VariantType::String, "", { "Option string passed to tracker" } },
331  } };
