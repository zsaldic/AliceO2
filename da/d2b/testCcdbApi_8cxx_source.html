130 void countItems(const string& s, int& countObjects, int& countSubfolders)
131 {
132  countObjects = 0;
133  countSubfolders = 0;
134  std::stringstream ss(s);
135  std::string line;
136  bool subfolderMode = false;
137  while (std::getline(ss, line, '\n')) {
138  ltrim(line);
139  rtrim(line);
140  if (line.length() == 0) {
141  continue;
142  }
143 
144  if (line.find("subfolders") != std::string::npos) {
145  subfolderMode = true;
146  continue;
147  }
148 
149  if (subfolderMode) {
150  countSubfolders++;
151  if (line.find(']') != std::string::npos) {
152  break;
153  }
154  }
155 
156  if (line.find(R"("path")") == 0) {
157  countObjects++;
158  }
159  }
160 }
