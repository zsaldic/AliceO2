132 Bool_t Detector::ProcessHits(FairVolume* v)
133 {
134  // TODO Implement handling of parents and primary particle
135  Double_t eloss = fMC->Edep();
136  if (eloss < DBL_EPSILON)
137  return false; // only process hits which actually deposit some energy in the EMCAL
138  Geometry* geom = GetGeometry();
139  // Obtain detector ID
140  // This is not equal to the volume ID of the fair volume
141  // EMCAL geometry implementation in VMC works with a copy of the volume and placing it n-times into the mother volume
142  // via translation / rotation, so the copy index is the index of a tower / module / supermodule node within a mother
143  // volume Additional care needs to be taken for the supermodule index: The copy is connected to a certain supermodule
144  // type, which differs for various parts of the detector
145  Int_t copyEta, copyPhi, copyMod, copySmod;
146  fMC->CurrentVolID(copyEta); // Tower in module - x-direction
147  fMC->CurrentVolOffID(1, copyPhi); // Tower in module - y-direction
148  fMC->CurrentVolOffID(3, copyMod); // Module in supermodule
149  fMC->CurrentVolOffID(
150  4, copySmod); // Supermodule in EMCAL - attention, with respect to a given supermodule type (offsets needed)
151  std::string smtype(fMC->CurrentVolOffName(4));
152  int offset(0);
153  if (smtype == "SM3rd") {
154  offset = 10;
155  } else if (smtype == "DCSM") {
156  offset = 12;
157  } else if (smtype == "DCEXT") {
158  offset = 18;
159  }
160  LOG(DEBUG3) << "Supermodule copy " << copySmod << ", module copy " << copyMod << ", y-dir " << copyPhi << ", x-dir "
161  << copyEta << ", supermodule ID " << copySmod + offset - 1;
162  LOG(DEBUG3) << "path " << fMC->CurrentVolPath();
163  LOG(DEBUG3) << "Name of the supermodule type " << fMC->CurrentVolOffName(4) << ", Module name "
164  << fMC->CurrentVolOffName(3);
165 
166  Int_t partID = fMC->GetStack()->GetCurrentTrackNumber(),
167  parent = fMC->GetStack()->GetCurrentTrack()->GetMother(0),
168  detID = geom->GetAbsCellId(offset + copySmod - 1, copyMod - 1, copyPhi - 1, copyEta - 1);
169 
170  Double_t lightyield(eloss);
171  if (fMC->TrackCharge())
172  lightyield = CalculateLightYield(eloss, fMC->TrackStep(), fMC->TrackCharge());
173  lightyield /= geom->GetSampling();
174 
175  auto o2stack = static_cast<o2::Data::Stack*>(fMC->GetStack());
176  const bool isDaughterOfSeenTrack = o2stack->isTrackDaughterOf(partID, mCurrentTrackID);
177  if (!isDaughterOfSeenTrack || detID != mCurrentCellID || !mCurrentHit) {
178  // Condition for new hit:
179  // - Processing different track
180  // - Inside different cell
181  // - First track of the event
182  // std::cout << "New track / cell started";
183  Float_t posX, posY, posZ, momX, momY, momZ, energy;
184  fMC->TrackPosition(posX, posY, posZ);
185  fMC->TrackMomentum(momX, momY, momZ, energy);
186  Double_t estart = fMC->Etot(), time = fMC->TrackTime() * 1e9; // time in ns
187 
189  mCurrentHit = AddHit(partID, parent, 0, estart, detID, Point3D<float>(posX, posY, posZ),
190  Vector3D<float>(momX, momY, momZ), time, lightyield);
191  o2stack->addHit(GetDetId());
192  mCurrentTrackID = partID;
193  mCurrentCellID = detID;
194 
195  } else {
196  // std::cout << "Adding energy to the current hit";
197  mCurrentHit->SetEnergyLoss(mCurrentHit->GetEnergyLoss() + lightyield);
198  }
199 
200  return true;
201 }

213 Double_t Detector::CalculateLightYield(Double_t energydeposit, Double_t tracklength, Int_t charge) const
214 {
215  if (charge == 0)
216  return energydeposit; // full energy deposit for neutral particles (photons)
217  // Apply Birk's law (copied from G3BIRK)
218 
219  Float_t birkC1Mod = 0;
220  if (mBirkC0 == 1) { // Apply correction for higher charge states
221  if (std::abs(charge) >= 2)
222  birkC1Mod = mBirkC1 * 7.2 / 12.6;
223  else
224  birkC1Mod = mBirkC1;
225  }
226 
227  Float_t dedxcm = 0.;
228  if (tracklength > 0)
229  dedxcm = 1000. * energydeposit / tracklength;
230  else
231  dedxcm = 0;
232 
233  return energydeposit / (1. + birkC1Mod * dedxcm + mBirkC2 * dedxcm * dedxcm);
234 }

263 {

266  Geometry* geom = GetGeometry();
267  std::string gn = geom->GetName();

272  Matrix(rotMatrixID, 90.0, 0., 90.0, 90.0, 0.0, 0.0);
273 
274  // Create the EMCAL Mother Volume (a polygone) within which to place the Detector and named XEN1
275  if (contains(gn, "WSUC")) { // TRD1 for WSUC facility
276  // Nov 25,2010

286  TVirtualMC::GetMC()->Gspos(geom->GetNameOfEMCALEnvelope(), 1, "WSUC", 0.0, 0.0, +265., rotMatrixID, "ONLY");
287  } else {
288  Double_t envelopA[10];
289  envelopA[0] = geom->GetArm1PhiMin(); // minimum phi angle
290  envelopA[1] = geom->GetArm1PhiMax() - geom->GetArm1PhiMin(); // angular range in phi
291  envelopA[2] = envelopA[1] / geom->GetPhiSuperModule(); // Section of that
292  envelopA[3] = 2; // 2: z coordinates

295  envelopA[6] = geom->GetEnvelop(1); // rmax at z1 - includes padding
296  envelopA[7] = geom->GetEnvelop(2) / 2.; // zmax includes padding

298  envelopA[8] = envelopA[5]; // radii are the same.
299  envelopA[9] = envelopA[6];

301  TVirtualMC::GetMC()->Gsvolu(geom->GetNameOfEMCALEnvelope(), "PGON", getMediumID(ID_AIR), envelopA,
302  10); // Polygone filled with air

305  << envelopA[6];
306  LOG(DEBUG2) << "ConstructGeometry: XU0 = " << envelopA[5] << ", " << envelopA[6];

309  TVirtualMC::GetMC()->Gspos(geom->GetNameOfEMCALEnvelope(), 1, "cave", 0.0, 0.0, 0.0, rotMatrixID, "ONLY");

313 void Detector::CreateShiskebabGeometry()
314 {
315  // TRD1
316  using boost::algorithm::contains;
317  Geometry* g = GetGeometry();
318  std::string gn = g->GetName();
319  std::transform(gn.begin(), gn.end(), gn.begin(), ::toupper);
320 
321  Double_t trd1Angle = g->GetTrd1Angle() * TMath::DegToRad(), tanTrd1 = TMath::Tan(trd1Angle / 2.);
322  // see AliModule::fFIdTmedArr
323  // fIdTmedArr = fIdtmed->GetArray() - 1599 ; // see AliEMCAL::::CreateMaterials()
324  // Int_t kIdAIR=1599, kIdPB = 1600, kIdSC = 1601, kIdSTEEL = 1603;
325  // idAL = 1602;
326  Double_t par[10], xpos = 0., ypos = 0., zpos = 0.;
327 
328  LOG(DEBUG2) << "Name of mother volume: " << g->GetNameOfEMCALEnvelope();
329  CreateSupermoduleGeometry(g->GetNameOfEMCALEnvelope());
330 
331  auto SMTypeList = g->GetEMCSystem();
332  auto tmpType = NOT_EXISTENT;
333  for (auto i : boost::irange(0, g->GetNumberOfSuperModules())) {
334  if (SMTypeList[i] == tmpType)
335  continue;
336  else
337  tmpType = SMTypeList[i];
338 
339  if (tmpType == EMCAL_STANDARD)
340  CreateEmcalModuleGeometry("SMOD", "EMOD"); // 18-may-05
341  else if (tmpType == EMCAL_HALF)
342  CreateEmcalModuleGeometry("SM10", "EMOD"); // Nov 1,2006 1/2 SM
343  else if (tmpType == EMCAL_THIRD)
344  CreateEmcalModuleGeometry("SM3rd", "EMOD"); // Feb 1,2012 1/3 SM
345  else if (tmpType == DCAL_STANDARD)
346  CreateEmcalModuleGeometry("DCSM", "EMOD"); // Mar 13, 2012, 6 or 10 DCSM
347  else if (tmpType == DCAL_EXT)
348  CreateEmcalModuleGeometry("DCEXT", "EMOD"); // Mar 13, 2012, DCAL extension SM
349  else
350  LOG(ERROR) << "Unkown SM Type!!";
351  }
352 
353  // Sensitive SC (2x2 tiles)
354  Double_t parSCM0[5] = { 0, 0, 0, 0 }, *dummy = nullptr, parTRAP[11];
355  if (!contains(gn, "V1")) {
356  Double_t wallThickness = g->GetPhiModuleSize() / g->GetNPHIdiv() - g->GetPhiTileSize();
357  for (Int_t i = 0; i < 3; i++)
358  parSCM0[i] = mParEMOD[i] - wallThickness;
359  parSCM0[3] = mParEMOD[3];
360  TVirtualMC::GetMC()->Gsvolu("SCM0", "TRD1", getMediumID(ID_AIR), parSCM0, 4);
361  TVirtualMC::GetMC()->Gspos("SCM0", 1, "EMOD", 0., 0., 0., 0, "ONLY");
362  } else {
363  Double_t wTh = g->GetLateralSteelStrip();
364  parSCM0[0] = mParEMOD[0] - wTh + tanTrd1 * g->GetTrd1AlFrontThick();
365  parSCM0[1] = mParEMOD[1] - wTh;
366  parSCM0[2] = mParEMOD[2] - wTh;
367  parSCM0[3] = mParEMOD[3] - g->GetTrd1AlFrontThick() / 2.;
368  TVirtualMC::GetMC()->Gsvolu("SCM0", "TRD1", getMediumID(ID_AIR), parSCM0, 4);
369  Double_t zshift = g->GetTrd1AlFrontThick() / 2.;
370  TVirtualMC::GetMC()->Gspos("SCM0", 1, "EMOD", 0., 0., zshift, 0, "ONLY");
371  //
372  CreateAlFrontPlate("EMOD", "ALFP");
373  }
374 
375  if (g->GetNPHIdiv() == 2 && g->GetNETAdiv() == 2) {
376  // Division to tile size - 1-oct-04
377  LOG(DEBUG2) << " Divide SCM0 on y-axis " << g->GetNETAdiv();
378  TVirtualMC::GetMC()->Gsdvn("SCMY", "SCM0", g->GetNETAdiv(), 2); // y-axis
379 
380  // Trapesoid 2x2
381  parTRAP[0] = parSCM0[3]; // dz
382  parTRAP[1] = TMath::ATan2((parSCM0[1] - parSCM0[0]) / 2., 2. * parSCM0[3]) * 180. / TMath::Pi(); // theta
383  parTRAP[2] = 0.; // phi
384 
385  // bottom
386  parTRAP[3] = parSCM0[2] / 2.; // H1
387  parTRAP[4] = parSCM0[0] / 2.; // BL1
388  parTRAP[5] = parTRAP[4]; // TL1
389  parTRAP[6] = 0.0; // ALP1
390 
391  // top
392  parTRAP[7] = parSCM0[2] / 2.; // H2
393  parTRAP[8] = parSCM0[1] / 2.; // BL2
394  parTRAP[9] = parTRAP[8]; // TL2
395  parTRAP[10] = 0.0; // ALP2
396 
397  LOG(DEBUG2) << " ** TRAP ** ";
398  for (Int_t i = 0; i < 11; i++)
399  LOG(DEBUG3) << " par[" << std::setw(2) << std::setprecision(2) << i << "] " << std::setw(9)
400  << std::setprecision(4) << parTRAP[i];
401 
402  TVirtualMC::GetMC()->Gsvolu("SCMX", "TRAP", getMediumID(ID_SC), parTRAP, 11);
403  xpos = +(parSCM0[1] + parSCM0[0]) / 4.;
404  TVirtualMC::GetMC()->Gspos("SCMX", 1, "SCMY", xpos, 0.0, 0.0, 0, "ONLY");
405 
406  // Using rotation because SCMX should be the same due to Pb tiles
407  xpos = -xpos;
408  Int_t rotMatrixID(-1);
409  Matrix(rotMatrixID, 90.0, 180., 90.0, 270.0, 0.0, 0.0);
410  TVirtualMC::GetMC()->Gspos("SCMX", 2, "SCMY", xpos, 0.0, 0.0, rotMatrixID, "ONLY");
411 
412  // put LED to the SCM0
413  const ShishKebabTrd1Module& mod = g->GetShishKebabTrd1Modules()[0];
414  Double_t tanBetta = mod.GetTanBetta();
415 
416  Int_t nr = 0;
417  ypos = 0.0;
418  Double_t xCenterSCMX = (parTRAP[4] + parTRAP[8]) / 2.;
419  if (!contains(gn, "V1")) {
420  par[1] = parSCM0[2] / 2; // y
421  par[2] = g->GetECPbRadThick() / 2.; // z
422  TVirtualMC::GetMC()->Gsvolu("PBTI", "BOX", getMediumID(ID_PB), dummy, 0);
423 
424  zpos = -mSampleWidth * g->GetNECLayers() / 2. + g->GetECPbRadThick() / 2.;
425  LOG(DEBUG2) << " Pb tiles ";
426 
427  for (Int_t iz = 0; iz < g->GetNECLayers(); iz++) {
428  par[0] = (parSCM0[0] + tanBetta * mSampleWidth * iz) / 2.;
429  xpos = par[0] - xCenterSCMX;
430  TVirtualMC::GetMC()->Gsposp("PBTI", ++nr, "SCMX", xpos, ypos, zpos, 0, "ONLY", par, 3);
431  LOG(DEBUG3) << iz + 1 << " xpos " << xpos << " zpos " << zpos << " par[0] " << par[0];
432  zpos += mSampleWidth;
433  }
434 
435  LOG(DEBUG2) << " Number of Pb tiles in SCMX " << nr;
436  } else {
437  // Oct 26, 2010
438  // First sheet of paper
439  par[1] = parSCM0[2] / 2.; // y
440  par[2] = g->GetTrd1BondPaperThick() / 2.; // z
441  par[0] = parSCM0[0] / 2.; // x
442  TVirtualMC::GetMC()->Gsvolu("PAP1", "BOX", getMediumID(ID_PAPER), par, 3);
443 
444  xpos = par[0] - xCenterSCMX;
445  zpos = -parSCM0[3] + g->GetTrd1BondPaperThick() / 2.;
446  TVirtualMC::GetMC()->Gspos("PAP1", 1, "SCMX", xpos, ypos, zpos, 0, "ONLY");
447 
448  for (auto iz : boost::irange(0, g->GetNECLayers() - 1)) {
449  nr = iz + 1;
450  Double_t dz = g->GetECScintThick() + g->GetTrd1BondPaperThick() + mSampleWidth * iz;
451 
452  // PB + 2 paper sheets
453  par[2] = g->GetECPbRadThick() / 2. + g->GetTrd1BondPaperThick(); // z
454  par[0] = (parSCM0[0] + tanBetta * dz) / 2.;
455  TString pa(Form("PA%2.2i", nr));
456  TVirtualMC::GetMC()->Gsvolu(pa.Data(), "BOX", getMediumID(ID_PAPER), par, 3);
457 
458  xpos = par[0] - xCenterSCMX;
459  zpos = -parSCM0[3] + dz + par[2];
460  TVirtualMC::GetMC()->Gspos(pa.Data(), 1, "SCMX", xpos, ypos, zpos, 0, "ONLY");
461 
462  // Pb
463  TString pb(Form("PB%2.2i", nr));
464  par[2] = g->GetECPbRadThick() / 2.; // z
465  TVirtualMC::GetMC()->Gsvolu(pb.Data(), "BOX", getMediumID(ID_PB), par, 3);
466  TVirtualMC::GetMC()->Gspos(pb.Data(), 1, pa.Data(), 0.0, 0.0, 0.0, 0, "ONLY");
467  }
468  }
469  }
