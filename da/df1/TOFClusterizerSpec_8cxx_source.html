31 // use the tasking system of DPL
32 // just need to implement 2 special methods init + run (there is no need to inherit from anything)
33 class TOFDPLClustererTask
34 {
35  using MCLabelContainer = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
36 
37  public:
38  void init(framework::InitContext& ic)
39  {
40  // nothing special to be set up
41  }
42 
43  void run(framework::ProcessingContext& pc)
44  {
45  static bool finished = false;
46  if (finished) {
47  return;
48  }
49  // get digit data
50  auto digits = pc.inputs().get<std::vector<std::vector<o2::tof::Digit>>*>("tofdigits");
51  auto digitlabels = pc.inputs().get<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>*>("tofdigitlabels");
52  mClusterer.setMCTruthContainer(&mClsLabels);
53 
54  // call actual clustering routine
55  mClustersArray.clear();
56  mClsLabels.clear();
57 
58  for (int i = 0; i < digits->size(); i++) {
59  printf("# TOF readout window for clusterization = %i\n", i);
60  auto digitsRO = digits->at(i);
61  mReader.setDigitArray(&digitsRO);
62  mClusterer.process(mReader, mClustersArray, &(digitlabels->at(i)));
63  }
64  LOG(INFO) << "TOF CLUSTERER : TRANSFORMED " << digits->size()
65  << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
66 
67  // send clusters
68  pc.outputs().snapshot(Output{ "TOF", "CLUSTERS", 0, Lifetime::Timeframe }, mClustersArray);
69  // send labels
70  pc.outputs().snapshot(Output{ "TOF", "CLUSTERSMCTR", 0, Lifetime::Timeframe }, mClsLabels);
71 
72  // declare done
73  finished = true;
74  pc.services().get<ControlService>().readyToQuit(false);
75  }
76 
77  private:
78  DigitDataReader mReader;
79  Clusterer mClusterer;
80 
81  std::vector<Cluster> mClustersArray;
82  MCLabelContainer mClsLabels;
83 };
