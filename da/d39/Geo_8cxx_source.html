31 void Geo::Init()

38  rotationAngles[ii] *= TMath::DegToRad();

42  rotationAngles[1] = 90. * TMath::DegToRad() + rotationAngles[5];
43 
44  for (Int_t ii = 0; ii < 3; ii++) {
45  mRotationMatrixSector[isector][ii][0] =
46  TMath::Sin(rotationAngles[2 * ii]) * TMath::Cos(rotationAngles[2 * ii + 1]);
47  mRotationMatrixSector[isector][ii][1] =
48  TMath::Sin(rotationAngles[2 * ii]) * TMath::Sin(rotationAngles[2 * ii + 1]);
49  mRotationMatrixSector[isector][ii][2] = TMath::Cos(rotationAngles[2 * ii]);
50  }

55  rotationAngles[2] = 0.;
56  rotationAngles[3] = 0.;
57  rotationAngles[4] = 90. * TMath::DegToRad();
58  rotationAngles[5] = 270. * TMath::DegToRad();
59  for (Int_t ii = 0; ii < 3; ii++) {
60  mRotationMatrixSector[NSECTORS][ii][0] =
61  TMath::Sin(rotationAngles[2 * ii]) * TMath::Cos(rotationAngles[2 * ii + 1]);

63  TMath::Sin(rotationAngles[2 * ii]) * TMath::Sin(rotationAngles[2 * ii + 1]);

87  rotationAngles[4] = -getAngles(iplate, istrip) * TMath::DegToRad();
88  rotationAngles[5] = 270. * TMath::DegToRad();
89  rotationAngles[2] = 90. * TMath::DegToRad() - rotationAngles[4];
90  }

343 Int_t Geo::fromPlateToStrip(Float_t* pos, Int_t iplate)
344 {

346  return -1;
347  }
348 
349  Int_t nstrips = 0;
350  switch (iplate) {
351  case 0:
352  case 4:

354  break;
355  case 1:
356  case 3:
357  nstrips = NSTRIPB;

364  constexpr Float_t HGLFY = HFILIY + 2 * HGLASSY; // heigth of GLASS+FISHLINE Layer

367  Float_t step[3];

370  for (Int_t istrip = 0; istrip < nstrips; istrip++) {

372

375  step[2] = -getDistances(iplate, istrip);
376  translate(posLoc2, step);
377  rotateToStrip(posLoc2, iplate, istrip);
378  if ((TMath::Abs(posLoc2[0]) <= STRIPLENGTH * 0.5) && (TMath::Abs(posLoc2[1]) <= HSTRIPY * 0.5) &&
379  (TMath::Abs(posLoc2[2]) <= WCPCBZ * 0.5)) {
380  step[0] = -0.5 * NPADX * XPAD;
381  step[1] = 0.;
382  step[2] = -0.5 * NPADZ * ZPAD;
383  translate(posLoc2, step);
384 
385  for (Int_t jj = 0; jj < 3; jj++)
386  pos[jj] = posLoc2[jj];
387

389  }
390  }

449 Int_t Geo::getPlate(const Float_t* pos)
450 {
451  //
452  // Returns the Plate index
453  //
454 
455  Int_t iPlate = -1;
456 
457  Float_t yLocal = pos[1];
458  Float_t zLocal = pos[2];
459 
460  Float_t deltaRhoLoc = (RMAX - RMIN) * 0.5 - MODULEWALLTHICKNESS + yLocal;
461  Float_t deltaZetaLoc = TMath::Abs(zLocal);
462 
463  Float_t deltaRHOmax = 0.;
464 
465  if (TMath::Abs(zLocal) >= EXTERINTERMODBORDER1 && TMath::Abs(zLocal) <= EXTERINTERMODBORDER2) {
466  deltaRhoLoc -= LENGTHEXINMODBORDER;
467  deltaZetaLoc = EXTERINTERMODBORDER2 - deltaZetaLoc;
468  deltaRHOmax = (RMAX - RMIN) * 0.5 - MODULEWALLTHICKNESS - 2. * LENGTHEXINMODBORDER; // old 5.35, new 4.8
469 
470  if (deltaRhoLoc > deltaZetaLoc * deltaRHOmax / (INTERCENTRMODBORDER2 - INTERCENTRMODBORDER1)) {
471  if (zLocal < 0)
472  iPlate = 0;
473  else
474  iPlate = 4;
475  } else {
476  if (zLocal < 0)
477  iPlate = 1;
478  else
479  iPlate = 3;
480  }
481  } else if (TMath::Abs(zLocal) >= INTERCENTRMODBORDER1 && TMath::Abs(zLocal) <= INTERCENTRMODBORDER2) {
482  deltaRhoLoc -= LENGTHINCEMODBORDERD;
483  deltaZetaLoc = deltaZetaLoc - INTERCENTRMODBORDER1;
484  deltaRHOmax = (RMAX - RMIN) * 0.5 - MODULEWALLTHICKNESS - 2. * LENGTHINCEMODBORDERD; // old 0.39, new 0.2
485 
486  if (deltaRhoLoc > deltaZetaLoc * deltaRHOmax / (INTERCENTRMODBORDER2 - INTERCENTRMODBORDER1))
487  iPlate = 2;
488  else {
489  if (zLocal < 0)
490  iPlate = 1;
491  else
492  iPlate = 3;
493  }
494  }
495 
496  if (zLocal > -ZLENA * 0.5 && zLocal < -EXTERINTERMODBORDER2)
497  iPlate = 0;
498  else if (zLocal > -EXTERINTERMODBORDER1 && zLocal < -INTERCENTRMODBORDER2)
499  iPlate = 1;
500  else if (zLocal > -INTERCENTRMODBORDER1 && zLocal < INTERCENTRMODBORDER1)
501  iPlate = 2;
502  else if (zLocal > INTERCENTRMODBORDER2 && zLocal < EXTERINTERMODBORDER1)
503  iPlate = 3;
504  else if (zLocal > EXTERINTERMODBORDER2 && zLocal < ZLENA * 0.5)
505  iPlate = 4;
506 
507  return iPlate;
508 }
