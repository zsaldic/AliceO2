109  // only do it, if not in pure trackSeeding mode

111  auto& conf = o2::conf::SimConfig::Instance();
112  auto chunks = conf.getInternalChunkSize();
113  if (chunks != -1) {
114  if (primcounter % chunks == 0) {
115  static int counter = 1;
116  auto seed = counter + 10;
117  gRandom->SetSeed(seed);
118  counter++;
119  }
120  }
121  }
122  primcounter++;
123 }
