27 CompletionPolicy CompletionPolicyHelpers::defineByName(std::string const&name, CompletionPolicy::CompletionOp op) {
28  auto matcher = [name](DeviceSpec const &device) -> bool {
29  return device.name == name;
30  };
31  auto callback = [op](gsl::span<PartRef const> const &inputs) -> CompletionPolicy::CompletionOp {
32  return op;
33  };
34  switch (op) {
35  case CompletionPolicy::CompletionOp::Consume:
36  return CompletionPolicy{"always-consume", matcher, callback };
37  break;
38  case CompletionPolicy::CompletionOp::Process:
39  return CompletionPolicy{"always-process", matcher, callback };
40  break;
41  case CompletionPolicy::CompletionOp::Wait:
42  return CompletionPolicy{"always-wait", matcher, callback };
43  break;
44  case CompletionPolicy::CompletionOp::Discard:
45  return CompletionPolicy{"always-discard", matcher, callback };
46  break;
47  }
48  O2_BUILTIN_UNREACHABLE();

51 CompletionPolicy CompletionPolicyHelpers::consumeWhenAll() {
52  auto matcher = [](DeviceSpec const &) -> bool { return true; };
53  auto callback = [](gsl::span<PartRef const> const &inputs) -> CompletionPolicy::CompletionOp {
54  for (auto &input : inputs) {
55  if (input.header == nullptr && input.payload == nullptr) {
56  return CompletionPolicy::CompletionOp::Wait;
57  }
58  }
59  return CompletionPolicy::CompletionOp::Consume;
60  };
61  return CompletionPolicy{"consume-all", matcher, callback };

77 CompletionPolicy CompletionPolicyHelpers::processWhenAny() {
78  auto matcher = [](DeviceSpec const &) -> bool { return true; };
79  auto callback = [](gsl::span<PartRef const> const &inputs) -> CompletionPolicy::CompletionOp {
80  size_t present = 0;
81  for (auto &input : inputs) {
82  if (input.header != nullptr) {
83  present++;
84  }
85  }
86  if (present == inputs.size()) {
87  return CompletionPolicy::CompletionOp::Consume;
88  } else if (present == 0) {
89  return CompletionPolicy::CompletionOp::Wait;
90  }
91  return CompletionPolicy::CompletionOp::Process;
92  };
93  return CompletionPolicy{"process-any", matcher, callback };
94 }
