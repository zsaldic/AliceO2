40 int LookUp::groupFinder(int nRow, int nCol)
41 {
42  int row_index = nRow / TopologyDictionary::RowClassSpan;
43  if (nRow % TopologyDictionary::RowClassSpan == 0) {
44  row_index--;
45  }
46  int col_index = nCol / TopologyDictionary::ColClassSpan;
47  if (nCol % TopologyDictionary::RowClassSpan == 0) {
48  col_index--;
49  }
50  if (row_index > TopologyDictionary::MaxNumberOfClasses || col_index > TopologyDictionary::MaxNumberOfClasses) {
51  return TopologyDictionary::NumberOfRareGroups - 1;
52  } else {
53  return row_index * TopologyDictionary::MaxNumberOfClasses + col_index;
54  }
55 }

57 int LookUp::findGroupID(int nRow, int nCol, const unsigned char patt[Cluster::kMaxPatternBytes])
58 {
59  int nBits = nRow * nCol;
60  // Small topology
61  if (nBits < 9) {
62  int ID = mDictionary.mSmallTopologiesLUT[(nCol - 1) * 255 + (int)patt[0]];
63  if (ID >= 0)
64  return ID;
65  else { //small rare topology (inside groups)
66  int index = groupFinder(nRow, nCol);
67  return (mTopologiesOverThreshold + index);
68  }
69  }
70  // Big topology
71  unsigned long hash = ClusterTopology::getCompleteHash(nRow, nCol, patt);
72  auto ret = mDictionary.mFinalMap.find(hash);
73  if (ret != mDictionary.mFinalMap.end())
74  return ret->second;
75  else { // Big rare topology (inside groups)
76  int index = groupFinder(nRow, nCol);
77  return (mTopologiesOverThreshold + index);
