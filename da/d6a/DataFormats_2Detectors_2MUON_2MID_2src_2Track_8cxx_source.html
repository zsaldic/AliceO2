79 bool Track::propagateToZ(float zPosition)
80 {
83 
84  // Nothing to be done if we're already at zPosition
85  if (mPosition.z() == zPosition) {
86  return false;
87  }
88 
89  // Propagate the track position parameters
90  float dZ = zPosition - mPosition.z();
91  float newX = mPosition.x() + mDirection.x() * dZ;
92  float newY = mPosition.y() + mDirection.y() * dZ;
93  setPosition(newX, newY, zPosition);
94 
95  // Propagate the covariance matrix
96  std::array<float, 6> newCovParams;
97  float dZ2 = dZ * dZ;
98  for (int idx = 0; idx < 2; ++idx) {
99  int slopeIdx = idx + 2;
100  int covIdx = idx + 4;
101  // s_x^2 -> s_x^2 + 2*cov(x,slopeX)*dZ + s_slopeX^2*dZ^2
102  newCovParams[idx] =
103  mCovarianceParameters[idx] + 2. * mCovarianceParameters[covIdx] * dZ + mCovarianceParameters[slopeIdx] * dZ2;
104  // cov(x,slopeX) -> cov(x,slopeX) + s_slopeX^2*dZ
105  newCovParams[covIdx] = mCovarianceParameters[covIdx] + mCovarianceParameters[slopeIdx] * dZ;
106  // s_slopeX^2 -> s_slopeX^2
107  newCovParams[slopeIdx] = mCovarianceParameters[slopeIdx];
108  }
109 
110  mCovarianceParameters.swap(newCovParams);
111 
112  return true;
113 }

116 bool Track::isCompatible(const Track& track, float chi2Cut) const
117 {
119  if (track.getPosition().z() != getPosition().z()) {
120  Track copyTrack(track);
121  copyTrack.propagateToZ(getPosition().z());
122  return isCompatible(copyTrack, chi2Cut);
123  }

125  // // method 1: chi2 calculation accounting for covariance between slope and position.
126  // // This is the full calculation. However, if we have two parallel tracks with same x
127  // // but different ( > Nsigmas) y position, the algorithm will return true since the
128  // // difference in one of the parameters is compensated by the similarity of the others.
129  // // This is probably not what we need.
130  // double chi2 = 0.;
131  // double pos1[2] = { getPosition().x(), getPosition().y() };
132  // double dir1[2] = { getDirection().x(), getDirection().y() };
133  // double pos2[2] = { track.getPosition().x(), track.getPosition().y() };
134  // double dir2[2] = { track.getDirection().x(), track.getDirection().y() };
135  //
136  // for (int icoor = 0; icoor < 2; ++icoor) {
137  // double diffPos = pos1[icoor] - pos2[icoor];
138  // double diffSlope = dir1[icoor] - dir2[icoor];
139  // double varPos = mCovarianceParameters[icoor] + track.mCovarianceParameters[icoor];
140  // double varSlope = mCovarianceParameters[icoor + 2] + track.mCovarianceParameters[icoor + 2];
141  // double cov = mCovarianceParameters[icoor + 4] + track.mCovarianceParameters[icoor + 4];
142  // chi2 += (diffPos * diffPos * varSlope + diffSlope * diffSlope * varPos - 2. * diffPos * diffSlope * cov) /
143  // (varPos * varSlope - cov * cov);
144  // }
145  //
146  // return (chi2 / 4.) < chi2Cut;
147 
148  // method 2: apply the cut on each parameter
149  // This method avoids the issue of method 1
150  double p1[4] = { getPosition().x(), getPosition().y(), getDirection().x(), getDirection().y() };
151  double p2[4] = { track.getPosition().x(), track.getPosition().y(), track.getDirection().x(),
152  track.getDirection().y() };
153  for (int ipar = 0; ipar < 4; ++ipar) {
154  double diff = p1[ipar] - p2[ipar];
155  if (diff * diff / (mCovarianceParameters[ipar] + track.mCovarianceParameters[ipar]) > chi2Cut) {
156  return false;
157  };
158  }
159  return true;
160 }
