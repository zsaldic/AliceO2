136  {
137  if (digit1.getChannel() != digit2.getChannel()) {
138  return false;
139  }
140

142  return false;
143  }
144

146  if (std::abs(digit1.getTDC() - digit2.getTDC()) > Geo::DEADTIMETDC) {
147  return false;
148  }
149  return true;
150  }
