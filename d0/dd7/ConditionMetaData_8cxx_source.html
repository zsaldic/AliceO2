75 void ConditionMetaData::addDateToComment()
76 {
77  // add the date to the comment.
78  // This method is supposed to be useful if called at the time when the object
79  // is created, so that later it can more easily be tracked, in particular
80  // when the date of the file can be lost or when one is interested in the
81  // date of creation, irrespective of a later copy of it
82 
83  TTimeStamp ts(time(nullptr));
84  TString comment(getComment());
85  comment += Form("\tDate of production: %s\n", ts.AsString());
86  comment.Remove(comment.Last('+'));
87  setComment(comment);

90 void ConditionMetaData::printConditionMetaData()
91 {
92  // print the object's metaData
93 
94  TString message;
95  if (mObjectClassName != "") {
96  message += TString::Format("\tObject's class name: %s\n", mObjectClassName.Data());
97  }
98  if (mResponsible != "") {
99  message += TString::Format("\tResponsible: %s\n", mResponsible.Data());
100  }
101  if (mBeamPeriod != 0) {
102  message += TString::Format("\tBeam period: %d\n", mBeamPeriod);
103  }
104  if (mAliRootVersion != "") {
105  message += TString::Format("\tAliRoot version: %s\n", mAliRootVersion.Data());
106  }
107  if (mComment != "") {
108  message += TString::Format("\tComment: %s\n", mComment.Data());
109  }
110  if (mProperties.GetEntries() > 0) {
111  message += "\tProperties key names:";
112 
113  TIter iter(mProperties.GetTable());
114  TPair *aPair;
115  while ((aPair = (TPair *) iter.Next())) {
116  message += TString::Format("\t\t%s\n", ((TObjString *) aPair->Key())->String().Data());
117  }
118  }
119  message += '\n';
120  Printf("**** Object's ConditionMetaData parameters **** \n%s", message.Data());
121 }
