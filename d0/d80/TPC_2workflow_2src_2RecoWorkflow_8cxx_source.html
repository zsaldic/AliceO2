69 framework::WorkflowSpec getWorkflow(std::vector<int> const& tpcSectors, std::vector<int> const& laneConfiguration,
70  bool propagateMC, unsigned nLanes, std::string const& cfgInput, std::string const& cfgOutput)
71 {
72  InputType inputType;
73 
74  try {
75  inputType = InputMap.at(cfgInput);
76  } catch (std::out_of_range&) {
77  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
78  }
79  std::vector<OutputType> outputTypes;
80  try {
81  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
82  } catch (std::out_of_range&) {
83  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
84  }
85  auto isEnabled = [&outputTypes](OutputType type) {
86  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
87  };
88 
89  if (inputType == InputType::Raw && isEnabled(OutputType::Digits)) {
90  throw std::invalid_argument("input/output type mismatch, can not produce 'digits' from 'raw'");
91  }
92  if (inputType == InputType::DecodedClusters && (isEnabled(OutputType::Digits) || isEnabled(OutputType::Raw))) {
93  throw std::invalid_argument("input/output type mismatch, can not produce 'digits', nor 'raw' from 'decoded-clusters");
94  }
95 
96  WorkflowSpec specs;
97 
98  if (inputType == InputType::Digits) {
99  specs.emplace_back(o2::TPC::getPublisherSpec(PublisherConf{
100  "tpc-digit-reader",
101  "o2sim",
102  { "digitbranch", "TPCDigit", "Digit branch" },
103  { "mcbranch", "TPCDigitMCTruth", "MC label branch" },
104  OutputSpec{ "TPC", "DIGITS" },
105  OutputSpec{ "TPC", "DIGITSMCTR" },
106  tpcSectors,
107  laneConfiguration,
108  },
109  propagateMC));
110  } else if (inputType == InputType::Raw) {
111  specs.emplace_back(o2::TPC::getPublisherSpec(PublisherConf{
112  "tpc-raw-cluster-reader",
113  "tpcraw",
114  { "databranch", "TPCClusterHw", "Branch with raw clusters" },
115  { "mcbranch", "TPCClusterHwMCTruth", "MC label branch" },
116  OutputSpec{ "TPC", "CLUSTERHW" },
117  OutputSpec{ "TPC", "CLUSTERHWMCLBL" },
118  tpcSectors,
119  laneConfiguration,
120  },
121  propagateMC));
122  } else if (inputType == InputType::DecodedClusters) {
123  specs.emplace_back(o2::TPC::getPublisherSpec(PublisherConf{
124  "tpc-decoded-cluster-reader",
125  "tpcrec",
126  { "clusterbranch", "TPCClusterNative", "Branch with decoded clusters" },
127  { "clustermcbranch", "TPCClusterNativeMCTruth", "MC label branch" },
128  OutputSpec{ "TPC", "CLUSTERNATIVE" },
129  OutputSpec{ "TPC", "CLNATIVEMCLBL" },
130  tpcSectors,
131  laneConfiguration,
132  },
133  propagateMC));
134  }
135 
136  // output matrix
137  bool runTracker = isEnabled(OutputType::Tracks);
138  bool runDecoder = runTracker || isEnabled(OutputType::DecodedClusters);
139  bool runClusterer = runDecoder || isEnabled(OutputType::Raw);
140 
141  // input matrix
142  runClusterer &= inputType == InputType::Digitizer || inputType == InputType::Digits;
143  runDecoder &= runClusterer || inputType == InputType::Raw;
144  runTracker &= runDecoder || inputType == InputType::DecodedClusters;
145 
146  WorkflowSpec parallelProcessors;
148  //
149  // clusterer process(es)
150  //
151  //
152  if (runClusterer) {
153  parallelProcessors.push_back(o2::TPC::getClustererSpec(propagateMC, inputType == InputType::Digitizer));
154  }
155 
157  //
158  // cluster decoder process(es)
159  //
160  //
161  if (runDecoder) {
162  parallelProcessors.push_back(o2::TPC::getClusterDecoderRawSpec(propagateMC));
163  }
164 
166  //
167  // set up parallel TPC lanes
168  //
169  WorkflowSpec lanePocessors;
170  std::swap(lanePocessors, parallelProcessors);
171  auto const& conf = laneConfiguration;
172  size_t confSize = laneConfiguration.size();
173  size_t lanes = nLanes < confSize ? nLanes : confSize;
174  for (auto laneProcess : lanePocessors) {
175  size_t confIndex = 0;
176  size_t moreInputs = confSize % nLanes;
177  size_t inputMultiplicity = confSize / lanes;
178  if (moreInputs > 0) {
179  inputMultiplicity += 1;
180  }
181  auto amendProcess = [conf, lanes, &confIndex, &inputMultiplicity, &moreInputs](DataProcessorSpec& spec, size_t) {
182  auto inputs = std::move(spec.inputs);
183  auto outputs = std::move(spec.outputs);
184  spec.inputs.reserve(inputMultiplicity);
185  spec.outputs.reserve(inputMultiplicity);
186  for (size_t inputNo = 0; inputNo < inputMultiplicity; ++inputNo) {
187  for (auto& input : inputs) {
188  spec.inputs.push_back(input);
189  spec.inputs.back().binding += std::to_string(inputNo);
190  DataSpecUtils::updateMatchingSubspec(spec.inputs.back(), conf[confIndex + inputNo]);
191  }
192  for (auto& output : outputs) {
193  spec.outputs.push_back(output);
194  spec.outputs.back().binding.value += std::to_string(inputNo);
195  spec.outputs.back().subSpec = conf[confIndex + inputNo];
196  }
197  }
198  confIndex += inputMultiplicity;
199  if (moreInputs > 0) {
200  --moreInputs;
201  if (moreInputs == 0) {
202  --inputMultiplicity;
203  }
204  }
205  };
206 
207  if (nLanes > 1) {
208  // add multiple processes and distribute inputs among them
209  parallelProcessors = parallel(laneProcess, lanes, amendProcess);
210  specs.insert(specs.end(), parallelProcessors.begin(), parallelProcessors.end());
211  } else if (nLanes == 1) {
212  // add one single process with all the inputs
213  amendProcess(laneProcess, 0);
214  specs.push_back(laneProcess);
215  }
216  }

219  //
220  // generation of processor specs for various types of outputs
221  // based on generic RootTreeWriter and MakeRootTreeWriterSpec generator
222  //
223  // -------------------------------------------------------------------------------------------
224  // the callbacks for the RootTreeWriter
225  //
226  // The generic writer needs a way to associate incoming data with the individual branches for
227  // the TPC sectors. The sector number is transmitted as part of the sector header, the callback
228  // finds the corresponding index in the vector of configured sectors
229  auto getIndex = [tpcSectors](o2::framework::DataRef const& ref) {
230  auto const* tpcSectorHeader = o2::framework::DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(ref);
231  if (!tpcSectorHeader) {
232  throw std::runtime_error("TPC sector header missing in header stack");
233  }
234  if (tpcSectorHeader->sector < 0) {
235  // special data sets, don't write
236  return ~(size_t)0;
237  }
238  size_t index = 0;
239  for (auto const& sector : tpcSectors) {
240  if (sector == tpcSectorHeader->sector) {
241  return index;
242  }
243  ++index;
244  }
245  throw std::runtime_error("sector " + std::to_string(tpcSectorHeader->sector) + " not configured for writing");
246  };
247  auto getName = [tpcSectors](std::string base, size_t index) {
248  return base + "_" + std::to_string(tpcSectors.at(index));
249  };
250 
251  // check if the process is ready to quit
252  // this is decided upon the meta information in the TPC sector header, the operation is set as
253  // a negative number in the sector member, -2 indicates no-operation, -1 indicates end-of-data
254  // see also PublisherSpec.cxx
255  // in this workflow, the EOD is sent after the last real data, and all inputs will receive EOD,
256  // so it is enough to check on the first occurence
257  // FIXME: this will be changed once DPL can propagate control events like EOD
258  auto checkReady = [](o2::framework::DataRef const& ref, bool& isReady) {
259  auto const* tpcSectorHeader = o2::framework::DataRefUtils::getHeader<o2::TPC::TPCSectorHeader*>(ref);
260  // sector number -1 indicates end-of-data
261  if (tpcSectorHeader != nullptr) {
262  isReady = tpcSectorHeader->sector == -1;
263  // indicate normal processing if not ready and skip if ready
264  if (isReady) {
265  return MakeRootTreeWriterSpec::TerminationCondition::Action::SkipProcessing;
266  }
267  }
268  return MakeRootTreeWriterSpec::TerminationCondition::Action::DoProcessing;
269  };

283  input.binding += std::to_string(laneConfiguration[index]);

287  def.keys = mergeInputs(def.keys, laneConfiguration.size(), amendInput);
