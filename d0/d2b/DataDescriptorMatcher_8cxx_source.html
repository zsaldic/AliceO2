51 bool OriginValueMatcher::match(header::DataHeader const& header, VariableContext& context) const
52 {
53  if (auto ref = std::get_if<ContextRef>(&mValue)) {
54  auto& variable = context.get(ref->index);
55  if (auto value = std::get_if<std::string>(&variable)) {
56  return strncmp(header.dataOrigin.str, value->c_str(), 4) == 0;
57  }
58  auto maxSize = strnlen(header.dataOrigin.str, 4);
59  context.put({ ref->index, std::string(header.dataOrigin.str, maxSize) });
60  return true;
61  } else if (auto s = std::get_if<std::string>(&mValue)) {
62  return strncmp(header.dataOrigin.str, s->c_str(), 4) == 0;
63  }
64  throw std::runtime_error("Mismatching type for variable");
65 }

193 // actual polymorphic matcher which is able to cast the pointer to the correct
194 // kind of header.
195 bool DataDescriptorMatcher::match(char const* d, VariableContext& context) const
196 {
197  bool leftValue = false, rightValue = false;
198 
199  // FIXME: Using std::visit is not API compatible due to a new
200  // exception being thrown. This is the ABI compatible version.
201  // Replace with:
202  //
203  // auto eval = [&d](auto&& arg) -> bool {
204  // using T = std::decay_t<decltype(arg)>;
205  // if constexpr (std::is_same_v<T, std::unique_ptr<DataDescriptorMatcher>>) {
206  // return arg->match(d, context);
207  // if constexpr (std::is_same_v<T, ConstantValueMatcher>) {
208  // return arg->match(d);
209  // } else {
210  // return arg.match(d, context);
211  // }
212  // };
213  // switch (mOp) {
214  // case Op::Or:
215  // return std::visit(eval, mLeft) || std::visit(eval, mRight);
216  // case Op::And:
217  // return std::visit(eval, mLeft) && std::visit(eval, mRight);
218  // case Op::Xor:
219  // return std::visit(eval, mLeft) ^ std::visit(eval, mRight);
220  // case Op::Just:
221  // return std::visit(eval, mLeft);
222  // }
223  // When we drop support for macOS 10.13
224  if (auto pval0 = std::get_if<OriginValueMatcher>(&mLeft)) {
225  auto dh = o2::header::get<header::DataHeader*>(d);
226  if (dh == nullptr) {
227  throw std::runtime_error("Cannot find DataHeader");
228  }
229  leftValue = pval0->match(*dh, context);
230  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&mLeft)) {
231  auto dh = o2::header::get<header::DataHeader*>(d);
232  if (dh == nullptr) {
233  throw std::runtime_error("Cannot find DataHeader");
234  }
235  leftValue = pval1->match(*dh, context);
236  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&mLeft)) {
237  auto dh = o2::header::get<header::DataHeader*>(d);
238  if (dh == nullptr) {
239  throw std::runtime_error("Cannot find DataHeader");
240  }
241  leftValue = pval2->match(*dh, context);
242  } else if (auto pval3 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&mLeft)) {
243  leftValue = (*pval3)->match(d, context);
244  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&mLeft)) {
245  leftValue = pval4->match();
246  } else if (auto pval5 = std::get_if<StartTimeValueMatcher>(&mLeft)) {
247  auto dph = o2::header::get<DataProcessingHeader*>(d);
248  if (dph == nullptr) {
249  throw std::runtime_error("Cannot find DataProcessingHeader");
250  }
251  leftValue = pval5->match(*dph, context);
252  } else {
253  throw std::runtime_error("Bad parsing tree");
254  }
255  // Common speedup.
256  if (mOp == Op::And && leftValue == false) {
257  return false;
258  }
259  if (mOp == Op::Or && leftValue == true) {
260  return true;
261  }
262  if (mOp == Op::Just) {
263  return leftValue;
264  }
265 
266  if (auto pval0 = std::get_if<OriginValueMatcher>(&mRight)) {
267  auto dh = o2::header::get<header::DataHeader*>(d);
268  rightValue = pval0->match(*dh, context);
269  } else if (auto pval1 = std::get_if<DescriptionValueMatcher>(&mRight)) {
270  auto dh = o2::header::get<header::DataHeader*>(d);
271  rightValue = pval1->match(*dh, context);
272  } else if (auto pval2 = std::get_if<SubSpecificationTypeValueMatcher>(&mRight)) {
273  auto dh = o2::header::get<header::DataHeader*>(d);
274  rightValue = pval2->match(*dh, context);
275  } else if (auto pval3 = std::get_if<std::unique_ptr<DataDescriptorMatcher>>(&mRight)) {
276  rightValue = (*pval3)->match(d, context);
277  } else if (auto pval4 = std::get_if<ConstantValueMatcher>(&mRight)) {
278  rightValue = pval4->match();
279  } else if (auto pval5 = std::get_if<StartTimeValueMatcher>(&mRight)) {
280  auto dph = o2::header::get<DataProcessingHeader*>(d);
281  rightValue = pval5->match(*dph, context);
282  }
283  // There are cases in which not having a rightValue might be legitimate,
284  // so we do not throw an exception.
285  switch (mOp) {
286  case Op::Or:
287  return leftValue || rightValue;
288  case Op::And:
289  return leftValue && rightValue;
290  case Op::Xor:
291  return leftValue ^ rightValue;
292  case Op::Just:
293  return leftValue;
294  }
295  throw std::runtime_error("Bad parsing tree");
296 };
