158 int PreClusterBlock::readBuffer()
159 {
162 
163  assert(!mWriteMode);
164 
165  // make sure mNPreClusters was assigned
166  if (mNPreClusters == nullptr) {
167  return -EILSEQ;
168  }
169 
170  // initialize status
171  int status = 0;
172 
173  // rolling pre-cluster
174  PreClusterStruct preCluster = {};
175 
176  // loop over
177  for (int i = 0; i < *mNPreClusters; ++i) {
178 
179  // store number of digits from buffer and increment
180  uint16_t* nDigits = nullptr;
181  if (mCurrentSize < mSize) {
182  nDigits = reinterpret_cast<uint16_t*>(mBuffer);
183  mBuffer = (reinterpret_cast<uint16_t*>(mBuffer) + 1);
184  mCurrentSize += SSizeOfUShort;
185  } else {
186  LOG(ERROR) << "Cannot read the expected number of preclusters.";
187  status = -EILSEQ;
188  break;
189  }
190 
191  // read the digits
192  if (nDigits && (*nDigits) > 0 && mCurrentSize + (*nDigits) * SSizeOfDigit <= mSize) {
193 
194  auto digit = reinterpret_cast<DigitStruct*>(mBuffer);
195  mBuffer = (reinterpret_cast<DigitStruct*>(mBuffer) + (*nDigits));
196  mCurrentSize += (*nDigits) * SSizeOfDigit;
197 
198  // store
199  preCluster.nDigits = *nDigits;
200  preCluster.digits = digit;
201  mPreClusters.push_back(preCluster);
202 
203  } else {
204 
205  if (*nDigits == 0) {
206  LOG(ERROR) << "The precluster cannot contain 0 digit.";
207  } else {
208  LOG(ERROR) << "Cannot read the expected number of digits.";
209  }
210 
211  status = -EILSEQ;
212  break;
213  }
214  }
215 
216  // sanity check on read size
217  if (mCurrentSize != mSize && status >= 0) {
218  LOG(ERROR) << "The number of bytes read differs from the buffer size";
219  status = -EILSEQ;
220  }
221 
222  // reset in case of negative status
223  if (status < 0) {
224  mCurrentSize = mSize + 1;
225  mNPreClusters = nullptr;
226  mPreClusters.clear();
227  }
228 
229  return status;
230 }
