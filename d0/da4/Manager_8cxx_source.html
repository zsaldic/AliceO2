150 Bool_t Manager::initFromSnapshot(const char *snapshotFileName, Bool_t overwrite)
151 {
152  // initialize manager from a CDB snapshot, that is add the entries
153  // to the entries map and the ids to the ids list taking them from
154  // the map and the list found in the input file
155 
156  // if the manager is locked it cannot initialize from a snapshot
157  if (mLock) {
158  LOG(ERROR) << "Being locked I cannot initialize from the snapshot!";
159  return kFALSE;
160  }
161 
162  // open the file
163  TString snapshotFile(snapshotFileName);
164  if (snapshotFile.BeginsWith("alien://")) {
165  if (!gGrid) {
166  TGrid::Connect("alien://", "");
167  if (!gGrid) {
168  LOG(ERROR) << "Connection to alien failed!";
169  return kFALSE;
170  }
171  }
172  }
173 
174  TFile *f = TFile::Open(snapshotFileName);
175  if (!f || f->IsZombie()) {
176  LOG(ERROR) << "Cannot open file " << snapshotFileName;
177  return kFALSE;
178  }
179 
180  // retrieve entries' map from snapshot file
181  TMap *entriesMap = nullptr;
182  TIter next(f->GetListOfKeys());
183  TKey *key;
184  while ((key = (TKey *) next())) {
185  if (strcmp(key->GetClassName(), "TMap") != 0) {
186  continue;
187  }
188  entriesMap = (TMap *) key->ReadObj();
189  break;
190  }
191  if (!entriesMap || entriesMap->GetEntries() == 0) {
192  LOG(ERROR) << "Cannot get valid map of CDB entries from snapshot file";
193  return kFALSE;
194  }
195 
196  // retrieve ids' list from snapshot file
197  TList *idsList = nullptr;
198  TIter nextKey(f->GetListOfKeys());
199  TKey *keyN;
200  while ((keyN = (TKey *) nextKey())) {
201  if (strcmp(keyN->GetClassName(), "TList") != 0) {
202  continue;
203  }
204  idsList = (TList *) keyN->ReadObj();
205  break;
206  }
207  if (!idsList || idsList->GetEntries() == 0) {
208  LOG(ERROR) << "Cannot get valid list of CDB entries from snapshot file";
209  return kFALSE;
210  }
211 
212  // Add each (entry,id) from the snapshot to the memory: entry to the cache, id to the list of ids.
213  // If "overwrite" is false: add the entry to the cache and its id to the list of ids
214  // only if neither of them is already there.
215  // If "overwrite" is true: write the snapshot entry,id in any case. If something
216  // was already there for that calibration type, remove it and issue a warning
217  TIter iterObj(entriesMap->GetTable());
218  TPair *pair = nullptr;
219  Int_t nAdded = 0;
220  while ((pair = dynamic_cast<TPair *>(iterObj.Next()))) {
221  TObjString *os = (TObjString *) pair->Key();
222  TString path = os->GetString();
223  TIter iterId(idsList);
224  ConditionId *id = nullptr;
225  ConditionId *correspondingId = nullptr;
226  while ((id = dynamic_cast<ConditionId *>(iterId.Next()))) {
227  TString idpath(id->getPathString());
228  if (idpath == path) {
229  correspondingId = id;
230  break;
231  }
232  }
233  if (!correspondingId) {
234  LOG(ERROR) << R"(id for ")" << path.Data()
235  << R"(" not found in the snapshot (while entry was). This entry is skipped!)";
236  break;
237  }
238  Bool_t cached = mConditionCache.Contains(path.Data());
239  Bool_t registeredId = kFALSE;
240  TIter iter(mIds);
241  ConditionId *idT = nullptr;
242  while ((idT = dynamic_cast<ConditionId *>(iter.Next()))) {
243  if (idT->getPathString() == path) {
244  registeredId = kTRUE;
245  break;
246  }
247  }
248 
249  if (overwrite) {
250  if (cached || registeredId) {
251  LOG(WARNING) << R"(An entry was already cached for ")" << path.Data()
252  << R"(". Removing it before caching from snapshot)";
253  unloadFromCache(path.Data());
254  }
255  mConditionCache.Add(pair->Key(), pair->Value());
256  mIds->Add(id);
257  nAdded++;
258  } else {
259  if (cached || registeredId) {
260  LOG(WARNING) << R"(An entry was already cached for ")" << path.Data()
261  << R"(". Not adding this object from snapshot)";
262  } else {
263  mConditionCache.Add(pair->Key(), pair->Value());
264  mIds->Add(id);
265  nAdded++;
266  }
267  }
268  }
269 
270  // mCondition is the new owner of the cache
271  mConditionCache.SetOwnerKeyValue(kTRUE, kTRUE);
272  entriesMap->SetOwnerKeyValue(kFALSE, kFALSE);
273  mIds->SetOwner(kTRUE);
274  idsList->SetOwner(kFALSE);
275  LOG(INFO) << nAdded << " new (entry,id) cached. Total number " << mConditionCache.GetEntries();
276 
277  f->Close();
278  delete f;
279 
280  return kTRUE;
281 }

408 void Manager::alienToCvmfsUri(TString &uriString) const
409 {
410  // convert alien storage uri to local:///cvmfs storage uri (called when OCDB_PATH is set)
411 
412  TObjArray *arr = uriString.Tokenize('?');
413  TIter iter(arr);
414  TObjString *str = nullptr;
415  TString entryKey = "";
416  TString entryValue = "";
417  TString newUriString = "";
418  while ((str = (TObjString *) iter.Next())) {
419  TString entry(str->String());
420  Int_t indeq = entry.Index('=');
421  entryKey = entry(0, indeq + 1);
422  entryValue = entry(indeq + 1, entry.Length() - indeq);
423 
424  if (entryKey.Contains("folder", TString::kIgnoreCase)) {
425 
426  TRegexp re_RawFolder("^/alice/data/20[0-9]+/OCDB");
427  TRegexp re_MCFolder("^/alice/simulation/2008/v4-15-Release");
428  TString rawFolder = entryValue(re_RawFolder);
429  TString mcFolder = entryValue(re_MCFolder);
430  if (!rawFolder.IsNull()) {
431  entryValue.Replace(0, 6, "/cvmfs/alice-ocdb.cern.ch/calibration");
432  // entryValue.Replace(entryValue.Length()-4, entryValue.Length(), "");
433  } else if (!mcFolder.IsNull()) {
434  entryValue.Replace(0, 36, "/cvmfs/alice-ocdb.cern.ch/calibration/MC");
435  } else {
436  LOG(FATAL) << "Environment variable for cvmfs OCDB folder set for an invalid OCDB storage:\n "
437  << entryValue.Data();
438  }
439  } else {
440  newUriString += entryKey;
441  }
442  newUriString += entryValue;
443  newUriString += '?';
444  }
445  newUriString.Prepend("local://");
446  newUriString.Remove(TString::kTrailing, '?');
447  uriString = newUriString;
448 }

480 {

482

484  // the requested storage, return it

683 void Manager::validateCvmfsCase() const
684 {
685  // The OCDB_PATH variable contains the path to the directory in /cvmfs/ which is
686  // an AliRoot tag based snapshot of the AliEn file catalogue (e.g.
687  // /cvmfs/alice.cern.ch/x86_64-2.6-gnu-4.1.2/Packages/OCDB/v5-05-76-AN).
688  // The directory has to contain:
689  // 1) <data|MC>/20??.list.gz gzipped text files listing the OCDB files (seen by that AliRoot tag)
690  // 2) bin/getOCDBFilesPerRun.sh (shell+awk) script extracting from 1) the list
691  // of valid files for the given run.
692 
693  if (!mCvmfsOcdb.BeginsWith("/cvmfs"))
694  LOG(FATAL) << "OCDB_PATH set to an invalid path: " << mCvmfsOcdb.Data();
695 
696  TString cvmfsUri(mCvmfsOcdb);
697  gSystem->ExpandPathName(cvmfsUri);
698  if (gSystem->AccessPathName(cvmfsUri))
699  LOG(FATAL) << "OCDB_PATH set to an invalid path: " << cvmfsUri.Data();
700
