86 inline void DigitTime::addDigit(const MCCompLabel& label, const CRU& cru, GlobalPadNumber globalPad, float signal)
87 {
88  auto& paddigit = mGlobalPads[globalPad];
89  if (paddigit.getID() == -1) {
90  // this means we have a new digit
91  paddigit.setID(mDigitCounter++);
92  // could also register this pad in a vector of digits
93  }
94  paddigit.addDigit(label, signal, mLabels);
95  mCommonMode[cru.gemStack()] += signal;
96 }

106 inline float DigitTime::getCommonMode(const CRU& cru) const
107 {
108  static const Mapper& mapper = Mapper::instance();
109  const auto gemStack = static_cast<int>(cru.gemStack());
110  const auto nPads = mapper.getNumberOfPads(gemStack);
111  return mCommonMode[gemStack] /
112  static_cast<float>(nPads);
113 }
