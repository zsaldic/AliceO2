37 void DataSampling::GenerateInfrastructure(WorkflowSpec& workflow, const std::string& policiesSource, size_t threads)
38 {
39  LOG(DEBUG) << "Generating Data Sampling infrastructure...";
40  Dispatcher dispatcher(createDispatcherName(), policiesSource);
41  Options options;
42 
43  std::unique_ptr<ConfigurationInterface> cfg = ConfigurationFactory::getConfiguration(policiesSource);
44  auto policiesTree = cfg->getRecursive("dataSamplingPolicies");
45 
46  for (auto&& policyConfig : policiesTree) {
47 
48  DataSamplingPolicy policy(policyConfig.second);
49 
50  // Dispatcher gets connected to any available outputs, which are listed in policies (active or not).
51  // policies partially fulfilled are also taken into account - the user might need data from different FLPs,
52  // but corresponding to the same event.
53  // If there is a strong need, it can be changed to subscribe to every output in the workflow, which would allow to
54  // modify input streams during the runtime.
55 
56  LOG(DEBUG) << "Checking if the topology can provide any data for policy '" << policy.getName() << "'.";
57 
58  bool dataFound = false;
59  for (const auto& dataProcessor : workflow) {
60  for (const auto& externalOutput : dataProcessor.outputs) {
61  InputSpec candidateInputSpec{
62  "doesnt-matter", //externalOutput.binding.value,
63  externalOutput.origin,
64  externalOutput.description,
65  externalOutput.subSpec,
66  externalOutput.lifetime
67  };
68 
69  if (policy.match(candidateInputSpec)) {
70  Output output = policy.prepareOutput(candidateInputSpec);
71  OutputSpec outputSpec{
72  output.origin,
73  output.description,
74  output.subSpec,
75  output.lifetime
76  };
77 
78  dispatcher.registerPath({ candidateInputSpec, outputSpec });
79  dataFound = true;
80  LOG(DEBUG) << " - found " << externalOutput << ", it will be published in " << outputSpec;
81  }
82  }
83  }
84  if (dataFound && !policy.getFairMQOutputChannel().empty()) {
85  options.push_back({ "channel-config", VariantType::String, policy.getFairMQOutputChannel().c_str(), { "Out-of-band channel config" } });
86  LOG(DEBUG) << " - registering output FairMQ channel '" << policy.getFairMQOutputChannel() << "'";
87  }
88  }
89 
90  DataProcessorSpec spec;
91  spec.name = dispatcher.getName();
92  spec.inputs = dispatcher.getInputSpecs();
93  spec.outputs = dispatcher.getOutputSpecs();
94  spec.algorithm = adaptFromTask<Dispatcher>(std::move(dispatcher));
95  spec.maxInputTimeslices = threads;
96  spec.labels = { { "DataSampling" }, { "Dispatcher" } };
97  spec.options = options;
98 
99  workflow.emplace_back(std::move(spec));
100 }

120 std::vector<InputSpec> DataSampling::InputSpecsForPolicy(ConfigurationInterface* const config, const std::string& policyName)
121 {
122  std::vector<InputSpec> inputs;
123  auto policiesTree = config->getRecursive("dataSamplingPolicies");
124 
125  for (auto&& policyConfig : policiesTree) {
126  if (policyConfig.second.get<std::string>("id") == policyName) {
127  DataSamplingPolicy policy(policyConfig.second);
128  if (policy.getSubSpec() == -1) {
129  //fixme: support it, when wildcards are available
130  LOG(WARNING) << "InputSpecsForPolicy does not support subscriptions to all subSpecs yet.";
131  }
132  for (const auto& path : policy.getPathMap()) {
133  inputs.push_back(
134  InputSpec{
135  path.second.binding.value,
136  path.second.origin,
137  path.second.description,
138  path.second.subSpec,
139  path.second.lifetime });
140  }
141  break;
142  }
143  }
144  return inputs;
145 }
