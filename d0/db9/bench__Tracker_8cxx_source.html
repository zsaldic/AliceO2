27 std::vector<o2::mid::Cluster2D> generateTestData(int nTracks, o2::mid::TrackGenerator& trackGen,
28  const o2::mid::HitFinder& hitFinder, const o2::mid::Mapping& mapping)
29 {
30  o2::mid::Mapping::MpStripIndex stripIndex;
31  o2::mid::MpArea area;
32  std::vector<o2::mid::Cluster2D> clusters;
33  o2::mid::Cluster2D cl;
34  std::vector<o2::mid::Track> tracks = trackGen.generate(nTracks);
35  for (auto& track : tracks) {
36  for (int ich = 0; ich < 4; ++ich) {
37  std::vector<std::pair<int, Point3D<float>>> pairs = hitFinder.getLocalPositions(track, ich);
38  bool isFired = false;
39  for (auto& pair : pairs) {
40  int deId = pair.first;
41  float xPos = pair.second.x();
42  float yPos = pair.second.y();
43  stripIndex = mapping.stripByPosition(xPos, yPos, 0, deId, false);
44  if (!stripIndex.isValid()) {
45  continue;
46  }
47  cl.deId = deId;
48  area = mapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
49  cl.yCoor = area.getCenterY();
50  float halfSize = area.getHalfSizeY();
51  cl.sigmaY2 = halfSize * halfSize / 3.;
52  stripIndex = mapping.stripByPosition(xPos, yPos, 1, deId, false);
53  area = mapping.stripByLocation(stripIndex.strip, 1, stripIndex.line, stripIndex.column, deId);
54  cl.xCoor = area.getCenterX();
55  halfSize = area.getHalfSizeX();
56  cl.sigmaX2 = halfSize * halfSize / 3.;
57  clusters.push_back(cl);
58  } // loop on fired pos
59  } // loop on chambers
60  } // loop on tracks
61  return clusters;
62 }
