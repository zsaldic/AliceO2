40 DataProcessorSpec getPublisherSpec(PublisherConf const& config, bool propagateMC)
41 {
42  if (config.tpcSectors.size() == 0 || config.outputIds.size() == 0) {
43  throw std::invalid_argument("need TPC sector and output id configuration");

48  std::vector<int> outputIds;

55  auto initFunction = [config, propagateMC](InitContext& ic) {
56  // get the option from the init context
57  auto filename = ic.options().get<std::string>("infile");
58  auto treename = ic.options().get<std::string>("treename");
59  auto clbrName = ic.options().get<std::string>(config.databranch.option.c_str());
60  auto mcbrName = ic.options().get<std::string>(config.mcbranch.option.c_str());
61  auto nofEvents = ic.options().get<int>("nevents");
62  auto publishingMode = nofEvents == -1 ? RootTreeReader::PublishingMode::Single : RootTreeReader::PublishingMode::Loop;
63 
64  auto processAttributes = std::make_shared<ProcessAttributes>();
65  {
66  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
67  auto& sectors = processAttributes->sectors;
68  auto& activeSectors = processAttributes->activeSectors;
69  auto& readers = processAttributes->readers;
70  auto& outputIds = processAttributes->outputIds;
71 
72  sectors = config.tpcSectors;
73  outputIds = config.outputIds;
74  for (auto const& s : sectors) {
75  // set the mask of active sectors
76  if (s >= NSectors) {
77  std::string message = std::string("invalid sector range specified, allowed 0-") + std::to_string(NSectors - 1);
78  // FIXME should probably be FATAL, but this doesn't seem to be handled in the DPL control flow
79  // at least the process is not marked dead in the DebugGUI
80  LOG(ERROR) << message;
81  throw std::invalid_argument(message);
82  }
83  activeSectors |= (uint64_t)0x1 << s;
84  }
85 
86  // set up the tree interface
87  // TODO: parallelism on sectors needs to be implemented as selector in the reader
88  // the data is now in parallel branches, as first attempt use an array of readers
89  constexpr auto persistency = Lifetime::Timeframe;
90  auto outputId = outputIds.begin();
91  for (size_t sector = 0; sector < NSectors; ++sector) {
92  if ((activeSectors & ((uint64_t)0x1 << sector)) == 0) {
93  continue;
94  }
95  o2::header::DataHeader::SubSpecificationType subSpec = *outputId;
96  std::string clusterbranchname = clbrName + "_" + std::to_string(sector);
97  std::string mcbranchname = mcbrName + "_" + std::to_string(sector);
98  auto& dto = config.dataoutput;
99  auto& mco = config.mcoutput;
100  if (propagateMC) {
101  readers[sector] = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
102  filename.c_str(), // input file name
103  nofEvents, // number of entries to publish
104  publishingMode,
105  Output{ dto.origin, dto.description, subSpec, persistency },
106  clusterbranchname.c_str(), // name of cluster branch
107  Output{ mco.origin, mco.description, subSpec, persistency },
108  mcbranchname.c_str() // name of mc label branch
109  );
110  } else {
111  readers[sector] = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
112  filename.c_str(), // input file name
113  nofEvents, // number of entries to publish
114  publishingMode,
115  Output{ dto.origin, dto.description, subSpec, persistency },
116  clusterbranchname.c_str() // name of cluster branch
117  );
118  }
119  if (++outputId == outputIds.end()) {
120  outputId = outputIds.begin();
121  }
122  }
123  }
124 
125  // set up the processing function
126  // using by-copy capture of the worker instance shared pointer
127  // the shared pointer makes sure to clean up the instance when the processing
128  // function gets out of scope
129  // FIXME: wanted to use it = sectors.begin() in the variable capture but the iterator
130  // is const and can not be incremented
131  auto processingFct = [processAttributes, index = std::make_shared<int>(0), propagateMC](ProcessingContext& pc)
132  {
133  if (processAttributes->finished) {
134  return;
135  }
136 
137  auto publish = [&processAttributes, &index, &pc, propagateMC]() {
138  auto& sectors = processAttributes->sectors;
139  auto& activeSectors = processAttributes->activeSectors;
140  auto& readers = processAttributes->readers;
141  while (*index < sectors.size() && !readers[sectors[*index]]) {
142  // probably more efficient to use a vector of valid readers instead of the fixed array with
143  // possibly invalid entries
144  ++(*index);
145  }
146  if (*index == sectors.size()) {
147  // there is no valid reader for this call
148  return false;
149  }
150  auto sector = sectors[*index];
151  o2::TPC::TPCSectorHeader header{ sector };
152  header.activeSectors = activeSectors;
153  auto& r = *(readers[sector].get());
154 
155  // increment the reader and invoke it for the processing context
156  if (r.next()) {
157  // there is data, run the reader
158  r(pc, header);
159  } else {
160  // no more data, delete the reader
161  readers[sector].reset();
162  return false;
163  }
164  ++(*index);
165  return true;
166  };
167 
168  if (*index >= processAttributes->sectors.size()) {
169  *index = 0;
170  }
171  int operation = -2;
172  for (size_t lane = 0; lane < processAttributes->outputIds.size(); ++lane) {
173  if (!publish()) {
174  // need to publish a dummy packet
175  // FIXME define and use flags in the TPCSectorHeader, for now using the same schema as
176  // in digitizer workflow, -1 -> end of data, -2 noop
177  if (lane == 0) {
178  operation = -1;
179  }
180  o2::header::DataHeader::SubSpecificationType subSpec = processAttributes->outputIds[lane];
181  o2::TPC::TPCSectorHeader header{ operation };
182  pc.outputs().snapshot(OutputRef{ "output", subSpec, { header } }, subSpec);
183  if (propagateMC) {
184  pc.outputs().snapshot(OutputRef{ "outputMC", subSpec, { header } }, subSpec);
185  }
186  }
187  }
188 
189  if ((processAttributes->finished = (operation == -1)) && processAttributes->terminateOnEod) {
190  pc.services().get<ControlService>().readyToQuit(false);
191  }
192  };
193 
194  // return the actual processing function as a lambda function using variables
195  // of the init function
196  return processingFct;
197  };

199  auto createOutputSpecs = [&config, propagateMC]() {
200  std::vector<OutputSpec> outputSpecs;
201  for (size_t n = 0; n < config.outputIds.size(); ++n) {
202  o2::header::DataHeader::SubSpecificationType subSpec = config.outputIds[n];
203  outputSpecs.emplace_back(OutputSpec{ { "output" }, config.dataoutput.origin, config.dataoutput.description, subSpec, Lifetime::Timeframe });
204  if (propagateMC) {
205  outputSpecs.emplace_back(OutputSpec{ { "outputMC" }, config.mcoutput.origin, config.mcoutput.description, subSpec, Lifetime::Timeframe });
206  }
207  }
208  return std::move(outputSpecs);
209  };

211  auto& dtb = config.databranch;
212  auto& mcb = config.mcbranch;
213  return DataProcessorSpec{ config.processName.c_str(),
214  Inputs{}, // no inputs
215  { createOutputSpecs() },
216  AlgorithmSpec(initFunction),
217  Options{
218  { "infile", VariantType::String, "", { "Name of the input file" } },
219  { "treename", VariantType::String, config.defaultTreeName.c_str(), { "Name of input tree" } },
220  { dtb.option.c_str(), VariantType::String, dtb.defval.c_str(), { dtb.help.c_str() } },
221  { mcb.option.c_str(), VariantType::String, mcb.defval.c_str(), { mcb.help.c_str() } },
222  { "nevents", VariantType::Int, -1, { "number of events to run" } },
223  { "terminate-on-eod", VariantType::Bool, true, { "terminate on end-of-data" } },
224  } };
