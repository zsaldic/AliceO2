144 template <DigitzationMode MODE>
145 inline float SAMPAProcessing::makeSignal(float ADCcounts, const int sector, const int globalPadInSector,
146  float& pedestal, float& noise)
147 {
148  float signal = ADCcounts;
149  pedestal = getPedestal(sector, globalPadInSector);
150  noise = getNoise(sector, globalPadInSector);
151  switch (MODE) {
152  case DigitzationMode::FullMode: {
153  signal += noise;
154  signal += pedestal;
155  return getADCSaturation(signal);
156  break;
157  }
158  case DigitzationMode::SubtractPedestal: {
159  signal += noise;
160  signal += pedestal;
161  float signalSubtractPedestal = getADCSaturation(signal) - pedestal;
162  return signalSubtractPedestal;
163  break;
164  }
165  case DigitzationMode::NoSaturation: {
166  signal += noise;
167  signal += pedestal;
168  return signal;
169  break;
170  }
171  case DigitzationMode::PropagateADC: {
172  return signal;
173  break;
174  }
175  }
176  return signal;
177 }

187 template <typename T>
188 inline T SAMPAProcessing::getGamma4(T time, T startTime, T ADC) const
189 {
190  Vc::float_v tmp0 = (time - startTime) / mEleParam->getPeakingTime();
191  Vc::float_m cond = (tmp0 > 0);
192  Vc::float_v tmp;
193  tmp(cond) = tmp0;
194  Vc::float_v tmp2 = tmp * tmp;
195  return 55.f * ADC * Vc::exp(-4.f * tmp) * tmp2 * tmp2;
196 }
