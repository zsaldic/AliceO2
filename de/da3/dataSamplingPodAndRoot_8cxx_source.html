97  bool dataGood = true;
98  for (int j = 0; j < header->payloadSize / sizeof(FakeCluster); ++j) {
99  float diff = std::abs(-inputDataTpc[j].x - inputDataTpcProcessed[j].x) +
100  std::abs(2 * inputDataTpc[j].y - inputDataTpcProcessed[j].y) +
101  std::abs(inputDataTpc[j].z * inputDataTpc[j].q - inputDataTpcProcessed[j].z) +
102  std::abs(inputDataTpc[j].q - inputDataTpcProcessed[j].q);
103  if (diff > 1) {
104  dataGood = false;
105  break;
106  }
107  }

109  LOG(INFO) << "qcTaskTPC - received data is " << (dataGood ? "correct" : "wrong");

239 void someProcessingStageAlgorithm(ProcessingContext& ctx)
240 {
241  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
242  const FakeCluster* inputDataIts = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataITS").payload);
243 
244  auto processedTpcClusters =
245  ctx.outputs().make<FakeCluster>(Output{ "TPC", "CLUSTERS_P", 0 }, collectionChunkSize);
246  auto processedItsClusters =
247  ctx.outputs().make<FakeCluster>(Output{ "ITS", "CLUSTERS_P", 0 }, collectionChunkSize);
248 
249  int i = 0;
250  for (auto& cluster : processedTpcClusters) {
251  assert(i < collectionChunkSize);
252  cluster.x = -inputDataTpc[i].x;
253  cluster.y = 2 * inputDataTpc[i].y;
254  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
255  cluster.q = inputDataTpc[i].q;
256  i++;
257  }
258 
259  i = 0;
260  for (auto& cluster : processedItsClusters) {
261  assert(i < collectionChunkSize);
262  cluster.x = -inputDataIts[i].x;
263  cluster.y = 2 * inputDataIts[i].y;
264  cluster.z = inputDataIts[i].z * inputDataIts[i].q;
265  cluster.q = inputDataIts[i].q;
266  i++;
267  }
