40 void FLPSenderDevice::Run()
41 {
42  // base buffer, to be copied from for every timeframe body (zero-copy)
43  FairMQMessagePtr baseMsg(NewMessage(mEventSize));
44 
45  // store the channel reference to avoid traversing the map on every loop iteration
46  //FairMQChannel& dataInChannel = fChannels.at(fInChannelName).at(0);
47 
48  while (CheckCurrentState(RUNNING)) {
49  // - Get the SubtimeframeMetadata
50  // - Add the current FLP id to the SubtimeframeMetadata
51  // - Forward to the EPN the whole subtimeframe
52  FairMQParts subtimeframeParts;
53  if (Receive(subtimeframeParts, mInChannelName, 0, 100) <= 0)
54  continue;
55 
56  assert(subtimeframeParts.Size() != 0);
57  assert(subtimeframeParts.Size() >= 2);
58  const auto* dh = o2::header::get<header::DataHeader*>(subtimeframeParts.At(0)->GetData());
59  assert(strncmp(dh->dataDescription.str, "SUBTIMEFRAMEMD", 16) == 0);
60 
61  SubframeMetadata* sfm = reinterpret_cast<SubframeMetadata*>(subtimeframeParts.At(1)->GetData());
62  sfm->flpIndex = mIndex;
63 
64  mArrivalTime.push(steady_clock::now());
65  mSTFBuffer.push(move(subtimeframeParts));
66 
67  // if offset is 0 - send data out without staggering.
68  assert(mSTFBuffer.size() > 0);
69 
70  if (mSendOffset == 0 && mSTFBuffer.size() > 0) {
71  sendFrontData();
72  } else if (mSTFBuffer.size() > 0) {
73  if (duration_cast<milliseconds>(steady_clock::now() - mArrivalTime.front()).count() >= (mSendDelay * mSendOffset)) {
74  sendFrontData();
75  } else {
76  // LOG(INFO) << "buffering...";
77  }
78  }
79  }
80 }

82 inline void FLPSenderDevice::sendFrontData()
83 {
84  SubframeMetadata *sfm = static_cast<SubframeMetadata*>(mSTFBuffer.front().At(1)->GetData());
85  uint16_t currentTimeframeId = o2::DataFlow::timeframeIdFromTimestamp(sfm->startTime, sfm->duration);
86  if (mLastTimeframeId != -1) {
87  if (currentTimeframeId == mLastTimeframeId) {
88  LOG(ERROR) << "Sent same consecutive timeframe ids\n";
89  }
90  }
91  mLastTimeframeId = currentTimeframeId;
92 
93  // for which EPN is the message?
94  int direction = currentTimeframeId % mNumEPNs;
95  if (Send(mSTFBuffer.front(), mOutChannelName, direction, 0) < 0) {
96  LOG(ERROR) << "Failed to queue sub-timeframe #" << currentTimeframeId << " to EPN[" << direction << "]";
97  }
98  mSTFBuffer.pop();
99  mArrivalTime.pop();
100 }
