119  static o2::ITSMFT::ClusterShape* PixelsToClusterShape(std::vector<Pixel> v)
120  {
121  if (v.size() == 0)
122  return new o2::ITSMFT::ClusterShape();
123 
124  // To remove the multiple hits problem
125  Pixel p0 = v[0];
126  for (auto p = v.begin() + 1; p != v.end(); ++p) {
127  if (fabs(p0.GetRow() - p->GetRow()) > v.size() || fabs(p0.GetCol() - p->GetCol()) > v.size()) {
128  v.erase(p);
129  p--;
130  }
131  }
132 
133  vector<UInt_t> r;
134  vector<UInt_t> c;
135  UInt_t min_r = UINT_MAX, min_c = UINT_MAX;
136  for (auto& p : v) {
137  if (p.GetRow() < min_r)
138  min_r = p.GetRow();
139  if (p.GetCol() < min_c)
140  min_c = p.GetCol();
141  if (find(begin(r), end(r), p.GetRow()) == end(r))
142  r.push_back(p.GetRow());
143  if (find(begin(c), end(c), p.GetCol()) == end(c))
144  c.push_back(p.GetCol());
145  }
146 
147  UInt_t rows = r.size();
148  UInt_t cols = c.size();
149  std::vector<UInt_t> pindex;
150  for (auto& p : v) {
151  Int_t rd = p.GetRow() - min_r;
152  Int_t cd = p.GetCol() - min_c;
153  Int_t index = rd * cols + cd;
154  pindex.push_back(index);
155  }
156 
157  return new o2::ITSMFT::ClusterShape(rows, cols, pindex);
158  }

161  map<UInt_t, vector<Pixel>> m_pixels;
162 };
