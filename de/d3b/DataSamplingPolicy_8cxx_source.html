36 void DataSamplingPolicy::configure(const ptree& config)
37 {
38  mName = config.get<std::string>("id");
39  if (mName.size() > 14) {
40  LOG(WARNING) << "DataSamplingPolicy name '" << mName << "' is longer than 14 characters, trimming.";
41  mName.resize(14);
42  }
43 
44  // todo: get the sub spec range if there is a requirement and when dpl supports it
45  auto subSpecString = config.get<std::string>("subSpec");
46  mSubSpec = subSpecString.find_first_of("-*") != std::string::npos ? -1 : std::strtoull(subSpecString.c_str(), nullptr, 10);
47  mPaths.clear();
48  size_t outputId = 0;
49 
50  for (const auto& dataHeaderConfig : config.get_child("dataHeaders")) {
51 
52  header::DataOrigin origin;
53  header::DataDescription description;
54  origin.runtimeInit(dataHeaderConfig.second.get<std::string>("dataOrigin").c_str());
55  description.runtimeInit(dataHeaderConfig.second.get<std::string>("dataDescription").c_str());
56 
57  InputSpec inputSpec{
58  dataHeaderConfig.second.get<std::string>("binding"),
59  origin,
60  description,
61  mSubSpec
62  };
63 
64  OutputSpec outputSpec{
65  { dataHeaderConfig.second.get<std::string>("binding") },
66  createPolicyDataOrigin(),
67  createPolicyDataDescription(mName, outputId++),
68  mSubSpec
69  };
70 
71  mPaths.emplace(inputSpec, outputSpec);
72  if (outputId > 9) {
73  LOG(ERROR) << "Maximum 10 inputs in DataSamplingPolicy are supported";
74  break;
75  }
76  }
77 
78  mConditions.clear();
79  for (const auto& conditionConfig : config.get_child("samplingConditions")) {
80  mConditions.push_back(DataSamplingConditionFactory::create(conditionConfig.second.get<std::string>("condition")));
81  mConditions.back()->configure(conditionConfig.second);
82  }
83 
84  mFairMQOutputChannel = config.get_optional<std::string>("fairMQOutput").value_or("");
85 }

100 const Output DataSamplingPolicy::prepareOutput(const InputSpec& input) const
101 {
102  auto result = mPaths.find(input);
103  if (result != mPaths.end()) {
104  auto concrete = DataSpecUtils::asConcreteDataMatcher(input);
105  return Output{ result->second.origin, result->second.description, concrete.subSpec };
106  } else {
107  return Output{ header::gDataOriginInvalid, header::gDataDescriptionInvalid };
