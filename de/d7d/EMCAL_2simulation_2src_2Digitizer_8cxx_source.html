38 void Digitizer::process(const std::vector<Hit>& hits, std::vector<Digit>& digits)
39 {
40  digits.clear();
41  mDigits.clear();
42  mMCTruthContainer.clear();
43 
44  for (auto hit : hits) {
45  try {
46  Int_t LabelIndex = mMCTruthContainer.getIndexedSize();
47  Digit digit = hitToDigit(hit, LabelIndex);
48  Int_t id = digit.GetTower();
49 
50  if (id < 0 || id > mGeometry->GetNCells()) {
51  LOG(WARNING) << "tower index out of range: " << id << FairLogger::endl;
52  continue;
53  }
54 
55  Bool_t flag = false;
56  for (auto& digit0 : mDigits[id]) {
57  if (digit0.canAdd(digit)) {
58  digit0 += digit;
59  LabelIndex = digit0.GetLabel();
60  flag = true;
61  break;
62  }
63  }
64 
65  if (!flag) {
66  mDigits[id].push_front(digit);
67  }
68 
69  o2::EMCAL::MCLabel label(hit.GetTrackID(), mCurrEvID, mCurrSrcID, mEventTime);
70  mMCTruthContainer.addElementRandomAccess(LabelIndex, label);
71  auto labels = mMCTruthContainer.getLabels(LabelIndex);
72  std::sort(labels.begin(), labels.end());
73  } catch (InvalidPositionException& e) {
74  LOG(ERROR) << "Error in creating the digit: " << e.what() << FairLogger::endl;
75  }
76  }
77 
78  fillOutputContainer(digits);
79 }

104 void Digitizer::fillOutputContainer(std::vector<Digit>& digits)
105 {
106  std::forward_list<Digit> l;
107 
108  for (auto tower : mDigits) {
109  for (auto& digit : tower.second) {
110  l.push_front(digit);
111  }
112  }
113 
114  l.sort();
115 
116  for (auto digit : l) {
117  digits.push_back(digit);
118  }
119 
120  mMCTruthOutputContainer.clear();
121  for (int index = 0; index < mMCTruthContainer.getIndexedSize(); ++index) {
122  mMCTruthOutputContainer.addElements(index, mMCTruthContainer.getLabels(index));
123  }
124 }
