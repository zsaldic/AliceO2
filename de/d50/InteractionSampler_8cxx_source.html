17 void InteractionSampler::init()
18 {
19  // (re-)initialize sample and check parameters consistency
20 
21  int nBCSet = mBCFilling.getNBunches();
22  if (!nBCSet) {
23  LOG(WARNING) << "No bunch filling provided, impose default one" << FairLogger::endl;
24  mBCFilling.setDefault();
25  nBCSet = mBCFilling.getNBunches();
26  }
27 
28  if (mMuBC < 0. && mIntRate < 0.) {
29  LOG(WARNING) << "No IR or muBC is provided, setting default IR" << FairLogger::endl;
30  mIntRate = DefIntRate;
31  }
32 
33  if (mMuBC > 0.) {
34  mIntRate = mMuBC * nBCSet * o2::constants::lhc::LHCRevFreq;
35  LOG(INFO) << "Deducing IR=" << mIntRate << "Hz from " << nBCSet << " BCs at mu=" << mMuBC << FairLogger::endl;
36  } else {
37  mMuBC = mIntRate / (nBCSet * o2::constants::lhc::LHCRevFreq);
38  LOG(INFO) << "Deducing mu=" << mMuBC << " per BC from IR=" << mIntRate << " with " << nBCSet << " BCs"
39  << FairLogger::endl;
40  }
41 
42  mBCMin = 0;
43  mBCMax = -1;
44  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
45  if (mBCFilling.testBC(i)) {
46  if (mBCMin > i) {
47  mBCMin = i;
48  }
49  if (mBCMax < i) {
50  mBCMax = i;
51  }
52  }
53  }
54  double muexp = TMath::Exp(-mMuBC);
55  mProbInteraction = 1. - muexp;
56  mMuBCZTRed = mMuBC * muexp / mProbInteraction;
57  mBCCurrent = mBCMin + gRandom->Integer(mBCMax - mBCMin + 1);
58  mIntBCCache = 0;
59  mOrbit = 0;
60 }

75 o2::InteractionTimeRecord InteractionSampler::generateCollisionTime()
76 {
77  // generate single interaction record
78  if (mIntRate < 0) {
79  init();
80  }
81 
82  if (mIntBCCache < 1) { // do we still have interaction in current BC?
83  mIntBCCache = simulateInteractingBC(); // decide which BC interacts and N collisions
84  }
85  double timeInt = mTimeInBC.back() + o2::InteractionTimeRecord::bc2ns(mBCCurrent, mOrbit);
86  mTimeInBC.pop_back();
87  mIntBCCache--;
88 
89  o2::InteractionTimeRecord tmp(timeInt);
90 
91  return o2::InteractionTimeRecord(timeInt);
92 }

95 int InteractionSampler::simulateInteractingBC()
96 {
97  // Returns number of collisions assigned to selected BC
98 
99  do {
100  nextCollidingBC(); // pick next interacting bunch
101  } while (gRandom->Rndm() > mProbInteraction); // skip BCs w/o collisions
102 
103  // once BC is decided, enforce at least one interaction
104  int ncoll = genPoissonZT();
105  // assign random time withing a bunch
106  for (int i = ncoll; i--;) {
107  double tInBC = 0; // tInBC should be in the vicinity of the BC
108  do {
109  tInBC = gRandom->Gaus(0.,mBCTimeRMS);
110  } while (std::abs(tInBC) > o2::constants::lhc::LHCBunchSpacingNS / 2.1);
111  mTimeInBC.push_back(tInBC);
112  }
113  if (ncoll > 1) { // sort in DECREASING time order (we are reading vector from the end)
114  std::sort(mTimeInBC.begin(), mTimeInBC.end(), [](const float a, const float b) { return a > b; });
115  }
116  return ncoll;
117 }
