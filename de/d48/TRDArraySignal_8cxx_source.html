94 void TRDArraySignal::allocate(int nrow, int ncol, int ntime)
95 {
96  //
97  // Allocates memory for an TRDArraySignal object with dimensions
98  // Row*NumberOfNecessaryMCMs*ADCchannelsInMCM*Time
99  // To be consistent with AliTRDarrayADC
100  //
101 
102  mNrow = nrow;
103  mNcol = ncol;
104  mNtime = ntime;
105  int adcchannelspermcm = TRDFeeParam::getNadcMcm();
106  int padspermcm = TRDFeeParam::getNcolMcm();
107  int numberofmcms = mNcol / padspermcm;
108  mNumberOfChannels = numberofmcms * adcchannelspermcm;
109  mNdim = nrow * mNumberOfChannels * ntime;
110  if (mSignal.size() != mNdim)
111  mSignal.resize(mNdim);
112 
113  memset(&mSignal[0], 0, sizeof(mSignal[0]) * mNdim);
114 }

133 void TRDArraySignal::compress(float minval)
134 {
135  //
136  // Compress the vector, setting values equal or
137  // below minval to zero (minval>=0)
138  //
139 
140  int counter = 0;
141  int newDim = 0;
142  int j;
143  int r = 0;
144  int k = 0;
145 
146  std::vector<int> longArr(mNdim);
147  memset(&longArr[0], 0, sizeof(longArr[0]) * mNdim);
148 
149  //Initialize the vector
150 
151  for (int i = 0; i < mNdim; i++) {
152  j = 0;
153  if (mSignal[i] <= minval) {
154  for (k = i; k < mNdim; k++) {
155  if (mSignal[k] <= minval) {
156  j = j + 1;
157  longArr[r] = j;
158  } else {
159  break;
160  }
161  }
162  r = r + 1;
163  }
164  i = i + j;
165  }
166 
167  //Calculate the size of the compressed vector
168  for (int i = 0; i < mNdim; i++) {
169  if (longArr[i] != 0) {
170  counter = counter + longArr[i] - 1;
171  }
172  }
173  newDim = mNdim - counter; //New dimension
174 
175  //Fill the buffer of the compressed vector
176  std::vector<float> buffer(mNdim);
177  memset(&buffer[0], 0, sizeof(buffer[0]) * mNdim);
178  int counterTwo = 0;
179 
180  //Write the new vector
181  int g = 0;
182  for (int i = 0; i < newDim; i++) {
183  if (counterTwo < mNdim) {
184  if (mSignal[counterTwo] > minval) {
185  buffer[i] = mSignal[counterTwo];
186  }
187  if (mSignal[counterTwo] <= minval) {
188  buffer[i] = -(longArr[g]);
189  counterTwo = counterTwo + longArr[g] - 1;
190  g++;
191  }
192  counterTwo++;
193  }
194  }
195 
196  //Copy the buffer
197  if (mSignal.size() != newDim) {
198  mSignal.resize(newDim);
199  }
200  LOG(debug) << "Compressed ArraySignal by " << mNdim / newDim;
201  mNdim = newDim;
202  mSignal = buffer;
203  for (int i = 0; i < newDim; i++) {
204  mSignal[i] = buffer[i];
205  }
206 }

209 void TRDArraySignal::expand()
210 {
211  // Expand the vector
212 
213  //Check if the vector has not been already expanded
214  int verif = 0;
215  for (int i = 0; i < mNdim; i++) {
216  if (mSignal[i] < 0) {
217  verif++;
218  }
219  }
220 
221  if (verif == 0) {
222  return;
223  }
224 
225  int dimexp = 0;
226  std::vector<int> longArr(mNdim);
227 
228  memset(&longArr[0], 0, sizeof(longArr[0]) * mNdim);
229 
230  int r2 = 0;
231  for (int i = 0; i < mNdim; i++) {
232  if (mSignal[i] < 0) {
233  longArr[r2] = (int)(-mSignal[i]);
234  r2++;
235  }
236  }
237 
238  //Calculate new dimensions
239  for (int i = 0; i < mNdim; i++) {
240  if (longArr[i] != 0) {
241  dimexp = dimexp + longArr[i] - 1;
242  }
243  }
244  dimexp = dimexp + mNdim; //Dimension of the expanded vector
245 
246  //Write in the buffer the new vector
247  int contaexp = 0;
248  int h = 0;
249  std::vector<float> buffer(dimexp);
250 
251  memset(&buffer[0], 0, sizeof(buffer[0]) * mNdim);
252 
253  for (int i = 0; i < dimexp; i++) {
254  if (mSignal[contaexp] > 0) {
255  buffer[i] = mSignal[contaexp];
256  }
257  if (mSignal[contaexp] < 0) {
258  for (int j = 0; j < longArr[h]; j++) {
259  buffer[i + j] = 0;
260  }
261  i = i + longArr[h] - 1;
262  h++;
263  }
264  contaexp++;
265  }
266 
267  if (mSignal.size() != dimexp)
268  mSignal.resize(dimexp);
269  mNdim = dimexp;
270  mSignal = buffer;
271 }
