96  T&&... args)

98  float diff{ 0.f };

103  auto end = std::chrono::high_resolution_clock::now();
104 
105  std::chrono::duration<double, std::milli> diff_t{ end - start };
106  diff = diff_t.count();
107 
108  if (taskName == nullptr) {
109  ostream << diff << "\t";
110  } else {
111  ostream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms" << std::endl;
112  }
113  } else {
114  (this->*task)(std::forward<T>(args)...);
115  }
116 
117  return diff;
