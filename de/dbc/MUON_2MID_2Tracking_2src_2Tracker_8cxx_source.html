133 bool Tracker::processSide(bool isRight, bool isInward)
134 {
136  int firstCh = (isInward) ? 3 : 0;
137  int secondCh = (isInward) ? 2 : 1;
138  int rpcOffset1 = Constants::getDEId(isRight, firstCh, 0);
139  int rpcOffset2 = Constants::getDEId(isRight, secondCh, 0);
140 
141  // loop on RPCs in first plane
142  Track track;
143  for (int irpc = 0; irpc < 9; ++irpc) {
144  int deId1 = rpcOffset1 + irpc;
145  for (int icl1 = 0; icl1 < mNClusters[deId1]; ++icl1) {
146  // loop on clusters of the RPC in the first plane
147  auto& cl1 = mClusters[deId1][icl1];
148  int firstRpc = getFirstNeighbourRPC(irpc);
149  int lastRpc = getLastNeighbourRPC(irpc);
150  for (int irpc2 = firstRpc; irpc2 <= lastRpc; ++irpc2) {
151  // loop on (neighbour) RPCs in second plane
152  int deId2 = rpcOffset2 + irpc2;
153  for (int icl2 = 0; icl2 < mNClusters[deId2]; ++icl2) {
154  // loop on clusters of the RPC in the second plane
155  auto& cl2 = mClusters[deId2][icl2];
156 
157  if (!makeTrackSeed(track, cl1, cl2)) {
158  continue;
159  }
160 
161  track.setClusterMatched(firstCh, getClusterId(cl1.id, deId1));
162  track.setClusterMatched(secondCh, getClusterId(cl2.id, deId2));
163  track.setClusterMatched(3 - firstCh, 0);
164  track.setClusterMatched(3 - secondCh, 0);
165  LOG(DEBUG) << deId1 << " - " << deId2;
166  LOG(DEBUG) << "Position: " << track.getPosition();
167  // LOG(DEBUG) << "Covariance: " << track.getCovarianceParameters();
168  followTrack(track, isRight, isInward);
169  } // loop on clusters in second plane
170  } // loop on RPCs in second plane
171  } // loop on clusters in first plane
172  } // loop on RPCs in first plane
173  return true;
174 }

177 bool Tracker::makeTrackSeed(Track& track, const Cluster3D& cl1, const Cluster3D& cl2) const
178 {
180 
181  // First check if the delta_x between the two clusters is not too large
182  double dZ = cl2.position.z() - cl1.position.z();
183  double dZ2 = dZ * dZ;
184  double nonBendingSlope = (cl2.position.x() - cl1.position.x()) / dZ;
185  double nonBendingImpactParam = std::abs(cl2.position.x() - cl2.position.z() * nonBendingSlope);
186  double nonBendingImpactParamErr = std::sqrt(
187  (cl1.position.z() * cl1.position.z() * cl2.sigmaX2 + cl2.position.z() * cl2.position.z() * cl1.sigmaX2) / dZ2);
188  if ((nonBendingImpactParam - mSigmaCut * nonBendingImpactParamErr) > mImpactParamCut) {
189  LOG(DEBUG) << "NB slope: " << nonBendingSlope << " NB impact param: " << nonBendingImpactParam << " - " << mSigmaCut
190  << " * " << nonBendingImpactParamErr << " > " << mImpactParamCut;
191  return false;
192  }
193 
194  // Then start making the track (from 2 points)
195  track.setPosition(cl2.position.x(), cl2.position.y(), cl2.position.z());
196  track.setDirection(nonBendingSlope, (cl2.position.y() - cl1.position.y()) / dZ, 1.);
197  track.setCovarianceParameters(cl2.sigmaX2, // x-x
198  cl2.sigmaY2, // y-y
199  (cl1.sigmaX2 + cl2.sigmaX2) / dZ2, // slopeX-slopeX
200  (cl1.sigmaY2 + cl2.sigmaY2) / dZ2, // slopeY-slopeY
201  cl2.sigmaX2 / dZ, // x-slopeX
202  cl2.sigmaY2 / dZ); // y-slopeY
203 
204  return true;
205 }

208 bool Tracker::followTrack(const Track& track, bool isRight, bool isInward)
209 {
211  double bestChi2 = 2. * mSigmaCut * mSigmaCut;
212  int nFiredChambers = 0;
213  int chamberOrder[2];
214  chamberOrder[0] = isInward ? 1 : 2;
215  chamberOrder[1] = isInward ? 0 : 3;
216 
217  // Add the track to the list
218  if (mNTracks >= static_cast<unsigned long int>(mTracks.size())) {
219  mTracks.emplace_back(Track());
220  }
221  Track& bestTrack(mTracks[mNTracks]);
222 
223  // loop on next two chambers
224  for (int ich = 0; ich < 2; ++ich) {
225  findNextCluster(track, isRight, isInward, chamberOrder[ich], 0, 8, nFiredChambers, bestChi2, bestTrack);
226  if (nFiredChambers == 2) {
227  // We already found all clusters: no need to search for mono-cathodic clusters
228  // in the next chamber
229  break;
230  }
231  }
232 
233  if (nFiredChambers == 0) {
234  // No track found
235  return false;
236  }
237 
238  // Extrapolate to first cluster in MT11 and compute the chi2
239  finalizeTrack(bestTrack);
240 
241  // Add the track if it is not compatible or better than the ones we already have
242  return addTrack(bestTrack);
243 }

246 bool Tracker::findNextCluster(const Track& track, bool isRight, bool isInward, int chamber, int firstRPC, int lastRPC,
247  int& nFiredChambers, double& bestChi2, Track& bestTrack, double chi2, int depth) const
248 {
250  int nextChamber = (isInward) ? chamber - 1 : chamber + 1;
251  int rpcOffset = Constants::getDEId(isRight, chamber, 0);
252  Track newTrack;
253  for (int irpc = firstRPC; irpc <= lastRPC; ++irpc) {
254  int deId = rpcOffset + irpc;
255  for (int icl = 0; icl < mNClusters[deId]; ++icl) {
256  auto& cl = mClusters[deId][icl];
257  double addChi2AtCluster = tryOneCluster(track, cl, newTrack);
258  double sumChi2 = chi2 + addChi2AtCluster;
259  if (sumChi2 > bestChi2) {
260  continue;
261  }
262  if (nextChamber >= 0 && nextChamber <= 3) {
263  findNextCluster(newTrack, isRight, isInward, nextChamber, getFirstNeighbourRPC(irpc), getLastNeighbourRPC(irpc),
264  nFiredChambers, bestChi2, bestTrack, sumChi2, depth + 1);
265  }
266  if (depth >= nFiredChambers && sumChi2 < bestChi2) {
267  nFiredChambers = depth;
268  bestChi2 = sumChi2;
269  bestTrack = newTrack;
270  LOG(DEBUG) << "DeId " << deId << " cluster " << icl << " nFiredChambers " << nFiredChambers << " chi2 "
271  << bestChi2;
272  }
273  } // loop on clusters
274  } // loop on RPC
275 
276  return (nFiredChambers > 0);
277 }

280 double Tracker::tryOneCluster(const Track& track, const Cluster3D& cluster, Track& newTrack) const
281 {
287  double dZ = cluster.position.z() - track.getPosition().z();
288  double dZ2 = dZ * dZ;
289  double cpos[2] = { cluster.position.x(), cluster.position.y() };
290  double cerr2[2] = { cluster.sigmaX2, cluster.sigmaY2 };
291  double pos[2] = { track.getPosition().x(), track.getPosition().y() };
292  double newPos[2] = { 0., 0. };
293  double dist[2] = { 0., 0. };
294  double dir[2] = { track.getDirection().x(), track.getDirection().y() };
295  const std::array<float, 6> covParams = track.getCovarianceParameters();
296  for (int icoor = 0; icoor < 2; ++icoor) {
297  newPos[icoor] = pos[icoor] + dir[icoor] * dZ;
298  dist[icoor] = cpos[icoor] - newPos[icoor];
299  double err2 = covParams[icoor] + dZ2 * covParams[icoor + 2] + 2. * dZ * covParams[icoor + 4] + cerr2[icoor];
300  double distMax = mSigmaCut * std::sqrt(2. * err2) + 4.;
301  if (std::abs(dist[icoor]) > distMax) {
302  LOG(DEBUG) << "Coordinate " << icoor << " cl " << cpos[icoor] << " tr " << newPos[icoor] << " err "
303  << std::sqrt(err2);
304  return 2. * mMaxChi2;
305  }
306  }
307 
308  newTrack = track;
309  newTrack.propagateToZ(cluster.position.z());
310  newTrack.setClusterMatched(Constants::getChamber(cluster.deId), getClusterId(cluster.id, cluster.deId));
311 
312  return runKalmanFilter(newTrack, cluster);
313 }

316 double Tracker::runKalmanFilter(Track& track, const Cluster3D& cluster) const
317 {
320 
321  double pos[2] = { track.getPosition().x(), track.getPosition().y() };
322  double dir[2] = { track.getDirection().x(), track.getDirection().y() };
323  double clusPos[2] = { cluster.position.x(), cluster.position.y() };
324 
325  std::array<float, 6> newCovParams;
326  const std::array<float, 6> covParams = track.getCovarianceParameters();
327  double newPos[2], newDir[2];
328  double clusterSigma[2] = { cluster.sigmaX2, cluster.sigmaY2 };
329  double chi2 = 0.;
330  for (int idx = 0; idx < 2; ++idx) {
331  int slopeIdx = idx + 2;
332  int covIdx = idx + 4;
333 
334  double den = clusterSigma[idx] + covParams[idx];
335  assert(den != 0.);
336  // if ( den == 0. ) return 2.*mMaxChi2;
337 
338  // Compute the new covariance
339  // cov -> (W+U)^-1
340  // where W = (old_cov)^-1
341  // and U = inverse of uncertainties of cluster
342  // s_x^2 -> s_x^2 * cl_s_x^2 / (s_x^2 + cl_s_x^2)
343  newCovParams[idx] = covParams[idx] * clusterSigma[idx] / den;
344  // cov(x,slopeX) -> cov(x,slopeX) * cl_s_x^2 / (s_x^2 + cl_s_x^2)
345  newCovParams[covIdx] = covParams[covIdx] * clusterSigma[idx] / den;
346  // s_slopeX^2 -> s_slopeX^2 - cov(x,slopeX)^2 / (s_x^2 + cl_s_x^2)
347  newCovParams[slopeIdx] = covParams[slopeIdx] - covParams[covIdx] * covParams[covIdx] / den;
348 
349  double diff = clusPos[idx] - pos[idx];
350 
351  // New parameters: p' = ((W+U)^-1)U(m-p) + p
352  // x -> x + ( cl_x - x ) * s_x^2 / (s_x^2 + cl_s_x^2)
353  newPos[idx] = pos[idx] + diff * covParams[idx] / den;
354  // slopeX -> slopeX + ( cl_x - x ) * s_slopeX^2 / (s_x^2 + cl_s_x^2)
355  newDir[idx] = dir[idx] + diff * covParams[covIdx] / den;
356 
357  chi2 += diff * diff / den;
358  }
359 
360  // Save the new parameters
361  track.setPosition(newPos[0], newPos[1], cluster.position.z());
362  track.setDirection(newDir[0], newDir[1], 1.);
363 
364  // Save the new parameters covariance matrix
365  track.setCovarianceParameters(newCovParams);
366 
367  return chi2;
368 }

371 void Tracker::finalizeTrack(Track& track)
372 {
375  int ndf = 0;
376  double chi2 = 0.;
377  for (int ich = 3; ich >= 0; --ich) {
378  int matchedClusterIdx = track.getClusterMatched(ich);
379  if (matchedClusterIdx == 0) {
380  continue;
381  }
382  ++ndf;
383  int deId = matchedClusterIdx / 1000;
384  int icl = matchedClusterIdx % 1000 - 1;
385  Cluster3D& cl(mClusters[deId][icl]);
386  track.propagateToZ(cl.position.z());
387  double clPos[2] = { cl.position.x(), cl.position.y() };
388  double clErr2[2] = { cl.sigmaX2, cl.sigmaY2 };
389  double trackPos[2] = { track.getPosition().x(), track.getPosition().y() };
390  double trackCov[2] = { track.getCovarianceParameter(Track::CovarianceParamIndex::VarX),
391  track.getCovarianceParameter(Track::CovarianceParamIndex::VarY) };
392  for (int icoor = 0; icoor < 2; ++icoor) {
393  double diff = trackPos[icoor] - clPos[icoor];
394  chi2 += diff * diff / (trackCov[icoor] + clErr2[icoor]);
395  }
396  }
397  track.setChi2(chi2);
398  track.setNDF(ndf);
399  LOG(DEBUG) << track;
400 }

403 bool Tracker::addTrack(const Track& track)
404 {
409 
410  float chi2OverNDF = track.getChi2OverNDF();
411  // We divide the chi2 by two since we want to consider only the uncertainty
412  // on one of the two tracks. We further reduce to 0.4 since we want to account
413  // for the case where one of the two reconstructed tracks has a much better precision
414  // of the other
415  float chi2Cut = 0.4 * mSigmaCut * mSigmaCut;
416  for (int itrack = 0; itrack < mNTracks; ++itrack) {
417  int nCommonClusters = 0;
418  auto& checkTrack = mTracks[itrack];
419  for (int ich = 0; ich < 4; ++ich) {
420  if (track.getClusterMatched(ich) == checkTrack.getClusterMatched(ich)) {
421  ++nCommonClusters;
422  }
423  }
424  if (nCommonClusters == 4) {
425  return false;
426  }
427  if (track.isCompatible(checkTrack, chi2Cut)) {
428  // The new track is compatible with an existing one
429  if (chi2OverNDF < checkTrack.getChi2OverNDF()) {
430  // The new track is more precise than the old one: replace it!
431  LOG(DEBUG) << "Replacing track " << checkTrack << "\n with " << track;
432  checkTrack = track;
433  return true;
434  } else {
435  LOG(DEBUG) << "Rejecting track " << track << "\n compatible with " << checkTrack;
436  // The new track is less precise than the old one: reject it
437  return false;
438  }
439  }
440  }
441 
442  // The new track is not compatible with the previous ones: add the track to the list
443  ++mNTracks;
444  return true;
445 }
