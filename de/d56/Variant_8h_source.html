117 // Poor man variant class. Does not take ownership of anything passed to it.
118 // FIXME: we should really use C++17 std::variant when it
119 // comes about
120 class Variant
121 {
122  using storage_t = std::aligned_union<8, int, int64_t, const char*, float, double, bool>::type;
123 
124  public:
125  Variant(VariantType type = VariantType::Unknown) : mType{ type } {}
126 
127  template <typename T>
128  Variant(T value) : mType{ variant_trait<T>::type() }
129  {
130  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
131  }
132 
133  template <typename T>
134  Variant(std::initializer_list<T>) : mType{ VariantType::Unknown }
135  {
136  static_assert(sizeof(T) == 0,
137  "brace-enclosed initializer list forbidden for Variant"
138  "\n did you accidentally put braces around the default value?");
139  }
140 
141  Variant(const Variant& other) : mType(other.mType)
142  {
143  // In case this is a string we need to duplicate it to avoid
144  // double deletion.
145  if (mType == variant_trait<const char*>::type()) {
146  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
147  } else {
148  mStore = other.mStore;
149  }
150  }
151 
152  Variant(Variant&& other) : mType(other.mType)
153  {
154  // In case this is a string we need to duplicate it to avoid
155  // double deletion.
156  if (mType == variant_trait<const char*>::type()) {
157  mStore = other.mStore;
158  *reinterpret_cast<char**>(&(other.mStore)) = nullptr;
159  } else {
160  mStore = other.mStore;
161  }
162  }
163 
164  ~Variant()
165  {
166  // In case we allocated a string out of bound, we
167  // should delete it.
168  if (mType == variant_trait<const char*>::type() || mType == variant_trait<char*>::type()) {
169  free(*reinterpret_cast<void**>(&mStore));
170  }
171  }
172 
173  void operator=(const Variant& other)
174  {
175  if (mType == variant_trait<const char*>::type()) {
176  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
177  } else {
178  mStore = other.mStore;
179  }
180  }
181 
182  template <typename T>
183  T get() const
184  {
185  if (mType != variant_trait<T>::type()) {
186  throw std::runtime_error("Mismatch between types");
187  }
188  return variant_helper<storage_t, T>::get(&mStore);
189  }
190 
191  template <typename T>
192  void set(T value)
193  {
194  return variant_helper<storage_t, T>::set(mStore, value);
195  }
196 
197  VariantType type() const { return mType; }
198 
199  private:
200  friend std::ostream& operator<<(std::ostream& oss, Variant const& val);
201  storage_t mStore;
202  VariantType mType;
203 };
