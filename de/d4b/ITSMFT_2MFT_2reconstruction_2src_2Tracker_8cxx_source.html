66  end = std::chrono::system_clock::now();
67  diff = end - start;
68  LOG(INFO) << "Processing time for single frame " << mROFrame << " : " << diff.count() << " s" << FairLogger::endl;
69 
70  start = end;

72  mROFrame++; // expect incremented frame in following clusters
73  }
74  }

127 Int_t Tracker::loadClusters(const std::vector<Cluster>& clusters)
128 {
129 
130  Int_t numOfClusters = clusters.size();
131  int nLoaded = 0;
132 
133  if (mContinuousMode) { // check the ROFrame in cont. mode
134  for (auto& c : clusters) {
135  if (c.getROFrame() != mROFrame) {
136  continue;
137  }
138  nLoaded++;
139  Int_t layer = mGeom->getLayer(c.getSensorID());
140  if (!sLayers[layer].insertCluster(&c)) {
141  continue;
142  }
143  }
144  } else { // do not check the ROFrame in triggered mode
145  for (auto& c : clusters) {
146  nLoaded++;
147  Int_t layer = mGeom->getLayer((Int_t)(c.getSensorID()));
148  if (!sLayers[layer].insertCluster(&c)) {
149  continue;
150  }
151  }
152  }
153 
154  if (nLoaded) {
155  std::vector<std::future<void>> fut;
156  for (Int_t l = 0; l < Constants::LayersNumber; l += mNumOfThreads) {
157  for (Int_t t = 0; t < mNumOfThreads; t++) {
158  if ((l + t) >= Constants::LayersNumber)
159  break;
160  auto f = std::async(std::launch::async, &Tracker::Layer::init, sLayers + (l + t));
161  fut.push_back(std::move(f));
162  }
163  for (Int_t t = 0; t < fut.size(); t++)
164  fut[t].wait();
165  }
166  }
167 
168  return nLoaded;
169 }
