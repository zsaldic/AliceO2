32  // number of bunches and the 40 MHz clock with 25 ns bunch spacing
33  // gives revolution time of 89.1 us and 11.223345 kHz
34  // this depends on the assumption that the particles are moving effectively
35  // at speed of light. There are also documents specifying the orbit time
36  // to 89.4 us
37  // Note: avoid to define the revolution frequency and use the integral numbers
38  // for bunch places and bunch spacing in nano seconds
39  // TODO: this eventually needs to be configurable
40  const int gNumberOfBunches = 3564;
41  const int gBunchSpacingNanoSec = 25;
42  const int gOrbitTimeNanoSec = std::ratio<gNumberOfBunches*gBunchSpacingNanoSec>::num;

44  using OrbitPrecision = std::integral_constant<int, 0>;
45  using BunchPrecision = std::integral_constant<int, 1>;

47  // the type of the clock tick depends on whether to use also the bunches
48  // as substructure of the orbit.
49  // a trait class to extrat the properties of the clock, namely the type
50  // of the tick and the period
51  template <typename T>
52  struct Property {
53  // the default does not specify anything and is never going to be used
54  };
55  template <>
56  struct Property<OrbitPrecision> {
57  using rep = uint32_t;
58  // avoid rounding errors by using the integral numbers in the std::ratio
59  // template to define the period
60  using period = std::ratio_multiply<std::ratio<gOrbitTimeNanoSec>, std::nano>;
61  };
62  template <>
63  struct Property<BunchPrecision> {
64  using rep = uint64_t;
65  // this is effectively the LHC clock and the ratio is the
66  // bunch spacing
67  using period = std::ratio_multiply<std::ratio<gBunchSpacingNanoSec>, std::nano>;
68  };
69 };

95  time_point now() noexcept {
96  // tp1 - tp2 results in a duration, we use to create a time_point with characteristics
