52 DataProcessorSpec getTOFDigitizerSpec(int channel)
53 {
54  // setup of some data structures shared between init and processing functions
55  // (a shared pointer is used since then automatic cleanup is guaranteed with a lifetime beyond
56  // one process call)
57  auto simChains = std::make_shared<std::vector<TChain*>>();
58 
59  // the instance of the actual digitizer
60  auto digitizer = std::make_shared<o2::tof::Digitizer>();
61 
62  // containers for digits and labels
63  auto digits = std::make_shared<std::vector<o2::tof::Digit>>();
64  auto digitsAccum = std::make_shared<std::vector<o2::tof::Digit>>(); // accumulator for all digits
65  auto labels = std::make_shared<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>();
66 
67  // the actual processing function which get called whenever new data is incoming
68  auto process = [simChains, digitizer, digits, digitsAccum, labels, channel](ProcessingContext& pc) {
69  static bool finished = false;
70  if (finished) {
71  return;
72  }
73  // RS: at the moment using hardcoded flag for continuos readout
74  static o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::CONTINUOUS;
75 
76  // read collision context from input
77  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
78  auto& timesview = context->getEventRecords();
79  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES";
80 
81  // if there is nothing to do ... return
82  if (timesview.size() == 0) {
83  return;
84  }
85 
86  TStopwatch timer;
87  timer.Start();
88 
89  LOG(INFO) << " CALLING TOF DIGITIZATION ";
90 
91  static std::vector<o2::tof::HitType> hits;
92  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum;
93 
94  auto& eventParts = context->getEventParts();
95  // loop over all composite collisions given from context
96  // (aka loop over all the interaction records)
97  for (int collID = 0; collID < timesview.size(); ++collID) {
98  digitizer->setEventTime(timesview[collID].timeNS);
99 
100  // for each collision, loop over the constituents event and source IDs
101  // (background signal merging is basically taking place here)
102  for (auto& part : eventParts[collID]) {
103  digitizer->setEventID(part.entryID);
104  digitizer->setSrcID(part.sourceID);
105 
106  // get the hits for this event and this source
107  hits.clear();
108  retrieveHits(*simChains.get(), "TOFHit", part.sourceID, part.entryID, &hits);
109 
110  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found " << hits.size() << " hits ";
111 
112  // call actual digitization procedure
113  labels->clear();
114  digits->clear();
115  digitizer->process(&hits, digits.get());
116  // copy digits into accumulator
117  //std::copy(digits->begin(), digits->end(), std::back_inserter(*digitsAccum.get()));
118  //labelAccum.mergeAtBack(*labels);
119  LOG(INFO) << "Have " << digits->size() << " digits ";
120  }
121  }
122  if (digitizer->isContinuous()) {
123  digits->clear();
124  labels->clear();
125  digitizer->flushOutputContainer(*digits.get());
126  LOG(INFO) << "FLUSHING LEFTOVER STUFF " << digits->size();
127  // copy digits into accumulator
128  //std::copy(digits->begin(), digits->end(), std::back_inserter(*digitsAccum.get()));
129  //labelAccum.mergeAtBack(*labels);
130  }
131 
132  // temporary accumulate vector of vecotors of digits in a single vector
133  // to be replace once we will be able to write the vector of vectors as different TTree entries
134  std::vector<std::vector<Digit>>* digitsVectOfVect = digitizer->getDigitPerTimeFrame();
135  std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>* mcLabVecOfVec = digitizer->getMCTruthPerTimeFrame();
136  for (Int_t i = 0; i < digitsVectOfVect->size(); i++) {
137  std::copy(digitsVectOfVect->at(i).begin(), digitsVectOfVect->at(i).end(), std::back_inserter(*digitsAccum.get()));
138  labelAccum.mergeAtBack(mcLabVecOfVec->at(i));
139  }
140 
141  LOG(INFO) << "Have " << labelAccum.getNElements() << " TOF labels ";
142  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
143  pc.outputs().snapshot(Output{ "TOF", "DIGITS", 0, Lifetime::Timeframe }, *digitsVectOfVect);
144  pc.outputs().snapshot(Output{ "TOF", "DIGITSMCTR", 0, Lifetime::Timeframe }, *mcLabVecOfVec);
145  LOG(INFO) << "TOF: Sending ROMode= " << roMode << " to GRPUpdater";
146  pc.outputs().snapshot(Output{ "TOF", "ROMode", 0, Lifetime::Timeframe }, roMode);
147 
148  timer.Stop();
149  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
150 
151  // we should be only called once; tell DPL that this process is ready to exit
152  pc.services().get<ControlService>().readyToQuit(false);
153  finished = true;
154  };
155 
156  // init function returning the lambda taking a ProcessingContext
157  auto initIt = [simChains, process, digitizer, labels](InitContext& ctx) {
158  // setup the input chain for the hits
159  simChains->emplace_back(new TChain("o2sim"));
160 
161  // add the main (background) file
162  simChains->back()->AddFile(ctx.options().get<std::string>("simFile").c_str());
163 
164  // maybe add a particular signal file
165  auto signalfilename = ctx.options().get<std::string>("simFileS");
166  if (signalfilename.size() > 0) {
167  simChains->emplace_back(new TChain("o2sim"));
168  simChains->back()->AddFile(signalfilename.c_str());
169  }
170 
171  // make sure that the geometry is loaded (TODO will this be done centrally?)
172  if (!gGeoManager) {
173  o2::Base::GeometryManager::loadGeometry();
174  }
175 
176  // init digitizer
177  digitizer->init();
178  const bool isContinuous = ctx.options().get<int>("pileup");
179  LOG(INFO) << "CONTINUOUS " << isContinuous;
180  digitizer->setContinuous(isContinuous);
181  digitizer->setMCTruthContainer(labels.get());
182 
183  // return the actual processing function which is now setup/configured
184  return process;
185  };

187  // create the full data processor spec using
188  // a name identifier
189  // input description
190  // algorithmic description (here a lambda getting called once to setup the actual processing function)
191  // options that can be used for this processor (here: input file names where to take the hits)
192  return DataProcessorSpec{
193  "TOFDigitizer", Inputs{ InputSpec{ "collisioncontext", "SIM", "COLLISIONCONTEXT",
194  static_cast<SubSpecificationType>(channel), Lifetime::Timeframe } },
195  Outputs{ OutputSpec{ "TOF", "DIGITS", 0, Lifetime::Timeframe },
196  OutputSpec{ "TOF", "DIGITSMCTR", 0, Lifetime::Timeframe },
197  OutputSpec{ "TOF", "ROMode", 0, Lifetime::Timeframe } },
198  AlgorithmSpec{ initIt },
199  Options{ { "simFile", VariantType::String, "o2sim.root", { "Sim (background) input filename" } },
200  { "simFileS", VariantType::String, "", { "Sim (signal) input filename" } },
201  { "pileup", VariantType::Int, 1, { "whether to run in continuous time mode" } } }
202  // I can't use VariantType::Bool as it seems to have a problem
203  };
204 }
