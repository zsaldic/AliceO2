145  DataRef getByPos(int pos) const {
146  if (pos * 2 + 1 > mSpan.size() || pos < 0) {
147  throw std::runtime_error("Unknown message requested at position " + std::to_string(pos));
148  }
149  if (pos > mInputsSchema.size()) {
150  throw std::runtime_error("Unknown schema at position" + std::to_string(pos));
151  }
152  if (mSpan.get(pos * 2) != nullptr && mSpan.get(pos * 2 + 1) != nullptr) {
153  return DataRef{ &mInputsSchema[pos].matcher,
154  mSpan.get(pos * 2),
155  mSpan.get(pos * 2 + 1) };
156  } else {
157  return DataRef{ &mInputsSchema[pos].matcher, nullptr, nullptr };
158  }
159  }
