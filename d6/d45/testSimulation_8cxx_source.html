59 std::vector<Hit> generateHits(size_t nHits, int deId, const o2::mid::Mapping& mapping, const o2::mid::GeometryTransformer geoTrans)
60 {
61  std::vector<Hit> hits;
62  std::random_device rd;
63  std::mt19937 mt(rd());
64  std::uniform_real_distribution<double> distX(-127.5, 127.5);
65  std::uniform_real_distribution<double> distY(-40., 40.);
66  while (hits.size() < nHits) {
67  Point3D<float> point(distX(mt), distY(mt), 0.);
68  if (!mapping.stripByPosition(point.x(), point.y(), 0, deId, false).isValid()) {
69  continue;
70  }
71  Point3D<float> globalPoint = SIMUL::geoTrans.localToGlobal(deId, point);
72  hits.emplace_back(hits.size(), deId, globalPoint, globalPoint);
73  }
74  return hits;
75 }

77 bool checkLabel(const ColumnData& digit, MCLabel& label, std::string& errorMessage)
78 {
79  std::stringstream ss;
80  ss << digit << "\n";
81  bool isOk = true;
82  if (label.getDEId() != digit.deId) {
83  isOk = false;
84  ss << "label deId " << label.getDEId() << " != " << digit.deId << "\n";
85  }
86  if (label.getColumnId() != digit.columnId) {
87  isOk = false;
88  ss << "label columnId " << label.getColumnId() << " != " << digit.columnId << "\n";
89  }
90  int firstStrip = label.getFirstStrip();
91  int lastStrip = label.getLastStrip();
92  int cathode = label.getCathode();
93  int nLines = (cathode == 0) ? 4 : 1;
94  for (int iline = 0; iline < nLines; ++iline) {
95  for (int istrip = 0; istrip < 16; ++istrip) {
96  int currStrip = MCLabel::getStrip(istrip, iline);
97  bool isFired = digit.isStripFired(istrip, cathode, iline);
98  if (isFired != (currStrip >= firstStrip && currStrip <= lastStrip)) {
99  isOk = false;
100  ss << "Cathode: " << cathode << " firstStrip: " << firstStrip << " lastStrip: " << lastStrip;
101  ss << " line: " << iline << " strip: " << istrip << " fired: " << isFired << "\n";
102  }
103  }
104  }
105  errorMessage = ss.str();
106  return isOk;

125 BOOST_DATA_TEST_CASE(MID_SimulChain, boost::unit_test::data::make(getDEList()), deId)
126 {
127  // In this test, we generate a cluster from one impact point and we reconstruct it.
128  // If the impact point is in the RPC, the digitizer will return a list of strips,
129  // that are close to each other by construction.
130  // We will therefore have exactly one reconstructed cluster.
131  // It is worth noting, however, that the clustering algorithm is designed to
132  // produce two clusters if the BP and NBP do not overlap.
133  // The digitizer produces superposed BP and NBP strips only if the response parameters
134  // are the same for both.
135  // Otherwise we can have from 1 to 3 clusters produced.
136 
137  std::vector<ColumnDataMC> digitStoreMC;
138  o2::dataformats::MCTruthContainer<MCLabel> digitLabelsMC;
139  std::vector<o2::mid::ColumnData> digitStore;
140  o2::dataformats::MCTruthContainer<MCLabel> digitLabels;
141 
142  for (int ievent = 0; ievent < 100; ++ievent) {
143  auto hits = generateHits(1, deId, SIMUL::mapping, SIMUL::geoTrans);
144  std::stringstream ss;
145  int nGenClusters = 1, nRecoClusters = 0;
146  SIMUL::digitizer.process(hits, digitStoreMC, digitLabelsMC);
147  SIMUL::digitsMerger.process(digitStoreMC, digitLabelsMC, digitStore, digitLabels);
148  SIMUL::preClusterizer.process(digitStore);
149  SIMUL::clusterizer.process(SIMUL::preClusterizer.getPreClusters());
150  nRecoClusters = SIMUL::clusterizer.getNClusters();
151  ss << "nRecoClusters: " << nRecoClusters << " nGenClusters: " << nGenClusters << "\n";
152  for (auto& col : digitStore) {
153  ss << col << "\n";
154  }
155  ss << "\n Clusters:\n";
156  for (auto it = SIMUL::clusterizer.getClusters().begin(); it < SIMUL::clusterizer.getClusters().begin() + nRecoClusters; ++it) {
157  ss << "pos: (" << it->xCoor << ", " << it->yCoor << ") sigma2: (" << it->sigmaX2 << ", " << it->sigmaY2 << ")\n";
158  }
159 
160  int nColumns = digitStore.size();
161 
162  if (SIMUL::params.getParB(0, deId) == SIMUL::params.getParB(1, deId) && nColumns <= 2) {
163  BOOST_TEST((nRecoClusters == nGenClusters), ss.str());
164  } else {
165  BOOST_TEST((nRecoClusters >= nGenClusters && nRecoClusters <= nColumns), ss.str());
166  }
167  }
168 }

170 BOOST_DATA_TEST_CASE(MID_Digitizer, boost::unit_test::data::make(getDEList()), deId)
171 {
172  // In this test we generate 100 hits
173  // We then tests that the MC labels are correctly assigned
174  auto hits = generateHits(100, deId, SIMUL::mapping, SIMUL::geoTrans);
175  std::vector<ColumnDataMC> digitStoreMC;
176  o2::dataformats::MCTruthContainer<MCLabel> digitLabelsMC;
177  SIMUL::digitizer.process(hits, digitStoreMC, digitLabelsMC);
178  // We check that we have as many sets of labels as digits
179  BOOST_TEST(digitStoreMC.size() == digitLabelsMC.getIndexedSize());
180  for (size_t idig = 0; idig < digitLabelsMC.getIndexedSize(); ++idig) {
181  auto labels = digitLabelsMC.getLabels(idig);
182  auto digit = digitStoreMC[idig];
183  // Then for each label we check that the parameters correctly identify the digit
184  for (auto label : labels) {
185  std::string errorMessage;
186  BOOST_TEST(checkLabel(digit, label, errorMessage), errorMessage.c_str());
187  }
188  }
189 
190  // We then test the merging of the 100 hits
191  std::vector<ColumnData> digitStore;
192  o2::dataformats::MCTruthContainer<MCLabel> digitLabels;
193  SIMUL::digitsMerger.process(digitStoreMC, digitLabelsMC, digitStore, digitLabels);
194  // The number of merged digits should be smaller than the number of input digits
195  BOOST_TEST(digitStore.size() <= digitStoreMC.size());
196  // We check that we have as many sets of labels as digits
197  BOOST_TEST(digitStore.size() == digitLabels.getIndexedSize());
198  // We check that we do not discard any label in the merging
199  BOOST_TEST(digitLabels.getNElements() == digitLabelsMC.getNElements());
200  for (auto digit : digitStoreMC) {
201  bool isMergedDigit = false;
202  for (auto col : digitStore) {
203  if (digit.deId == col.deId && digit.columnId == col.columnId) {
205  isMergedDigit = true;
206  // Finally we check that all input patterns are contained in the merged ones
207  for (int iline = 0; iline < 4; ++iline) {
208  BOOST_TEST(((digit.getBendPattern(iline) & col.getBendPattern(iline)) == digit.getBendPattern(iline)));
209  }
210  BOOST_TEST(((digit.getNonBendPattern() & col.getNonBendPattern()) == digit.getNonBendPattern()));
211  }
212  }
213  // Check that all digits get merged
214  BOOST_TEST(isMergedDigit);
215  }
216 }
