43 bool Digitizer::addBPStrips(double xPos, double yPos, int deId, double prob, double xOffset)
44 {
46 
47  // The offset is used to check the strips in the bending plane
48  // placed in the board close to the fired one
49  // If the response says that the strip is too far away, there is no need to check further
50  if (!mResponse.isFired(prob, std::abs(xOffset), 0, deId)) {
51  return false;
52  }
53 
54  Mapping::MpStripIndex stripIndex = mMapping.stripByPosition(xPos + 1.01 * xOffset, yPos, 0, deId, false);
55  // Check if the point is still inside the RPC
56  if (!stripIndex.isValid()) {
57  return false;
58  }
59  addStrip(stripIndex, 0, deId);
60  MpArea area = mMapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
61  std::array<double, 2> dist = { area.getYmax() - yPos, yPos - area.getYmin() };
62  addNeighbours(stripIndex, 0, deId, prob, dist, xOffset);
63  return true;
64 }

67 bool Digitizer::addNeighbours(const Mapping::MpStripIndex& stripIndex, int cathode, int deId, double prob,
68  const std::array<double, 2>& initialDist, double xOffset)
69 {
71 
72  double xOffset2 = xOffset * xOffset;
73  for (int idir = 0; idir < 2; ++idir) {
74  // Search for neighbours in the two directions
75  // up and down for the BP, right and left for the NBP
76  double dist = initialDist[idir];
77  Mapping::MpStripIndex neigh = mMapping.nextStrip(stripIndex, cathode, deId, idir);
78  while (neigh.isValid() && mResponse.isFired(prob, std::sqrt(dist * dist + xOffset2), cathode, deId)) {
79  addStrip(neigh, cathode, deId);
80  dist += mMapping.getStripSize(neigh.strip, cathode, neigh.column, deId);
81  neigh = mMapping.nextStrip(neigh, cathode, deId, idir);
82  }
83  }
84  return true;
85 }

88 bool Digitizer::hitToDigits(const Hit& hit)
89 {
91 
92  // Convert point from global to local coordinates
93  auto midPt = hit.middlePoint();
94  int deId = hit.GetDetectorID();
95  Point3D<double> localPoint = mTransformer.globalToLocal(deId, (double)midPt.x(), (double)midPt.y(), (double)midPt.z());
96 
97  // First get the touched BP strip
98  Mapping::MpStripIndex stripIndex = mMapping.stripByPosition(localPoint.x(), localPoint.y(), 0, deId);
99 
100  // Check if the chamber was efficient
101  bool isEfficientBP, isEfficientNBP;
102  if (!mEfficiencyResponse.isEfficient(deId, stripIndex.column, stripIndex.line, isEfficientBP, isEfficientNBP)) {
103  return false;
104  }
105 
106  // Digitize if the RPC was efficient
107  mDigits.clear();
108  double prob = mRandom(mGenerator);
109 
110  if (isEfficientBP) {
111  addStrip(stripIndex, 0, deId);
112  MpArea area = mMapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
113  // This is the distance between the hit point and the edges of the strip along y
114  std::array<double, 2> dist = { area.getYmax() - localPoint.y(), localPoint.y() - area.getYmin() };
115  addNeighbours(stripIndex, 0, deId, prob, dist);
116  // Search for neighbours in the close column toward inside
117  addBPStrips(localPoint.x(), localPoint.y(), deId, prob, area.getXmin() - localPoint.x());
118  // Search for neighbours in the close column toward outside
119  addBPStrips(localPoint.x(), localPoint.y(), deId, prob, area.getXmax() - localPoint.x());
120  }
121 
122  // Then check the touched NBP strip
123  if (isEfficientNBP) {
124  stripIndex = mMapping.stripByPosition(localPoint.x(), localPoint.y(), 1, deId);
125  addStrip(stripIndex, 1, deId);
126  MpArea area = mMapping.stripByLocation(stripIndex.strip, 1, stripIndex.line, stripIndex.column, deId);
127  // This is the distance between the hit point and the edges of the strip along x
128  std::array<double, 2> dist = { area.getXmax() - localPoint.x(), localPoint.x() - area.getXmin() };
129  addNeighbours(stripIndex, 1, deId, prob, dist);
130  }
131 
132  return true;
133 }

136 void Digitizer::process(const std::vector<Hit>& hits, std::vector<ColumnDataMC>& digitStore, o2::dataformats::MCTruthContainer<MCLabel>& mcContainer)
137 {
139  // mMCTruthContainer.clear();
140  digitStore.clear();
141  mcContainer.clear();
142  int firstStrip = 0, lastStrip = 0;
143 
144  for (auto& hit : hits) {
145  hitToDigits(hit);
146 
147  for (auto& digit : mDigits) {
148  digitStore.emplace_back(digit);
149  for (int icathode = 0; icathode < 2; ++icathode) {
150  if (getLabelLimits(icathode, digit, firstStrip, lastStrip)) {
151  MCLabel label(hit.GetTrackID(), mEventID, mSrcID, digit.deId, digit.columnId, icathode, firstStrip, lastStrip);
152  mcContainer.addElement(digitStore.size() - 1, label);
153  }
154  }
155  }
156  }
157 }

160 bool Digitizer::getLabelLimits(int cathode, const ColumnDataMC& col, int& firstStrip, int& lastStrip) const
161 {
163  int nLines = (cathode == 0) ? 4 : 1;
164  int invalid = 9999;
165  firstStrip = invalid, lastStrip = invalid;
166  for (int iline = 0; iline < nLines; ++iline) {
167  for (int istrip = 0; istrip < 16; ++istrip) {
168  if (col.isStripFired(istrip, cathode, iline)) {
169  lastStrip = MCLabel::getStrip(istrip, iline);
170  if (firstStrip == invalid) {
171  firstStrip = lastStrip;
172  }
173  } else if (firstStrip != invalid) {
174  return true;
175  }
176  }
177  }
178  return (firstStrip != invalid);
179 }
