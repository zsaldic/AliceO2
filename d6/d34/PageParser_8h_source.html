271  size_t setElement(size_t position, const value_type& element) const {
272  // write functionality not yet implemented for grouped elements
273  assert(std::is_void<GroupType>::value);
274  // check if we are at the end
275  if (position >= mSize) {
276  assert(position == mSize);
277  return mSize;
278  }
279 
280  // check if there is space for one element

301  bool getElement(size_t& position, value_type& element) const {
302  // check if we are at the end
303  if (position >= mSize) {
304  assert(position == mSize);
305  position = mSize;
306  return false;
307  }
308 
309  // handle group if defined
310  if (!std::is_void<GroupType>::value) {
311  if (mNGroupElements == 0) {
312  // new group has to be read from the buffer
313  do {
314  if ((position % page_size) == 0) {
315  position += sizeof(PageHeaderType);
316  }
317  if ((position % page_size) != sizeof(PageHeaderType)) {
318  // forward to the next page
319  position += page_size - (position % page_size) + sizeof(PageHeaderType);
320  if (position > mSize) {
321  //this is probably a valid condition as the group header can just
322  //indicate zero clusters
323  //throw std::runtime_error("");
324  position = mSize;
325  return false;
326  }
327  }
328  const_cast<PageParser*>(this)->mNGroupElements = readGroupHeader(position, mGroupHeader);
329  position += pageparser::sizeofGroupHeader<GroupType>();
330  } while (mNGroupElements == 0);
331 
332  size_t nPages = 0;
333  size_t required = pageparser::sizeofGroupHeader<GroupType>() + mNGroupElements * sizeof(value_type);
334  do {
335  // the block of elements can go beyond the current page, find out
336  // how many additional pages are required
337  required += sizeof(PageHeaderType);
338  ++nPages;
339  } while (required > nPages * page_size);
340  required -= sizeof(PageHeaderType) + pageparser::sizeofGroupHeader<GroupType>();
341  if (position + required > mSize) {
342  throw std::runtime_error("format error: the number of group elements "
343  "does not fit into the remaining buffer");
344  }
345  }
346  // now we will read one element
347  const_cast<PageParser*>(this)->mNGroupElements -= 1;;
348  }
349 
350  // check if there is space for one element
351  if (position + sizeof(value_type) > mSize) {
352  // FIXME: not sure if this is considered an error condition if
353  // no groups are used, i.e. the buffer should have the correct size
354  // and no extra space after the last element
355  position = mSize;
356  return false;
357  }
358 
359  auto source = mBuffer + position;
360  auto target = reinterpret_cast<BufferType*>(&element);
361  position += copy<SourceInPageBuffer>(source, target, page_size - (position % page_size));
362  return true;
363  }

365  // copy data, depending on compile time switch, either source or target
366  // pointer are treated as pointer in the raw page, i.e. can be additionally
367  // incremented by the page header
368  template<typename SwitchT>
369  size_t copy(const BufferType* source, BufferType* target, size_t pageCapacity) const
370  {
371  size_t position = 0;
372  auto copySize = sizeof(value_type);
373  // choose which of the pointers needs additional PageHeader offsets
374  auto pageOffsetTarget = SwitchT::value? &target : const_cast<BufferType**>(&source);
375  if (pageCapacity == page_size) {
376  // skip the page header at beginning of page
377  position += sizeof(PageHeaderType);
378  pageCapacity -= sizeof(PageHeaderType);
379  *pageOffsetTarget += sizeof(PageHeaderType);
380  }
381  if (copySize > pageCapacity) {
382  // object is split at the page boundary, copy the part
383  // in the current page first
384  copySize = pageCapacity;
385  }
386  if (copySize > 0) {
387  memcpy(target, source, copySize);
388  position += copySize;
389  source += copySize;
390  target += copySize;
391  }
392  copySize = sizeof(value_type) - copySize;
393  if (copySize > 0) {
394  // skip page header at beginning of new page and copy
395  // remaining part of the element
396  position += sizeof(PageHeaderType);
397  *pageOffsetTarget += sizeof(PageHeaderType);
398  memcpy(target, source, copySize);
399  position += copySize;
400  }
401  return position;
402  }
