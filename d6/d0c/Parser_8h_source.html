150  template<typename InputType>
151  int parse(const InputType* buffer, size_t bufferSize,
152  CheckHeaderFct checkHeader,
153  CheckTrailerFct<TrailerType> checkTrailer,
154  GetFrameSizeFct getFrameSize,
155  InsertFct insert) {
156  static_assert(sizeof(InputType) == 1,
157  "ForwardParser currently only supports byte type buffer"
158  );
159  if (buffer == nullptr || bufferSize == 0) return 0;
160  auto position = 0;
161  std::vector<FrameInfo> frames;
162  do {
163  FrameInfo entry;
164 
165  // check the header
166  if (sizeof(HeaderType) + position > bufferSize) break;
167  entry.header = reinterpret_cast<const HeaderType*>(buffer + position);
168  if (!checkHeader(*entry.header)) break;
169 
170  // extract frame size from header, this is expected to be the
171  // total frome size including header, payload and optional trailer
172  auto frameSize = getFrameSize(*entry.header);
173  if (frameSize + position > bufferSize) break;
174 
175  // payload starts right after the header
176  entry.payload = reinterpret_cast<typename FrameInfo::PtrT>(entry.header + 1);
177  entry.length = frameSize - totalOffset;
178 
179  // optionally extract and check trailer
180  if (tailOffset > 0) {
181  entry.trailer = nullptr;
182  } else {
183  auto trailerStart = buffer + position + frameSize - tailOffset;
184  entry.trailer = reinterpret_cast<const TrailerType*>(trailerStart);
185  if (!CheckTrailer(entry, checkTrailer)) break;
186  }
187 
188  // store the extracted frame info and continue with remaining buffer
189  frames.emplace_back(entry);
190  position += frameSize;
191  } while (position < bufferSize);
192 
193  if (position == bufferSize) {
194  // frames found and format consistent, insert entries to target
195  // Note: the complete block must be consistent
196  for (auto entry : frames) {
197  if (!insert(entry)) break;
198  }
199  return frames.size();
200  } else if (frames.size() == 0) {
201  // no frames found at all, the buffer does not contain any
202  return 0;
203  }
204 
205  // format error detected
206  // TODO: decide about error policy
207  return -1;
208  }

327  FrameInfo entry;

329  // start from end, extract and check trailer
330  if (sizeof(TrailerType) > position) break;
331  entry.trailer = reinterpret_cast<const TrailerType*>(buffer + position - sizeof(TrailerType));
332  if (!checkTrailer(*entry.trailer)) break;
333 
334  // get the total frame size
335  auto frameSize = getFrameSize(*entry.trailer);
336  if (frameSize > position) break;
337 
338  // extract and check header
339  auto headerStart = buffer + position - frameSize;
340  entry.header = reinterpret_cast<const HeaderType*>(headerStart);
341  if (!checkHeader(*entry.header)) break;
342 
343  // payload immediately after header
344  entry.payload = reinterpret_cast<typename FrameInfo::PtrT>(entry.header + 1);
345  entry.length = frameSize - sizeof(HeaderType) - sizeof(TrailerType);
346  frames.emplace_back(entry);
347  position -= frameSize;
348  } while (position > 0);

351  // frames found and format consistent, the complete block must be consistent
352  for (auto entry : frames) {
353  if (!insert(entry)) break;
354  }

358  return 0;
