41 Int_t CalibPedestal::updateROC(const Int_t roc, const Int_t row, const Int_t pad,
42  const Int_t timeBin, const Float_t signal)
43 {
44  Int_t adcValue = Int_t(signal);
45  if (timeBin < mFirstTimeBin || timeBin > mLastTimeBin)
46  return 0;
47  if (adcValue < mADCMin || adcValue > mADCMax)
48  return 0;
49 
50  const GlobalPadNumber padInROC = mMapper.getPadNumberInROC(PadROCPos(roc, row, pad));
51  Int_t bin = padInROC * mNumberOfADCs + (adcValue - mADCMin);
52  vectorType& adcVec = *getVector(ROC(roc), kTRUE);
53  ++(adcVec[bin]);
54 
55  //printf("bin: %5d, val: %.2f\n", bin, adcVec[bin]);
56 
57  return 0;
58 }

78 void CalibPedestal::analyse()
79 {
80  ROC roc;
81 
82  std::vector<float> fitValues;
83 
84  for (auto& vecPtr : mADCdata) {
85  auto vec = vecPtr.get();
86  if (!vec) {
87  ++roc;
88  continue;
89  }
90 
91  CalROC& calROCPedestal = mPedestal.getCalArray(roc);
92  CalROC& calROCNoise = mNoise.getCalArray(roc);
93 
94  float* array = vec->data();
95 
96  const size_t numberOfPads = (roc.rocType() == RocType::IROC) ? mMapper.getPadsInIROC() : mMapper.getPadsInOROC();
97 
98  float pedestal{};
99  float noise{};
100 
101  for (Int_t ichannel = 0; ichannel < numberOfPads; ++ichannel) {
102  size_t offset = ichannel * mNumberOfADCs;
103  if (mStatisticsType == StatisticsType::GausFit) {
104  fitGaus(mNumberOfADCs, array + offset, float(mADCMin), float(mADCMax + 1), fitValues);
105  pedestal = fitValues[1];
106  noise = fitValues[2];
107  } else if (mStatisticsType == StatisticsType::MeanStdDev) {
108  StatisticsData data = getStatisticsData(array + offset, mNumberOfADCs, double(mADCMin), double(mADCMax));
109  pedestal = data.mCOG;
110  noise = data.mStdDev;
111  }
112 
113  calROCPedestal.setValue(ichannel, pedestal);
114  calROCNoise.setValue(ichannel, noise);
115 
116  //printf("roc: %2d, channel: %4d, pedestal: %.2f, noise: %.2f\n", roc.getRoc(), ichannel, pedestal, noise);
117  }
118 
119  ++roc;
120  }
121 }
