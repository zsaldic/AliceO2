94 int runSim(std::string transport, std::string primaddress, std::string mergeraddress)
95 {
96  auto factory = FairMQTransportFactory::CreateTransportFactory(transport);
97  auto primchannel = FairMQChannel{ "primary-get", "req", factory };
98  primchannel.Connect(primaddress);
99  primchannel.ValidateChannel();
100 
101  auto datachannel = FairMQChannel{ "simdata", "push", factory };
102  datachannel.Connect(mergeraddress);
103  datachannel.ValidateChannel();
104  // the channels are setup
105 
106  // init the sim object
107  auto sim = getDevice();
108  sim->lateInit();
109 
110  // the simplified runloop
111  while (sim->Kernel(primchannel, datachannel)) {
112  }
113  LOG(INFO) << "simulation is done";
114  return 0;
115 }

117 void pinToCPU(int cpuid)
118 {
119 // MacOS does not support this API so we add a protection
120 #ifndef __APPLE__
121  auto affinity = getenv("ALICE_CPUAFFINITY");
122  if (affinity) {
123  pthread_t thread;
124 
125  thread = pthread_self();
126 
127  cpu_set_t cpuset;
128  CPU_ZERO(&cpuset);
129  CPU_SET(cpuid, &cpuset);
130 
131  auto s = pthread_setaffinity_np(thread, sizeof(cpu_set_t), &cpuset);
132  if (s != 0) {
133  LOG(WARNING) << "FAILED TO SET PTHREAD AFFINITY";
134  }
135 
136  /* Check the actual affinity mask assigned to the thread */
137  s = pthread_getaffinity_np(thread, sizeof(cpu_set_t), &cpuset);
138  if (s != 0) {
139  LOG(WARNING) << "FAILED TO GET PTHREAD AFFINITY";
140  }
141 
142  for (int j = 0; j < CPU_SETSIZE; j++) {
143  if (CPU_ISSET(j, &cpuset)) {
144  LOG(INFO) << "ENABLED CPU " << j;
145  }
146  }
147  }
148 #else
149  LOG(WARN) << "CPU AFFINITY NOT IMPLEMENTED ON APPLE";
150 #endif
151 }

153 int main(int argc, char* argv[])
154 {
155  auto internalfork = getenv("ALICE_SIMFORKINTERNAL");
156  if (internalfork) {
157  // retrieve published port numbers
158  auto serverportenv = getenv("ALICE_O2SIM_SERVERPORT");
159  if (!serverportenv) {
160  LOG(FATAL) << "NEED SERVER PORT PUBLISHED";
161  }
162  auto mergerportenv = getenv("ALICE_O2SIM_MERGERPORT");
163  if (!mergerportenv) {
164  LOG(FATAL) << "NEED MERGER PORT PUBLISHED";
165  }
166 
167  std::string serveraddress("tcp://localhost:" + std::string(serverportenv));
168  std::string mergeraddress("tcp://localhost:" + std::string(mergerportenv));
169  auto host = getenv("ALICE_SIMMAINHOST");
170  if (host) {
171  // argv[1] is supposed to be an IP address or hostname
172  serveraddress = "tcp://" + std::string(host) + ":" + std::string(serverportenv);
173  mergeraddress = "tcp://" + std::string(host) + ":" + std::string(mergerportenv);
174  }
175  LOG(INFO) << serveraddress;
176  LOG(INFO) << mergeraddress;
177 
178  // This is a solution based on initializing the simulation once
179  // and then fork the process to share the simulation memory across
180  // many processes. Here we are not using FairMQDevices and just setup
181  // some channels manually and do our own runloop.
182 
183  // we init the simulation first
184  std::unique_ptr<FairRunSim> simrun;
185  // TODO: take the addresses from somewhere else
186  if (!initializeSim("zeromq", serveraddress, simrun)) {
187  LOG(ERROR) << "Could not initialize simulation";
188  return 1;
189  }
190 
191  // should be factored out?
192  int nworkers = std::max(1u, std::thread::hardware_concurrency() / 2);
193  auto f = getenv("ALICE_NSIMWORKERS");
194  if (f) {
195  nworkers = atoi(f);
196  }
197  LOG(INFO) << "Running with " << nworkers << " sim workers ";
198 
199  // then we fork and create a device in each fork
200  for (int i = 0; i < nworkers; ++i) {
201  // we use the current process as one of the workers as it has nothing else to do
202  auto pid = (i == nworkers - 1) ? 0 : fork();
203  if (pid == 0) {
204  // we will try to pin each worker to a particular CPU
205  // this can be made configurable via enviroment variables??
206  pinToCPU(i);
207 
208  runSim("zeromq", serveraddress, mergeraddress);
209 
210  _exit(0);
211  }
212  }
213  int status;
214  wait(&status); /* only the parent waits */
215  _exit(0);
216  } else {
217  // This the solution where we setup an ordinary FairMQDevice
218  // (each if which will setup its own simulation). Parallelism
219  // is achieved outside by instantiating multiple device processes.
220  _exit(initAndRunDevice(argc, argv));
221  }
222 }
