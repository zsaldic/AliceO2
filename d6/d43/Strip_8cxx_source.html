59 Int_t Strip::addDigit(Int_t channel, Int_t tdc, Int_t tot, Int_t bc, Int_t lbl)
60 {
61 
62  // return the MC label. We pass it also as argument, but it can change in
63  // case the digit was merged
64 
65  auto key = Digit::getOrderingKey(channel, bc, tdc); // the digits are ordered first per channel, then inside the channel per BC, then per time
66  auto dig = findDigit(key);
67  if (dig) {
68  lbl = dig->getLabel(); // getting the label from the already existing digit
69  dig->merge(tdc, tot); // merging to the existing digit
70  } else {
71  auto digIter = mDigits.emplace(std::make_pair(key, Digit(channel, tdc, tot, bc, lbl)));
72  }
73 
74  return lbl;
75 }
