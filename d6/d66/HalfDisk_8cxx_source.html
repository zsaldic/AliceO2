99 void HalfDisk::createLadders()
100 {
101 
102  LOG(DEBUG1) << "CreateLadders: start building ladders" << FairLogger::endl;
103  for (Int_t iLadder = 0; iLadder < mSegmentation->getNLadders(); iLadder++) {
104 
105  LadderSegmentation* ladderSeg = mSegmentation->getLadder(iLadder);
106  if (!ladderSeg)
107  Fatal("CreateLadders", Form("No Segmentation found for ladder %d ", iLadder), 0, 0);
108  auto* ladder = new Ladder(ladderSeg);
109  TGeoVolume* ladVol = ladder->createVolume();
110 
111  // Position of the center on the ladder volume in the ladder coordinate system
112  TGeoBBox* shape = (TGeoBBox*)ladVol->GetShape();
113  Double_t center[3];
114  center[0] = shape->GetDX();
115  center[1] = shape->GetDY();
116  center[2] = shape->GetDZ();
117 
118  Double_t master[3];
119  ladderSeg->getTransformation()->LocalToMaster(center, master);
120  Int_t ladderId = Geometry::instance()->getLadderID(ladderSeg->GetUniqueID());
121 
122  mHalfDiskVolume->AddNode(ladVol, ladderId, new TGeoCombiTrans(master[0], master[1], master[2],
123  ladderSeg->getTransformation()->GetRotation()));
124 
125  delete ladder;
126  }
127 }
