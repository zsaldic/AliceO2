38 void CookedTrackerDPL::init(InitContext& ic)
39 {
40  auto nthreads = ic.options().get<int>("nthreads");
41  mTracker.setNumberOfThreads(nthreads);
42  auto filename = ic.options().get<std::string>("grp-file");
43  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
44  if (grp) {
45  mGRP.reset(grp);
46  o2::Base::Propagator::initFieldFromGRP(grp);
47  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
48 
49  o2::Base::GeometryManager::loadGeometry();
50  o2::ITS::GeometryTGeo* geom = o2::ITS::GeometryTGeo::Instance();
51  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L, o2::TransformType::T2GRot,
52  o2::TransformType::T2G));
53  mTracker.setGeometry(geom);
54 
55  double origD[3] = { 0., 0., 0. };
56  mTracker.setBz(field->getBz(origD));
57 
58  bool continuous = mGRP->isDetContinuousReadOut("ITS");
59  LOG(INFO) << "ITSCookedTracker RO: continuous=" << continuous;
60  mTracker.setContinuousMode(continuous);
61  } else {
62  LOG(ERROR) << "Cannot retrieve GRP from the " << filename.c_str() << " file !";
63  mState = 0;
64  }
65  mState = 1;
66 }

68 void CookedTrackerDPL::run(ProcessingContext& pc)
69 {
70  if (mState != 1)
71  return;
72 
73  auto compClusters = pc.inputs().get<const std::vector<o2::ITSMFT::CompClusterExt>>("compClusters");
74  auto clusters = pc.inputs().get<const std::vector<o2::ITSMFT::Cluster>>("clusters");
75  auto labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
76  auto rofs = pc.inputs().get<const std::vector<o2::ITSMFT::ROFRecord>>("ROframes");
77 
78  LOG(INFO) << "ITSCookedTracker pulled " << clusters.size() << " clusters, "
79  << labels->getIndexedSize() << " MC label objects , in "
80  << rofs.size() << " RO frames";
81 
82  std::vector<o2::ITS::TrackITS> tracks;
83  o2::dataformats::MCTruthContainer<o2::MCCompLabel> trackLabels;
84  mTracker.setMCTruthContainers(labels.get(), &trackLabels);
85 
86  std::vector<std::array<Double_t, 3>> vertices; //FIXME : run an actual vertex finder !
87  vertices.push_back({ 0., 0., 0. });
88  mTracker.setVertices(vertices);
89 
90  mTracker.process(clusters, tracks);
91 
92  LOG(INFO) << "ITSCookedTracker pushed " << tracks.size() << " tracks";
93  pc.outputs().snapshot(Output{ "ITS", "TRACKS", 0, Lifetime::Timeframe }, tracks);
94  pc.outputs().snapshot(Output{ "ITS", "TRACKSMCTR", 0, Lifetime::Timeframe }, trackLabels);
95 
96  mState = 2;
97  //pc.services().get<ControlService>().readyToQuit(true);
98 }
