139  TString idPath = filename(0, filename.Last('/'));
140  id.setPath(idPath);

143  }
144 
145  filename = filename(idPath.Length() + 1, filename.Length() - idPath.Length());
146 
147  Ssiz_t mSize;
148  // valid filename: Run#firstRun_#lastRun_v#version_s0.root
149  TRegexp keyPattern("^Run[0-9]+_[0-9]+_v[0-9]+_s0.root$");
150  keyPattern.Index(filename, &mSize);
151  if (!mSize) {
152 
153  // TODO backward compatibility ... maybe remove later!
154  Ssiz_t oldmSize;
155  TRegexp oldKeyPattern("^Run[0-9]+_[0-9]+_v[0-9]+.root$");
156  oldKeyPattern.Index(filename, &oldmSize);
157  if (!oldmSize) {
158  LOG(DEBUG) << "Bad filename <" << filename.Data() << ">.";
159  return kFALSE;
160  } else {
161  LOG(DEBUG) << "Old filename format <" << filename.Data() << ">.";
162  id.setSubVersion(-11); // TODO trick to ensure backward compatibility
163  }
164 
165  } else {
166  id.setSubVersion(-1); // TODO trick to ensure backward compatibility
167  }
168 
169  filename.Resize(filename.Length() - sizeof(".root") + 1);

171  TObjArray *strArray = (TObjArray *) filename.Tokenize("_");

174  id.setFirstRun(atoi(firstRunString.Data() + 3));
175  id.setLastRun(atoi(((TObjString *) strArray->At(1))->GetString()));
