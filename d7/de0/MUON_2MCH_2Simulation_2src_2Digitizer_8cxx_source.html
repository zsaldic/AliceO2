103 int Digitizer::processHit(const Hit& hit, int detID, double event_time)
104 {
105  Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
106 
107  Response& resp = response(isStation1(detID));
108 
109  //convert energy to charge
110  auto charge = resp.etocharge(hit.GetEnergyLoss());
111  auto time = hit.GetTime(); //to be used for pile-up
112 
113  //transformation from global to local
114  auto t = o2::mch::getTransformation(detID, *gGeoManager);
115  Point3D<float> lpos;
116  t.MasterToLocal(pos, lpos);
117 
118  auto anodpos = resp.getAnod(lpos.X());
119  auto fracplane = resp.chargeCorr();
120  auto chargebend = fracplane * charge;
121  auto chargenon = charge / fracplane;
122 
123  //borders of charge gen.
124  auto xMin = anodpos - resp.getQspreadX() * 0.5;
125  auto xMax = anodpos + resp.getQspreadX() * 0.5;
126  auto yMin = lpos.Y() - resp.getQspreadY() * 0.5;
127  auto yMax = lpos.Y() + resp.getQspreadY() * 0.5;
128 
129  //get segmentation for detector element
130  auto& seg = segmentation(detID);
131  auto localX = anodpos;
132  auto localY = lpos.Y();
133 
134  //get area for signal induction from segmentation
135  //single pad as check
136  int padidbendcent = 0;
137  int padidnoncent = 0;
138  int ndigits = 0;
139 
140  bool padexists = seg.findPadPairByPosition(localX, localY, padidbendcent, padidnoncent);
141  if (!padexists) {
142  LOG(ERROR) << "Did not find _any_ pad for localX,Y=" << localX << "," << localY;
143  return 0;
144  }
145 
146  seg.forEachPadInArea(xMin, yMin, xMax, yMax, [&resp, &digits = this->mDigits, chargebend, chargenon, localX, localY, &seg, &ndigits ](int padid) {
147  auto dx = seg.padSizeX(padid) * 0.5;
148  auto dy = seg.padSizeY(padid) * 0.5;
149  auto xmin = (localX - seg.padPositionX(padid)) - dx;
150  auto xmax = xmin + dx;
151  auto ymin = (localY - seg.padPositionY(padid)) - dy;
152  auto ymax = ymin + dy;
153  auto q = resp.chargePadfraction(xmin, xmax, ymin, ymax);
154  if (seg.isBendingPad(padid)) {
155  q *= chargebend;
156  } else {
157  q *= chargenon;
158  }
159  auto signal = resp.response(q);
160  digits.emplace_back(padid, signal);
161  ++ndigits;
162  });
163  return ndigits;
164 }

167 void Digitizer::fillOutputContainer(std::vector<Digit>& digits)
168 {
169  // filling the digit container
170  if (mDigits.empty())
171  return;
172 
173  auto itBeg = mDigits.begin();
174  auto iter = itBeg;
175  for (; iter != mDigits.end(); ++iter) {
176  digits.emplace_back(*iter);
177  }
178  mDigits.erase(itBeg, iter);
179  mMCTruthOutputContainer.clear();
180  for (int index = 0; index < mTrackLabels.getIndexedSize(); ++index) {
181  mMCTruthOutputContainer.addElements(index, mTrackLabels.getLabels(index));
182  }
