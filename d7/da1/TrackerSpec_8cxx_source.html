60 void TrackerDPL::init(InitContext& ic)
61 {
62  auto filename = ic.options().get<std::string>("grp-file");
63  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
64  if (grp) {
65  mGRP.reset(grp);
66  o2::Base::Propagator::initFieldFromGRP(grp);
67  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
68 
69  o2::Base::GeometryManager::loadGeometry();
70  o2::ITS::GeometryTGeo* geom = o2::ITS::GeometryTGeo::Instance();
71  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L, o2::TransformType::T2GRot,
72  o2::TransformType::T2G));
73 
74  mTracker = std::make_unique<o2::ITS::Tracker>(&mTraits);
75  double origD[3] = { 0., 0., 0. };
76  mTracker->setBz(field->getBz(origD));
77  } else {
78  LOG(ERROR) << "Cannot retrieve GRP from the " << filename.c_str() << " file !";
79  mState = 0;
80  }
81  mState = 1;
82 }

84 void TrackerDPL::run(ProcessingContext& pc)
85 {
86  if (mState != 1)
87  return;
88 
89  auto compClusters = pc.inputs().get<const std::vector<o2::ITSMFT::CompClusterExt>>("compClusters");
90  auto clusters = pc.inputs().get<const std::vector<o2::ITSMFT::Cluster>>("clusters");
91  auto labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
92  auto rofs = pc.inputs().get<const std::vector<o2::ITSMFT::ROFRecord>>("ROframes");
93 
94  LOG(INFO) << "ITSTracker pulled " << clusters.size() << " clusters, "
95  << labels->getIndexedSize() << " MC label objects , in "
96  << rofs.size() << " RO frames";
97 
98  std::vector<o2::ITS::TrackITS> tracks;
99  o2::dataformats::MCTruthContainer<o2::MCCompLabel> trackLabels;
100  std::vector<o2::ITS::TrackITS> allTracks;
101  o2::dataformats::MCTruthContainer<o2::MCCompLabel> allTrackLabels;
102 
103  std::uint32_t roFrame = 0;
104  o2::ITS::ROframe event(0);
105 
106  bool continuous = mGRP->isDetContinuousReadOut("ITS");
107  LOG(INFO) << "ITSTracker RO: continuous=" << continuous;
108 
109  if (continuous) {
110  int nclLeft = clusters.size();
111  while (nclLeft > 0) {
112  int nclUsed = o2::ITS::IOUtils::loadROFrameData(roFrame, event, &clusters, labels.get());
113  if (nclUsed) {
114  LOG(INFO) << "ROframe: " << roFrame << ", clusters left: " << nclLeft;
115  event.addPrimaryVertex(0.f, 0.f, 0.f); //FIXME : run an actual vertex finder !
116  mTracker->setROFrame(roFrame);
117  mTracker->clustersToTracks(event);
118  tracks.swap(mTracker->getTracks());
119  LOG(INFO) << "Found tracks: " << tracks.size();
120  trackLabels = mTracker->getTrackLabels();
121  std::copy(tracks.begin(), tracks.end(), std::back_inserter(allTracks));
122  allTrackLabels.mergeAtBack(trackLabels);
123  nclLeft -= nclUsed;
124  }
125  roFrame++;
126  }
127  } else {
128  o2::ITS::IOUtils::loadEventData(event, &clusters, labels.get());
129  event.addPrimaryVertex(0.f, 0.f, 0.f); //FIXME : run an actual vertex finder !
130  mTracker->clustersToTracks(event);
131  allTracks.swap(mTracker->getTracks());
132  allTrackLabels = mTracker->getTrackLabels();
133  }
134 
135  LOG(INFO) << "ITSTracker pushed " << allTracks.size() << " tracks";
136  pc.outputs().snapshot(Output{ "ITS", "TRACKS", 0, Lifetime::Timeframe }, allTracks);
137  pc.outputs().snapshot(Output{ "ITS", "TRACKSMCTR", 0, Lifetime::Timeframe }, allTrackLabels);
138 
139  mState = 2;
140  //pc.services().get<ControlService>().readyToQuit(true);
141 }
