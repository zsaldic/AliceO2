31 void fakePayload(std::vector<o2::byte>& buffer, std::function<void(T&, int)> filler, int numOfElements)
32 {
33  auto payloadSize = sizeof(T)*numOfElements;
34  LOG(INFO) << "Payload size " << payloadSize << "\n";
35  buffer.resize(buffer.size() + payloadSize);
36 
37  T *payload = reinterpret_cast<T*>(buffer.data() + sizeof(HeartbeatHeader));
38  for (int i = 0; i < numOfElements; ++i) {
39  new (payload + i) T();
40  // put some random toy time stamp to each cluster
41  filler(payload[i], i);
42  }
43 }

63 void DataPublisherDevice::InitTask()
64 {
65  mInputChannelName = GetConfig()->GetValue<std::string>(OptionKeyInputChannelName);
66  mOutputChannelName = GetConfig()->GetValue<std::string>(OptionKeyOutputChannelName);
67  // TODO: this turned out to be a missing feature in the description fields of
68  // the different headers. The idea is to use a field of n bytes as an integer
69  // but create the integers from a char sequence. This was thought to be done
70  // at compile time, but here we want to have it configurable at runtime.
71  //
72  // GetConfig()->GetValue<std::string>(OptionKeyDataDescription).c_str()
73  // GetConfig()->GetValue<std::string>(OptionKeyDataOrigin).c_str()
74  //
75  // Still needed:
76  // * a registry for descriptors (probably for different descriptor types of
77  // even the same size) to ensure uniqueness
78  // * create the unsigned integer value once from the configurable string and
79  // check in the registry
80  // * constructors and assignment operators taking the integer type as argument
81  if (GetConfig()->GetValue<std::string>(OptionKeyDataDescription) == "TPCCLUSTER")
82  {
83  mDataDescription = DataDescription("CLUSTERS");
84  mDataOrigin = DataOrigin("TPC");
85  }
86  else if (GetConfig()->GetValue<std::string>(OptionKeyDataDescription) == "ITSRAW")
87  {
88  mDataDescription = DataDescription("CLUSTERS");
89  mDataOrigin = DataOrigin("ITS");
90  }
91  mSubSpecification = GetConfig()->GetValue<SubSpecificationT>(OptionKeySubspecification);
92  mFileName = GetConfig()->GetValue<std::string>(OptionKeyFileName);
93 
94  OnData(mInputChannelName.c_str(), &DataPublisherDevice::HandleData);
95 
96  // reserve space for the HBH at the beginning
97  mFileBuffer.resize(sizeof(o2::header::HeartbeatHeader));
98 
99  if (!mFileName.empty()) {
100  AppendFile(mFileName.c_str(), mFileBuffer);
101  }
102  else if (mDataDescription == DataDescription("CLUSTERS")
103  && mDataOrigin == DataOrigin("TPC"))
104  {
105  auto f = [](TPCTestCluster &cluster, int idx) {cluster.timeStamp = idx;};
106  fakePayload<TPCTestCluster>(mFileBuffer, f, 1000);
107  LOG(INFO) << "Payload size (after) " << mFileBuffer.size() << "\n";
108  // For the moment, add the data as another part to this message
109  }
110  else if (mDataDescription == DataDescription("CLUSTERS")
111  && mDataOrigin == DataOrigin("ITS"))
112  {
113  auto f = [](ITSRawData &cluster, int idx) {cluster.timeStamp = idx;};
114  fakePayload<ITSRawData>(mFileBuffer, f, 500);
115  }
116 
117  mFileBuffer.resize(mFileBuffer.size() + sizeof(o2::header::HeartbeatTrailer));
118  auto* hbhOut = reinterpret_cast<o2::header::HeartbeatHeader*>(&mFileBuffer[0]);
119  auto* hbtOut = reinterpret_cast<o2::header::HeartbeatTrailer*>(&mFileBuffer[mFileBuffer.size() - sizeof(o2::header::HeartbeatTrailer)]);
120  *hbhOut = o2::header::HeartbeatHeader();
121  *hbtOut = o2::header::HeartbeatTrailer();
122 }

131 bool DataPublisherDevice::HandleO2LogicalBlock(const byte* headerBuffer,
132  size_t headerBufferSize,
133  const byte* dataBuffer,
134  size_t dataBufferSize)
135 {
136  // AliceO2::header::hexDump("data buffer", dataBuffer, dataBufferSize);
137  const auto* dataHeader = o2::header::get<o2::header::DataHeader*>(headerBuffer);
138  const auto* hbfEnvelope = o2::header::get<o2::header::HeartbeatFrameEnvelope*>(headerBuffer);
139 
140  // TODO: not sure what the return value is supposed to indicate, it's
141  // not handled in O2Device::ForEach at the moment
142  // indicate that the block has not been processed by a 'false'
143  if (!dataHeader ||
144  (dataHeader->dataDescription) != o2::header::gDataDescriptionHeartbeatFrame) return false;
145 
146  if (!hbfEnvelope) {
147  LOG(ERROR) << "no heartbeat frame envelope header found";
148  return false;
149  }
150 
151  // TODO: consistency checks
152  // hbfEnvelope->header;
153  // hbfEnvelope->trailer;
154  // - block type in both HBH and HBT
155  // - HBH size + payload size (specified in HBT) + HBT size == dataBufferSize
156  // - dynamically adjust start of the trailer (if this contains more than one
157  // 64 bit word
158 
159  // TODO: make tool for reading and manipulation of the HeartbeatFrame/Envelop
160 
161 
162  // assume everything valid
163  // write the HBH and HBT as envelop to the buffer of the file data
164  auto* hbhOut = reinterpret_cast<o2::header::HeartbeatHeader*>(&mFileBuffer[0]);
165  auto* hbtOut = reinterpret_cast<o2::header::HeartbeatTrailer*>(&mFileBuffer[mFileBuffer.size() - sizeof(o2::header::HeartbeatTrailer)]);
166 
167  // copy HBH and HBT, but set the length explicitely to 1
168  // TODO: handle all kinds of corner cases, or add an assert
169  *hbhOut = hbfEnvelope->header;
170  hbhOut->headerLength = 1;
171  *hbtOut = hbfEnvelope->trailer;
172  hbtOut->dataLength = mFileBuffer.size() - sizeof(o2::header::HeartbeatFrameEnvelope);
173 
174  // top level subframe header, the DataHeader is going to be used with
175  // configured description, origin and sub specification
176  o2::header::DataHeader dh;
177  dh.dataDescription = mDataDescription;
178  dh.dataOrigin = mDataOrigin;
179  dh.subSpecification = mSubSpecification;
180  dh.payloadSize = mFileBuffer.size();
181 
182  O2Message outgoing;
183 
184  LOG(DEBUG) << "Sending buffer of size " << mFileBuffer.size() << "\n";
185  LOG(DEBUG) << "Orbit number " << hbhOut->orbit << "\n";
186  // build multipart message from header and payload
187  // TODO: obviously there is a lot to do here, avoid copying etc, this
188  // is just a proof of principle
189  // NewSimpleMessage(mFileBuffer) does not work with the vector
190 
191 
192  // TODO: fix payload size in dh
193  auto *buffer = new char[mFileBuffer.size()];
194  memcpy(buffer, mFileBuffer.data(), mFileBuffer.size());
195  o2::Base::addDataBlock(outgoing, dh, NewMessage(buffer, mFileBuffer.size(),
196  [](void* data, void* hint) { delete[] reinterpret_cast<char*>(data); }, nullptr));
197 
198  // send message
199  Send(outgoing, mOutputChannelName.c_str());
200  outgoing.fParts.clear();
201 
202  return true;
203 }

205 bool DataPublisherDevice::AppendFile(const char* name, std::vector<o2::byte>& buffer)
206 {
207  bool result = true;
208  std::ifstream ifile(name, std::ifstream::binary);
209  if (ifile.bad()) return false;
210 
211  // get length of file:
212  ifile.seekg (0, ifile.end);
213  int length = ifile.tellg();
214  ifile.seekg (0, ifile.beg);
215 
216  // allocate memory:
217  int position = buffer.size();
218  buffer.resize(buffer.size() + length);
219 
220  // read data as a block:
221  ifile.read(reinterpret_cast<char*>(&buffer[position]),length);
222  if (!(result = ifile.good())) {
223  LOG(ERROR) << "failed to read " << length << " byte(s) from file " << name << std::endl;
224  }
225 
226  ifile.close();
227 
228  return result;
229 }
