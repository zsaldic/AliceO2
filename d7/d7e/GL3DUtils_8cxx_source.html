30 GLuint compileShaders(const char* VertexSourcePointer, const char* FragmentSourcePointer)
31 {
32  // Create the shaders
33  GLuint VertexShaderID = glCreateShader(GL_VERTEX_SHADER);
34  GLuint FragmentShaderID = glCreateShader(GL_FRAGMENT_SHADER);
35 
36  GLint Result = GL_FALSE;
37  int InfoLogLength;
38 
39  // Compile Vertex Shader
40  glShaderSource(VertexShaderID, 1, &VertexSourcePointer, NULL);
41  glCompileShader(VertexShaderID);
42 
43  // Check Vertex Shader
44  glGetShaderiv(VertexShaderID, GL_COMPILE_STATUS, &Result);
45  glGetShaderiv(VertexShaderID, GL_INFO_LOG_LENGTH, &InfoLogLength);
46 
47  if (InfoLogLength > 0) {
48  std::vector<char> VertexShaderErrorMessage(InfoLogLength + 1);
49  glGetShaderInfoLog(VertexShaderID, InfoLogLength, NULL, &VertexShaderErrorMessage[0]);
50  throw std::runtime_error(&VertexShaderErrorMessage[0]);
51  }
52 
53  // Compile Fragment Shader
54  glShaderSource(FragmentShaderID, 1, &FragmentSourcePointer, NULL);
55  glCompileShader(FragmentShaderID);
56 
57  // Check Fragment Shader
58  glGetShaderiv(FragmentShaderID, GL_COMPILE_STATUS, &Result);
59  glGetShaderiv(FragmentShaderID, GL_INFO_LOG_LENGTH, &InfoLogLength);
60  if (InfoLogLength > 0) {
61  std::vector<char> FragmentShaderErrorMessage(InfoLogLength + 1);
62  glGetShaderInfoLog(FragmentShaderID, InfoLogLength, NULL, &FragmentShaderErrorMessage[0]);
63  throw std::runtime_error(&FragmentShaderErrorMessage[0]);
64  }
65 
66  // Link the program
67  GLuint ProgramID = glCreateProgram();
68  glAttachShader(ProgramID, VertexShaderID);
69  glAttachShader(ProgramID, FragmentShaderID);
70  glLinkProgram(ProgramID);
71 
72  // Check the program
73  glGetProgramiv(ProgramID, GL_LINK_STATUS, &Result);
74  glGetProgramiv(ProgramID, GL_INFO_LOG_LENGTH, &InfoLogLength);
75  if (InfoLogLength > 0) {
76  std::vector<char> ProgramErrorMessage(InfoLogLength + 1);
77  glGetProgramInfoLog(ProgramID, InfoLogLength, NULL, &ProgramErrorMessage[0]);
78  throw std::runtime_error(&ProgramErrorMessage[0]);
79  }
80 
81  glDetachShader(ProgramID, VertexShaderID);
82  glDetachShader(ProgramID, FragmentShaderID);
83 
84  glDeleteShader(VertexShaderID);
85  glDeleteShader(FragmentShaderID);
86 
87  return ProgramID;
88 }
