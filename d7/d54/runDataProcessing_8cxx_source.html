209 int createPipes(int maxFd, int* pipes)
210 {
211  auto p = pipe(pipes);
212  maxFd = maxFd > pipes[0] ? maxFd : pipes[0];
213  maxFd = maxFd > pipes[1] ? maxFd : pipes[1];
214 
215  if (p == -1) {
216  std::cerr << "Unable to create PIPE: ";
217  switch (errno) {
218  case EFAULT:
219  assert(false && "EFAULT while reading from pipe");
220  break;
221  case EMFILE:
222  std::cerr << "Too many active descriptors";
223  break;
224  case ENFILE:
225  std::cerr << "System file table is full";
226  break;
227  default:
228  std::cerr << "Unknown PIPE" << std::endl;
229  };
230  // Kill immediately both the parent and all the children
231  kill(-1 * getpid(), SIGKILL);
232  }
233  return maxFd;
234 }

249 void spawnDevice(DeviceSpec const& spec, std::map<int, size_t>& socket2DeviceInfo, DeviceControl& control,
250  DeviceExecution& execution, std::vector<DeviceInfo>& deviceInfos, int& maxFd, fd_set& childFdset)
251 {
252  int childstdout[2];
253  int childstderr[2];
254 
255  maxFd = createPipes(maxFd, childstdout);
256  maxFd = createPipes(maxFd, childstderr);
257 
258  // If we have a framework id, it means we have already been respawned
259  // and that we are in a child. If not, we need to fork and re-exec, adding
260  // the framework-id as one of the options.
261  pid_t id = 0;
262  id = fork();
263  // We are the child: prepare options and reexec.
264  if (id == 0) {
265  // We allow being debugged and do not terminate on SIGTRAP
266  signal(SIGTRAP, SIG_IGN);
267 
268  // We do not start the process if control.noStart is set.
269  if (control.stopped) {
270  kill(getpid(), SIGSTOP);
271  }
272 
273  // This is the child. We close the read part of the pipe, stdout
274  // and dup2 the write part of the pipe on it. Then we can restart.
275  close(childstdout[0]);
276  close(childstderr[0]);
277  close(STDOUT_FILENO);
278  close(STDERR_FILENO);
279  dup2(childstdout[1], STDOUT_FILENO);
280  dup2(childstderr[1], STDERR_FILENO);
281  execvp(execution.args[0], execution.args.data());
282  }
283 
284  // This is the parent. We close the write end of
285  // the child pipe and and keep track of the fd so
286  // that we can later select on it.
287  struct sigaction sa_handle_int;
288  sa_handle_int.sa_handler = handle_sigint;
289  sigemptyset(&sa_handle_int.sa_mask);
290  sa_handle_int.sa_flags = SA_RESTART;
291  if (sigaction(SIGINT, &sa_handle_int, nullptr) == -1) {
292  perror("Unable to install signal handler");
293  exit(1);
294  }
295 
296  std::cout << "Starting " << spec.id << " on pid " << id << "\n";
297  DeviceInfo info;
298  info.pid = id;
299  info.active = true;
300  info.readyToQuit = false;
301  info.historySize = 1000;
302  info.historyPos = 0;
303  info.maxLogLevel = LogParsingHelpers::LogLevel::Debug;
304  info.dataRelayerViewIndex = Metric2DViewIndex{ "data_relayer", 0, 0, {} };
305  info.variablesViewIndex = Metric2DViewIndex{ "matcher_variables", 0, 0, {} };
306  info.queriesViewIndex = Metric2DViewIndex{ "data_queries", 0, 0, {} };
307 
308  socket2DeviceInfo.insert(std::make_pair(childstdout[0], deviceInfos.size()));
309  socket2DeviceInfo.insert(std::make_pair(childstderr[0], deviceInfos.size()));
310  deviceInfos.emplace_back(info);
311  // Let's add also metrics information for the given device
312  gDeviceMetricsInfos.emplace_back(DeviceMetricsInfo{});
313 
314  close(childstdout[1]);
315  close(childstderr[1]);
316  FD_SET(childstdout[0], &childFdset);
317  FD_SET(childstderr[0], &childFdset);
318 }

334 void processChildrenOutput(DriverInfo& driverInfo, DeviceInfos& infos, DeviceSpecs const& specs,
335  DeviceControls& controls, std::vector<DeviceMetricsInfo>& metricsInfos)
336 {
337  // Wait for children to say something. When they do
338  // print it.
339  fd_set fdset;
340  timeval timeout;
341  timeout.tv_sec = 0;
342  timeout.tv_usec = 16666; // This should be enough to allow 60 HZ redrawing.
343  memcpy(&fdset, &driverInfo.childFdset, sizeof(fd_set));
344  int numFd = select(driverInfo.maxFd, &fdset, nullptr, nullptr, &timeout);
345  if (numFd == 0) {
346  return;
347  }
348  for (int si = 0; si < driverInfo.maxFd; ++si) {
349  if (FD_ISSET(si, &fdset)) {
350  assert(driverInfo.socket2DeviceInfo.find(si) != driverInfo.socket2DeviceInfo.end());
351  auto& info = infos[driverInfo.socket2DeviceInfo[si]];
352 
353  bool fdActive = getChildData(si, info);
354  // If the pipe was closed due to the process exiting, we
355  // can avoid the select.
356  if (!fdActive) {
357  info.active = false;
358  close(si);
359  FD_CLR(si, &driverInfo.childFdset);
360  }
361  --numFd;
362  }
363  // FIXME: no need to check after numFd gets to 0.
364  }
365  // Display part. All you need to display should actually be in
366  // `infos`.
367  // TODO: split at \n
368  // TODO: update this only once per 1/60 of a second or
369  // things like this.
370  // TODO: have multiple display modes
371  // TODO: graphical view of the processing?
372  assert(infos.size() == controls.size());
373  std::smatch match;
374  ParsedMetricMatch metricMatch;
375  std::string token;
376  const std::string delimiter("\n");
377  bool hasNewMetric = false;
378  for (size_t di = 0, de = infos.size(); di < de; ++di) {
379  DeviceInfo& info = infos[di];
380  DeviceControl& control = controls[di];
381  DeviceMetricsInfo& metrics = metricsInfos[di];
382 
383  if (info.unprinted.empty()) {
384  continue;
385  }
386 
387  auto s = info.unprinted;
388  size_t pos = 0;
389  info.history.resize(info.historySize);
390  info.historyLevel.resize(info.historySize);
391 
392  auto updateMetricsViews =
393  Metric2DViewIndex::getUpdater({ &info.dataRelayerViewIndex,
394  &info.variablesViewIndex,
395  &info.queriesViewIndex });
396 
397  auto newMetricCallback = [&updateMetricsViews, &driverInfo, &metricsInfos, &hasNewMetric](std::string const& name, MetricInfo const& metric, int value, size_t metricIndex) {
398  updateMetricsViews(name, metric, value, metricIndex);
399  hasNewMetric = true;
400  };
401 
402  while ((pos = s.find(delimiter)) != std::string::npos) {
403  token = s.substr(0, pos);
404  auto logLevel = LogParsingHelpers::parseTokenLevel(token);
405 
406  // Check if the token is a metric from SimpleMetricsService
407  // if yes, we do not print it out and simply store it to be displayed
408  // in the GUI.
409  // Then we check if it is part of our Poor man control system
410  // if yes, we execute the associated command.
411  if (DeviceMetricsHelper::parseMetric(token, metricMatch)) {
412  // We use this callback to cache which metrics are needed to provide a
413  // the DataRelayer view.
414  DeviceMetricsHelper::processMetric(metricMatch, metrics, newMetricCallback);
415  } else if (logLevel == LogParsingHelpers::LogLevel::Info && parseControl(token, match)) {
416  auto command = match[1];
417  auto validFor = match[2];
418  LOG(DEBUG) << "Found control command " << command << " from pid " << info.pid << " valid for " << validFor;
419  if (command == "QUIT") {
420  if (validFor == "ALL") {
421  for (auto& deviceInfo : infos) {
422  deviceInfo.readyToQuit = true;
423  }
424  }
425  // in case of "ME", fetch the pid and modify only matching deviceInfos
426  if (validFor == "ME") {
427  for (auto& deviceInfo : infos) {
428  if (deviceInfo.pid == info.pid) {
429  deviceInfo.readyToQuit = true;
430  }
431  }
432  }
433  }
434  } else if (!control.quiet && (strstr(token.c_str(), control.logFilter) != nullptr) &&
435  logLevel >= control.logLevel) {
436  assert(info.historyPos >= 0);
437  assert(info.historyPos < info.history.size());
438  info.history[info.historyPos] = token;
439  info.historyLevel[info.historyPos] = logLevel;
440  info.historyPos = (info.historyPos + 1) % info.history.size();
441  std::cout << "[" << info.pid << "]: " << token << std::endl;
442  }
443  // We keep track of the maximum log error a
444  // device has seen.
445  if (logLevel > info.maxLogLevel && logLevel > LogParsingHelpers::LogLevel::Info &&
446  logLevel != LogParsingHelpers::LogLevel::Unknown) {
447  info.maxLogLevel = logLevel;
448  }
449  if (logLevel == LogParsingHelpers::LogLevel::Error) {
450  info.lastError = token;
451  }
452  s.erase(0, pos + delimiter.length());
453  }
454  info.unprinted = s;
455  }
456  if (hasNewMetric) {
457  hasNewMetric = false;
458  updateMetricsNames(driverInfo, metricsInfos);
459  }
460  // FIXME: for the gui to work correctly I would actually need to
461  // run the loop more often and update whenever enough time has
462  // passed.
463 }

692 // This is the handler for the parent inner loop.
693 int runStateMachine(DataProcessorSpecs const& workflow, DriverControl& driverControl, DriverInfo& driverInfo,
694  std::vector<DeviceMetricsInfo>& metricsInfos, std::string frameworkId)
695 {
696  DeviceSpecs deviceSpecs;
697  DeviceInfos infos;
698  DeviceControls controls;
699  DeviceExecutions deviceExecutions;
700  auto resourceManager = std::make_unique<SimpleResourceManager>(driverInfo.startPort, driverInfo.portRange);
701 
702  void* window = nullptr;
703  decltype(gui::getGUIDebugger(infos, deviceSpecs, metricsInfos, driverInfo, controls, driverControl)) debugGUICallback;
704 
705  // An empty frameworkId means this is the driver, so we initialise the GUI
706  if (driverInfo.batch == false && frameworkId.empty()) {
707  window = initGUI("O2 Framework debug GUI");
708  }
709  if (driverInfo.batch == false && window == nullptr) {
710  LOG(WARN) << "Could not create GUI. Switching to batch mode. Do you have GLFW on your system?";
711  driverInfo.batch = true;
712  }
713  bool guiQuitRequested = false;
714 
715  auto frameLast = std::chrono::high_resolution_clock::now();
716  auto inputProcessingLast = frameLast;
717  // FIXME: I should really have some way of exiting the
718  // parent..
719  DriverState current;
720  while (true) {
721  // If control forced some transition on us, we push it to the queue.
722  if (driverControl.forcedTransitions.empty() == false) {
723  for (auto transition : driverControl.forcedTransitions) {
724  driverInfo.states.push_back(transition);
725  }
726  driverControl.forcedTransitions.resize(0);
727  }
728  // In case a timeout was requested, we check if we are running
729  // for more than the timeout duration and exit in case that's the case.
730  {
731  auto currentTime = std::chrono::steady_clock::now();
732  std::chrono::duration<double> diff = currentTime - driverInfo.startTime;
733  if ((graceful_exit == false) && (driverInfo.timeout > 0) && (diff.count() > driverInfo.timeout)) {
734  LOG(INFO) << "Timout ellapsed. Requesting to quit.";
735  graceful_exit = true;
736  }
737  }
738  // Move to exit loop if sigint was sent we execute this only once.
739  if (graceful_exit == true && driverInfo.sigintRequested == false) {
740  driverInfo.sigintRequested = true;
741  driverInfo.states.resize(0);
742  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
743  driverInfo.states.push_back(DriverState::GUI);
744  }
745  // If one of the children dies and sigint was not requested
746  // we should decide what to do.
747  if (sigchld_requested == true && driverInfo.sigchldRequested == false) {
748  driverInfo.sigchldRequested = true;
749  pruneGUI(driverInfo.states);
750  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
751  driverInfo.states.push_back(DriverState::GUI);
752  }
753  if (driverInfo.states.empty() == false) {
754  current = driverInfo.states.back();
755  } else {
756  current = DriverState::UNKNOWN;
757  }
758  driverInfo.states.pop_back();
759  switch (current) {
760  case DriverState::INIT:
761  LOG(INFO) << "Initialising O2 Data Processing Layer";
762 
763  // Install signal handler for quitting children.
764  driverInfo.sa_handle_child.sa_handler = &handle_sigchld;
765  sigemptyset(&driverInfo.sa_handle_child.sa_mask);
766  driverInfo.sa_handle_child.sa_flags = SA_RESTART | SA_NOCLDSTOP;
767  if (sigaction(SIGCHLD, &driverInfo.sa_handle_child, nullptr) == -1) {
768  perror(nullptr);
769  exit(1);
770  }
771  FD_ZERO(&(driverInfo.childFdset));
772 
777  driverInfo.states.push_back(DriverState::RUNNING);
778  driverInfo.states.push_back(DriverState::GUI);
779  // driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
780  LOG(INFO) << "O2 Data Processing Layer initialised. We brake for nobody.";
781  break;
782  case DriverState::MATERIALISE_WORKFLOW:
783  try {
784  std::vector<ComputingResource> resources = resourceManager->getAvailableResources();
785  DeviceSpecHelpers::dataProcessorSpecs2DeviceSpecs(workflow, driverInfo.channelPolicies, driverInfo.completionPolicies, deviceSpecs, resources);
786  // This should expand nodes so that we can build a consistent DAG.
787  } catch (std::runtime_error& e) {
788  std::cerr << "Invalid workflow: " << e.what() << std::endl;
789  return 1;
790  }
791  break;
792  case DriverState::DO_CHILD:
793  for (auto& spec : deviceSpecs) {
794  if (spec.id == frameworkId) {
795  return doChild(driverInfo.argc, driverInfo.argv, spec);
796  }
797  }
798  LOG(ERROR) << "Unable to find component with id " << frameworkId;
799  break;
800  case DriverState::REDEPLOY_GUI:
801  // The callback for the GUI needs to be recalculated every time
802  // the deployed configuration changes, e.g. a new device
803  // has been added to the topology.
804  // We need to recreate the GUI callback every time we reschedule
805  // because getGUIDebugger actually recreates the GUI state.
806  if (window) {
807  debugGUICallback = gui::getGUIDebugger(infos, deviceSpecs, metricsInfos, driverInfo, controls, driverControl);
808  }
809  break;
810  case DriverState::SCHEDULE:
811  // FIXME: for the moment modifying the topology means we rebuild completely
812  // all the devices and we restart them. This is also what DDS does at
813  // a larger scale. In principle one could try to do a delta and only
814  // restart the data processors which need to be restarted.
815  LOG(INFO) << "Redeployment of configuration asked.";
816  controls.resize(deviceSpecs.size());
817  deviceExecutions.resize(deviceSpecs.size());
818 
819  DeviceSpecHelpers::prepareArguments(driverInfo.argc, driverInfo.argv, driverControl.defaultQuiet,
820  driverControl.defaultStopped, deviceSpecs,
821  driverInfo.workflowOptions, deviceExecutions, controls);
822  for (size_t di = 0; di < deviceSpecs.size(); ++di) {
823  spawnDevice(deviceSpecs[di], driverInfo.socket2DeviceInfo, controls[di], deviceExecutions[di], infos,
824  driverInfo.maxFd, driverInfo.childFdset);
825  }
826  driverInfo.maxFd += 1;
827  assert(infos.empty() == false);
828  LOG(INFO) << "Redeployment of configuration done.";
829  break;
830  case DriverState::RUNNING:
831  // Calculate what we should do next and eventually
832  // show the GUI
833  if (guiQuitRequested ||
834  (driverInfo.terminationPolicy == TerminationPolicy::QUIT && (checkIfCanExit(infos) == true))) {
835  // Something requested to quit. This can be a user
836  // interaction with the GUI or (if --completion-policy=quit)
837  // it could mean that the workflow does not have anything else to do.
838  // Let's update the GUI one more time and then EXIT.
839  LOG(INFO) << "Quitting";
840  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
841  driverInfo.states.push_back(DriverState::GUI);
842  } else if (infos.size() != deviceSpecs.size()) {
843  // If the number of deviceSpecs is different from
844  // the DeviceInfos it means the speicification
845  // does not match what is running, so we need to do
846  // further scheduling.
847  driverInfo.states.push_back(DriverState::RUNNING);
848  driverInfo.states.push_back(DriverState::GUI);
849  driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
850  driverInfo.states.push_back(DriverState::GUI);
851  driverInfo.states.push_back(DriverState::SCHEDULE);
852  driverInfo.states.push_back(DriverState::GUI);
853  } else if (deviceSpecs.size() == 0) {
854  LOG(INFO) << "No device resulting from the workflow. Quitting.";
855  // If there are no deviceSpecs, we exit.
856  driverInfo.states.push_back(DriverState::EXIT);
857  } else {
858  driverInfo.states.push_back(DriverState::RUNNING);
859  driverInfo.states.push_back(DriverState::GUI);
860  }
861  {
862  usleep(1000); // We wait for 1 millisecond between one processing
863  // and the other.
864  auto inputProcessingStart = std::chrono::high_resolution_clock::now();
865  auto inputProcessingLatency = inputProcessingStart - inputProcessingLast;
866  processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
867  auto inputProcessingEnd = std::chrono::high_resolution_clock::now();
868  driverInfo.inputProcessingCost = std::chrono::duration_cast<std::chrono::milliseconds>(inputProcessingEnd - inputProcessingStart).count();
869  driverInfo.inputProcessingLatency = std::chrono::duration_cast<std::chrono::milliseconds>(inputProcessingLatency).count();
870  inputProcessingLast = inputProcessingStart;
871  }
872  break;
873  case DriverState::GUI:
874  if (window) {
875  auto frameStart = std::chrono::high_resolution_clock::now();
876  auto frameLatency = frameStart - frameLast;
877  // We want to render at ~60 frames per second, so latency needs to be ~16ms
878  if (std::chrono::duration_cast<std::chrono::milliseconds>(frameLatency).count() > 20) {
879  guiQuitRequested = (pollGUI(window, debugGUICallback) == false);
880  auto frameEnd = std::chrono::high_resolution_clock::now();
881  driverInfo.frameCost = std::chrono::duration_cast<std::chrono::milliseconds>(frameEnd - frameStart).count();
882  driverInfo.frameLatency = std::chrono::duration_cast<std::chrono::milliseconds>(frameLatency).count();
883  frameLast = frameStart;
884  }
885  }
886  break;
887  case DriverState::QUIT_REQUESTED:
888  LOG(INFO) << "QUIT_REQUESTED" << std::endl;
889  guiQuitRequested = true;
890  killChildren(infos, SIGTERM);
891  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
892  driverInfo.states.push_back(DriverState::GUI);
893  break;
894  case DriverState::HANDLE_CHILDREN:
895  // I allow queueing of more sigchld only when
896  // I process the previous call
897  sigchld_requested = false;
898  driverInfo.sigchldRequested = false;
899  processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
900  processSigChild(infos);
901  if (areAllChildrenGone(infos) == true &&
902  (guiQuitRequested || (checkIfCanExit(infos) == true) || graceful_exit)) {
903  // We move to the exit, regardless of where we were
904  driverInfo.states.resize(0);
905  driverInfo.states.push_back(DriverState::EXIT);
906  driverInfo.states.push_back(DriverState::GUI);
907  } else if (areAllChildrenGone(infos) == false &&
908  (guiQuitRequested || checkIfCanExit(infos) == true || graceful_exit)) {
909  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
910  driverInfo.states.push_back(DriverState::GUI);
911  } else {
912  driverInfo.states.push_back(DriverState::GUI);
913  }
914  break;
915  case DriverState::EXIT:
916  return calculateExitCode(infos);
917  case DriverState::PERFORM_CALLBACKS:
918  for (auto& callback : driverControl.callbacks) {
919  callback(deviceSpecs, deviceExecutions);
920  }
921  driverControl.callbacks.clear();
922  driverInfo.states.push_back(DriverState::GUI);
923  break;
924  default:
925  LOG(ERROR) << "Driver transitioned in an unknown state. Shutting down.";
926  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
927  }
928  }
929 }
