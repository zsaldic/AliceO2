308 //____________________________________________________________
309 inline float TrackPar::getP2Inv() const
310 {
311  // return the inverted track momentum^2
312  return getQ2Pt() * getQ2Pt() / (1.f + getTgl() * getTgl());
313 }

315 //____________________________________________________________
316 inline float TrackPar::getPInv() const
317 {
318  // return the inverted track momentum^2
319  return fabs(getQ2Pt()) / sqrtf(1.f + getTgl() * getTgl());
320 }

322 //____________________________________________________________
323 inline float TrackPar::getP2() const
324 {
325  // return the track momentum^2

328 }
