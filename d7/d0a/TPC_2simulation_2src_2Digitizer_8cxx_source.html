46 void Digitizer::process(const std::vector<o2::TPC::HitGroup>& hits,
47  const int eventID, const int sourceID)
48 {
49  const static Mapper& mapper = Mapper::instance();
50  const static ParameterDetector& detParam = ParameterDetector::defaultInstance();
51  const static ParameterElectronics& eleParam = ParameterElectronics::defaultInstance();
52  const static ParameterGEM& gemParam = ParameterGEM::defaultInstance();
53 
54  static GEMAmplification& gemAmplification = GEMAmplification::instance();
55  gemAmplification.updateParameters();
56  static ElectronTransport& electronTransport = ElectronTransport::instance();
57  electronTransport.updateParameters();
58  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
59  sampaProcessing.updateParameters();
60 
61  const int nShapedPoints = eleParam.getNShapedPoints();
62  const auto amplificationMode = gemParam.getAmplificationMode();
63  static std::vector<float> signalArray;
64  signalArray.resize(nShapedPoints);
65 
67  mDigitContainer.reserve(sampaProcessing.getTimeBinFromTime(mEventTime));
68 
69  for (auto& hitGroup : hits) {
70  const int MCTrackID = hitGroup.GetTrackID();
71  for (size_t hitindex = 0; hitindex < hitGroup.getSize(); ++hitindex) {
72  const auto& eh = hitGroup.getHit(hitindex);
73 
74  GlobalPosition3D posEle(eh.GetX(), eh.GetY(), eh.GetZ());
75 
76  // Distort the electron position in case space-charge distortions are used
77  if (mUseSCDistortions) {
78  mSpaceChargeHandler->distortElectron(posEle);
79  }
80 
83  if (electronTransport.isCompletelyOutOfSectorCoarseElectronDrift(posEle, mSector)) {
84  continue;
85  }
86 
88  const int nPrimaryElectrons = static_cast<int>(eh.GetEnergyLoss());
89  const float hitTime = eh.GetTime() * 0.001;
90  float driftTime = 0.f;
91 
93 
95  for (int iEle = 0; iEle < nPrimaryElectrons; ++iEle) {
96 
98  const GlobalPosition3D posEleDiff = electronTransport.getElectronDrift(posEle, driftTime);
99  const float absoluteTime = driftTime + mEventTime + hitTime;
100 
102  if (electronTransport.isElectronAttachment(driftTime)) {
103  continue;
104  }
105 
107  if (std::abs(posEleDiff.Z()) > detParam.getTPClength()) {
108  continue;
109  }
110 
112  if (mapper.isOutOfSector(posEleDiff, mSector)) {
113  continue;
114  }
115 
117  const DigitPos digiPadPos = mapper.findDigitPosFromGlobalPosition(posEleDiff, mSector);
118  if (!digiPadPos.isValid()) {
119  continue;
120  }
121 
123  if (digiPadPos.getCRU().sector() != mSector) {
124  continue;
125  }
126 
128  const int nElectronsGEM = gemAmplification.getStackAmplification(digiPadPos.getCRU(), digiPadPos.getPadPos(), amplificationMode);
129  if (nElectronsGEM == 0) {
130  continue;
131  }
132 
133  const GlobalPadNumber globalPad = mapper.globalPadNumber(digiPadPos.getGlobalPadPos());
134  const float ADCsignal = sampaProcessing.getADCvalue(static_cast<float>(nElectronsGEM));
135  const MCCompLabel label(MCTrackID, eventID, sourceID);
136  sampaProcessing.getShapedSignal(ADCsignal, absoluteTime, signalArray);
137  for (float i = 0; i < nShapedPoints; ++i) {
138  const float time = absoluteTime + i * eleParam.getZBinWidth();
139  mDigitContainer.addDigit(label, digiPadPos.getCRU(), sampaProcessing.getTimeBinFromTime(time), globalPad,
140  signalArray[i]);
141  }
143  }
145  }
146  }
147 }
