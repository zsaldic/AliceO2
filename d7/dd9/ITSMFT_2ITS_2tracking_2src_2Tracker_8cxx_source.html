376 bool Tracker::fitTrack(const ROframe& event, TrackITS& track, int start, int end, int step)
377 {
378  track.setChi2(0);
379  for (int iLayer{ start }; iLayer != end; iLayer += step) {
380  if (track.getClusterIndex(iLayer) == Constants::ITS::UnusedIndex) {
381  continue;
382  }
383  const TrackingFrameInfo& trackingHit = event.getTrackingFrameInfoOnLayer(iLayer).at(track.getClusterIndex(iLayer));
384 
385  if (!track.rotate(trackingHit.alphaTrackingFrame))
386  return false;
387 
388  if (!track.propagateTo(trackingHit.xTrackingFrame, getBz()))
389  return false;
390 
391  track.setChi2(track.getChi2() +
392  track.getPredictedChi2(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame));
393  if (!track.TrackParCov::update(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame))
394  return false;
395 
396  const float xx0 = (iLayer > 2) ? 0.008f : 0.003f; // Rough layer thickness
397  constexpr float radiationLength = 9.36f; // Radiation length of Si [cm]
398  constexpr float density = 2.33f; // Density of Si [g/cm^3]
399  if (!track.correctForMaterial(xx0, xx0 * radiationLength * density, true))
400  return false;
401  }
402  return true;

564 track::TrackParCov Tracker::buildTrackSeed(const Cluster& cluster1, const Cluster& cluster2,
565  const Cluster& cluster3, const TrackingFrameInfo& tf3)
566 {
567  const float ca = std::cos(tf3.alphaTrackingFrame), sa = std::sin(tf3.alphaTrackingFrame);
568  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
569  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
570  const float z1 = cluster1.zCoordinate;
571  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
572  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
573  const float z2 = cluster2.zCoordinate;
574  const float x3 = tf3.xTrackingFrame;
575  const float y3 = tf3.positionTrackingFrame[0];
576  const float z3 = tf3.positionTrackingFrame[1];
577 
578  const float crv = MathUtils::computeCurvature(x1, y1, x2, y2, x3, y3);
579  const float x0 = MathUtils::computeCurvatureCentreX(x1, y1, x2, y2, x3, y3);
580  const float tgl12 = MathUtils::computeTanDipAngle(x1, y1, x2, y2, z1, z2);
581  const float tgl23 = MathUtils::computeTanDipAngle(x2, y2, x3, y3, z2, z3);
582 
583  const float fy = 1. / (cluster2.rCoordinate - cluster3.rCoordinate);
584  const float& tz = fy;
