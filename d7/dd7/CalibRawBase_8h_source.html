160 inline CalibRawBase::ProcessStatus CalibRawBase::processEvent(int eventNumber)
161 {
162  if (mGBTFrameContainers.size()) {
163  return processEventGBT();
164  } else if (mRawReaders.size()) {
165  return processEventRawReader(eventNumber);
166  } else if (mRawReadersCRU.size()) {
167  return processEventRawReaderCRU();
168  } else {
169  return ProcessStatus::NoReaders;
170  }
171 }

174 inline CalibRawBase::ProcessStatus CalibRawBase::processEventGBT()
175 {
176  if (!mGBTFrameContainers.size())
177  return ProcessStatus::NoReaders;
178  resetEvent();
179 
180  // loop over raw readers, fill digits for 500 time bins and process
181  // digits
182 
183  const int nRowIROC = mMapper.getNumberOfRowsROC(0);
184 
185  ProcessStatus status = ProcessStatus::Ok;
186 
187  std::vector<Digit> digits(80);
188  for (auto& reader_ptr : mGBTFrameContainers) {
189  auto reader = reader_ptr.get();
190  int readTimeBins = 0;
191  for (int i = 0; i < mTimeBinsPerCall; ++i) {
192  if (reader->getData(digits)) {
193  for (auto& digi : digits) {
194  CRU cru(digi.getCRU());
195  const int roc = cru.roc();
196  // TODO: OROC case needs subtraction of number of pad rows in IROC
197  const PadRegionInfo& regionInfo = mMapper.getPadRegionInfo(cru.region());
198  const PartitionInfo& partInfo = mMapper.getPartitionInfo(cru.partition());
199 
200  // row is local in region (CRU)
201  const int row = digi.getRow();
202  const int pad = digi.getPad();
203  if (row == 255 || pad == 255)
204  continue;
205 
206  int rowOffset = 0;
207  switch (mPadSubset) {
208  case PadSubset::ROC: {
209  rowOffset = regionInfo.getGlobalRowOffset();
210  rowOffset -= (cru.rocType() == RocType::OROC) * nRowIROC;
211  break;
212  }
213  case PadSubset::Region: {
214  break;
215  }
216  case PadSubset::Partition: {
217  rowOffset = regionInfo.getGlobalRowOffset();
218  rowOffset -= partInfo.getGlobalRowOffset();
219  break;
220  }
221  }
222 
223  // modify row depending on the calibration type used
224  const int timeBin = i; //digi.getTimeStamp();
225  const float signal = digi.getChargeFloat();
226 
227  updateCRU(cru, row, pad, timeBin, signal);
228  updateROC(roc, row + rowOffset, pad, timeBin, signal);
229  }
230  ++readTimeBins;
231  }
232 
233  digits.clear();
234  }
235 
236  // set status, don't overwrite decision
237  if (status == ProcessStatus::Ok) {
238  if (readTimeBins == 0) {
239  return ProcessStatus::NoMoreData;
240  } else if (readTimeBins < mTimeBinsPerCall) {
241  status = ProcessStatus::Truncated;
242  }
243  }
244 
245  // notify that one raw reader processing finalized for this event
246  endReader();
247  }
248 
249  endEvent();
250  ++mNevents;
251  return status;
252 }
