43 NodeColor
44  decideColorForNode(const DeviceInfo& info)
45 {
46  NodeColor result;
47  if (info.active == false) {
48  result.normal = PaletteHelpers::RED;
49  result.hovered = PaletteHelpers::SHADED_RED;
50  return result;
51  }
52  switch (info.maxLogLevel) {
53  case LogParsingHelpers::LogLevel::Error:
54  result.normal = PaletteHelpers::SHADED_RED;
55  result.hovered = PaletteHelpers::RED;
56  result.title = PaletteHelpers::RED;
57  result.title_hovered = PaletteHelpers::DARK_RED;
58  break;
59  case LogLevel::Warning:
60  result.normal = PaletteHelpers::SHADED_YELLOW;
61  result.hovered = PaletteHelpers::YELLOW;
62  result.title = PaletteHelpers::YELLOW;
63  result.title_hovered = PaletteHelpers::DARK_YELLOW;
64  break;
65  case LogLevel::Info:
66  result.normal = PaletteHelpers::SHADED_GREEN;
67  result.hovered = PaletteHelpers::GREEN;
68  result.title = PaletteHelpers::GREEN;
69  result.title_hovered = PaletteHelpers::DARK_GREEN;
70  break;
71  default:
72  result.normal = PaletteHelpers::GRAY;
73  result.hovered = PaletteHelpers::LIGHT_GRAY;
74  result.title = PaletteHelpers::GRAY;
75  result.title_hovered = PaletteHelpers::BLACK;
76  break;
77  }
78  return result;
79 }

120 // Private helper struct for the layout of the graph
121 struct NodePos {
122  ImVec2 pos;
123  static ImVec2 GetInputSlotPos(ImVector<Node> const& infos, ImVector<NodePos> const& positions, int nodeId, int slot_no)
124  {
125  ImVec2 const& pos = positions[nodeId].pos;
126  ImVec2 const& size = infos[nodeId].Size;
127  float inputsCount = infos[nodeId].InputsCount;
128  return ImVec2(pos.x, pos.y + size.y * ((float)slot_no + 1) / (inputsCount + 1));
129  }
130  static ImVec2 GetOutputSlotPos(ImVector<Node> const& infos, ImVector<NodePos> const& positions, int nodeId, int slot_no)
131  {
132  ImVec2 const& pos = positions[nodeId].pos;
133  ImVec2 const& size = infos[nodeId].Size;
134  float outputsCount = infos[nodeId].OutputsCount;
135  return ImVec2(pos.x + size.x, pos.y + size.y * ((float)slot_no + 1) / (outputsCount + 1));
136  }
137 };
