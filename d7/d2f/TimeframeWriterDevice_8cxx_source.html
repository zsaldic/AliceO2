53 void TimeframeWriterDevice::Run()
54 {
55  boost::filesystem::path p(mOutFileName);
56  size_t streamedTimeframes = 0;
57  bool needsNewFile = true;
58  while (CheckCurrentState(RUNNING) && mFileCount < mMaxFiles) {
59  // In case we need to process more than one file,
60  // the filename is split in basename and extension
61  // and we call the files `<basename><count>.<extension>`.
62  if (needsNewFile) {
63  std::string filename = mOutFileName;
64  if (mMaxFiles > 1) {
65  std::string base_path(mOutFileName, 0, mOutFileName.find_last_of("."));
66  std::string extension(mOutFileName, mOutFileName.find_last_of("."));
67  filename = base_path + std::to_string(mFileCount) + extension;
68  }
69  LOG(INFO) << "Opening " << filename << " for output\n";
70  mFile.open(filename.c_str(), std::ofstream::out | std::ofstream::binary);
71  needsNewFile = false;
72  }
73 
74  FairMQParts timeframeParts;
75  if (Receive(timeframeParts, mInChannelName, 0, 100) <= 0)
76  continue;
77 
78  streamTimeframe(mFile, timeframeParts);
79  if ((mFile.tellp() > mMaxFileSize) || (streamedTimeframes++ > mMaxTimeframes))
80  {
81  mFile.flush();
82  mFile.close();
83  mFileCount++;
84  needsNewFile = true;
85  }
86  }
87 }
