322 void FillMaxHists(Int_t type = 0)
323 {
324  //
325  // type: 0 fill side and sector, 1: fill sector only
326  //
327  Float_t kEpsilon = 0.000000000001;
328  CalPad* pad = mEvDisp.getCalPadMax();
329  TH2F* hSide = nullptr;
330  TH2F* hROC = nullptr;
331  ResetHists(type);
332  const int runNumber = TString(gSystem->Getenv("RUN_NUMBER")).Atoi();
333  //const int eventNumber = mEvDisp.getNumberOfProcessedEvents() - 1;
334  const int eventNumber = mEvDisp.getPresentEventNumber();
335  for (Int_t iROC = 0; iROC < 72; iROC++) {
336  // TODO: remove again at some point
337  if (iROC % 36 != 0) {
338  continue;
339  }
340  hROC = mHMaxOROC;
341  hSide = mHMaxC;
342  if (iROC < 36)
343  hROC = mHMaxIROC;
344  if (iROC % 36 < 18)
345  hSide = mHMaxA;
346  if (iROC % 36 == mSelectedSector % 36) {
347  TString title = Form("Max Values %cROC %c%02d (%02d)", (iROC < 36) ? 'I' : 'O', (iROC % 36 < 18) ? 'A' : 'C', iROC % 18, iROC);
348  //TString title = Form("Max Values Run %d Event %d", runNumber, eventNumber);
349  if (hROC)
350  hROC->SetTitle(title.Data());
351  }
352  auto& calRoc = pad->getCalArray(iROC);
353  const int nRows = mMapper.getNumberOfRowsROC(iROC);
354  for (Int_t irow = 0; irow < nRows; irow++) {
355  const int nPads = mMapper.getNumberOfPadsInRowROC(iROC, irow);
356  for (Int_t ipad = 0; ipad < nPads; ipad++) {
357  Float_t value = calRoc.getValue(irow, ipad);
358  //printf("iROC: %02d, sel: %02d, row %02d, pad: %02d, value: %.5f\n", iROC, mSelectedSector, irow, ipad, value);
359  if (TMath::Abs(value) > kEpsilon) {
360  if (!type && hSide) {
361  const GlobalPosition2D global2D = mMapper.getPadCentre(PadSecPos(Sector(iROC % 36), PadPos(irow, ipad + (iROC >= 36) * mMapper.getNumberOfRowsROC(iROC))));
362  Int_t binx = 1 + TMath::Nint((global2D.X() + 250.) * hSide->GetNbinsX() / 500.);
363  Int_t biny = 1 + TMath::Nint((global2D.Y() + 250.) * hSide->GetNbinsY() / 500.);
364  hSide->SetBinContent(binx, biny, value);
365  }
366  const int nPads = mMapper.getNumberOfPadsInRowROC(iROC, irow);
367  const Int_t cpad = ipad - nPads / 2;
368  if ((iROC % 36 == mSelectedSector % 36) && hROC) {
369  //printf(" ->>> Fill: iROC: %02d, sel: %02d, row %02d, pad: %02d, value: %.5f\n", iROC, mSelectedSector, irow, ipad, value);
370  hROC->Fill(irow, cpad, value);
371  }
372  }
373  if ((iROC % 36 == mSelectedSector % 36) && hROC) {
374  hROC->SetUniqueID(iROC);
375  }
376  }
377  }
378  }
379  if (!type) {
380  Update("MaxValsA;MaxValsC");
381  }
382 
383  Update("MaxValsI;MaxValsO");
384 }

405 Int_t FindROCFromXY(const Float_t x, const Float_t y, const Int_t side)
406 {
407  //
408  //
409  //
410 
411  Float_t r = TMath::Sqrt(x * x + y * y);
412  static const float innerWall = mMapper.getPadCentre(PadPos(0, 0)).X() - 5.;
413  static const float outerWall = mMapper.getPadCentre(PadPos(151, 0)).X() + 5.;
414  static const float outerIROC = mMapper.getPadCentre(PadPos(62, 0)).X();
415  static const float innerOROC = mMapper.getPadCentre(PadPos(63, 0)).X();
416  static const float betweenROC = (outerIROC + innerOROC) / 2.;
417  //check radial boundary
418  if (r < innerWall || r > outerWall)
419  return -1;
420  //check for IROC or OROC
421  Int_t type = 0;
422  if (r > betweenROC)
423  type = 1;
424  Int_t alpha = TMath::Nint(TMath::ATan2(y, x) / TMath::Pi() * 180);
425  // printf("%6.3f %6.3f %03d\n",x, y, alpha);
426  if (alpha < 0)
427  alpha += 360;
428  Int_t roc = alpha / 20 + side * 18 + type * 36;
429  return roc;
430 }
