18 // Composed Label to encode MC track id, event it comes from and the source (file)

20 class MCCompLabel
21 {
22  private:
23  static constexpr ULong64_t ul0x1 = 0x1;
24  static constexpr ULong64_t NotSet = 0xffffffffffffffff;
25 
26  ULong64_t mLabel = NotSet;
27 
28  public:
29  // number of bits reserved for MC track ID, DON'T modify this, since the
30  // track ID might be negative
31  static constexpr int nbitsTrackID = sizeof(int) * 8;
32  static constexpr int nbitsEvID = 19; // number of bits reserved for MC event ID
33  static constexpr int nbitsSrcID = 8; // number of bits reserved for MC source ID
34  // the rest of the bits is reserved at the moment
35 
36  // check if the fields are defined consistently
37  static_assert(nbitsTrackID == sizeof(int) * 8, "TrackID must have int size");
38  static_assert(nbitsTrackID + nbitsEvID + nbitsSrcID <= sizeof(ULong64_t) * 8,
39  "Fields cannot be stored in 64 bits");
40 
41  // mask to extract MC track ID
42  static constexpr ULong64_t maskTrackID = (ul0x1 << nbitsTrackID) - 1;
43  // mask to extract MC track ID
44  static constexpr ULong64_t maskEvID = (ul0x1 << nbitsEvID) - 1;
45  // mask to extract MC track ID
46  static constexpr ULong64_t maskSrcID = (ul0x1 << nbitsSrcID) - 1;
47  // mask for all used fields
48  static constexpr ULong64_t maskFull = (ul0x1 << (nbitsTrackID + nbitsEvID + nbitsSrcID)) - 1;
49 
50  MCCompLabel(int trackID, int evID = 0, int srcID = 0) { set(trackID, evID, srcID); }
51  MCCompLabel() = default;
52  ~MCCompLabel() = default;
53 
54  // check if label was assigned
55  bool isSet() const { return mLabel != NotSet; }
56  // check if label was not assigned
57  bool isEmpty() const { return mLabel == NotSet; }
58  // check if label was assigned with non-negaive trackID
59  bool isPosTrackID() const { return getTrackID() >= 0; }
60  // return 1 if the tracks are the same and have labels>=0,
61  // 0 if the tracks are the same but at least one of them has label<0
62  // -1 otherwhise
63  int compare(const MCCompLabel& other) const
64  {
65  if (getEventID() != other.getEventID() || getSourceID() != other.getSourceID()) {
66  return -1;
67  }
68  int tr1 = getTrackID(), tr2 = other.getTrackID();
69  return (tr1 == tr2) ? 1 : ((tr1 = -tr2) ? 0 : -1);
70  }
71 
72  // conversion operator
73  operator ULong64_t() const { return mLabel; }
74  // allow to retrieve bare label
75  ULong64_t getRawValue() const { return mLabel; }
76 
77  // comparison operator, compares only label, not evential weight info
78  bool operator==(const MCCompLabel& other) const { return (mLabel & maskFull) == (other.mLabel & maskFull); }
79  // invalidate
80  void unset() { mLabel = NotSet; }
81  void set(int trackID, int evID, int srcID)
82  {
84  mLabel = (maskTrackID & static_cast<ULong64_t>(static_cast<UInt_t>(trackID))) |
85  (maskEvID & static_cast<ULong64_t>(evID)) << nbitsTrackID |
86  (maskSrcID & static_cast<ULong64_t>(srcID)) << (nbitsTrackID + nbitsEvID);
87  }
88 
89  int getTrackID() const { return static_cast<int>(mLabel & maskTrackID); }
90  int getEventID() const { return (mLabel >> nbitsTrackID) & maskEvID; }
91  int getSourceID() const { return (mLabel >> (nbitsTrackID + nbitsEvID)) & maskSrcID; }
92  void get(int& trackID, int& evID, int& srcID)
93  {
95  trackID = getTrackID();
96  evID = getEventID();
97  srcID = getSourceID();
98  }
