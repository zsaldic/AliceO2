62  int getCRU(const Sector& sec, GlobalPadNumber globalPad)
63  {
64  const auto row = mMapGlobalPadToPadPos[globalPad].getRow();
65  const auto nCRUPerSector = mMapPadRegionInfo.size();
66  int region = 0;
67  for (size_t i = 1; i < nCRUPerSector; ++i) {
68  if (row < mMapPadRegionInfo[i].getGlobalRowOffset()) {
69  break;
70  }
71  ++region;
72  }
73 
74  return int(sec * nCRUPerSector + region);
75  }

380  {
381  const double cs = std::cos(alpha), sn = std::sin(alpha);

572 inline bool Mapper::isOutOfSector(GlobalPosition3D posEle, const Sector& sector, const float margin) const
573 {
574  int secRight = int(sector);
575  int secLeft = int(Sector::getLeft(sector));
576  const float dSectorBoundaryRight = -SinsPerSectorNotShifted[secRight % SECTORSPERSIDE] * posEle.X() + CosinsPerSectorNotShifted[secRight % SECTORSPERSIDE] * posEle.Y();
577  const float dSectorBoundaryLeft = -SinsPerSectorNotShifted[secLeft % SECTORSPERSIDE] * posEle.X() + CosinsPerSectorNotShifted[secLeft % SECTORSPERSIDE] * posEle.Y();
578 
579  if ((dSectorBoundaryLeft > 0 && dSectorBoundaryRight < 0) || (dSectorBoundaryLeft < 0 && dSectorBoundaryRight > 0)) {
580  return false;
581  }
582  if (std::abs(dSectorBoundaryLeft) > margin && std::abs(dSectorBoundaryRight) > margin) {
583  return true;
584  }
585  return false;
586 }
