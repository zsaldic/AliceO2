179 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
180 constexpr T String2(const char (&str)[N])
181 {
182  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
183  static_assert(N >= pos, "Position is larger than the length of string");
184  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
185  "String parameter is longer than the size of the data type"
186  );
187  return((T) str[0+pos] |
188  (str[0+pos] && sizeof(T) >= 2 ? ((T) str[1+pos] << (sizeof(T) >= 2 ? 8 : 0) |
189  (str[1+pos] && sizeof(T) >= 4 ? ((T) str[2+pos] << (sizeof(T) >= 4 ? 16 : 0) |
190  (str[2+pos] && sizeof(T) >= 4 ? ((T) str[3+pos] << (sizeof(T) >= 4 ? 24 : 0) |
191  (str[3+pos] && sizeof(T) >= 8 ? ((T) str[4+pos] << (sizeof(T) >= 8 ? 32 : 0) |
192  (str[4+pos] && sizeof(T) >= 8 ? ((T) str[5+pos] << (sizeof(T) >= 8 ? 40 : 0) |
193  (str[5+pos] && sizeof(T) >= 8 ? ((T) str[6+pos] << (sizeof(T) >= 8 ? 48 : 0) |
194  (str[6+pos] && sizeof(T) >= 8 ? ((T) str[7+pos] << (sizeof(T) >= 8 ? 56 : 0) )
195  : 0)) : 0)) : 0)) : 0)) : 0)) : 0)) : 0));
196 }

294  void runtimeInit(const char* string, short length = -1) {
295  char* target = str;
296  char* targetEnd = target;
297  if (length >= 0 && length < (int)N) targetEnd += length;
298  else targetEnd += N;
299  const char* source = string;
300  for ( ; source != nullptr && target < targetEnd && *source !=0; ++target, ++source) *target = *source;
301  targetEnd = str + N;
302  for ( ; target < targetEnd; ++target) *target = 0;
303  // require the string to be not longer than the descriptor size
304  if (source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N))) {
305  } else {
306  throw std::invalid_argument("argument must not be longer than descriptor size");
307  }
308  }

464 template <typename HeaderType, typename std::enable_if_t<std::is_pointer<HeaderType>::value, int> = 0>
465 auto get(const o2::byte* buffer, size_t /*len*/ = 0)
466 {
467  using HeaderConstPtrType = const typename std::remove_pointer<HeaderType>::type*;
468  using HeaderValueType = typename std::remove_pointer<HeaderType>::type;
469 
470  const BaseHeader* current = BaseHeader::get(buffer);
471  if (!current)
472  return HeaderConstPtrType{ nullptr };
473  if (current->description == HeaderValueType::sHeaderType)
474  return reinterpret_cast<HeaderConstPtrType>(current);
475  while ((current = current->next())) {
476  if (current->description == HeaderValueType::sHeaderType)
477  return reinterpret_cast<HeaderConstPtrType>(current);
478  }
479  return HeaderConstPtrType{ nullptr };
480 }
