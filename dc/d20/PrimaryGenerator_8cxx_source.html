161 Bool_t PrimaryGenerator::embedInto(TString fname)
162 {
166  if (mEmbedFile && mEmbedFile->IsOpen()) {
167  LOG(ERROR) << "Another embedding file is currently open";
168  return kFALSE;
169  }
170 
172  mEmbedFile = TFile::Open(fname);
173  if (!mEmbedFile || !mEmbedFile->IsOpen()) {
174  LOG(ERROR) << "Cannot open file for embedding: " << fname;
175  return kFALSE;
176  }
177 
179  mEmbedTree = (TTree*)mEmbedFile->Get("o2sim");
180  if (!mEmbedTree) {
181  LOG(ERROR) << R"(Cannot find "o2sim" tree for embedding in )" << fname;
182  return kFALSE;
183  }
184 
186  mEmbedEntries = mEmbedTree->GetEntries();
187  if (mEmbedEntries <= 0) {
188  LOG(ERROR) << "Invalid number of entries found in tree for embedding: " << mEmbedEntries;
189  return kFALSE;
190  }
191 
193  mEmbedEvent = new MCEventHeader;
194  mEmbedTree->SetBranchAddress("MCEventHeader.", &mEmbedEvent);
195 
197  return kTRUE;
198 }
