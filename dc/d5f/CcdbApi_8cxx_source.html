50 void CcdbApi::store(TObject* rootObject, std::string path, std::map<std::string, std::string> metadata,
51  long startValidityTimestamp, long endValidityTimestamp)
52 {
53  // Serialize the object
54  TMessage message(kMESS_OBJECT);
55  message.Reset();
56  message.WriteObjectAny(rootObject, rootObject->IsA());
57 
58  // Prepare
59  long sanitizedStartValidityTimestamp = startValidityTimestamp;
60  if (startValidityTimestamp == -1) {
61  cout << "Start of Validity not set, current timestamp used." << endl;
62  sanitizedStartValidityTimestamp = getCurrentTimestamp();
63  }
64  long sanitizedEndValidityTimestamp = endValidityTimestamp;
65  if (endValidityTimestamp == -1) {
66  cout << "End of Validity not set, start of validity plus 1 year used." << endl;
67  sanitizedEndValidityTimestamp = getFutureTimestamp(60 * 60 * 24 * 365);
68  }
69  string fullUrl = getFullUrlForStorage(path, metadata, sanitizedStartValidityTimestamp, sanitizedEndValidityTimestamp);
70 
71  // Curl preparation
72  CURL* curl;
73  struct curl_httppost* formpost = nullptr;
74  struct curl_httppost* lastptr = nullptr;
75  struct curl_slist* headerlist = nullptr;
76  static const char buf[] = "Expect:";
77  // todo : what is the correct file name ?
78  string tmpFileName = string(rootObject->GetName()) + "_" + getTimestampString(getCurrentTimestamp()) + ".root";
79  curl_formadd(&formpost,
80  &lastptr,
81  CURLFORM_COPYNAME, "send",
82  CURLFORM_BUFFER, tmpFileName.c_str(),
83  CURLFORM_BUFFERPTR, message.Buffer(),
84  CURLFORM_BUFFERLENGTH, message.Length(),
85  CURLFORM_END);
86 
87  curl = curl_easy_init();
88  headerlist = curl_slist_append(headerlist, buf);
89  if (curl != nullptr) {
90  /* what URL that receives this POST */
91  curl_easy_setopt(curl, CURLOPT_URL, fullUrl.c_str());
92  curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headerlist);
93  curl_easy_setopt(curl, CURLOPT_HTTPPOST, formpost);
94 
95  /* Perform the request, res will get the return code */
96  CURLcode res = curl_easy_perform(curl);
97  /* Check for errors */
98  if (res != CURLE_OK) {
99  fprintf(stderr, "curl_easy_perform() failed: %s\n",
100  curl_easy_strerror(res));
101  }
102 
103  /* always cleanup */
104  curl_easy_cleanup(curl);
105 
106  /* then cleanup the formpost chain */
107  curl_formfree(formpost);
108  /* free slist */
109  curl_slist_free_all(headerlist);
110  }
111 }

113 string CcdbApi::getFullUrlForStorage(const string& path, const map<string, string>& metadata,
114  long startValidityTimestamp, long endValidityTimestamp)
115 {
116  // Prepare timestamps
117  string startValidityString = getTimestampString(startValidityTimestamp < 0 ? getCurrentTimestamp() : startValidityTimestamp);
118  string endValidityString = getTimestampString(endValidityTimestamp < 0 ? getFutureTimestamp(60 * 60 * 24 * 365) : endValidityTimestamp);
119  // Build URL
120  string fullUrl = mUrl + "/" + path + "/" + startValidityString + "/" + endValidityString + "/";
121  // Add metadata
122  for (auto& kv : metadata) {
123  fullUrl += kv.first + "=" + kv.second + "&";
124  }
125  return fullUrl;

159 static size_t WriteMemoryCallback(void* contents, size_t size, size_t nmemb, void* userp)
160 {
161  size_t realsize = size * nmemb;
162  auto* mem = (struct MemoryStruct*)userp;
163 
164  mem->memory = (char*)realloc(mem->memory, mem->size + realsize + 1);
165  if (mem->memory == nullptr) {
166  printf("not enough memory (realloc returned NULL)\n");
167  return 0;
168  }
169 
170  memcpy(&(mem->memory[mem->size]), contents, realsize);
171  mem->size += realsize;
172  mem->memory[mem->size] = 0;
173 
174  return realsize;
175 }

177 TObject* CcdbApi::retrieve(std::string path, std::map<std::string, std::string> metadata,
178  long timestamp)
179 {
180  // Note : based on https://curl.haxx.se/libcurl/c/getinmemory.html
181  // Thus it does not comply to our coding guidelines as it is a copy paste.
182 
183  string fullUrl = getFullUrlForRetrieval(path, metadata, timestamp);
184 
185  // Prepare CURL
186  CURL* curl_handle;
187  CURLcode res;
188  struct MemoryStruct chunk {
189  (char*)malloc(1) /*memory*/, 0 /*size*/
190  };
191  TObject* result = nullptr;
192 
193  /* init the curl session */
194  curl_handle = curl_easy_init();
195 
196  /* specify URL to get */
197  curl_easy_setopt(curl_handle, CURLOPT_URL, fullUrl.c_str());
198 
199  /* send all data to this function */
200  curl_easy_setopt(curl_handle, CURLOPT_WRITEFUNCTION, WriteMemoryCallback);
201 
202  /* we pass our 'chunk' struct to the callback function */
203  curl_easy_setopt(curl_handle, CURLOPT_WRITEDATA, (void*)&chunk);
204 
205  /* some servers don't like requests that are made without a user-agent
206  field, so we provide one */
207  curl_easy_setopt(curl_handle, CURLOPT_USERAGENT, "libcurl-agent/1.0");
208 
209  /* if redirected , we tell libcurl to follow redirection */
210  curl_easy_setopt(curl_handle, CURLOPT_FOLLOWLOCATION, 1L);
211 
212  /* get it! */
213  res = curl_easy_perform(curl_handle);
214 
215  /* check for errors */
216  if (res != CURLE_OK) {
217  fprintf(stderr, "curl_easy_perform() failed: %s\n",
218  curl_easy_strerror(res));
219  } else {
220  /*
221  * Now, our chunk.memory points to a memory block that is chunk.size
222  * bytes big and contains the remote file.
223  */
224 
225  // printf("%lu bytes retrieved\n", (long) chunk.size);
226 
227  long response_code;
228  res = curl_easy_getinfo(curl_handle, CURLINFO_RESPONSE_CODE, &response_code);
229  if ((res == CURLE_OK) && (response_code != 404)) {
230  TMessage mess(kMESS_OBJECT);
231  mess.SetBuffer(chunk.memory, chunk.size, kFALSE);
232  mess.SetReadMode();
233  mess.Reset();
234  result = (TObject*)(mess.ReadObjectAny(mess.GetClass()));
235  if (result == nullptr) {
236  cerr << "couldn't retrieve the object " << path << endl;
237  }
238  } else {
239  cerr << "invalid URL : " << fullUrl << endl;
240  }
241 
242  // Print data
243  // cout << "size : " << chunk.size << endl;
244  // cout << "data : " << endl;
245  // char* mem = (char*)chunk.memory;
246  // for (int i = 0 ; i < chunk.size/4 ; i++) {
247  // cout << mem;
248  // mem += 4;
249  // }
250  }
251 
252  /* cleanup curl stuff */
253  curl_easy_cleanup(curl_handle);
254 
255  free(chunk.memory);
256 
257  return result;
258 }

308 long CcdbApi::getFutureTimestamp(int secondsInFuture)
309 {
310  std::chrono::seconds sec(secondsInFuture);
311  auto future = std::chrono::system_clock::now() + sec;
312  auto future_ms = std::chrono::time_point_cast<std::chrono::milliseconds>(future);
313  auto epoch = future_ms.time_since_epoch();
314  auto value = std::chrono::duration_cast<std::chrono::milliseconds>(epoch);
315  return value.count();
316 }
