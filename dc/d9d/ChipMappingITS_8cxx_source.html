42 ChipMappingITS::ChipMappingITS()
43 {
44  // init chips info
45 
46  /*
47  FEE ID field in RDH is 16-bit wide
48  Each RU has 10-bit DIPSWITCH, use bits 9:2 as an 8bit ID field to be mapped to FEE ID field in RDH
49 
50  | Lr |Stave Count| Bin. pref | Range of Binary Addresses
51  | |= RU Count | Stave addr|
52  |----------------------------------------------------------
53  | L0 | 12 |b 0000xxxx | 0000_0000 : 0000_1011 (0 – 11)
54  | L1 | 16 |b 0001xxxx | 0001_0000 : 0001_1111 (0 – 15)
55  | L2 | 20 |b 001xxxxx | 001_00000 : 001_10011 (0 – 19)
56  | L3 | 24 |b 010xxxxx | 010_00000 : 010_10111 (0 – 23)
57  | L4 | 30 |b 011xxxxx | 011_00000 : 011_11101 (0 – 29)
58  | L5 | 42 |b 10xxxxxx | 10_000000 : 10_101001 (0 – 41)
59  | L6 | 48 |b 11xxxxxx | 11_000000 : 11_101111 (0 – 47)
60 
61  FEEId format:
62  15|14 12|11 10 |9 8|7 6|5 0|
63  0| Layer | Reserve | Fiber | 00 | StaveNumber |
64 
65  */
66 
67  uint32_t maxRUHW = composeFEEId(NLayers - 1, NStavesOnLr[NLayers - 1], NLinks - 1); // Max possible FEE ID
68  assert(maxRUHW < 0xffff);
69  mFEEId2RUSW.resize(maxRUHW + 1, 0xff);
70 
71  // IB: single cable per chip
72  int ctrChip = 0;
73  mChipInfoEntrySB[IB] = ctrChip;
74  mCableHW2SW[IB].resize(NChipsPerStaveSB[IB], 0xff);
75  mCableHWFirstChip[IB].resize(NChipsPerStaveSB[IB], 0xff);
76  for (int i = 0; i < NChipsPerStaveSB[IB]; i++) {
77  auto& cInfo = mChipsInfo[ctrChip++];
78  cInfo.id = i;
79  cInfo.moduleHW = 0;
80  cInfo.moduleSW = 0;
81  cInfo.chipOnModuleSW = i;
82  cInfo.chipOnModuleHW = i;
83  cInfo.cableHW = i;
84  cInfo.cableSW = i;
85  cInfo.chipOnCable = 0; // every chip is master
86  mCableHW2SW[IB][cInfo.cableHW] = cInfo.cableSW; //1-to-1 mapping
87  mCableHWFirstChip[IB][i] = 0; // stave and module are the same
88  }
89 
90  for (int bid = MB; bid <= OB; bid++) { // MB and OB staves have similar layout
91  mChipInfoEntrySB[bid] = ctrChip;
92  mCableHW2SW[bid].resize(0xff, 0xff);
93  mCableHWFirstChip[bid].resize(0xff, 0xff);
94 
95  for (int i = 0; i < NChipsPerStaveSB[bid]; i++) {
96  auto& cInfo = mChipsInfo[ctrChip++];
97  int hstave = i / (NChipsPerStaveSB[bid] / 2);
98  cInfo.id = i;
99  cInfo.moduleSW = i / NChipsPerModuleSB[bid]; // SW module ID (within the stave)
100  cInfo.moduleHW = 1 + cInfo.moduleSW % (NModulesPerStaveSB[bid] / 2); // identification within the HS
101 
102  cInfo.chipOnModuleSW = i % NChipsPerModuleSB[bid];
103  cInfo.chipOnModuleHW = ChipOBModSW2HW[cInfo.chipOnModuleSW];
104 
105  uint8_t connector = (hstave << 1) + (cInfo.chipOnModuleSW < (NChipsPerModuleSB[bid] / 2) ? 0 : 1);
106  cInfo.cableHW = (connector << 3) + (cInfo.moduleHW - 1);
107  cInfo.cableSW = (cInfo.moduleHW - 1) + connector * (NModulesPerStaveSB[bid] / 2);
108  cInfo.chipOnCable = cInfo.chipOnModuleSW % (NChipsPerModuleSB[bid] / 2); // each cable serves half module
109  mCableHW2SW[bid][cInfo.cableHW] = cInfo.cableSW;
110  if (cInfo.chipOnCable == 0) {
111  mCableHWFirstChip[bid][cInfo.cableHW] = cInfo.moduleSW * NChipsPerModuleSB[bid];
112  }
113  }
114  }
115 
116  int ctrStv = 0;
117  uint16_t chipCount = 0;
118  for (int ilr = 0; ilr < NLayers; ilr++) {
119  for (int ist = 0; ist < NStavesOnLr[ilr]; ist++) {
120  auto& sInfo = mStavesInfo[ctrStv];
121  sInfo.idSW = ctrStv++;
122 
123  // map FEEIds (RU read out by at most 3 GBT links) to SW ID
124  sInfo.idHW = composeFEEId(ilr, ist, 0); // FEEId for link 0
125  mFEEId2RUSW[sInfo.idHW] = sInfo.idSW;
126  for (int lnk = 1; lnk < NLinks; lnk++) {
127  mFEEId2RUSW[composeFEEId(ilr, ist, lnk)] = sInfo.idSW;
128  }
129  sInfo.layer = ilr;
130  sInfo.ruType = RUTypeLr[ilr];
131  sInfo.nCables = NCablesPerStaveSB[sInfo.ruType];
132  sInfo.firstChipIDSW = chipCount;
133  chipCount += NChipsPerStaveSB[sInfo.ruType];
134  }
135  }
136  assert(ctrStv == getNRUs());
137 }

158 void ChipMappingITS::expandChipInfoSW(int idSW, int& lay, int& sta, int& ssta, int& mod, int& chipInMod) const
159 {
160  // convert SW chip ID to detailed info SW info
161  ChipInfo chi;
162  getChipInfoSW(idSW, chi);
163  const auto staveInfo = getRUInfoSW(chi.ru);
164  lay = staveInfo->layer;
165  sta = staveInfo->idSW;
166  mod = chi.chOnRU->moduleSW;
167  chipInMod = chi.chOnRU->chipOnModuleSW;
168  ssta = lay < 3 || (mod < (NModulesPerStaveSB[chi.ruType] >> 2)) ? 0 : 1;
169 }

172 void ChipMappingITS::expandChipInfoHW(int idSW, int& lay, int& sta, int& ssta, int& mod, int& chipInMod) const
173 {
174  // convert SW chip ID to detailed info HW info
175  ChipInfo chi;
176  getChipInfoSW(idSW, chi);
177  const auto staveInfo = getRUInfoSW(chi.ru);
178  lay = staveInfo->layer;
179  sta = staveInfo->idSW - getFirstStavesOnLr(lay); // stave relative to layer
180  mod = lay < 3 ? 0 : 1 + (chi.chOnRU->moduleSW % NModulesAlongStaveSB[chi.ruType]); // module on the substave
181  ssta = chi.chOnRU->moduleSW / NModulesAlongStaveSB[chi.ruType];
182  chipInMod = chi.chOnRU->chipOnModuleHW;
183 }

186 std::string ChipMappingITS::getChipNameHW(int idSW) const
187 {
188  // convert global SW chip ID to name in HW conventions
189  int lay, sta, ssta, mod, cinmod;
190  expandChipInfoHW(idSW, lay, sta, ssta, mod, cinmod);
191  std::stringstream strs;
192  strs << 'L' << lay << '_' << std::setfill('0') << std::setw(2) << sta;
193  if (lay > 2) {
194  strs << (ssta ? 'U' : 'L') << "_M" << mod;
195  }
196  strs << "_C" << std::setfill('0') << std::setw(2) << cinmod;
197 
198  return strs.str();
