401 void ImDrawList::UpdateClipRect()
402 {
403  // If current command is used with different settings we need to add a new command
404  const ImVec4 curr_clip_rect = GetCurrentClipRect();
405  ImDrawCmd* curr_cmd = CmdBuffer.Size > 0 ? &CmdBuffer.Data[CmdBuffer.Size-1] : NULL;
406  if (!curr_cmd || (curr_cmd->ElemCount != 0 && memcmp(&curr_cmd->ClipRect, &curr_clip_rect, sizeof(ImVec4)) != 0) || curr_cmd->UserCallback != NULL)
407  {
408  AddDrawCmd();
409  return;
410  }
411 
412  // Try to merge with previous command if it matches, else use current command
413  ImDrawCmd* prev_cmd = CmdBuffer.Size > 1 ? curr_cmd - 1 : NULL;
414  if (curr_cmd->ElemCount == 0 && prev_cmd && memcmp(&prev_cmd->ClipRect, &curr_clip_rect, sizeof(ImVec4)) == 0 && prev_cmd->TextureId == GetCurrentTextureId() && prev_cmd->UserCallback == NULL)
415  CmdBuffer.pop_back();

519 void ImDrawList::ChannelsMerge()
520 {
521  // Note that we never use or rely on channels.Size because it is merely a buffer that we never shrink back to 0 to keep all sub-buffers ready for use.
522  if (_ChannelsCount <= 1)
523  return;
524 
525  ChannelsSetCurrent(0);
526  if (CmdBuffer.Size && CmdBuffer.back().ElemCount == 0)
527  CmdBuffer.pop_back();
528 
529  int new_cmd_buffer_count = 0, new_idx_buffer_count = 0;
530  for (int i = 1; i < _ChannelsCount; i++)
531  {
532  ImDrawChannel& ch = _Channels[i];
533  if (ch.CmdBuffer.Size && ch.CmdBuffer.back().ElemCount == 0)
534  ch.CmdBuffer.pop_back();
535  new_cmd_buffer_count += ch.CmdBuffer.Size;
536  new_idx_buffer_count += ch.IdxBuffer.Size;
537  }
538  CmdBuffer.resize(CmdBuffer.Size + new_cmd_buffer_count);
539  IdxBuffer.resize(IdxBuffer.Size + new_idx_buffer_count);
540 
541  ImDrawCmd* cmd_write = CmdBuffer.Data + CmdBuffer.Size - new_cmd_buffer_count;
542  _IdxWritePtr = IdxBuffer.Data + IdxBuffer.Size - new_idx_buffer_count;
543  for (int i = 1; i < _ChannelsCount; i++)
544  {
545  ImDrawChannel& ch = _Channels[i];
546  if (int sz = ch.CmdBuffer.Size) { memcpy(cmd_write, ch.CmdBuffer.Data, sz * sizeof(ImDrawCmd)); cmd_write += sz; }
547  if (int sz = ch.IdxBuffer.Size) { memcpy(_IdxWritePtr, ch.IdxBuffer.Data, sz * sizeof(ImDrawIdx)); _IdxWritePtr += sz; }
548  }
549  UpdateClipRect(); // We call this instead of AddDrawCmd(), so that empty channels won't produce an extra draw call.
550  _ChannelsCount = 1;
551 }

626 void ImDrawList::AddPolyline(const ImVec2* points, const int points_count, ImU32 col, bool closed, float thickness)
627 {

629  return;
630 
631  const ImVec2 uv = _Data->TexUvWhitePixel;
632 
633  int count = points_count;
634  if (!closed)
635  count = points_count-1;
636 
637  const bool thick_line = thickness > 1.0f;
638  if (Flags & ImDrawListFlags_AntiAliasedLines)
639  {
640  // Anti-aliased stroke
641  const float AA_SIZE = 1.0f;
642  const ImU32 col_trans = col & ~IM_COL32_A_MASK;
643 
644  const int idx_count = thick_line ? count*18 : count*12;
645  const int vtx_count = thick_line ? points_count*4 : points_count*3;
646  PrimReserve(idx_count, vtx_count);
647 
648  // Temporary buffer
649  ImVec2* temp_normals = (ImVec2*)alloca(points_count * (thick_line ? 5 : 3) * sizeof(ImVec2));
650  ImVec2* temp_points = temp_normals + points_count;
651 
652  for (int i1 = 0; i1 < count; i1++)
653  {
654  const int i2 = (i1+1) == points_count ? 0 : i1+1;
655  ImVec2 diff = points[i2] - points[i1];
656  diff *= ImInvLength(diff, 1.0f);
657  temp_normals[i1].x = diff.y;
658  temp_normals[i1].y = -diff.x;
659  }
660  if (!closed)
661  temp_normals[points_count-1] = temp_normals[points_count-2];
662 
663  if (!thick_line)
664  {
665  if (!closed)
666  {
667  temp_points[0] = points[0] + temp_normals[0] * AA_SIZE;

676  {

917 {
918  float dx = x4 - x1;
919  float dy = y4 - y1;
920  float d2 = ((x2 - x4) * dy - (y2 - y4) * dx);
921  float d3 = ((x3 - x4) * dy - (y3 - y4) * dx);
922  d2 = (d2 >= 0) ? d2 : -d2;
923  d3 = (d3 >= 0) ? d3 : -d3;
924  if ((d2+d3) * (d2+d3) < tess_tol * (dx*dx + dy*dy))
925  {
926  path->push_back(ImVec2(x4, y4));
927  }
928  else if (level < 10)
929  {
930  float x12 = (x1+x2)*0.5f, y12 = (y1+y2)*0.5f;
931  float x23 = (x2+x3)*0.5f, y23 = (y2+y3)*0.5f;

1702  int total_ranges_count = 0;
1703  for (int input_i = 0; input_i < atlas->ConfigData.Size; input_i++)
1704  {
1705  ImFontConfig& cfg = atlas->ConfigData[input_i];
1706  if (!cfg.GlyphRanges)
1707  cfg.GlyphRanges = atlas->GetGlyphRangesDefault();
1708  for (const ImWchar* in_range = cfg.GlyphRanges; in_range[0] && in_range[1]; in_range += 2, total_ranges_count++)
1709  total_glyphs_count += (in_range[1] - in_range[0]) + 1;
1710  }
1711 
1712  // We need a width for the skyline algorithm. Using a dumb heuristic here to decide of width. User can override TexDesiredWidth and TexGlyphPadding if they wish.
1713  // Width doesn't really matter much, but some API/GPU have texture size limitations and increasing width can decrease height.
1714  atlas->TexWidth = (atlas->TexDesiredWidth > 0) ? atlas->TexDesiredWidth : (total_glyphs_count > 4000) ? 4096 : (total_glyphs_count > 2000) ? 2048 : (total_glyphs_count > 1000) ? 1024 : 512;
1715  atlas->TexHeight = 0;
1716 
1717  // Start packing
1718  const int max_tex_height = 1024*32;
1719  stbtt_pack_context spc = {};
1720  if (!stbtt_PackBegin(&spc, NULL, atlas->TexWidth, max_tex_height, 0, atlas->TexGlyphPadding, NULL))
1721  return false;
1722  stbtt_PackSetOversampling(&spc, 1, 1);
1723 
1724  // Pack our extra data rectangles first, so it will be on the upper-left corner of our texture (UV will have small values).
1725  ImFontAtlasBuildPackCustomRects(atlas, spc.pack_info);
1726 
1727  // Initialize font information (so we can error without any cleanup)
1728  struct ImFontTempBuildData
1729  {
1730  stbtt_fontinfo FontInfo;
1731  stbrp_rect* Rects;
1732  int RectsCount;
1733  stbtt_pack_range* Ranges;
1734  int RangesCount;
1735  };
1736  ImFontTempBuildData* tmp_array = (ImFontTempBuildData*)ImGui::MemAlloc((size_t)atlas->ConfigData.Size * sizeof(ImFontTempBuildData));
1737  for (int input_i = 0; input_i < atlas->ConfigData.Size; input_i++)
1738  {
1739  ImFontConfig& cfg = atlas->ConfigData[input_i];
1740  ImFontTempBuildData& tmp = tmp_array[input_i];
1741  IM_ASSERT(cfg.DstFont && (!cfg.DstFont->IsLoaded() || cfg.DstFont->ContainerAtlas == atlas));
1742 
1743  const int font_offset = stbtt_GetFontOffsetForIndex((unsigned char*)cfg.FontData, cfg.FontNo);
1744  IM_ASSERT(font_offset >= 0 && "FontData is incorrect, or FontNo cannot be found.");
1745  if (!stbtt_InitFont(&tmp.FontInfo, (unsigned char*)cfg.FontData, font_offset))
1746  {
1747  atlas->TexWidth = atlas->TexHeight = 0; // Reset output on failure
1748  ImGui::MemFree(tmp_array);
1749  return false;
1750  }
1751  }
1752 
1753  // Allocate packing character data and flag packed characters buffer as non-packed (x0=y0=x1=y1=0)
1754  int buf_packedchars_n = 0, buf_rects_n = 0, buf_ranges_n = 0;
1755  stbtt_packedchar* buf_packedchars = (stbtt_packedchar*)ImGui::MemAlloc(total_glyphs_count * sizeof(stbtt_packedchar));
1756  stbrp_rect* buf_rects = (stbrp_rect*)ImGui::MemAlloc(total_glyphs_count * sizeof(stbrp_rect));
1757  stbtt_pack_range* buf_ranges = (stbtt_pack_range*)ImGui::MemAlloc(total_ranges_count * sizeof(stbtt_pack_range));
1758  memset(buf_packedchars, 0, total_glyphs_count * sizeof(stbtt_packedchar));
1759  memset(buf_rects, 0, total_glyphs_count * sizeof(stbrp_rect)); // Unnecessary but let's clear this for the sake of sanity.
1760  memset(buf_ranges, 0, total_ranges_count * sizeof(stbtt_pack_range));
1761 
1762  // First font pass: pack all glyphs (no rendering at this point, we are working with rectangles in an infinitely tall texture at this point)
1763  for (int input_i = 0; input_i < atlas->ConfigData.Size; input_i++)
1764  {
1765  ImFontConfig& cfg = atlas->ConfigData[input_i];
1766  ImFontTempBuildData& tmp = tmp_array[input_i];
1767 
1768  // Setup ranges
1769  int font_glyphs_count = 0;
1770  int font_ranges_count = 0;
1771  for (const ImWchar* in_range = cfg.GlyphRanges; in_range[0] && in_range[1]; in_range += 2, font_ranges_count++)
1772  font_glyphs_count += (in_range[1] - in_range[0]) + 1;
1773  tmp.Ranges = buf_ranges + buf_ranges_n;
1774  tmp.RangesCount = font_ranges_count;
1775  buf_ranges_n += font_ranges_count;
1776  for (int i = 0; i < font_ranges_count; i++)
1777  {
1778  const ImWchar* in_range = &cfg.GlyphRanges[i * 2];
1779  stbtt_pack_range& range = tmp.Ranges[i];
1780  range.font_size = cfg.SizePixels;
1781  range.first_unicode_codepoint_in_range = in_range[0];
1782  range.num_chars = (in_range[1] - in_range[0]) + 1;
1783  range.chardata_for_range = buf_packedchars + buf_packedchars_n;
1784  buf_packedchars_n += range.num_chars;
1785  }
1786 
1787  // Gather the sizes of all rectangle we need
1788  tmp.Rects = buf_rects + buf_rects_n;
1789  tmp.RectsCount = font_glyphs_count;
1790  buf_rects_n += font_glyphs_count;
1791  stbtt_PackSetOversampling(&spc, cfg.OversampleH, cfg.OversampleV);
1792  int n = stbtt_PackFontRangesGatherRects(&spc, &tmp.FontInfo, tmp.Ranges, tmp.RangesCount, tmp.Rects);
1793  IM_ASSERT(n == font_glyphs_count);
1794 
1795  // Detect missing glyphs and replace them with a zero-sized box instead of relying on the default glyphs
1796  // This allows us merging overlapping icon fonts more easily.
1797  int rect_i = 0;
1798  for (int range_i = 0; range_i < tmp.RangesCount; range_i++)
1799  for (int char_i = 0; char_i < tmp.Ranges[range_i].num_chars; char_i++, rect_i++)
1800  if (stbtt_FindGlyphIndex(&tmp.FontInfo, tmp.Ranges[range_i].first_unicode_codepoint_in_range + char_i) == 0)
1801  tmp.Rects[rect_i].w = tmp.Rects[rect_i].h = 0;
1802 
1803  // Pack
1804  stbrp_pack_rects((stbrp_context*)spc.pack_info, tmp.Rects, n);
1805 
1806  // Extend texture height
1807  // Also mark missing glyphs as non-packed so we don't attempt to render into them
1808  for (int i = 0; i < n; i++)
1809  {
1810  if (tmp.Rects[i].w == 0 && tmp.Rects[i].h == 0)
1811  tmp.Rects[i].was_packed = 0;
1812  if (tmp.Rects[i].was_packed)
1813  atlas->TexHeight = ImMax(atlas->TexHeight, tmp.Rects[i].y + tmp.Rects[i].h);
1814  }
1815  }
1816  IM_ASSERT(buf_rects_n == total_glyphs_count);
1817  IM_ASSERT(buf_packedchars_n == total_glyphs_count);
1818  IM_ASSERT(buf_ranges_n == total_ranges_count);
1819 
1820  // Create texture
1821  atlas->TexHeight = (atlas->Flags & ImFontAtlasFlags_NoPowerOfTwoHeight) ? (atlas->TexHeight + 1) : ImUpperPowerOfTwo(atlas->TexHeight);
1822  atlas->TexUvScale = ImVec2(1.0f / atlas->TexWidth, 1.0f / atlas->TexHeight);
1823  atlas->TexPixelsAlpha8 = (unsigned char*)ImGui::MemAlloc(atlas->TexWidth * atlas->TexHeight);
1824  memset(atlas->TexPixelsAlpha8, 0, atlas->TexWidth * atlas->TexHeight);
1825  spc.pixels = atlas->TexPixelsAlpha8;
1826  spc.height = atlas->TexHeight;
1827 
1828  // Second pass: render font characters
1829  for (int input_i = 0; input_i < atlas->ConfigData.Size; input_i++)
1830  {
1831  ImFontConfig& cfg = atlas->ConfigData[input_i];
1832  ImFontTempBuildData& tmp = tmp_array[input_i];
1833  stbtt_PackSetOversampling(&spc, cfg.OversampleH, cfg.OversampleV);
1834  stbtt_PackFontRangesRenderIntoRects(&spc, &tmp.FontInfo, tmp.Ranges, tmp.RangesCount, tmp.Rects);
1835  if (cfg.RasterizerMultiply != 1.0f)
1836  {
1837  unsigned char multiply_table[256];
1838  ImFontAtlasBuildMultiplyCalcLookupTable(multiply_table, cfg.RasterizerMultiply);
1839  for (const stbrp_rect* r = tmp.Rects; r != tmp.Rects + tmp.RectsCount; r++)
1840  if (r->was_packed)
1841  ImFontAtlasBuildMultiplyRectAlpha8(multiply_table, spc.pixels, r->x, r->y, r->w, r->h, spc.stride_in_bytes);
1842  }
1843  tmp.Rects = NULL;
1844  }
1845 
1846  // End packing
1847  stbtt_PackEnd(&spc);
1848  ImGui::MemFree(buf_rects);
1849  buf_rects = NULL;
1850 
1851  // Third pass: setup ImFont and glyphs for runtime
1852  for (int input_i = 0; input_i < atlas->ConfigData.Size; input_i++)
1853  {
1854  ImFontConfig& cfg = atlas->ConfigData[input_i];
1855  ImFontTempBuildData& tmp = tmp_array[input_i];
1856  ImFont* dst_font = cfg.DstFont; // We can have multiple input fonts writing into a same destination font (when using MergeMode=true)
1857  if (cfg.MergeMode)
1858  dst_font->BuildLookupTable();
1859 
1860  const float font_scale = stbtt_ScaleForPixelHeight(&tmp.FontInfo, cfg.SizePixels);
1861  int unscaled_ascent, unscaled_descent, unscaled_line_gap;
1862  stbtt_GetFontVMetrics(&tmp.FontInfo, &unscaled_ascent, &unscaled_descent, &unscaled_line_gap);
1863 
1864  const float ascent = ImFloor(unscaled_ascent * font_scale + ((unscaled_ascent > 0.0f) ? +1 : -1));
1865  const float descent = ImFloor(unscaled_descent * font_scale + ((unscaled_descent > 0.0f) ? +1 : -1));
1866  ImFontAtlasBuildSetupFont(atlas, dst_font, &cfg, ascent, descent);
1867  const float off_x = cfg.GlyphOffset.x;
1868  const float off_y = cfg.GlyphOffset.y + (float)(int)(dst_font->Ascent + 0.5f);
1869 
1870  for (int i = 0; i < tmp.RangesCount; i++)
1871  {
1872  stbtt_pack_range& range = tmp.Ranges[i];
1873  for (int char_idx = 0; char_idx < range.num_chars; char_idx += 1)
1874  {
1875  const stbtt_packedchar& pc = range.chardata_for_range[char_idx];
1876  if (!pc.x0 && !pc.x1 && !pc.y0 && !pc.y1)
1877  continue;
1878 
1879  const int codepoint = range.first_unicode_codepoint_in_range + char_idx;
1880  if (cfg.MergeMode && dst_font->FindGlyphNoFallback((unsigned short)codepoint))
1881  continue;
1882 
1883  stbtt_aligned_quad q;
1884  float dummy_x = 0.0f, dummy_y = 0.0f;
1885  stbtt_GetPackedQuad(range.chardata_for_range, atlas->TexWidth, atlas->TexHeight, char_idx, &dummy_x, &dummy_y, &q, 0);
1886  dst_font->AddGlyph((ImWchar)codepoint, q.x0 + off_x, q.y0 + off_y, q.x1 + off_x, q.y1 + off_y, q.s0, q.t0, q.s1, q.t1, pc.xadvance);
1887  }
1888  }
1889  }
1890 
1891  // Cleanup temporaries
1892  ImGui::MemFree(buf_packedchars);
1893  ImGui::MemFree(buf_ranges);
1894  ImGui::MemFree(tmp_array);
1895 
1896  ImFontAtlasBuildFinish(atlas);
1897 
1898  return true;
1899 }

1951 static void ImFontAtlasBuildRenderDefaultTexData(ImFontAtlas* atlas)
1952 {
1953  IM_ASSERT(atlas->CustomRectIds[0] >= 0);
1954  IM_ASSERT(atlas->TexPixelsAlpha8 != NULL);
1955  ImFontAtlas::CustomRect& r = atlas->CustomRects[atlas->CustomRectIds[0]];
1956  IM_ASSERT(r.ID == FONT_ATLAS_DEFAULT_TEX_DATA_ID);
1957  IM_ASSERT(r.IsPacked());
1958 
1959  const int w = atlas->TexWidth;
1960  if (!(atlas->Flags & ImFontAtlasFlags_NoMouseCursors))
1961  {
1962  // Render/copy pixels
1963  IM_ASSERT(r.Width == FONT_ATLAS_DEFAULT_TEX_DATA_W_HALF * 2 + 1 && r.Height == FONT_ATLAS_DEFAULT_TEX_DATA_H);
1964  for (int y = 0, n = 0; y < FONT_ATLAS_DEFAULT_TEX_DATA_H; y++)
1965  for (int x = 0; x < FONT_ATLAS_DEFAULT_TEX_DATA_W_HALF; x++, n++)
1966  {
1967  const int offset0 = (int)(r.X + x) + (int)(r.Y + y) * w;
1968  const int offset1 = offset0 + FONT_ATLAS_DEFAULT_TEX_DATA_W_HALF + 1;
1969  atlas->TexPixelsAlpha8[offset0] = FONT_ATLAS_DEFAULT_TEX_DATA_PIXELS[n] == '.' ? 0xFF : 0x00;
1970  atlas->TexPixelsAlpha8[offset1] = FONT_ATLAS_DEFAULT_TEX_DATA_PIXELS[n] == 'X' ? 0xFF : 0x00;
1971  }
1972  }
1973  else
1974  {
1975  IM_ASSERT(r.Width == 2 && r.Height == 2);
1976  const int offset = (int)(r.X) + (int)(r.Y) * w;
1977  atlas->TexPixelsAlpha8[offset] = atlas->TexPixelsAlpha8[offset + 1] = atlas->TexPixelsAlpha8[offset + w] = atlas->TexPixelsAlpha8[offset + w + 1] = 0xFF;
1978  }
1979  atlas->TexUvWhitePixel = ImVec2((r.X + 0.5f) * atlas->TexUvScale.x, (r.Y + 0.5f) * atlas->TexUvScale.y);
1980 }
