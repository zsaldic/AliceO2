31 void broadcastMessage(FairMQDevice &device, o2::header::Stack &&headerStack, FairMQMessagePtr &&payloadMessage, int index) {
32  for (auto &channelInfo : device.fChannels) {
33  auto channel = channelInfo.first;
34  assert(channelInfo.second.size() == 1);
35  // FIXME: I need to make sure the input channel is not used... For the moment
36  // I rely on the convention dpl channels start with "from_".
37  if (strncmp(channel.data(), "from_", 5) != 0) {
38  continue;
39  }
41  if (strncmp(channel.data(), "from_internal-dpl-", 18) == 0) {
42  continue;
43  }
44 
45  // FIXME: this assumes there is only one output from here... This should
46  // really do the matchmaking between inputs and output channels.
47  auto channelAlloc = o2::pmr::getTransportAllocator(channelInfo.second[index].Transport());
48  FairMQMessagePtr headerMessage = o2::pmr::getMessage(std::move(headerStack), channelAlloc);
49 
50  FairMQParts out;
51  out.AddPart(std::move(headerMessage));
52  out.AddPart(std::move(payloadMessage));
53  device.Send(out, channel, index);
54  }
55 }

58 InjectorFunction o2DataModelAdaptor(OutputSpec const &spec, uint64_t startTime, uint64_t step) {
59  auto timesliceId = std::make_shared<size_t>(startTime);
60  return [timesliceId, step](FairMQDevice &device, FairMQParts &parts, int index) {
61  for (size_t i = 0; i < parts.Size()/2; ++i) {
62  auto dh = o2::header::get<DataHeader*>(parts.At(i * 2)->GetData());
63 
64  DataProcessingHeader dph{*timesliceId, 0};
65  o2::header::Stack headerStack{*dh, dph};
66  broadcastMessage(device, std::move(headerStack), std::move(parts.At(i*2+1)), index);
67  auto oldTimesliceId = *timesliceId;
68  *timesliceId += 1;
69  }
70  };

73 InjectorFunction incrementalConverter(OutputSpec const &spec, uint64_t startTime, uint64_t step) {
74  auto timesliceId = std::make_shared<size_t>(startTime);
75 
76  return [timesliceId, spec, step](FairMQDevice &device, FairMQParts &parts, int index) {
77  // We iterate on all the parts and we send them two by two,
78  // adding the appropriate O2 header.
79  for (size_t i = 0; i < parts.Size(); ++i) {
80  DataHeader dh;
81  dh.dataOrigin = spec.origin;
82  dh.dataDescription = spec.description;
83  dh.subSpecification = spec.subSpec;
84  dh.payloadSize = parts.At(i)->GetSize();
85 
86  DataProcessingHeader dph{*timesliceId, 0};
87  *timesliceId += step;
88  //we have to move the incoming data
89  o2::header::Stack headerStack{dh, dph};
90 
91  broadcastMessage(device, std::move(headerStack), std::move(parts.At(i)), index);
92  }
93  };
94 }

96 DataProcessorSpec specifyExternalFairMQDeviceProxy(char const *name,
97  std::vector<OutputSpec> const &outputs,
98  char const *channelConfig,
99  std::function<void(FairMQDevice &device,
100  FairMQParts& inputs,
101  int index)> converter) {
102  DataProcessorSpec spec;
103  spec.name = strdup(name);
104  spec.inputs = {};
105  spec.outputs = outputs;
106  // The Init method will register a new "Out of band" channel and
107  // attach an OnData to it which is responsible for converting incoming
108  // messages into DPL messages.
109  spec.algorithm = AlgorithmSpec{[converter, channel = spec.name](InitContext &ctx) {
110  auto device = ctx.services().get<RawDeviceService>().device();
111  assert(device);
112 
113  // Converter should pump messages
114  auto handler = [device,converter](FairMQParts &inputs, int idx) {
115  converter(*device, inputs, idx);
116  return true;
117  };
118 
119  device->OnData(channel, handler);
120  return [](ProcessingContext &) {};
121  }
122  };
123  const char *d = strdup((std::string("name=") + name + "," + channelConfig).c_str());
124  spec.options = {
125  ConfigParamSpec{"channel-config", VariantType::String, d, {"Out-of-band channel config"}}
126  };
127  return spec;
128 }
