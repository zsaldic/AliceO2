66 TGeoVolume* Ladder::createVolume()
67 {
68 
69  Int_t nChips = mSegmentation->getNSensors();
70 
71  // Create the flex
72  mFlex = new Flex(mSegmentation);
73  Double_t flexLength = nChips * (SegmentationAlpide::SensorSizeCols + Geometry::sSensorInterspace) +
74  Geometry::sLadderOffsetToEnd + Geometry::sSensorSideOffset;
75  Double_t shiftY =
76  4 * Geometry::sSensorTopOffset + SegmentationAlpide::SensorSizeRows - Geometry::sFlexHeight / 2; // to be verified!!
77  TGeoVolumeAssembly* flexVol = mFlex->makeFlex(mSegmentation->getNSensors(), flexLength);
78  mLadderVolume->AddNode(flexVol, 1, new TGeoTranslation(flexLength / 2 + Geometry::sSensorSideOffset / 2, shiftY,
79  Geometry::sFlexThickness / 2 - 2 * (Geometry::sKaptonOnCarbonThickness + Geometry::sKaptonGlueThickness)));
80 
81  // Create the CMOS Sensors
82  createSensors();
83 
84  return mLadderVolume;
85 }

90 void Ladder::createSensors()
91 {
92 
93  Geometry* mftGeom = Geometry::instance();
94 
95  // Create Shapes
96 
97  // sensor = sensitive volume
98  TString namePrefixS = "MFTSensor";
99 
100  // chip = sensor + readout
101  TString namePrefixC =
102  Form("MFT_C_%d_%d_%d", mftGeom->getHalfID(mSegmentation->GetUniqueID()),
103  mftGeom->getDiskID(mSegmentation->GetUniqueID()), mftGeom->getLadderID(mSegmentation->GetUniqueID()));
104 
105  // the MFT glue
106  TString namePrefixG =
107  Form("MFT_G_%d_%d_%d", mftGeom->getHalfID(mSegmentation->GetUniqueID()),
108  mftGeom->getDiskID(mSegmentation->GetUniqueID()), mftGeom->getLadderID(mSegmentation->GetUniqueID()));
109 
110  TGeoMedium* kMedGlue = gGeoManager->GetMedium("MFT_SE4445$");
111 
112  TGeoVolume* glue = gGeoManager->MakeBox(
113  namePrefixG.Data(), kMedGlue, (SegmentationAlpide::SensorSizeCols - Geometry::sGlueEdge) / 2.,
114  (SegmentationAlpide::SensorSizeRows - Geometry::sGlueEdge) / 2., Geometry::sGlueThickness / 2.);
115  glue->SetVisibility(kTRUE);
116  glue->SetLineColor(kRed - 10);
117  glue->SetLineWidth(1);
118  glue->SetFillColor(glue->GetLineColor());
119  glue->SetFillStyle(4000); // 0% transparent
120 
121  // common with ITS
122  TGeoVolume* chipVol = AlpideChip::createChip(Geometry::sChipThickness / 2., Geometry::sSensorThickness / 2.,
123  namePrefixC, namePrefixS, kFALSE);
124 
125  // chipVol->Print();
126 
127  for (int ichip = 0; ichip < mSegmentation->getNSensors(); ichip++) {
128 
129  ChipSegmentation* chipSeg = mSegmentation->getSensor(ichip);
130  TGeoCombiTrans* chipPos = chipSeg->getTransformation();
131  TGeoCombiTrans* chipPosGlue = chipSeg->getTransformation();
132 
133  // Position of the center on the chip in the chip coordinate system
134  Double_t pos[3] = { SegmentationAlpide::SensorSizeCols / 2., SegmentationAlpide::SensorSizeRows / 2.,
135  Geometry::sChipThickness / 2. - Geometry::sGlueThickness - 2 * (Geometry::sKaptonOnCarbonThickness + Geometry::sKaptonGlueThickness) };
136 
137  Double_t posglue[3] = { SegmentationAlpide::SensorSizeCols / 2., SegmentationAlpide::SensorSizeRows / 2.,
138  Geometry::sGlueThickness / 2 - Geometry::sChipThickness - 2 * (Geometry::sKaptonOnCarbonThickness + Geometry::sKaptonGlueThickness) };
139 
140  Double_t master[3];
141  Double_t masterglue[3];
142  chipPos->LocalToMaster(pos, master);
143  chipPosGlue->LocalToMaster(posglue, masterglue);
144 
145  TGeoBBox* shape = (TGeoBBox*)mLadderVolume->GetShape();
146  master[0] -= shape->GetDX();
147  master[1] -= shape->GetDY();
148  master[2] -= shape->GetDZ();
149 
150  masterglue[0] -= shape->GetDX();
151  masterglue[1] -= shape->GetDY();
152  masterglue[2] -= shape->GetDZ();
153 
154  LOG(DEBUG1) << "CreateSensors " << Form("adding chip %s_%d ", namePrefixS.Data(), ichip) << FairLogger::endl;
155  // chipPos->Print();
156 
157  TGeoTranslation* trans = new TGeoTranslation(master[0], master[1], master[2]);
158  TGeoHMatrix* final = new TGeoHMatrix((*trans) * (Geometry::sTransMFT2ITS).Inverse());
159  mLadderVolume->AddNode(chipVol, ichip, final);
160  mLadderVolume->AddNode(glue, ichip, new TGeoTranslation(masterglue[0], masterglue[1], masterglue[2]));
161  }
162 }
