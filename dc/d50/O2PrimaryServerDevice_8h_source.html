135  bool HandleRequest(FairMQMessagePtr& request, int /*index*/)
136  {
137  LOG(INFO) << "GOT A REQUEST WITH SIZE " << request->GetSize();
138  std::string requeststring(static_cast<char*>(request->GetData()), request->GetSize());
139 
140  if (requeststring.compare("configrequest") == 0) {
141  return HandleConfigRequest(request);
142  }
143 
144  else if (requeststring.compare("primrequest") != 0) {
145  LOG(INFO) << "unknown request\n";
146  return true;
147  }
148 
149  // we only need the initialized generator at this moment
150  if (mGeneratorInitThread.joinable()) {
151  mGeneratorInitThread.join();
152  }
153 
154  static int counter = 0;
155  if (counter >= mMaxEvents && mNeedNewEvent) {
156  return false;
157  }
158 
159  LOG(INFO) << "Received request for work ";

161  mStack.Reset();
162  mPrimGen.GenerateEvent(&mStack);
163  mNeedNewEvent = false;
164  mPartCounter = 0;
165  counter++;
166  }
167 
168  auto& prims = mStack.getPrimaries();
169  auto numberofparts = (int)std::ceil(prims.size() / (1. * mChunkGranularity));
170  // number of parts should be at least 1 (even if empty)
171  numberofparts = std::max(1, numberofparts);
172 
173  o2::Data::PrimaryChunk m;
174  o2::Data::SubEventInfo i;
175  i.eventID = counter;
176  i.maxEvents = mMaxEvents;
177  i.part = mPartCounter + 1;
178  i.nparts = numberofparts;
179  i.seed = counter + mInitialSeed;
180  i.index = m.mParticles.size();
181  i.mMCEventHeader = mEventHeader;
182  m.mSubEventInfo = i;
183 
184  //auto startoffset = (mPartCounter + 1) * mChunkGranularity;

186  //auto startiter = prims.begin() + mPartCounter * mChunkGranularity;

191  int endindex = prims.size() - mPartCounter * mChunkGranularity;
192  int startindex = prims.size() - (mPartCounter + 1) * mChunkGranularity;
193  if (startindex < 0) {
