50 void Digitizer::process(const std::vector<Hit>* hits, int evID, int srcID)
51 {
52  // digitize single event, the time must have been set beforehand
53 
54  LOG(INFO) << "Digitizing " << mGeometry->getName() << " hits of entry " << evID << " from source "
55  << srcID << " at time " << mEventTime + mParams.getTimeOffset() << " (TOff.= "
56  << mParams.getTimeOffset() << " ROFrame= " << mNewROFrame << ")"
57  << " cont.mode: " << isContinuous()
58  << " Min/Max ROFrames " << mROFrameMin << "/" << mROFrameMax << FairLogger::endl;
59 
60  // is there something to flush ?
61  if (mNewROFrame > mROFrameMin) {
62  fillOutputContainer(mNewROFrame - 1); // flush out all frame preceding the new one
63  }
64 
65  int nHits = hits->size();
66  std::vector<int> hitIdx(nHits);
67  std::iota(std::begin(hitIdx), std::end(hitIdx), 0);
68  // sort hits to improve memory access
69  std::sort(hitIdx.begin(), hitIdx.end(),
70  [hits](auto lhs, auto rhs) {
71  return (*hits)[lhs].GetDetectorID() < (*hits)[rhs].GetDetectorID();
72  });
73  for (int i : hitIdx) {
74  processHit((*hits)[i], mROFrameMax, evID, srcID);
75  }
76  // in the triggered mode store digits after every MC event
77  // TODO: in the real triggered mode this will not be needed, this is actually for the
78  // single event processing only
79  if (!mParams.isContinuous()) {
80  fillOutputContainer(mROFrameMax);
81  }
82 }

85 void Digitizer::setEventTime(double t)
86 {
87  // assign event time in ns
88  mEventTime = t;
89  // to randomize the RO phase wrt the event time we use a random offset
90  if (mParams.isContinuous()) { // in continuous mode we set the offset only in the very beginning
91  if (!mParams.isTimeOffsetSet()) { // offset is initially at -inf
92  mParams.setTimeOffset(0);
93  }
94  } else { // in the triggered mode we start from 0 ROFrame in every event, is this correct?
95  mParams.setTimeOffset(mEventTime); // + mParams.getROFrameLength() * (gRandom->Rndm() - 0.5));
96  mROFrameMin = 0; // so we reset the frame counters
97  mROFrameMax = 0;
98  }
99 
100  mEventTime -= mParams.getTimeOffset(); // subtract common offset
101  if (mEventTime < 0.) {
102  mEventTime = 0.;
103  } else if (mEventTime > UINT_MAX * mParams.getROFrameLength()) {
104  LOG(FATAL) << "ROFrame for event time " << t << " exceeds allowe maximum " << UINT_MAX << FairLogger::endl;
105  }
106 
107  // RO frame corresponding to provided time
108  mNewROFrame = static_cast<UInt_t>(mEventTime * mParams.getROFrameLengthInv());
109 
110  if (mNewROFrame < mROFrameMin) {
111  LOG(FATAL) << "New ROFrame (time=" << t << ") precedes currently cashed " << mROFrameMin << FairLogger::endl;
112  }
113 
114  if (mParams.isContinuous() && mROFrameMax < mNewROFrame) {
115  mROFrameMax = mNewROFrame - 1; // all frames up to this are finished
116  }
117 }

120 void Digitizer::fillOutputContainer(UInt_t frameLast)
121 {
122  // fill output with digits from min.cached up to requested frame, generating the noise beforehand
123  if (frameLast > mROFrameMax) {
124  frameLast = mROFrameMax;
125  }
126  // make sure all buffers for extra digits are created up to the maxFrame
127  getExtraDigBuffer(mROFrameMax);
128 
129  LOG(INFO) << "Filling " << mGeometry->getName() << " digits output for RO frames " << mROFrameMin << ":"
130  << frameLast << FairLogger::endl;
131 
132  o2::ITSMFT::ROFRecord rcROF;
133 
134  // we have to write chips in RO increasing order, therefore have to loop over the frames here
135  for (; mROFrameMin <= frameLast; mROFrameMin++) {
136  rcROF.setROFrame(mROFrameMin);
137  rcROF.getROFEntry().setIndex(mDigits->size()); // start of current ROF in digits
138 
139  auto& extra = *(mExtraBuff.front().get());
140  for (auto& chip : mChips) {
141  chip.addNoise(mROFrameMin, mROFrameMin, &mParams);
142  auto& buffer = chip.getPreDigits();
143  if (buffer.empty()) {
144  continue;
145  }
146  auto itBeg = buffer.begin();
147  auto iter = itBeg;
148  ULong64_t maxKey = chip.getOrderingKey(mROFrameMin + 1, 0, 0) - 1; // fetch digits with key below that
149  for (; iter != buffer.end(); ++iter) {
150  if (iter->first > maxKey) {
151  break; // is the digit ROFrame from the key > the max requested frame
152  }
153  auto& preDig = iter->second; // preDigit
154  if (preDig.charge >= mParams.getChargeThreshold()) {
155  int digID = mDigits->size();
156  mDigits->emplace_back(chip.getChipIndex(), mROFrameMin, preDig.row, preDig.col, preDig.charge);
157  mMCLabels->addElement(digID, preDig.labelRef.label);
158  auto& nextRef = preDig.labelRef; // extra contributors are in extra array
159  while (nextRef.next >= 0) {
160  nextRef = extra[nextRef.next];
161  mMCLabels->addElement(digID, nextRef.label);
162  }
163  }
164  }
165  buffer.erase(itBeg, iter);
166  }
167  // finalize ROF record
168  rcROF.setNROFEntries(mDigits->size() - rcROF.getROFEntry().getIndex()); // number of digits
169  rcROF.getBCData().setFromNS(mROFrameMin * mParams.getROFrameLength() + mParams.getTimeOffset());
170  if (mROFRecords) {
171  mROFRecords->push_back(rcROF);
172  }
173  extra.clear(); // clear container for extra digits of the mROFrameMin ROFrame
174  // and move it as a new slot in the end
175  mExtraBuff.emplace_back(mExtraBuff.front().release());
176  mExtraBuff.pop_front();
177  }
178 }

181 void Digitizer::processHit(const o2::ITSMFT::Hit& hit, UInt_t& maxFr, int evID, int srcID)
182 {
183  // convert single hit to digits
184 
185  double hTime0 = hit.GetTime() * sec2ns + mEventTime; // time from the RO start, in ns
186 
187  // calculate RO Frame for this hit
188  if (hTime0 < 0) {
189  hTime0 = 0.;
190  }
191  float tTot = mParams.getSignalShape().getMaxDuration();
192  // frame of the hit signal start
193  UInt_t roFrame = UInt_t(hTime0 * mParams.getROFrameLengthInv());
194  // frame of the hit signal end: in the triggered mode we read just 1 frame
195  UInt_t roFrameMax = mParams.isContinuous() ? UInt_t((hTime0 + tTot) * mParams.getROFrameLengthInv()) : roFrame;
196  int nFrames = roFrameMax + 1 - roFrame;
197  if (roFrameMax > maxFr) {
198  maxFr = roFrameMax; // if signal extends beyond current maxFrame, increase the latter
199  }
200  // delay of the signal start wrt 1st ROF start
201  float timeInROF = float(hTime0 - (roFrame * mParams.getROFrameLength()));
202 
203  // here we start stepping in the depth of the sensor to generate charge diffision
204  float nStepsInv = mParams.getNSimStepsInv();
205  int nSteps = mParams.getNSimSteps();
206  const auto& matrix = mGeometry->getMatrixL2G(hit.GetDetectorID());
207  Vector3D<float> xyzLocS(matrix ^ (hit.GetPosStart())); // start position in sensor frame
208  Vector3D<float> xyzLocE(matrix ^ (hit.GetPos())); // end position in sensor frame
209  Vector3D<float> step(xyzLocE);
210  step -= xyzLocS;
211  step *= nStepsInv; // position increment at each step
212  // the electrons will injected in the middle of each step
213  Vector3D<float> stepH(step * 0.5);
214  xyzLocS += stepH;
215  xyzLocE -= stepH;
216 
217  int rowS = -1, colS = -1, rowE = -1, colE = -1, nSkip = 0;
218  // get entrance pixel row and col
219  while (!Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), rowS, colS)) { // guard-ring ?
220  if (++nSkip >= nSteps) {
221  return; // did not enter to sensitive matrix
222  }
223  xyzLocS += step;
224  }
225  // get exit pixel row and col
226  while (!Segmentation::localToDetector(xyzLocE.X(), xyzLocE.Z(), rowE, colE)) { // guard-ring ?
227  if (++nSkip >= nSteps) {
228  return; // did not enter to sensitive matrix
229  }
230  xyzLocE -= step;
231  }
232  // estimate the limiting min/max row and col where the non-0 response is possible
233  if (rowS > rowE) {
234  std::swap(rowS, rowE);
235  }
236  if (colS > colE) {
237  std::swap(colS, colE);
238  }
239  rowS -= AlpideRespSimMat::NPix / 2;
240  rowE += AlpideRespSimMat::NPix / 2;
241  if (rowS < 0) {
242  rowS = 0;
243  }
244  if (rowE >= Segmentation::NRows) {
245  rowE = Segmentation::NRows - 1;
246  }
247  colS -= AlpideRespSimMat::NPix / 2;
248  colE += AlpideRespSimMat::NPix / 2;
249  if (colS < 0) {
250  colS = 0;
251  }
252  if (colE >= Segmentation::NCols) {
253  colE = Segmentation::NCols - 1;
254  }
255  int rowSpan = rowE - rowS + 1, colSpan = colE - colS + 1; // size of plaquet where some response is expected
256 
257  float respMatrix[rowSpan][colSpan]; // response accumulated here
258  std::fill(&respMatrix[0][0], &respMatrix[0][0] + rowSpan * colSpan, 0.f);
259 
260  float nElectrons = hit.GetEnergyLoss() * mParams.getEnergyToNElectrons(); // total number of deposited electrons
261  nElectrons *= nStepsInv; // N electrons injected per step
262  if (nSkip) {
263  nSteps -= nSkip;
264  }
265  //
266  int rowPrev = -1, colPrev = -1, row, col;
267  float cRowPix = 0.f, cColPix = 0.f; // local coordinated of the current pixel center
268 
269  const o2::ITSMFT::AlpideSimResponse* resp = mParams.getAlpSimResponse();
270 
271  // take into account that the AlpideSimResponse has min/max thickness non-symmetric around 0
272  xyzLocS.SetY(xyzLocS.Y() + resp->getDepthShift());
273 
274  // collect charge in evey pixel which might be affected by the hit
275  for (int iStep = nSteps; iStep--;) {
276  // Get the pixel ID
277  Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), row, col);
278  if (row != rowPrev || col != colPrev) { // update pixel and coordinates of its center
279  if (!Segmentation::detectorToLocal(row, col, cRowPix, cColPix)) {
280  continue; // should not happen
281  }
282  rowPrev = row;
283  colPrev = col;
284  }
285  bool flipCol, flipRow;
286  // note that response needs coordinates along column row (locX) (locZ) then depth (locY)
287  auto rspmat = resp->getResponse(xyzLocS.X() - cRowPix, xyzLocS.Z() - cColPix, xyzLocS.Y(), flipRow, flipCol);
288 
289  xyzLocS += step;
290  if (!rspmat) {
291  continue;
292  }
293 
294  for (int irow = AlpideRespSimMat::NPix; irow--;) {
295  int rowDest = row + irow - AlpideRespSimMat::NPix / 2 - rowS; // destination row in the respMatrix
296  if (rowDest < 0 || rowDest >= rowSpan) {
297  continue;
298  }
299  for (int icol = AlpideRespSimMat::NPix; icol--;) {
300  int colDest = col + icol - AlpideRespSimMat::NPix / 2 - colS; // destination column in the respMatrix
301  if (colDest < 0 || colDest >= colSpan) {
302  continue;
303  }
304  respMatrix[rowDest][colDest] += rspmat->getValue(irow, icol, flipRow, flipCol);
305  }
306  }
307  }
308 
309  // fire the pixels assuming Poisson(n_response_electrons)
310  o2::MCCompLabel lbl(hit.GetTrackID(), evID, srcID);
311  auto& chip = mChips[hit.GetDetectorID()];
312 
313  for (int irow = rowSpan; irow--;) {
314  UShort_t rowIS = irow + rowS;
315  for (int icol = colSpan; icol--;) {
316  float nEleResp = respMatrix[irow][icol];
317  if (!nEleResp) {
318  continue;
319  }
320  int nEle = gRandom->Poisson(nElectrons * nEleResp); // total charge in given pixel
321  // ignore charge which have no chance to fire the pixel
322  if (nEle < mParams.getMinChargeToAccount()) {
323  continue;
324  }
325  UShort_t colIS = icol + colS;
326  //
327  registerDigits(chip, roFrame, timeInROF, nFrames, rowIS, colIS, nEle, lbl);
328  }
329  }
330 }

333 void Digitizer::registerDigits(ChipDigitsContainer& chip, UInt_t roFrame, float tInROF, int nROF,
334  UShort_t row, UShort_t col, int nEle, o2::MCCompLabel& lbl)
335 {
336  // Register digits for given pixel, accounting for the possible signal contribution to
337  // multiple ROFrame. The signal starts at time tInROF wrt the start of provided roFrame
338  // In every ROFrame we check the collected signal during strobe
339 
340  float tStrobe = mParams.getStrobeDelay() - tInROF; // strobe start wrt signal start
341  for (int i = 0; i < nROF; i++) {
342  UInt_t roFr = roFrame + i;
343  int nEleROF = mParams.getSignalShape().getCollectedCharge(nEle, tStrobe, tStrobe + mParams.getStrobeLength());
344  tStrobe += mParams.getROFrameLength(); // for the next ROF
345 
346  // discard too small contributions, they have no chance to produce a digit
347  if (nEleROF < mParams.getMinChargeToAccount()) {
348  continue;
349  }
350  if (roFr > mEventROFrameMax)
351  mEventROFrameMax = roFr;
352  if (roFr < mEventROFrameMin)
353  mEventROFrameMin = roFr;
354  auto key = chip.getOrderingKey(roFr, row, col);
355  PreDigit* pd = chip.findDigit(key);
356  if (!pd) {
357  chip.addDigit(key, roFr, row, col, nEleROF, lbl);
358  } else { // there is already a digit at this slot, account as PreDigitExtra contribution
359  pd->charge += nEleROF;
360  if (pd->labelRef.label == lbl) { // don't store the same label twice
361  continue;
362  }
363  ExtraDig* extra = getExtraDigBuffer(roFr);
364  int& nxt = pd->labelRef.next;
365  bool skip = false;
366  while (nxt >= 0) {
367  if ((*extra)[nxt].label == lbl) { // don't store the same label twice
368  skip = true;
369  break;
370  }
371  nxt = (*extra)[nxt].next;
372  }
373  if (skip) {
374  continue;
375  }
376  // new predigit will be added in the end of the chain
377  nxt = extra->size();
378  extra->emplace_back(lbl);
379  }
380  }
