39 VertexerBase::VertexerBase(const ROframe& event) : mEvent{ event }, mAverageClustersRadii{ std::array<float, 3>{ 0.f, 0.f, 0.f } }
40 {
41  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {
42  const auto& currentLayer{ event.getClustersOnLayer(iLayer) };
43  const size_t clustersNum{ currentLayer.size() };
44  mClusters[iLayer].clear();
45  if (clustersNum > mClusters[iLayer].capacity()) {
46  mClusters[iLayer].reserve(clustersNum);
47  }
48  for (size_t iCluster{ 0 }; iCluster < clustersNum; ++iCluster) {
49  mClusters[iLayer].emplace_back(iLayer, currentLayer.at(iCluster));
50  }
51  if (mClusters[iLayer].size() != 0) {
52  const float inverseNumberOfClusters{ 1.f / mClusters[iLayer].size() };
53  for (auto& cluster : mClusters[iLayer]) {
54  mAverageClustersRadii[iLayer] += cluster.rCoordinate;
55  }
56  mAverageClustersRadii[iLayer] *= inverseNumberOfClusters;
57  } else {
58  mAverageClustersRadii[iLayer] = LayersRCoordinate()[iLayer];
59  }
60  }
61  mDeltaRadii10 = mAverageClustersRadii[1] - mAverageClustersRadii[0];
62  mDeltaRadii21 = mAverageClustersRadii[2] - mAverageClustersRadii[1];
63  mMaxDirectorCosine3 =
64  LayersZCoordinate()[2] / std::sqrt(LayersZCoordinate()[2] * LayersZCoordinate()[2] +
65  (mDeltaRadii10 + mDeltaRadii21) * (mDeltaRadii10 + mDeltaRadii21));
66 }

160 void VertexerBase::findTracklets(const bool useMCLabel)
161 {
162  if (mVertexerInitialised) {
163  std::vector<std::pair<int, int>> clusters0, clusters2;
164  std::vector<bool> usedCluster2Flags, usedCluster0Flags;
165  usedCluster2Flags.resize(mClusters[2].size(), false);
166  usedCluster0Flags.resize(mClusters[0].size(), false);
167 
168  for (int iBin1{ 0 }; iBin1 < ZBins * PhiBins; ++iBin1) {
169 
170  int ZBinLow0{ std::max(0, ZBins - static_cast<int>(std::ceil((ZBins - iBin1 % ZBins + 1) *
171  (mDeltaRadii21 + mDeltaRadii10) / mDeltaRadii10))) };
172  int ZBinHigh0{ std::min(
173  static_cast<int>(std::ceil((iBin1 % ZBins + 1) * (mDeltaRadii21 + mDeltaRadii10) / mDeltaRadii21)),
174  ZBins - 1) };
175  // int ZBinLow0 { 0 };
176  // int ZBinHigh0 { ZBins -1 };
177 
178  int PhiBin1{ static_cast<int>(iBin1 / ZBins) };
179  clusters0 = selectClusters(
180  mIndexTables[0],
181  std::array<int, 4>{ ZBinLow0, (PhiBin1 - mPhiSpan < 0) ? PhiBins + (PhiBin1 - mPhiSpan) : PhiBin1 - mPhiSpan,
182  ZBinHigh0,
183  (PhiBin1 + mPhiSpan > PhiBins) ? PhiBin1 + mPhiSpan - PhiBins : PhiBin1 + mPhiSpan });
184 
185  for (int iCluster1{ mIndexTables[1][iBin1] }; iCluster1 < mIndexTables[1][iBin1 + 1]; ++iCluster1) {
186  bool trackFound = false;
187  for (int iRow0{ 0 }; iRow0 < clusters0.size(); ++iRow0) {
188  for (int iCluster0{ std::get<0>(clusters0[iRow0]) };
189  iCluster0 < std::get<0>(clusters0[iRow0]) + std::get<1>(clusters0[iRow0]); ++iCluster0) {
190  if (usedCluster0Flags[iCluster0])
191  continue;
192  if (std::abs(mClusters[0][iCluster0].phiCoordinate - mClusters[1][iCluster1].phiCoordinate) < mPhiCut ||
193  std::abs(mClusters[0][iCluster0].phiCoordinate - mClusters[1][iCluster1].phiCoordinate) >
194  TwoPi - mPhiCut) {
195  float ZProjection{ mClusters[0][iCluster0].zCoordinate +
196  (mAverageClustersRadii[2] - mClusters[0][iCluster0].rCoordinate) *
197  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
198  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate) };
199  float ZProjectionInner{ mClusters[0][iCluster0].zCoordinate +
200  (mClusters[0][iCluster0].rCoordinate) *
201  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
202  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate) };
203  if (std::abs(ZProjection) > (LayersZCoordinate()[0] + mZCut))
204  continue;
205  int ZProjectionBin{ (ZProjection < -LayersZCoordinate()[0])
206  ? 0
207  : (ZProjection > LayersZCoordinate()[0]) ? ZBins - 1
208  : getZBinIndex(2, ZProjection) };
209  int ZBinLow2{ (ZProjectionBin - mZSpan < 0) ? 0 : ZProjectionBin - mZSpan };
210  int ZBinHigh2{ (ZProjectionBin + mZSpan > ZBins - 1) ? ZBins - 1 : ZProjectionBin + mZSpan };
211  // int ZBinLow2 { 0 };
212  // int ZBinHigh2 { ZBins - 1 };
213  int PhiBinLow2{ (PhiBin1 - mPhiSpan < 0) ? PhiBins + PhiBin1 - mPhiSpan : PhiBin1 - mPhiSpan };
214  int PhiBinHigh2{ (PhiBin1 + mPhiSpan > PhiBins - 1) ? PhiBin1 + mPhiSpan - PhiBins : PhiBin1 + mPhiSpan };
215  // int PhiBinLow2{ 0 };
216  // int PhiBinHigh2{ PhiBins - 1 };
217  clusters2 =
218  selectClusters(mIndexTables[2], std::array<int, 4>{ ZBinLow2, PhiBinLow2, ZBinHigh2, PhiBinHigh2 });
219  for (int iRow2{ 0 }; iRow2 < clusters2.size(); ++iRow2) {
220  for (int iCluster2{ std::get<0>(clusters2[iRow2]) };
221  iCluster2 < std::get<0>(clusters2[iRow2]) + std::get<1>(clusters2[iRow2]); ++iCluster2) {
222  if (usedCluster2Flags[iCluster2])
223  continue;
224  float ZProjectionRefined{
225  mClusters[0][iCluster0].zCoordinate +
226  (mClusters[2][iCluster2].rCoordinate - mClusters[0][iCluster0].rCoordinate) *
227  (mClusters[1][iCluster1].zCoordinate - mClusters[0][iCluster0].zCoordinate) /
228  (mClusters[1][iCluster1].rCoordinate - mClusters[0][iCluster0].rCoordinate)
229  };
230  bool testMC{ !useMCLabel ||
231  (mEvent.getClusterLabels(0, mClusters[0][iCluster0]).getTrackID() ==
232  mEvent.getClusterLabels(2, mClusters[2][iCluster2]).getTrackID() &&
233  mEvent.getClusterLabels(0, mClusters[0][iCluster0]).getTrackID() ==
234  mEvent.getClusterLabels(1, mClusters[1][iCluster1]).getTrackID()) };
235  float absDeltaPhi{ std::abs(mClusters[2][iCluster2].phiCoordinate -
236  mClusters[1][iCluster1].phiCoordinate) };
237  float absDeltaZ{ std::abs(mClusters[2][iCluster2].zCoordinate - ZProjectionRefined) };
238  if (absDeltaZ < mZCut && ((absDeltaPhi < mPhiCut || std::abs(absDeltaPhi - TwoPi) < mPhiCut) && testMC)) {
239  mTracklets.emplace_back(Line{
240  std::array<float, 3>{ mClusters[0][iCluster0].xCoordinate, mClusters[0][iCluster0].yCoordinate,
241  mClusters[0][iCluster0].zCoordinate },
242  std::array<float, 3>{ mClusters[1][iCluster1].xCoordinate, mClusters[1][iCluster1].yCoordinate,
243  mClusters[1][iCluster1].zCoordinate } });
244  if (std::abs(mTracklets.back().cosinesDirector[2]) < mMaxDirectorCosine3) {
245  usedCluster0Flags[iCluster0] = true;
246  usedCluster2Flags[iCluster2] = true;
247  trackFound = true;
248  break;
249  } else {
250  mTracklets.pop_back();
251  }
252  }
253  }
254  if (trackFound)
255  break;
256  }
257  }
258  if (trackFound)
259  break;
260  }
261  if (trackFound)
262  break;
263  }
264  }
265  }
266  mTrackletsFound = true;
267  }

270 void VertexerBase::findVertices()
271 {
272  if (mTrackletsFound) {
273  const int numTracklets{ static_cast<int>(mTracklets.size()) };
274  std::vector<bool> usedTracklets{};
275  usedTracklets.resize(mTracklets.size(), false);
276  for (int tracklet1{ 0 }; tracklet1 < numTracklets; ++tracklet1) {
277  if (usedTracklets[tracklet1])
278  continue;
279  for (int tracklet2{ tracklet1 + 1 }; tracklet2 < numTracklets; ++tracklet2) {
280  if (usedTracklets[tracklet2])
281  continue;
282  if (Line::getDCA(mTracklets[tracklet1], mTracklets[tracklet2]) <= mPairCut) {
283  mTrackletClusters.emplace_back(tracklet1, mTracklets[tracklet1], tracklet2, mTracklets[tracklet2]);
284  std::array<float, 3> tmpVertex{ mTrackletClusters.back().getVertex() };
285  if (tmpVertex[0] * tmpVertex[0] + tmpVertex[1] * tmpVertex[1] > 4.f) {
286  mTrackletClusters.pop_back();
287  break;
288  }
289  usedTracklets[tracklet1] = true;
290  usedTracklets[tracklet2] = true;
291  for (int tracklet3{ 0 }; tracklet3 < numTracklets; ++tracklet3) {
292  if (usedTracklets[tracklet3])
293  continue;
294  if (Line::getDistanceFromPoint(mTracklets[tracklet3], tmpVertex) < mPairCut) {
295  mTrackletClusters.back().add(tracklet3, mTracklets[tracklet3]);
296  usedTracklets[tracklet3] = true;
297  tmpVertex = mTrackletClusters.back().getVertex();
298  }
299  }
300  break;
301  }
302  }
303  }
304  std::sort(mTrackletClusters.begin(), mTrackletClusters.end(),
305  [](ClusterLines& cluster1, ClusterLines& cluster2) { return cluster1.getSize() > cluster2.getSize(); });
306  int noClusters{ static_cast<int>(mTrackletClusters.size()) };
307  for (int iCluster1{ 0 }; iCluster1 < noClusters; ++iCluster1) {
308  std::array<float, 3> vertex1{ mTrackletClusters[iCluster1].getVertex() };
309  std::array<float, 3> vertex2{};
310  for (int iCluster2{ iCluster1 + 1 }; iCluster2 < noClusters; ++iCluster2) {
311  vertex2 = mTrackletClusters[iCluster2].getVertex();
312  if (std::abs(vertex1[2] - vertex2[2]) < mClusterCut) {
313 
314  float distance{ (vertex1[0] - vertex2[0]) * (vertex1[0] - vertex2[0]) +
315  (vertex1[1] - vertex2[1]) * (vertex1[1] - vertex2[1]) +
316  (vertex1[2] - vertex2[2]) * (vertex1[2] - vertex2[2]) };
317  if (distance <= mPairCut * mPairCut) {
318  for (auto label : mTrackletClusters[iCluster2].getLabels()) {
319  mTrackletClusters[iCluster1].add(label, mTracklets[label]);
320  vertex1 = mTrackletClusters[iCluster1].getVertex();
321  }
322  }
323  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster2);
324  --iCluster2;
325  --noClusters;
326  }
327  }
328  }
329  for (int iCluster{ 0 }; iCluster < noClusters; ++iCluster) {
330  if (mTrackletClusters[iCluster].getSize() < mClusterContributorsCut && noClusters > 1) {
331  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster);
332  noClusters--;
333  continue;
334  }
335  float dist{ 0. };
336  for (auto& line : mTrackletClusters[iCluster].mLines) {
337  dist += Line::getDistanceFromPoint(line, mTrackletClusters[iCluster].getVertex()) /
338  mTrackletClusters[iCluster].getSize();
339  }
340  if (mTrackletClusters[iCluster].getVertex()[0] * mTrackletClusters[iCluster].getVertex()[0] +
341  mTrackletClusters[iCluster].getVertex()[1] * mTrackletClusters[iCluster].getVertex()[1] <
342  1.98 * 1.98) {
343  mVertices.emplace_back(
344  Point3D<float>{ mTrackletClusters[iCluster].getVertex()[0], mTrackletClusters[iCluster].getVertex()[1],
345  mTrackletClusters[iCluster].getVertex()[2] },
346  mTrackletClusters[iCluster].getRMS2(), // Symm matrix. Diagonal: RMS2 components,
347  // off-diagonal: square mean of projections on planes.
348  mTrackletClusters[iCluster].getSize(), // Contributors
349  mTrackletClusters[iCluster].getAvgDistance2() // In place of chi2
350  );
351  mVertices.back().setTimeStamp(mROFrame);
352  mEvent.addPrimaryVertex(mVertices.back().getX(), mVertices.back().getY(), mVertices.back().getZ());
353  }
354  }
355  }
