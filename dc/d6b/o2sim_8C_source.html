31 FairRunSim* o2sim_init(bool asservice)
32 {
33  auto& confref = o2::conf::SimConfig::Instance();
34  // we can read from CCDB (for the moment faking with a TFile)
35  // o2::conf::ConfigurableParam::fromCCDB("params_ccdb.root", runid);
36 
37  // update the parameters from stuff given at command line
38  o2::conf::ConfigurableParam::updateFromString(confref.getKeyValueString());
39  // write the configuration file
40  o2::conf::ConfigurableParam::writeINI("o2sim_configuration.ini");
41 
42  // we can update the binary CCDB entry something like this ( + timestamp key )
43  // o2::conf::ConfigurableParam::toCCDB("params_ccdb.root");
44 
45  // set seed
46  auto seed = o2::utils::RngHelper::setGRandomSeed(confref.getStartSeed());
47  LOG(INFO) << "RNG INITIAL SEED " << seed;
48 
49  auto genconfig = confref.getGenerator();
50  FairRunSim* run = new o2::steer::O2RunSim(asservice);
51  run->SetImportTGeoToVMC(false); // do not import TGeo to VMC since the latter is built together with TGeo
52  run->SetSimSetup([confref]() { o2::SimSetup::setup(confref.getMCEngine().c_str()); });
53 
54  auto pid = getpid();
55  std::stringstream s;
56  s << confref.getOutPrefix();
57  if (asservice) {
58  s << "_" << pid;
59  }
60  s << ".root";
61 
62  std::string outputfilename = s.str();
63  run->SetOutputFile(outputfilename.c_str()); // Output file
64  run->SetName(confref.getMCEngine().c_str()); // Transport engine
65  run->SetIsMT(confref.getIsMT()); // MT mode
66 
68  auto header = new o2::dataformats::MCEventHeader();
69  run->SetMCEventHeader(header);
70 
71  // construct geometry / including magnetic field
72  build_geometry(run);
73 
74  // setup generator
75  auto embedinto_filename = confref.getEmbedIntoFileName();
76  auto primGen = new o2::eventgen::PrimaryGenerator();
77  if (!embedinto_filename.empty()) {
78  primGen->embedInto(embedinto_filename);
79  }
80  if (!asservice) {
81  o2::eventgen::GeneratorFactory::setPrimaryGenerator(confref, primGen);
82  }
83  run->SetGenerator(primGen);
84 
85  // Timer
86  TStopwatch timer;
87  timer.Start();
88 
89  // run init
90  run->Init();
91  finalize_geometry(run);
92  std::stringstream geomss;
93  geomss << "O2geometry";
94  if (asservice) {
95  geomss << "_" << pid;
96  }
97  geomss << ".root";
98  gGeoManager->Export(geomss.str().c_str());
99  if (asservice) {
100  // create a link of expected file name to actually produced file
101  // (deletes link if previously existing)
102  unlink("O2geometry.root");
103  if (symlink(geomss.str().c_str(), "O2geometry.root") != 0) {
104  LOG(ERROR) << "Failed to create simlink to geometry file";
105  }
106  }
107  std::time_t runStart = std::time(nullptr);
108 
109  // runtime database
110  bool kParameterMerged = true;
111  auto rtdb = run->GetRuntimeDb();
112  auto parOut = new FairParRootFileIo(kParameterMerged);
113 
114  std::stringstream s2;
115  s2 << confref.getOutPrefix();
116  if (asservice) {
117  s2 << "_" << pid;
118  }
119  s2 << "_par.root";
120  std::string parfilename = s2.str();
121  parOut->open(parfilename.c_str());
122  rtdb->setOutput(parOut);
123  rtdb->saveOutput();
124  rtdb->print();
125  o2::PDG::addParticlesToPdgDataBase(0);
126 
127  {
128  // store GRPobject
129  o2::parameters::GRPObject grp;
130  grp.setRun(run->GetRunId());
131  grp.setTimeStart(runStart);
132  grp.setTimeEnd(std::time(nullptr));
133  TObjArray* modArr = run->GetListOfModules();
134  TIter next(modArr);
135  FairModule* module = nullptr;
136  while ((module = (FairModule*)next())) {
137  o2::Base::Detector* det = dynamic_cast<o2::Base::Detector*>(module);
138  if (!det) {
139  continue; // not a detector
140  }
141  if (det->GetDetId() < o2::detectors::DetID::First) {
142  continue; // passive
143  }
144  if (det->GetDetId() > o2::detectors::DetID::Last) {
145  continue; // passive
146  }
147  grp.addDetReadOut(o2::detectors::DetID(det->GetDetId()));
148  }
149  grp.print();
150  printf("VMC: %p\n", TVirtualMC::GetMC());
151  auto field = dynamic_cast<o2::field::MagneticField*>(run->GetField());
152  if (field) {
153  o2::units::Current_t currDip = field->getCurrentDipole();
154  o2::units::Current_t currL3 = field->getCurrentSolenoid();
155  grp.setL3Current(currL3);
156  grp.setDipoleCurrent(currDip);
157  }
158  // save
159  std::string grpfilename = confref.getOutPrefix() + "_grp.root";
160  TFile grpF(grpfilename.c_str(), "recreate");
161  grpF.WriteObjectAny(&grp, grp.Class(), "GRP");
162  }
163  // todo: save beam information in the grp
164 
165  timer.Stop();
166  Double_t rtime = timer.RealTime();
167  Double_t ctime = timer.CpuTime();
168 
169  // extract max memory usage for init
170  FairSystemInfo sysinfo;
171  LOG(INFO) << "Init: Real time " << rtime << " s, CPU time " << ctime << "s";
172  LOG(INFO) << "Init: Memory used " << sysinfo.GetMaxMemory() << " MB";
173 
174  return run;
175 }

189 
190  // Finish

194 
195  // extract max memory usage

197
