202 bool MatchTPCITS::validateTPCMatch(int mtID)
203 {
204  auto& tpcMatch = mMatchesTPC[mtID];
205  auto& rcTPC = mMatchRecordsTPC[tpcMatch.first]; // best TPC->ITS match
206  if (rcTPC.nextRecID == Validated)
207  return false; // RS do we need this
208  // check if it is consistent with corresponding ITS->TPC match
209  auto& itsMatch = mMatchesITS[rcTPC.matchID]; // matchCand of partner ITS track
210  auto& rcITS = mMatchRecordsITS[itsMatch.first]; // best ITS->TPC match
211  if (rcTPC.nextRecID == Validated)
212  return false; // RS do we need this ?
213  if (rcITS.matchID == mtID) { // is best matching TPC track for this ITS track actually mtID?
214 
215  // unlink winner TPC track from all ITS candidates except winning one
216  int nextTPC = rcTPC.nextRecID;
217  while (nextTPC > MinusOne) {
218  auto& rcTPCrem = mMatchRecordsTPC[nextTPC];
219  removeTPCfromITS(mtID, rcTPCrem.matchID); // remove references on mtID from ITS match=rcTPCrem.matchID
220  nextTPC = rcTPCrem.nextRecID;
221  }
222  rcTPC.nextRecID = Validated;
223  int itsWinID = rcTPC.matchID;
224 
225  // unlink winner ITS match from all TPC matches using it
226  int nextITS = rcITS.nextRecID;
227  while (nextITS > MinusOne) {
228  auto& rcITSrem = mMatchRecordsITS[nextITS];
229  removeITSfromTPC(itsWinID, rcITSrem.matchID); // remove references on itsWinID from TPC match=rcITSrem.matchID
230  nextITS = rcITSrem.nextRecID;
231  }
232  rcITS.nextRecID = Validated;
233  return true;
234  }
235  return false;
236 }

317 bool MatchTPCITS::prepareTPCTracks()
318 {
320  mMatchRecordsTPC.clear();
321 
322  if (!loadTPCTracksNextChunk()) {
323  return false;
324  }
325 
326  int ntr = mTPCTracksArrayInp->size();
327 
328  mMatchesTPC.reserve(mMatchesTPC.size() + ntr);
329  // number of records might be actually more than N tracks!
330  mMatchRecordsTPC.reserve(mMatchRecordsTPC.size() + mMaxMatchCandidates * ntr);
331 
332  // copy the track params, propagate to reference X and build sector tables
333  mTPCWork.clear();
334  mTPCWork.reserve(ntr);
335  if (mMCTruthON) {
336  mTPCLblWork.clear();
337  mTPCLblWork.reserve(ntr);
338  }
339  for (int sec = o2::constants::math::NSectors; sec--;) {
340  mTPCSectIndexCache[sec].clear();
341  mTPCSectIndexCache[sec].reserve(100 + 1.2 * ntr / o2::constants::math::NSectors);
342  mTPCTimeBinStart[sec].clear();
343  }
344 
345  for (int it = 0; it < ntr; it++) {
346 
347  o2::TPC::TrackTPC& trcOrig = (*mTPCTracksArrayInp)[it];
348 
349  // make sure the track was propagated to inner TPC radius at the ref. radius
350  if (trcOrig.getX() > mXTPCInnerRef + 0.1)
351  continue; // failed propagation to inner TPC radius, cannot be matched
352 
353  // create working copy of track param
354  mTPCWork.emplace_back(static_cast<o2::track::TrackParCov&>(trcOrig), mCurrTPCTracksTreeEntry, it);
355  auto& trc = mTPCWork.back();
356  // propagate to matching Xref
357  if (!propagateToRefX(trc)) {
358  mTPCWork.pop_back(); // discard track whose propagation to mXRef failed
359  continue;
360  }
361  if (mMCTruthON) {
362  mTPCLblWork.emplace_back(mTPCTrkLabels->getLabels(it)[0]);
363  }
364 
365  float time0 = trcOrig.getTime0() - mNTPCBinsFullDrift;
366  trc.timeBins.set(time0 - trcOrig.getDeltaTBwd() - mTPCTimeEdgeTSafeMargin,
367  time0 + trcOrig.getDeltaTFwd() + mTPCTimeEdgeTSafeMargin);
368  // assign min max possible Z for this track which still respects the clusters A/C side
369  if (trcOrig.hasASideClustersOnly()) {
370  trc.zMin = trc.getZ() - trcOrig.getDeltaTBwd() * mTPCBin2Z;
371  trc.zMax = trc.getZ() + trcOrig.getDeltaTFwd() * mTPCBin2Z;
372  } else if (trcOrig.hasCSideClustersOnly()) {
373  trc.zMin = trc.getZ() - trcOrig.getDeltaTFwd() * mTPCBin2Z;
374  trc.zMax = trc.getZ() + trcOrig.getDeltaTBwd() * mTPCBin2Z;
375  }
376  // TODO : special treatment of tracks crossing the CE
377 
378  // cache work track index
379  mTPCSectIndexCache[o2::utils::Angle2Sector(trc.getAlpha())].push_back(mTPCWork.size() - 1);
380  }
381 
382  // sort tracks in each sector according to their timeMax
383  for (int sec = o2::constants::math::NSectors; sec--;) {
384  auto& indexCache = mTPCSectIndexCache[sec];
385  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " TPC tracks" << FairLogger::endl;
386  if (!indexCache.size())
387  continue;
388  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
389  auto& trcA = mTPCWork[a];
390  auto& trcB = mTPCWork[b];
391  return (trcA.timeBins.tmax - trcB.timeBins.tmax) < 0.;
392  });
393 
394  // build array of 1st entries with tmax corresponding to each ITS RO cycle
395  float tmax = mTPCWork[indexCache.back()].timeBins.tmax;
396  int nbins = 1 + tpcTimeBin2ITSROFrame(tmax);
397  auto& tbinStart = mTPCTimeBinStart[sec];
398  tbinStart.resize(nbins > 1 ? nbins : 1, -1);
399  tbinStart[0] = 0;
400  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
401  auto& trc = mTPCWork[indexCache[itr]];
402  int bTrc = tpcTimeBin2ITSROFrame(trc.timeBins.tmax);
403  if (bTrc < 0) {
404  continue;
405  }
406  if (tbinStart[bTrc] == -1) {
407  tbinStart[bTrc] = itr;
408  }
409  }
410  for (int i = 1; i < nbins; i++) {
411  if (tbinStart[i] == -1) { // fill gaps with preceding indices
412  tbinStart[i] = tbinStart[i - 1];
413  }
414  }
415  } // loop over tracks of single sector
416 
417  return true;
418 }

421 bool MatchTPCITS::prepareITSTracks()
422 {
423  // load next chunk of ITS data and prepare for matching
424  mMatchesITS.clear();
425  mITSWork.clear();
426  // number of records might be actually more than N tracks!
427  mMatchRecordsITS.clear(); // RS TODO reserve(mMatchRecordsITS.size() + mMaxMatchCandidates*ntr);
428  if (mMCTruthON) {
429  mITSLblWork.clear();
430  }
431  for (int sec = o2::constants::math::NSectors; sec--;) {
432  mITSSectIndexCache[sec].clear();
433  }
434 
435  while (loadITSTracksNextChunk()) {
436  int ntr = mITSTracksArrayInp->size();
437  for (int it = 0; it < ntr; it++) {
438  auto& trcOrig = (*mITSTracksArrayInp)[it];
439 
440  if (trcOrig.getParamOut().getX() < 1.) {
441  continue; // backward refit failed
442  }
443  // working copy of outer track param
444  mITSWork.emplace_back(static_cast<o2::track::TrackParCov&>(trcOrig.getParamOut()), mCurrITSTracksTreeEntry, it);
445  auto& trc = mITSWork.back();
446 
447  // TODO: why I did this?
448  if (!trc.rotate(o2::utils::Angle2Alpha(trc.getPhiPos()))) {
449  mITSWork.pop_back(); // discard failed track
450  continue;
451  }
452  // make sure the track is at the ref. radius
453  if (!propagateToRefX(trc)) {
454  mITSWork.pop_back(); // discard failed track
455  continue; // add to cache only those ITS tracks which reached ref.X and have reasonable snp
456  }
457  if (mMCTruthON) {
458  mITSLblWork.emplace_back(mITSTrkLabels->getLabels(it)[0]);
459  }
460 
461  float tmn = itsROFrame2TPCTimeBin(trcOrig.getROFrame());
462  trc.timeBins.set(tmn, tmn + mITSROFrame2TPCBin);
463  trc.roFrame = trcOrig.getROFrame();
464 
465  // cache work track index
466  int sector = o2::utils::Angle2Sector(trc.getAlpha());
467  mITSSectIndexCache[sector].push_back(mITSWork.size() - 1);
468 
469  // If the ITS track is very close to the sector edge, it may match also to a TPC track in the neighb. sector.
470  // For a track with Yr and Phir at Xr the distance^2 between the poisition of this track in the neighb. sector
471  // when propagated to Xr (in this neighbouring sector) and the edge will be (neglecting the curvature)
472  // [(Xr*tg(10)-Yr)/(tgPhir+tg70)]^2 / cos(70)^2 // for the next sector
473  // [(Xr*tg(10)+Yr)/(tgPhir-tg70)]^2 / cos(70)^2 // for the prev sector
474  // Distances to the sector edges in neighbourings sectors (at Xref in theit proper frames)
475  float tgp = trc.getSnp();
476  tgp /= std::sqrt((1.f - tgp) * (1.f + tgp)); // tan of track direction XY
477 
478  // sector up
479  float dy2Up = (mYMaxAtXRef - trc.getY()) / (tgp + Tan70);
480  if ((dy2Up * dy2Up * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector up
481  addTrackCloneForNeighbourSector(trc, sector < (o2::constants::math::NSectors - 1) ? sector + 1 : 0);
482  }
483  // sector down
484  float dy2Dn = (mYMaxAtXRef + trc.getY()) / (tgp - Tan70);
485  if ((dy2Dn * dy2Dn * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector down
486  addTrackCloneForNeighbourSector(trc, sector > 1 ? sector - 1 : o2::constants::math::NSectors - 1);
487  }
488  }
489  }
490  // sort tracks in each sector according to their time, then tgl
491  for (int sec = o2::constants::math::NSectors; sec--;) {
492  auto& indexCache = mITSSectIndexCache[sec];
493  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " ITS tracks" << FairLogger::endl;
494  if (!indexCache.size()) {
495  continue;
496  }
497  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
498  auto& trackA = mITSWork[a];
499  auto& trackB = mITSWork[b];
500  if (trackA.roFrame < trackB.roFrame) { // ITS tracks have the same time coverage
501  return true;
502  } else if (trackA.roFrame > trackB.roFrame) {
503  return false;
504  }
505  return trackA.getTgl() < trackB.getTgl();
506  });
507 
508  // build array of 1st entries with of each ITS RO cycle
509  int nbins = 1 + mITSWork[indexCache.back()].roFrame;
510  auto& tbinStart = mITSTimeBinStart[sec];
511  tbinStart.resize(nbins > 1 ? nbins : 1, -1);
512  tbinStart[0] = 0;
513  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
514  auto& trc = mITSWork[indexCache[itr]];
515  if (tbinStart[trc.roFrame] == -1) {
516  tbinStart[trc.roFrame] = itr;
517  }
518  }
519  for (int i = 1; i < nbins; i++) {
520  if (tbinStart[i] == -1) { // fill gaps with preceding indices
521  tbinStart[i] = tbinStart[i - 1];
522  }
523  }
524  } // loop over tracks of single sector
525  mMatchesITS.reserve(mITSWork.size());
526  mMatchRecordsITS.reserve(mITSWork.size() * mMaxMatchCandidates);
527 
528  return true;
529 }

574 //_____________________________________________________
575 void MatchTPCITS::doMatching(int sec)
576 {
578  auto& cacheITS = mITSSectIndexCache[sec]; // array of cached ITS track indices for this sector
579  auto& cacheTPC = mTPCSectIndexCache[sec]; // array of cached ITS track indices for this sector
580  auto& tbinStartTPC = mTPCTimeBinStart[sec]; // array of 1st TPC track with timeMax in ITS ROFrame
581  auto& tbinStartITS = mITSTimeBinStart[sec];
582  int nTracksTPC = cacheTPC.size(), nTracksITS = cacheITS.size();
583  if (!nTracksTPC || !nTracksITS) {
584  LOG(INFO) << "Matchng sector " << sec << " : N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS << " in sector "
585  << sec << FairLogger::endl;
586  return;
587  }
588 
590  float maxTDriftSafe = (mNTPCBinsFullDrift + mITSTPCTimeBinSafeMargin + mTPCTimeEdgeTSafeMargin);
591 
592  // get min ROFrame (in TPC time-bins) of ITS tracks currently in cache
593  auto minROFITS = mITSWork[cacheITS.front()].roFrame;
594 
595  if (minROFITS >= int(tbinStartTPC.size())) {
596  LOG(INFO) << "ITS min ROFrame " << minROFITS << " exceeds all cached TPC track ROF eqiuvalent "
597  << cacheTPC.size() - 1 << FairLogger::endl;
598  return;
599  }
600 
601  int nCheckTPCControl = 0, nCheckITSControl = 0, nMatchesControl = 0; // temporary
602 
603  int idxMinTPC = tbinStartTPC[minROFITS]; // index of 1st cached TPC track within cached ITS ROFrames
604  for (int itpc = idxMinTPC; itpc < nTracksTPC; itpc++) {
605  auto& trefTPC = mTPCWork[cacheTPC[itpc]];
606  // estimate ITS 1st ROframe bin this track may match to: TPC track are sorted according to their
607  // timeMax, hence the timeMax - MaxmNTPCBinsFullDrift are non-decreasing
608  int itsROBin = tpcTimeBin2ITSROFrame(trefTPC.timeBins.tmax - maxTDriftSafe);
609  if (itsROBin >= int(tbinStartITS.size())) { // time of TPC track exceeds the max time of ITS in the cache
610  break;
611  }
612  int iits0 = tbinStartITS[itsROBin];
613  nCheckTPCControl++;
614  for (auto iits = iits0; iits < nTracksITS; iits++) {
615  auto& trefITS = mITSWork[cacheITS[iits]];
616  // compare if the ITS and TPC tracks may overlap in time
617  if (trefTPC.timeBins.tmax < trefITS.timeBins.tmin) {
618  // since TPC tracks are sorted in timeMax and ITS tracks are sorted in timeMin
619  // all following ITS tracks also will not match
620  break;
621  }
622  if (trefTPC.timeBins.tmin > trefITS.timeBins.tmax) { // its bracket is fully before TPC bracket
623  continue;
624  }
625  nCheckITSControl++;
626  float chi2 = -1;
627  int rejFlag = compareITSTPCTracks(trefITS, trefTPC, chi2);
628 
629 #ifdef _ALLOW_DEBUG_TREES_
630  if (mDBGOut && ((rejFlag == Accept && isDebugFlag(MatchTreeAccOnly)) || isDebugFlag(MatchTreeAll))) {
631  fillITSTPCmatchTree(cacheITS[iits], cacheTPC[itpc], rejFlag, chi2);
632  }
633 #endif
634 
635  if (rejFlag == RejectOnTgl) {
636  // ITS tracks in each ROFrame are ordered in Tgl, hence if this check failed on Tgl check
637  // (i.e. tgl_its>tgl_tpc+tolerance), tnem all other ITS tracks in this ROFrame will also have tgl too large.
638  // Jump on the 1st ITS track of the next ROFrame
639  int rof = trefITS.roFrame;
640  bool stop = false;
641  do {
642  if (++rof >= int(tbinStartITS.size())) {
643  stop = true;
644  break; // no more ITS ROFrames in cache
645  }
646  iits = tbinStartITS[rof] - 1; // next track to be checked -1
647  } while (iits <= tbinStartITS[trefITS.roFrame]); // skip empty bins
648  if (stop) {
649  break;
650  }
651  continue;
652  }
653  if (rejFlag != Accept) {
654  continue;
655  }
656  mTimerReg.Start(false);
657  registerMatchRecordTPC(trefITS, trefTPC, chi2); // register matching candidate
658  mTimerReg.Stop();
659  nMatchesControl++;
660  }
661  }
662 
663  LOG(INFO) << "Match sector " << sec << " N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS
664  << " N TPC tracks checked: " << nCheckTPCControl << " (starting from " << idxMinTPC
665  << "), checks: " << nCheckITSControl << ", matches:" << nMatchesControl << FairLogger::endl;
666 }

690 //______________________________________________
691 bool MatchTPCITS::registerMatchRecordTPC(TrackLocITS& tITS, TrackLocTPC& tTPC, float chi2)
692 {
695 
696  auto& mtcTPC = getTPCMatchEntry(tTPC); // get matchCand structure of this TPC track, create if none
697  int nextID = mtcTPC.first; // get 1st matchRecord this matchCand refers to
698  if (nextID < 0) { // no matches yet, just add new record
699  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
700  mtcTPC.first = mMatchRecordsTPC.size(); // new record will be added in the end
701  mMatchRecordsTPC.emplace_back(tITS.matchID, chi2); // create new record with empty reference on next match
702  return true;
703  }
704 
705  int count = 0, topID = MinusOne;
706  do {
707  auto& nextMatchRec = mMatchRecordsTPC[nextID];
708  count++;
709  if (chi2 < nextMatchRec.chi2) { // need to insert new record before nextMatchRec?
710  if (count < mMaxMatchCandidates) {
711  break; // will insert in front of nextID
712  } else { // max number of candidates reached, will overwrite the last one
713  nextMatchRec.chi2 = chi2;
714  suppressMatchRecordITS(nextMatchRec.matchID, tTPC.matchID); // flag as disabled the overriden ITS match
715  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
716  nextMatchRec.matchID = tITS.matchID; // reuse the record of suppressed ITS match to store better one
717  return true;
718  }
719  }
720  topID = nextID; // check next match record
721  nextID = nextMatchRec.nextRecID;
722  } while (nextID > MinusOne);
723 
724  // if count == mMaxMatchCandidates, the max number of candidates was already reached, and the
725  // new candidated was either discarded (if its chi2 is worst one) or has overwritten worst
726  // existing candidate. Otherwise, we need to add new entry
727  if (count < mMaxMatchCandidates) {
728  if (topID < 0) { // the new match is top candidate
729  topID = mtcTPC.first = mMatchRecordsTPC.size(); // register new record as top one
730  } else { // there are better candidates
731  topID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC.size(); // register to his parent
732  }
733  // nextID==-1 will mean that the while loop run over all candidates->the new one is the worst (goes to the end)
734  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
735  mMatchRecordsTPC.emplace_back(tITS.matchID, chi2, nextID); // create new record with empty reference on next match
736  // make sure that after addition the number of candidates don't exceed allowed number
737  count++;
738  while (nextID > MinusOne) {
739  if (count > mMaxMatchCandidates) {
740  suppressMatchRecordITS(mMatchRecordsTPC[nextID].matchID, tTPC.matchID);
741  // exclude nextID record, w/o changing topID (which becomes the last record)
742  nextID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC[nextID].nextRecID;
743  continue;
744  }
745  count++;
746  topID = nextID;
747  nextID = mMatchRecordsTPC[nextID].nextRecID;
748  }
749  return true;
750  } else {
751  return false; // unless nextID was assigned OverrideExisting, new candidate was discarded
752  }
753 }

755 //______________________________________________
756 void MatchTPCITS::registerMatchRecordITS(TrackLocITS& tITS, int matchTPCID, float chi2)
757 {
759  auto& itsMatch = getITSMatchEntry(tITS); // if needed, create new entry
760  int nextRecord = itsMatch.first; // entry of 1st match record in mMatchRecordsITS
761  int idnew = mMatchRecordsITS.size();
762  mMatchRecordsITS.emplace_back(matchTPCID, chi2); // associate index of matchCand with this record
763  if (nextRecord < 0) { // this is the 1st match for this TPC track
764  itsMatch.first = idnew;
765  return;
766  }
767  // there are other matches for this ITS track, insert the new record preserving chi2 order
768  // navigate till last record or the one with worse chi2
769  int topID = MinusOne;
770  auto& newRecord = mMatchRecordsITS.back();
771  do {
772  auto& recITS = mMatchRecordsITS[nextRecord];
773  if (chi2 < recITS.chi2) { // insert before this one
774  newRecord.nextRecID = nextRecord; // new one will refer to old one it overtook
775  if (topID < 0) {
776  itsMatch.first = idnew; // the new one is the best match, the matchCand will refer to it
777  } else {
778  mMatchRecordsITS[topID].nextRecID = idnew; // new record will follow existing better one
779  }
780  return;
781  }
782  topID = nextRecord;
783  nextRecord = mMatchRecordsITS[nextRecord].nextRecID;
784  } while (nextRecord > MinusOne);
785 
786  // if we reached here, the new record should be added in the end
787  mMatchRecordsITS[topID].nextRecID = idnew; // register new link
788 }

790 //______________________________________________
791 int MatchTPCITS::compareITSTPCTracks(const TrackLocITS& tITS, const TrackLocTPC& tTPC, float& chi2) const
792 {
794  auto& trackTPC = tTPC;
795  auto& trackITS = tITS;
796  chi2 = -1.f;
797  int rejFlag = Accept;
798  float diff; // make rough check differences and their nsigmas
799 
800  // start with check on Tgl, since rjection on it will allow to profit from sorting
801  diff = trackITS.getParam(o2::track::kTgl) - trackTPC.getParam(o2::track::kTgl);
802  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kTgl], RejectOnTgl))) {
803  return rejFlag;
804  }
805  diff *= diff / (trackITS.getDiagError2(o2::track::kTgl) + trackTPC.getDiagError2(o2::track::kTgl));
806  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kTgl], RejectOnTgl + NSigmaShift))) {
807  return rejFlag;
808  }
809 
810  diff = trackITS.getParam(o2::track::kY) - trackTPC.getParam(o2::track::kY);
811  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kY], RejectOnY))) {
812  return rejFlag;
813  }
814  diff *= diff / (trackITS.getDiagError2(o2::track::kY) + trackTPC.getDiagError2(o2::track::kY));
815  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kY], RejectOnY + NSigmaShift))) {
816  return rejFlag;
817  }
818 
819  if (mCompareTracksDZ) { // in continuous mode we usually don't use DZ
820  diff = trackITS.getParam(o2::track::kZ) - trackTPC.getParam(o2::track::kZ);
821  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kZ], RejectOnZ))) {
822  return rejFlag;
823  }
824  diff *= diff / (trackITS.getDiagError2(o2::track::kZ) + trackTPC.getDiagError2(o2::track::kZ));
825  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kZ], RejectOnZ + NSigmaShift))) {
826  return rejFlag;
827  }
828  } else { // in continuous mode we use special check on allowed Z range
829  if (trackITS.getParam(o2::track::kZ) - tTPC.zMax > mCrudeAbsDiffCut[o2::track::kZ])
830  return RejectOnZ;
831  if (trackITS.getParam(o2::track::kZ) - tTPC.zMin < -mCrudeAbsDiffCut[o2::track::kZ])
832  return -RejectOnZ;
833  }
834 
835  diff = trackITS.getParam(o2::track::kSnp) - trackTPC.getParam(o2::track::kSnp);
836  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kSnp], RejectOnSnp))) {
837  return rejFlag;
838  }
839  diff *= diff / (trackITS.getDiagError2(o2::track::kSnp) + trackTPC.getDiagError2(o2::track::kSnp));
840  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kSnp], RejectOnSnp + NSigmaShift))) {
841  return rejFlag;
842  }
843 
844  diff = trackITS.getParam(o2::track::kQ2Pt) - trackTPC.getParam(o2::track::kQ2Pt);
845  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kQ2Pt], RejectOnQ2Pt))) {
846  return rejFlag;
847  }
848  diff *= diff / (trackITS.getDiagError2(o2::track::kQ2Pt) + trackTPC.getDiagError2(o2::track::kQ2Pt));
849  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kQ2Pt], RejectOnQ2Pt + NSigmaShift))) {
850  return rejFlag;
851  }
852  // calculate mutual chi2 excluding Z in continuos mode
853  chi2 = getPredictedChi2NoZ(tITS, tTPC);
854  if (chi2 > mCutMatchingChi2)
855  return RejectOnChi2;
856 
857  return Accept;
858 }

926 float MatchTPCITS::getPredictedChi2NoZ(const o2::track::TrackParCov& tr1, const o2::track::TrackParCov& tr2) const

952  if (!covMat.Invert()) {
953  LOG(ERROR) << "Cov.matrix inversion failed: " << covMat << FairLogger::endl;
954  return 2. * o2::track::HugeF;
955  }
956  double chi2diag = 0., chi2ndiag = 0.,
957  diff[o2::track::kNParams - 1] = { tr1.getParam(o2::track::kY) - tr2.getParam(o2::track::kY),
958  tr1.getParam(o2::track::kSnp) - tr2.getParam(o2::track::kSnp),

961  for (int i = o2::track::kNParams - 1; i--;) {
962  chi2diag += diff[i] * diff[i] * covMat(i, i);
963  for (int j = i; j--;) {
964  chi2ndiag += diff[i] * diff[j] * covMat(i, j);
965  }
966  }
967  return chi2diag + 2. * chi2ndiag;
968 }
