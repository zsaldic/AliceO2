48 DataAllocator::matchDataHeader(const Output& spec, size_t timeslice) {
49  // FIXME: we should take timeframeId into account as well.
50  for (auto &output : mAllowedOutputRoutes) {
51  if (DataSpecUtils::match(output.matcher, spec.origin, spec.description, spec.subSpec)
52  && ((timeslice % output.maxTimeslices) == output.timeslice)) {
53  return output.channel;
54  }
55  }
56  std::ostringstream str;
57  str << "Worker is not authorised to create message with "
58  << "origin(" << spec.origin.as<std::string>() << ")"
59  << "description(" << spec.description.as<std::string>() << ")"
60  << "subSpec(" << spec.subSpec << ")";
61  throw std::runtime_error(str.str());
62 }

152 void DataAllocator::adopt(const Output& spec, TableBuilder* tb)
153 {
154  std::string channel = matchDataHeader(spec, mTimingInfo->timeslice);
155  auto header = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodArrow, 0);
156  auto context = mContextRegistry->get<ArrowContext>();
157 
158  auto creator = [device = context->proxy().getDevice()](size_t s) -> std::unique_ptr<FairMQMessage> { return device->NewMessage(s); };
159  auto buffer = std::make_shared<FairMQResizableBuffer>(creator);
160 
164  std::shared_ptr<TableBuilder> p(tb);
165  auto finalizer = [payload = p](std::shared_ptr<FairMQResizableBuffer> b) -> void {
166  auto table = payload->finalize();
167 
168  auto stream = std::make_shared<arrow::io::BufferOutputStream>(b);
169  std::shared_ptr<arrow::ipc::RecordBatchWriter> writer;
170  auto outBatch = arrow::ipc::RecordBatchStreamWriter::Open(stream.get(), table->schema(), &writer);
171  auto outStatus = writer->WriteTable(*table);
172  if (outStatus.ok() == false) {
173  throw std::runtime_error("Unable to Write table");
174  }
175  };
176 
177  assert(context);
178  context->addBuffer(std::move(header), buffer, std::move(finalizer), channel);
179 }

205 Output DataAllocator::getOutputByBind(OutputRef&& ref)
206 {
207  if (ref.label.empty()) {
208  throw std::runtime_error("Invalid (empty) OutputRef provided.");
209  }
210  for (size_t ri = 0, re = mAllowedOutputRoutes.size(); ri != re; ++ri) {
211  if (mAllowedOutputRoutes[ri].matcher.binding.value == ref.label) {
212  auto spec = mAllowedOutputRoutes[ri].matcher;
213  return Output{ spec.origin, spec.description, ref.subSpec, spec.lifetime, std::move(ref.headerStack) };
214  }
215  }
216  throw std::runtime_error("Unable to find OutputSpec with label " + ref.label);
217  O2_BUILTIN_UNREACHABLE();
218 }
