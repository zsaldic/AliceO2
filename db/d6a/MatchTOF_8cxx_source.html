776 bool MatchTOF::propagateToRefXWithoutCov(o2::track::TrackParCov& trc, float xRef, float stepInCm, float bzField)
777 {
778  // propagate track to matching reference X without using the covariance matrix
779  // we create the copy of the track in a TrackPar object (no cov matrix)
780  o2::track::TrackPar trcNoCov(trc);
781  const float tanHalfSector = tan(o2::constants::math::SectorSpanRad / 2);
782  bool refReached = false;
783  float xStart = trcNoCov.getX();
784  // the first propagation will be from 2m, if the track is not at least at 2m
785  if (xStart < 50.)
786  xStart = 50.;
787  int istep = 1;
788  bool hasPropagated = trcNoCov.propagateParamTo(xStart + istep * stepInCm, bzField);
789  while (hasPropagated) {
790  if (trcNoCov.getX() > xRef) {
791  refReached = true; // we reached the 371cm reference
792  }
793  istep++;
794  if (fabs(trcNoCov.getY()) > trcNoCov.getX() * tanHalfSector) { // we are still in the same sector
795  // we need to rotate the track to go to the new sector
796  //Printf("propagateToRefX: changing sector");
797  auto alphaNew = o2::utils::Angle2Alpha(trcNoCov.getPhiPos());
798  if (!trcNoCov.rotateParam(alphaNew) != 0) {
799  // Printf("propagateToRefX: failed to rotate");
800  break; // failed (this line is taken from MatchTPCITS and the following comment too: RS: check effect on matching tracks to neighbouring sector)
801  }
802  }
803  if (refReached)
804  break;
805  hasPropagated = trcNoCov.propagateParamTo(xStart + istep * stepInCm, bzField);
806  }
807  // if (std::abs(trc.getSnp()) > MAXSNP) Printf("propagateToRefX: condition on snp not ok, returning false");
808  //Printf("propagateToRefX: snp of teh track is %f (--> %f grad)", trcNoCov.getSnp(), TMath::ASin(trcNoCov.getSnp())*TMath::RadToDeg());
809 
810  return refReached && std::abs(trcNoCov.getSnp()) < 0.95 && TMath::Abs(trcNoCov.getZ()) < Geo::MAXHZTOF; // Here we need to put MAXSNP
811 }
