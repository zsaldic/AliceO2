51 void Storage::getSelection(/*const*/ ConditionId *id)
52 {
53  // return required version and subversion from the list of selection criteria
54 
55  TIter iter(&mSelections);
56  ConditionId *aSelection;
57 
58  // loop on the list of selection criteria
59  while ((aSelection = (ConditionId *) iter.Next())) {
60  // check if selection element contains id's path and run (range)
61  if (aSelection->isSupersetOf(*id)) {
62  LOG(DEBUG) << "Using selection criterion: " << aSelection->ToString().Data() << " ";
63  // return required version and subversion
64 
65  id->setVersion(aSelection->getVersion());
66  id->setSubVersion(aSelection->getSubVersion());
67  return;
68  }
69  }
70 
71  // no valid element is found in the list of selection criteria -> return
72  LOG(DEBUG) << "Looking for objects with most recent version";
73  return;
74 }

104 void Storage::addSelection(const ConditionId &selection)
105 {
106  // add a selection criterion
107 
108  IdPath path = selection.getPath();
109  if (!path.isValid()) {
110  return;
111  }
112 
113  TIter iter(&mSelections);
114  const ConditionId *anId;
115  while ((anId = (ConditionId *) iter.Next())) {
116  if (selection.isSupersetOf(*anId)) {
117  LOG(WARNING) << "This selection is more general than a previous one and will hide it!";
118  LOG(WARNING) << (anId->ToString()).Data();
119  mSelections.AddBefore(anId, new ConditionId(selection));
120  return;
121  }
122  }
123  mSelections.AddFirst(new ConditionId(selection));
124 }

196 Condition *Storage::getObject(const ConditionId &query)
197 {
198  // get an Condition object from the database
199 
200  // check if query's path and runRange are valid
201  // query is invalid also if version is not specified and subversion is!
202  if (!query.isValid()) {
203  LOG(ERROR) << "Invalid query: " << query.ToString().Data();
204  return nullptr;
205  }
206 
207  // query is not specified if path contains wildcard or runrange = [-1,-1]
208  if (!query.isSpecified()) {
209  LOG(ERROR) << "Unspecified query: " << query.ToString().Data();
210  return nullptr;
211  }
212 
213  // This is needed otherwise TH1 objects (histos, TTree's) are lost when file is closed!
214  Bool_t oldStatus = TH1::AddDirectoryStatus();
215  TH1::AddDirectory(kFALSE);
216 
217  Condition *entry = getCondition(query);
218 
219  if (oldStatus != kFALSE) {
220  TH1::AddDirectory(kTRUE);
221  }
222 
223  // if drain storage is set, drain entry into drain storage
224  if (entry && (Manager::Instance())->isdrainSet()) {
225  Manager::Instance()->drain(entry);
226  }
227 
228  return entry;
229 }

245 TList *Storage::getAllObjects(const ConditionId &query)
246 {
247  // get multiple Condition objects from the database
248 
249  if (!query.isValid()) {
250  LOG(ERROR) << "Invalid query: " << query.ToString().Data();
251  return nullptr;
252  }
253 
254  if (query.isAnyRange()) {
255  LOG(ERROR) << "Unspecified run or runrange: " << query.ToString().Data();
256  return nullptr;
257  }
258 
259  // This is needed otherwise TH1 objects (histos, TTree's) are lost when file is closed!
260  Bool_t oldStatus = TH1::AddDirectoryStatus();
261  TH1::AddDirectory(kFALSE);
262 
263  TList *result = getAllEntries(query);
264 
265  if (oldStatus != kFALSE) {
266  TH1::AddDirectory(kTRUE);
267  }
268 
269  Int_t nEntries = result->GetEntries();
270 
271  LOG(INFO) << nEntries << " objects retrieved. Request was: " << query.ToString().Data();
272  for (int i = 0; i < nEntries; i++) {
273  Condition *entry = (Condition *) result->At(i);
274  LOG(INFO) << entry->getId().ToString().Data();
275  }
276 
277  // if drain storage is set, drain entries into drain storage
278  if ((Manager::Instance())->isdrainSet()) {
279  for (int i = 0; i < result->GetEntries(); i++) {
280  Condition *entry = (Condition *) result->At(i);
281  Manager::Instance()->drain(entry);
282  }
283  }

384 void Storage::queryStorages(Int_t run, const char *pathFilter, Int_t version, ConditionMetaData *md)
385 {
386  // query CDB for files valid for given run, and fill list mValidFileIds
387  // Actual query is done in virtual function queryValidFiles()
388  // If version is not specified, the query will fill mValidFileIds
389  // with highest versions
390 
391  mRun = run;
392 
393  mPathFilter = pathFilter;
394  if (!mPathFilter.isValid()) {
395  LOG(ERROR) << "Filter not valid: " << pathFilter;
396  mPathFilter = "*";
397  return;
398  }
399 
400  mVersion = version;
401 
402  LOG(INFO) << "Querying files valid for run " << mRun << R"( and path ")" << pathFilter << R"(" into CDB storage ")"
403  << mType.Data() << "://" << mBaseFolder.Data() << R"(")";
404 
405  // In mValidFileIds, clear id for the same 3level path, if any
406  LOG(DEBUG) << R"(Clearing list of CDB ConditionId's previously loaded for path ")" << pathFilter << R"(")";
407  IdPath filter(pathFilter);
408  for (Int_t i = mValidFileIds.GetEntries() - 1; i >= 0; --i) {
409  ConditionId *rmMe = dynamic_cast<ConditionId *>(mValidFileIds.At(i));
410  IdPath rmPath = rmMe->getPathString();
411  if (filter.isSupersetOf(rmPath)) {
412  LOG(DEBUG) << R"(Removing id ")" << rmPath.getPathString().Data() << R"(" matching: ")" << pathFilter << R"(")";
413  delete mValidFileIds.RemoveAt(i);
414  }
415  }
416 
417  if (mConditionMetaDataFilter) {
418  delete mConditionMetaDataFilter;
419  mConditionMetaDataFilter = nullptr;
420  }
421  if (md) {
422  mConditionMetaDataFilter = dynamic_cast<ConditionMetaData *>(md->Clone());
423  }
424 
425  queryValidFiles();
426 
427  LOG(INFO) << mValidFileIds.GetEntries() << " valid files found!";
428 }
