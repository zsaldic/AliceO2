82  // should go into a helper
83  // this function queries the sim config data and initializes the SimConfig singleton
84  // returns true if successful / false if not
85  static bool querySimConfig(FairMQChannel& channel)
86  {
87  auto text = new std::string("configrequest");
88  std::unique_ptr<FairMQMessage> request(channel.NewMessage(const_cast<char*>(text->c_str()),
89  text->length(), CustomCleanup, text));
90  std::unique_ptr<FairMQMessage> reply(channel.NewMessage());
91 
92  int timeoutinMS = 100000; // wait for 100s max
93  if (channel.Send(request, timeoutinMS) >= 0) {
94  LOG(INFO) << "Waiting for configuration answer ";
95  if (channel.Receive(reply, timeoutinMS) > 0) {
96  LOG(INFO) << "Configuration answer received, containing " << reply->GetSize() << " bytes ";
97 
98  // the answer is a TMessage containing the simulation Configuration
99  auto message = std::make_unique<o2::devices::TMessageWrapper>(reply->GetData(), reply->GetSize());
100  auto config = static_cast<o2::conf::SimConfigData*>(message.get()->ReadObjectAny(message.get()->GetClass()));
101  if (!config) {
102  return false;
103  }
104 
105  LOG(INFO) << "COMMUNICATED ENGINE " << config->mMCEngine;
106  auto& conf = o2::conf::SimConfig::Instance();
107  conf.resetFromConfigData(*config);
108  delete config;
109  } else {
110  LOG(ERROR) << "No configuration received within " << timeoutinMS << "ms\n";
111  return false;
112  }
113  } else {
114  LOG(ERROR) << "Could not send configuration request within " << timeoutinMS << "ms\n";
115  return false;
116  }
117  return true;
118  }

120  // initializes the simulation classes; queries the configuration on a given channel
121  static bool initSim(FairMQChannel& channel, std::unique_ptr<FairRunSim>& simptr)
122  {
123  if (!querySimConfig(channel)) {
124  return false;
125  }
126 
127  LOG(INFO) << "Setting up the simulation ...";
128  simptr = std::move(std::unique_ptr<FairRunSim>(o2sim_init(true)));
129  FairSystemInfo sysinfo;
130 
131  // to finish initialization (trigger further cross section table building etc) -- which especially
132  // G4 is doing at the first ProcessRun
133  // The goal is to have everything setup before we fork
134  TVirtualMC::GetMC()->ProcessRun(0);
135 
136  LOG(INFO) << "MEM-STAMP END OF SIM INIT" << sysinfo.GetCurrentMemory() / (1024. * 1024) << " "
137  << sysinfo.GetMaxMemory() << " MB\n";
138 
139  return true;
140  }

142  bool Kernel(FairMQChannel& requestchannel, FairMQChannel& dataoutchannel)
143  {
144  auto text = new std::string("primrequest");
145 
146  // create message object with a pointer to the data buffer,
147  // its size,
148  // custom deletion function (called when transfer is done),
149  // and pointer to the object managing the data buffer
150  FairMQMessagePtr request(requestchannel.NewMessage(const_cast<char*>(text->c_str()), // data
151  text->length(), // size
152  CustomCleanup,
153  text));
154  FairMQMessagePtr reply(dataoutchannel.NewMessage());
155 
156  mVMCApp->setSimDataChannel(&dataoutchannel);
157 
158  LOG(INFO) << "Requesting work ";
159  int timeoutinMS = 10000; // wait for 10s max -- we should have a more robust solution
160  if (requestchannel.Send(request, timeoutinMS) >= 0) {
161  LOG(INFO) << "Waiting for answer ";
162  // asking for primary generation
163 
164  if (requestchannel.Receive(reply, timeoutinMS) > 0) {
165  LOG(INFO) << "Answer received, containing " << reply->GetSize() << " bytes " << FairLogger::endl;
166 
167  // wrap incoming bytes as a TMessageWrapper which offers "adoption" of a buffer
168  auto message = new TMessageWrapper(reply->GetData(), reply->GetSize());
169  auto chunk = static_cast<o2::Data::PrimaryChunk*>(message->ReadObjectAny(message->GetClass()));
170 
171  mVMCApp->setPrimaries(chunk->mParticles);
172 
173  auto info = chunk->mSubEventInfo;
174  mVMCApp->setSubEventInfo(info);
175 
176  LOG(INFO) << "Processing " << chunk->mParticles.size() << FairLogger::endl;
177  gRandom->SetSeed(chunk->mSubEventInfo.seed);
178 
179  mVMC->ProcessRun(1);
180  FairSystemInfo sysinfo;
181  LOG(INFO) << "TIME-STAMP " << mTimer.RealTime() << "\t";
182  mTimer.Continue();
183  LOG(INFO) << "MEM-STAMP " << sysinfo.GetCurrentMemory() / (1024. * 1024) << " "
184  << sysinfo.GetMaxMemory() << " MB\n";
185  delete message;
186  delete chunk;
187  } else {
188  return false;
189  }
190  } else {
191  return false;
192  }
193  return true;
194  }
