38 // A container to hold and manage MC truth information/labels.
39 // The actual MCtruth type is a generic template type and can be supplied by the user
40 // It is meant to manage associations from one "dataobject" identified by an index into an array
41 // to multiple TruthElements

43 template <typename TruthElement>
44 class MCTruthContainer

46  private:

51  size_t getSize(uint dataindex) const
52  {
53  // calculate size / number of labels from a difference in pointed indices
54  const auto size = (dataindex < mHeaderArray.size() - 1)
55  ? mHeaderArray[dataindex + 1].index - mHeaderArray[dataindex].index
56  : mTruthArray.size() - mHeaderArray[dataindex].index;
57  return size;
58  }

106  // add element for a particular dataindex
107  // at the moment only strictly consecutive modes are supported
108  void addElement(uint dataindex, TruthElement const& element)
109  {
110  if (dataindex < mHeaderArray.size()) {
111  // look if we have something for this dataindex already
112  // must currently be the last one
113  if (dataindex != (mHeaderArray.size() - 1)) {
114  throw std::runtime_error("MCTruthContainer: unsupported code path");
115  }
116  } else {
117  // assert(dataindex == mHeaderArray.size());
118 
119  // add empty holes
120  int holes = dataindex - mHeaderArray.size();
121  assert(holes >= 0);
122  for (int i = 0; i < holes; ++i) {
123  mHeaderArray.emplace_back(mTruthArray.size());
124  }
125  // add a new one
126  mHeaderArray.emplace_back(mTruthArray.size());
127  }
128  auto& header = mHeaderArray[dataindex];
129  mTruthArray.emplace_back(element);
130  }

154  // Add element at last position or for a previous index
155  // (at random access position).
156  // This might be a slow process since data has to be moved internally
157  // so this function should be used with care.
158  void addElementRandomAccess(uint dataindex, TruthElement const& element)
159  {
160  if (dataindex >= mHeaderArray.size()) {
161  // a new dataindex -> push element at back
162 
163  // we still forbid to leave holes
164  assert(dataindex == mHeaderArray.size());
165 
166  mHeaderArray.resize(dataindex + 1);
167  mHeaderArray[dataindex] = mTruthArray.size();
168  mTruthArray.emplace_back(element);
169  } else {
170  // if appending at end use fast function
171  if (dataindex == mHeaderArray.size() - 1) {
172  addElement(dataindex, element);
173  return;
174  }
175 
176  // existing dataindex
177  // have to:
178  // a) move data;
179  // b) insert new element;
180  // c) adjust indices of all headers right to this
181  auto currentindex = mHeaderArray[dataindex].index;
182  auto lastindex = currentindex + getSize(dataindex);
183  assert(currentindex >= 0);
184 
185  // resize truth array
186  mTruthArray.resize(mTruthArray.size() + 1);
187  // move data (have to do from right to left)
188  for (int i = mTruthArray.size() - 1; i > lastindex; --i) {
189  mTruthArray[i] = mTruthArray[i - 1];
190  }
191  // insert new element
192  mTruthArray[lastindex] = element;
193 
194  // fix headers
195  for (uint i = dataindex + 1; i < mHeaderArray.size(); ++i) {
196  auto oldindex = mHeaderArray[i].index;
197  mHeaderArray[i].index = (oldindex != -1) ? oldindex + 1 : oldindex;
198  }
199  }
200  }

202  // merge another container to the back of this one
203  void mergeAtBack(MCTruthContainer<TruthElement> const& other)
204  {
205  const auto oldtruthsize = mTruthArray.size();
206  const auto oldheadersize = mHeaderArray.size();
207 
208  // copy from other
209  std::copy(other.mHeaderArray.begin(), other.mHeaderArray.end(), std::back_inserter(mHeaderArray));
210  std::copy(other.mTruthArray.begin(), other.mTruthArray.end(), std::back_inserter(mTruthArray));
211 
212  // adjust information of newly attached part
213  for (uint i = oldheadersize; i < mHeaderArray.size(); ++i) {
214  mHeaderArray[i].index += oldtruthsize;
215  }
216  }
