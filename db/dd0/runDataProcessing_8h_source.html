49 // This template magic allow users to customize the behavior of the process
50 // by (optionally) implementing a `configure` method which modifies one of the
51 // objects in question.
52 //
53 // For example it can be optionally implemented by the user to specify the
54 // channel policies for your setup. Use this if you want to customize the way
55 // your devices communicate between themself, e.g. if you want to use REQ/REP
56 // in place of PUB/SUB.
57 //
58 // The advantage of this approach is that we do not need to expose the
59 // configurability / configuration object to the user, unless he really wants to
60 // modify it. The drawback is that we need to declare the `customize` method
61 // before include this file.

92 int main(int argc, char** argv)
93 {
94  using namespace o2::framework;
95  using namespace boost::program_options;
96  // The 0 here is an int, therefore having the template matching in the
97  // SFINAE expression above fit better the version which invokes user code over
98  // the default one.
99 
100  std::vector<o2::framework::ConfigParamSpec> workflowOptions;
101  UserCustomizationsHelper::userDefinedCustomization(workflowOptions, 0);
102 
103  // The default policy is a catch all pub/sub setup to be consistent with the past.
104  std::vector<ChannelConfigurationPolicy> channelPolicies;
105 
106  UserCustomizationsHelper::userDefinedCustomization(channelPolicies, 0);
107  auto defaultChannelPolicies = ChannelConfigurationPolicy::createDefaultPolicies();
108  channelPolicies.insert(std::end(channelPolicies), std::begin(defaultChannelPolicies), std::end(defaultChannelPolicies));
109 
110  std::vector<CompletionPolicy> completionPolicies;
111  UserCustomizationsHelper::userDefinedCustomization(completionPolicies, 0);
112  auto defaultCompletionPolicies = CompletionPolicy::createDefaultPolicies();
113  completionPolicies.insert(std::end(completionPolicies), std::begin(defaultCompletionPolicies), std::end(defaultCompletionPolicies));
114 
115  std::unique_ptr<ParamRetriever> retriever{new BoostOptionsRetriever(workflowOptions, true, argc, argv)};
116  ConfigParamRegistry workflowOptionsRegistry(std::move(retriever));
117  ConfigContext configContext{workflowOptionsRegistry};
118  o2::framework::WorkflowSpec specs = defineDataProcessing(configContext);
119 
120  auto result = doMain(argc, argv, specs, channelPolicies, completionPolicies, workflowOptions, configContext);
121  LOG(INFO) << "Process " << getpid() << " is exiting.";
122  return result;
123 }
