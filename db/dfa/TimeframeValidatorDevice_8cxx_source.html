42 void o2::DataFlow::TimeframeValidatorDevice::Run()
43 {
44  while (CheckCurrentState(RUNNING)) {
45  FairMQParts timeframeParts;
46  if (Receive(timeframeParts, mInChannelName, 0, 100) <= 0)
47  continue;
48 
49  if (timeframeParts.Size() < 2)
50  LOG(ERROR) << "Expecting at least 2 parts\n";
51 
52  auto indexHeader = o2::header::get<header::DataHeader*>(timeframeParts.At(timeframeParts.Size() - 2)->GetData());
53  // FIXME: Provide iterator pair API for the index
54  // Index should really be something which provides an
55  // iterator pair API so that we can sort / find / lower_bound
56  // easily. Right now we simply use it a C-style array.
57  auto index = reinterpret_cast<IndexElement*>(timeframeParts.At(timeframeParts.Size() - 1)->GetData());
58 
59  // TODO: fill this with checks on time frame
60  LOG(INFO) << "This time frame has " << timeframeParts.Size() << " parts.\n";
61  auto indexEntries = indexHeader->payloadSize / sizeof(DataHeader);
62  if (indexHeader->dataDescription != DataDescription("TIMEFRAMEINDEX"))
63  LOG(ERROR) << "Could not find a valid index header\n";
64  LOG(INFO) << indexHeader->dataDescription.str << "\n";
65  LOG(INFO) << "This time frame has " << indexEntries << "entries in the index.\n";
66  if ((indexEntries * 2 + 2) != (timeframeParts.Size()))
67  LOG(ERROR) << "Mismatched index and received parts\n";
68 
69  // - Use the index to find out if we have TPC data
70  // - Get the part with the TPC data
71  // - Validate TPCCluster dummy data
72  // - Validate ITSRaw dummy data
73  int tpcIndex = -1;
74  int itsIndex = -1;
75 
76  for (int ii = 0; ii < indexEntries; ++ii) {
77  IndexElement &ie = index[ii];
78  assert(ie.second >= 0);
79  LOG(DEBUG) << ie.first.dataDescription.str << " "
80  << ie.first.dataOrigin.str << std::endl;
81  if ((ie.first.dataOrigin == header::gDataOriginTPC)
82  && (ie.first.dataDescription == header::gDataDescriptionClusters)) {
83  tpcIndex = ie.second;
84  }
85  if ((ie.first.dataOrigin == header::gDataOriginITS)
86  && (ie.first.dataDescription == header::gDataDescriptionClusters)) {
87  itsIndex = ie.second;
88  }
89  }
90 
91  if (tpcIndex < 0)
92  {
93  LOG(ERROR) << "Could not find expected TPC payload\n";
94  continue;
95  }
96  if (itsIndex < 0)
97  {
98  LOG(ERROR) << "Could not find expected ITS payload\n";
99  continue;
100  }
101  LOG(DEBUG) << "TPC Index " << tpcIndex << "\n";
102  LOG(DEBUG) << "ITS Index " << itsIndex << "\n";
103 
104  // Data header it at position - 1
105  auto tpcHeader = reinterpret_cast<DataHeader *>(timeframeParts.At(tpcIndex)->GetData());
106  if ((tpcHeader->dataDescription != header::gDataDescriptionClusters) ||
107  (tpcHeader->dataOrigin != header::gDataOriginTPC))
108  {
109  LOG(ERROR) << "Wrong data description. Expecting TPC - CLUSTERS, found "
110  << tpcHeader->dataOrigin.str << " - "
111  << tpcHeader->dataDescription.str << "\n";
112  continue;
113  }
114  auto tpcPayload = reinterpret_cast<TPCTestCluster *>(timeframeParts.At(tpcIndex + 1)->GetData());
115  if (tpcHeader->payloadSize % sizeof(TPCTestCluster))
116  LOG(ERROR) << "TPC - CLUSTERS Size Mismatch\n";
117  auto numOfClusters = tpcHeader->payloadSize / sizeof(TPCTestCluster);
118  for (size_t ci = 0 ; ci < numOfClusters; ++ci)
119  {
120  TPCTestCluster &cluster = tpcPayload[ci];
121  if (cluster.z != 1.5)
122  {
123  LOG(ERROR) << "TPC Data mismatch. Expecting z = 1.5 got " << cluster.z << "\n";
124  break;
125  }
126  if (cluster.timeStamp != ci)
127  {
128  LOG(ERROR) << "TPC Data mismatch. Expecting " << ci << " got " << cluster.timeStamp << "\n";
129  break;
130  }
131  }
132 
133  // Data header it at position - 1
134  auto itsHeader = reinterpret_cast<DataHeader *>(timeframeParts.At(itsIndex)->GetData());
135  if ((itsHeader->dataDescription != header::gDataDescriptionClusters)
136  || (itsHeader->dataOrigin != header::gDataOriginITS))
137  {
138  LOG(ERROR) << "Wrong data description. Expecting ITS - CLUSTERS, found "
139  << itsHeader->dataOrigin.str << " - " << itsHeader->dataDescription.str << "\n";
140  continue;
141  }
142  auto itsPayload = reinterpret_cast<ITSRawData*>(timeframeParts.At(itsIndex + 1)->GetData());
143  if (itsHeader->payloadSize % sizeof(ITSRawData))
144  LOG(ERROR) << "ITS - CLUSTERS Size Mismatch.\n";
145  numOfClusters = itsHeader->payloadSize / sizeof(ITSRawData);
146  for (size_t ci = 0 ; ci < numOfClusters; ++ci)
147  {
148  ITSRawData &cluster = itsPayload[ci];
149  if (cluster.timeStamp != ci)
150  {
151  LOG(ERROR) << "ITS Data mismatch. Expecting " << ci
152  << " got " << cluster.timeStamp << "\n";
153  break;
154  }
155  }
156  LOG(INFO) << "Everything is fine with received timeframe\n";
157 
158  }
159 }
