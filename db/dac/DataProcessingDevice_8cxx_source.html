183 bool DataProcessingDevice::handleData(FairMQParts& parts)
184 {
185  assert(mSpec.inputChannels.empty() == false);
186  assert(parts.Size() > 0);
187 
188  // Initial part. Let's hide all the unnecessary and have
189  // simple lambdas for each of the steps I am planning to have.
190  assert(!mSpec.inputs.empty());
191 
192  // These duplicate references are created so that each function
193  // does not need to know about the whole class state, but I can
194  // fine grain control what is exposed at each state.
195  auto& monitoringService = mServiceRegistry.get<Monitoring>();
196  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
197  ScopedExit metricFlusher([&monitoringService] {
198  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
199  monitoringService.flushBuffer(); });
200 
201  auto& device = *this;
202  auto& errorCount = mErrorCount;
203  auto& relayer = mRelayer;
204  auto& serviceRegistry = mServiceRegistry;
205 
206  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
207  // and we do a few stats. We bind parts as a lambda captured variable, rather
208  // than an input, because we do not want the outer loop actually be exposed
209  // to the implementation details of the messaging layer.
210  auto isValidInput = [&monitoringService, &parts]() -> bool {
211  // monitoringService.send({ (int)parts.Size(), "inputs/parts/total" });
212  monitoringService.send({ (int)parts.Size(), "inputs/parts/total" });
213 
214  if (parts.Size() % 2) {
215  return false;
216  }
217  for (size_t hi = 0; hi < parts.Size()/2; ++hi) {
218  auto pi = hi*2;
219  auto dh = o2::header::get<DataHeader*>(parts.At(pi)->GetData());
220  if (!dh) {
221  LOG(ERROR) << "Header is not a DataHeader?";
222  return false;
223  }
224  if (dh->payloadSize != parts.At(pi+1)->GetSize()) {
225  LOG(ERROR) << "DataHeader payloadSize mismatch";
226  return false;
227  }
228  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(pi)->GetData());
229  if (!dph) {
230  LOG(ERROR) << "Header stack does not contain DataProcessingHeader";
231  return false;
232  }
233  }
234  return true;
235  };

237  auto reportError = [&device](const char* message) {
238  device.error(message);
239  };
240 
241  auto putIncomingMessageIntoCache = [&parts,&relayer,&reportError]() {
242  // We relay execution to make sure we have a complete set of parts
243  // available.
244  for (size_t pi = 0; pi < (parts.Size()/2); ++pi) {
245  auto headerIndex = 2*pi;
246  auto payloadIndex = 2*pi+1;
247  assert(payloadIndex < parts.Size());
248  auto relayed = relayer.relay(std::move(parts.At(headerIndex)),
249  std::move(parts.At(payloadIndex)));
250  if (relayed == DataRelayer::WillNotRelay) {
251  reportError("Unable to relay part.");
252  return;
253  }
254  }
255  };
256 
257  // Second part. This is the actual outer loop we want to obtain, with
258  // implementation details which can be read. Notice how most of the state
259  // is actually hidden. For example we do not expose what "input" is. This
260  // will allow us to keep the same toplevel logic even if the actual meaning
261  // of input is changed (for example we might move away from multipart
262  // messages). Notice also that we need to act diffently depending on the
263  // actual CompletionOp we want to perform. In particular forwarding inputs
264  // also gets rid of them from the cache.
265  if (isValidInput() == false) {
266  reportError("Parts should come in couples. Dropping it.");
267  return true;
268  }
269  putIncomingMessageIntoCache();
270  return true;

275  // This is the actual hidden state for the outer loop. In case we decide we
276  // want to support multithreaded dispatching of operations, I can simply
277  // move these to some thread local store and the rest of the lambdas
278  // should work just fine.

301  // These duplicate references are created so that each function
302  // does not need to know about the whole class state, but I can
303  // fine grain control what is exposed at each state.
304  // FIXME: I should use a different id for this state.
305  auto& monitoringService = mServiceRegistry.get<Monitoring>();
306  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
307  ScopedExit metricFlusher([&monitoringService] {

311  auto reportError = [&device](const char* message) {

315  // For the moment we have a simple "immediately dispatch" policy for stuff
316  // in the cache. This could be controlled from the outside e.g. by waiting
317  // for a few sets of inputs to arrive before we actually dispatch the
318  // computation, however this can be defined at a later stage.
319  auto canDispatchSomeComputation = [&completed, &relayer]() -> bool {
320  completed = relayer.getReadyToProcess();
321  return completed.empty() == false;
322  };
323 
324  // We use this to get a list with the actual indexes in the cache which
325  // indicate a complete set of inputs. Notice how I fill the completed
326  // vector and return it, so that I can have a nice for loop iteration later
327  // on.
328  auto getReadyActions = [&relayer, &completed, &monitoringService]() -> std::vector<DataRelayer::RecordAction> {
329  int pendingInputs = (int)relayer.getParallelTimeslices() - completed.size();
330  monitoringService.send({ pendingInputs, "inputs/relayed/pending" });
331  if (completed.empty()) {
332  monitoringService.send({ 1, "inputs/relayed/incomplete" });
333  }
334  return completed;
335  };

337  // This is needed to convert from a pair of pointers to an actual DataRef
338  // and to make sure the ownership is moved from the cache in the relayer to
339  // the execution.
340  auto fillInputs = [&relayer, &inputsSchema, &currentSetOfInputs](TimesliceSlot slot) -> InputRecord {
341  currentSetOfInputs = std::move(relayer.getInputsForTimeslice(slot));
342  InputSpan span{ [&currentSetOfInputs](size_t i) -> char const* {
343  return currentSetOfInputs.at(i) ? static_cast<char const*>(currentSetOfInputs.at(i)->GetData()) : nullptr;
344  },
345  currentSetOfInputs.size() };
346  return InputRecord{ inputsSchema, std::move(span) };
347  };

349  // This is the thing which does the actual computation. No particular reason
350  // why we do the stateful processing before the stateless one.
351  // PROCESSING:{START,END} is done so that we can trigger on begin / end of processing
352  // in the GUI.
353  auto dispatchProcessing = [&processingCount, &allocator, &statefulProcess, &statelessProcess, &monitoringService,
354  &context, &rootContext, &stringContext, &rdfContext, &rawContext, &serviceRegistry, &device](TimesliceSlot slot, InputRecord& record) {

356  ProcessingContext processContext{record, serviceRegistry, allocator};
357  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_USER_CALLBACK);

359  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
360  processingCount++;
361  }
362  if (statelessProcess) {
363  ProcessingContext processContext{record, serviceRegistry, allocator};
364  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_USER_CALLBACK);
365  statelessProcess(processContext);

389  // I need a preparation step which gets the current timeslice id and
390  // propagates it to the various contextes (i.e. the actual entities which
391  // create messages) because the messages need to have the timeslice id into
392  // it.
393  auto prepareAllocatorForCurrentTimeSlice = [&timingInfo, &rootContext, &stringContext, &rdfContext, &rawContext, &context, &relayer, &timesliceIndex](TimesliceSlot i) {
394  auto timeslice = timesliceIndex.getTimesliceForSlot(i);
395  timingInfo.timeslice = timeslice.value;
396  rootContext.clear();
397  context.clear();
398  stringContext.clear();
399  rdfContext.clear();

401  };

403  // This is how we do the forwarding, i.e. we push
404  // the inputs which are shared between this device and others
405  // to the next one in the daisy chain.
406  // FIXME: do it in a smarter way than O(N^2)
407  auto forwardInputs = [&reportError, &forwards, &device, &currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
408  assert(record.size()*2 == currentSetOfInputs.size());
409  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
410  DataRef input = record.getByPos(ii);
411 
412  // If is now possible that the record is not complete when
413  // we forward it, because of a custom completion policy.
414  // this means that we need to skip the empty entries in the
415  // record for being forwarded.
416  if (input.header == nullptr || input.payload == nullptr) {
417  continue;
418  }
419  auto dh = o2::header::get<DataHeader*>(input.header);
420  if (!dh) {
421  reportError("Header is not a DataHeader?");
422  continue;
423  }
424  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
425  if (!dph) {
426  reportError("Header stack does not contain DataProcessingHeader");
427  continue;
428  }
429 
430  auto &header = currentSetOfInputs[ii*2];
431  auto &payload = currentSetOfInputs[ii*2+1];
432 
433  for (auto forward : forwards) {
434  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification)
435  && (dph->startTime % forward.maxTimeslices) == forward.timeslice) {
436 
437  if (header.get() == nullptr) {
438  LOG(ERROR) << "Missing header!";
439  continue;
440  }
441  auto fdph = o2::header::get<DataProcessingHeader*>(header.get()->GetData());
442  if (fdph == nullptr) {
443  LOG(ERROR) << "Forwarded data does not have a DataProcessingHeader";
444  continue;
445  }
446  auto fdh = o2::header::get<DataHeader*>(header.get()->GetData());
447  if (fdh == nullptr) {
448  LOG(ERROR) << "Forwarded data does not have a DataHeader";
449  continue;
450  }
451  FairMQParts forwardedParts;
452  forwardedParts.AddPart(std::move(header));
453  forwardedParts.AddPart(std::move(payload));
454  assert(forwardedParts.Size() == 2);
455  assert(o2::header::get<DataProcessingHeader*>(forwardedParts.At(0)->GetData()));
456  // FIXME: this should use a correct subchannel
457  device.Send(forwardedParts, forward.channel, 0);
458  }
459  }
460  }

463  // We use this to keep track of the latency of the first message we get for a given input record
464  // and of the last one.
465  struct InputLatency {
466  uint64_t minLatency;
467  uint64_t maxLatency;
468  };
469 
470  auto calculateInputRecordLatency = [](InputRecord const& record, auto now) -> InputLatency {
471  InputLatency result{ static_cast<uint64_t>(-1), 0 };
472 
473  auto currentTime = (uint64_t)std::chrono::duration<double, std::milli>(now.time_since_epoch()).count();
474  for (auto& item : record) {
475  auto* header = o2::header::get<DataProcessingHeader*>(item.header);
476  if (header == nullptr) {
477  continue;
478  }
479  auto partLatency = currentTime - header->creation;
480  result.minLatency = std::min(result.minLatency, partLatency);
481  result.maxLatency = std::max(result.maxLatency, partLatency);
482  }
483  return result;
484  };
485 
486  auto calculateTotalInputRecordSize = [](InputRecord const& record) -> int {
487  size_t totalInputSize = 0;
488  for (auto& item : record) {
489  auto* header = o2::header::get<DataHeader*>(item.header);
490  if (header == nullptr) {
491  continue;
492  }
493  totalInputSize += header->payloadSize;
494  }
495  return totalInputSize;
496  };
497 
498  if (canDispatchSomeComputation() == false) {
499  return false;
500  }
501 
502  for (auto action: getReadyActions()) {
503  if (action.op == CompletionPolicy::CompletionOp::Wait) {
504  continue;
505  }
506 
507  prepareAllocatorForCurrentTimeSlice(TimesliceSlot{ action.slot });
508  InputRecord record = fillInputs(action.slot);
509  if (action.op == CompletionPolicy::CompletionOp::Discard) {
510  if (forwards.empty() == false) {
511  forwardInputs(action.slot, record);
512  continue;
513  }
514  }
515  auto tStart = std::chrono::high_resolution_clock::now();
516  try {
517  for (size_t ai = 0; ai != record.size(); ai++) {
518  auto cacheId = action.slot.index * record.size() + ai;
519  auto state = record.isValid(ai) ? 2 : 0;
520  monitoringService.send({ state, "data_relayer/" + std::to_string(cacheId) });
521  }
522  dispatchProcessing(action.slot, record);
523  for (size_t ai = 0; ai != record.size(); ai++) {
524  auto cacheId = action.slot.index * record.size() + ai;
525  auto state = record.isValid(ai) ? 3 : 0;
526  monitoringService.send({ state, "data_relayer/" + std::to_string(cacheId) });
527  }
528  } catch(std::exception &e) {
529  errorHandling(e, record);
530  }
531  auto tEnd = std::chrono::high_resolution_clock::now();
532  double elapsedTimeMs = std::chrono::duration<double, std::milli>(tEnd - tStart).count();
533  auto totalProcessedSize = calculateTotalInputRecordSize(record);
534  auto latency = calculateInputRecordLatency(record, tStart);
535 
537  monitoringService.send({ totalProcessedSize, "dpl/processed_input_size_bytes" });
539  monitoringService.send({ elapsedTimeMs, "dpl/elapsed_time_ms" });
541  monitoringService.send({ (int)((totalProcessedSize / elapsedTimeMs) / 1000), "dpl/processing_rate_mb_s" });
544  monitoringService.send({ (int)latency.minLatency, "dpl/min_input_latency_ms" });
547  monitoringService.send({ (int)latency.maxLatency, "dpl/max_input_latency_ms" });
550  if (latency.maxLatency == 0) {
551  // avoid division by zero by assuming at least one ms of latency.
552  latency.maxLatency = 1;
553  }
554  monitoringService.send({ (int)((totalProcessedSize / latency.maxLatency) / 1000), "dpl/input_rate_mb_s" });
555 
556  // We forward inputs only when we consume them. If we simply Process them,
557  // we keep them for next message arriving.
558  if (action.op == CompletionPolicy::CompletionOp::Consume) {
559  if (forwards.empty() == false) {
560  forwardInputs(action.slot, record);
561  }
562  }
563  }
564 
565  return true;
