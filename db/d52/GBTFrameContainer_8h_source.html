251 inline void GBTFrameContainer::addGBTFrame(unsigned word3, unsigned word2, unsigned word1, unsigned word0)
252 {
253  if (!mEnableStoreGBTFrames && (mGBTFrames.size() > 1)) {
254  mGBTFrames[0] = mGBTFrames[1];
255  mGBTFrames[1].setData(word3, word2, word1, word0);
256  } else {
257  mGBTFrames.emplace_back(word3, word2, word1, word0);
258  }
259  processFrame(mGBTFrames.end()-1);
260 };
