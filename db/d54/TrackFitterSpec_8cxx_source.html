57  void run(framework::ProcessingContext& pc)
58  {
61 
62  // get the input buffer
63  auto msgIn = pc.inputs().get<gsl::span<char>>("tracks");
64  auto bufferPtr = msgIn.data();
65  int sizeLeft = msgIn.size();
66 
67  // create the output message
68  auto msgOut = pc.outputs().make<char>(Output{ "MCH", "REFITTRACKS", 0, Lifetime::Timeframe }, sizeLeft);
69  auto bufferPtrOut = msgOut.data();
70 
71  // copy header info
72  try {
73  copyHeader(bufferPtr, sizeLeft, bufferPtrOut);
74  } catch (exception const& e) {
75  throw;
76  }
77 
78  // get the number of tracks and copy it to the output message
79  if (sizeLeft < SSizeOfInt) {
80  throw out_of_range("missing number of tracks");
81  }
82  const int& nTracks = *reinterpret_cast<const int*>(bufferPtr);
83  bufferPtr += SSizeOfInt;
84  memcpy(bufferPtrOut, &nTracks, SSizeOfInt);
85  bufferPtrOut += SSizeOfInt;
86  sizeLeft -= SSizeOfInt;
87 
88  std::list<Cluster> clusters{};
89  for (int iTrack = 0; iTrack < nTracks; ++iTrack) {
90 
91  // get the input track
92  Track track{};
93  try {
94  readTrack(bufferPtr, sizeLeft, track, clusters);
95  } catch (exception const& e) {
96  throw;
97  }
98 
99  // refit the track
100  try {
101  mTrackFitter.fit(track);
102  } catch (exception const& e) {
103  throw runtime_error(std::string("Track fit failed: ") + e.what());
104  }
105 
106  // write the refitted track to the ouput message
107  writeTrack(track, bufferPtrOut);
108  }
109 
110  if (sizeLeft != 0) {
111  throw length_error("incorrect payload");
112  }
113  }

144  void readTrack(const char*& bufferPtr, int& sizeLeft, Track& track, std::list<Cluster>& clusters) const
145  {
149 
150  // skip the track parameters
151  if (sizeLeft < SSizeOfTrackParamStruct) {
152  throw out_of_range("missing track parameters");
153  }
154  bufferPtr += SSizeOfTrackParamStruct;
155  sizeLeft -= SSizeOfTrackParamStruct;
156 
157  // read number of clusters
158  if (sizeLeft < SSizeOfInt) {
159  throw out_of_range("missing number of clusters");
160  }
161  const int& nClusters = *reinterpret_cast<const int*>(bufferPtr);
162  if (nClusters > 20) {
163  throw length_error("too many (>20) clusters attached to the track");
164  }
165  bufferPtr += SSizeOfInt;
166  sizeLeft -= SSizeOfInt;
167 
168  for (int iCl = 0; iCl < nClusters; ++iCl) {
169 
170  // read cluster info
171  if (sizeLeft < SSizeOfClusterStruct) {
172  throw out_of_range("missing cluster");
173  }
174  clusters.emplace_back(*reinterpret_cast<const ClusterStruct*>(bufferPtr));
175  track.createParamAtCluster(clusters.back());
176  bufferPtr += SSizeOfClusterStruct;
177  sizeLeft -= SSizeOfClusterStruct;
178  }
179  }

182  void writeTrack(Track& track, char*& bufferPtrOut) const
183  {
185 
186  // fill track parameters
187  TrackParamStruct paramStruct = track.first().getTrackParamStruct();
188  memcpy(bufferPtrOut, &paramStruct, SSizeOfTrackParamStruct);
189  bufferPtrOut += SSizeOfTrackParamStruct;
190 
191  // fill number of clusters
192  int nClusters = track.getNClusters();
193  memcpy(bufferPtrOut, &nClusters, SSizeOfInt);
194  bufferPtrOut += SSizeOfInt;
195 
196  for (const auto& param : track) {
197 
198  // fill cluster info
199  ClusterStruct clusterStruct = param.getClusterPtr()->getClusterStruct();
200  memcpy(bufferPtrOut, &clusterStruct, SSizeOfClusterStruct);
201  bufferPtrOut += SSizeOfClusterStruct;
202  }
203  }
