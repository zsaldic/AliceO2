158 int GEMAmplification::getGEMMultiplication(int nElectrons, int GEM)
159 {
162  if (nElectrons < 1) {
164  return 0;
165  } else if (nElectrons > 500) {
169  return ((mRandomGaus.getNextValue() * std::sqrt(static_cast<float>(nElectrons)) *
170  mGasParam->getSigmaOverMu()) +
171  nElectrons) *
172  mGEMParam->getAbsoluteGain(GEM);
173  } else {
176  int electronsOut = 0;
177  for (int i = 0; i < nElectrons; ++i) {
178  electronsOut += mGain[GEM - 1].getNextValue();
179  }

184 int GEMAmplification::getElectronLosses(int nElectrons, float probability)
185 {
187  float electronsFloat = static_cast<float>(nElectrons);
188  if (nElectrons < 1 || probability < 0.00001) {
190  return 0;
191  } else if (probability > 0.99999) {
193  return nElectrons;
194  } else if (electronsFloat * probability >= 5.f && electronsFloat * (1.f - probability) >= 5.f) {
197  return (mRandomGaus.getNextValue() * std::sqrt(electronsFloat * probability * (1 - probability))) +
198  electronsFloat * probability + 0.5;
199  } else {
201  int nElectronsOut = 0;
202  for (int i = 0; i < nElectrons; ++i) {
203  if (mRandomFlat.getNextValue() < probability) {
204  ++nElectronsOut;
205  }
206  }
207  return nElectronsOut;
208  }
