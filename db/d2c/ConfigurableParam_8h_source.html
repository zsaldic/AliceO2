28 // Base class for a configurable parameter.
29 //
30 // A configurable parameter (ConfigurableParameter) is a simple class, defining
31 // a few (pod) properties/members which are registered
32 // in a global (boost) property tree / structure.
33 //
34 // The features that we provide here are:
35 // *) Automatic translation from C++ data description to INI/JSON/XML
36 // format via ROOT introspection and boost property trees and
37 // the possibility to readably save the configuration
38 // *) Serialization/Deserialization into ROOT binary blobs (for the purpose
39 // of writing/retrieving parameters from CCDB and to pass parameters along the processing chain)
40 // *) Automatic integration of sub-classes into a common configuration
41 // *) Be able to query properties from high level interfaces (just knowing
42 // *) Be able to set properties from high-level interfaces (and modifying the underlying
43 // C++ object)
44 // *) Automatic ability to modify parameters from the command-line
45 // *) Keeping track of the provenance of individual parameter values; The user is able
46 // to see whether is parameter is defaulted-code-value/coming-from-CCDB/coming-from-comandline
47 //
48 // Note that concrete parameter sub-classes **must** be implemented
49 // by inheriting from ConfigurableParamHelper and not from this class.
50 //
51 // ---------------------
52 // Example: To define a parameter class TPCGasParameters, one does the following:
53 //
54 // class TPCGasParamer : public ConfigurableParamHelper<TPCGasParameter>
55 // {
56 // public:
57 // double getGasDensity() const { return mGasDensity; }
58 // private: // define properties AND default values
59 // double mGasDensity = 1.23;
60 // int mGasMaterialID = 1;
61 //
62 // O2ParamDef(TPCGasParameter, TPCGas); // a macro implementing some magic
63 // }
64 //
65 //
66 // We can now query the parameters in various ways
67 // - All parameter classes are singletons and we can say: TPCGasParameter::Instance().getGasDensity();
68 // - We can query by key (using classname + parameter name) from the global registry:
69 // - ConfigurableParameter::getValueAs<double>("TPCGas", "mGasDensity");
70 //
71 // We can modify the parameters via the global registry together with an automatic syncing
72 // of the underlying C++ object:
73 // - ConfigurableParameter::setValue("TPCGas.mGasDensity", "0.5");
74 //
75 // - TPCGasParameter::Instance().getGasParameter() will now return 0.5;
76 //
77 // This feature allows to easily modify parameters at run-time via a textual representation
78 // (for example by giving strings on the command line)
79 //
80 // The collection of all parameter keys and values can be stored to a human/machine readable
81 // file
82 // - ConfigurableParameter::writeJSON("thisconfiguration.json")

122  template <typename T>
123  static void setValue(std::string const& mainkey, std::string const& subkey, T x)
124  {
125  auto key = mainkey + "." + subkey;
126  if (sPtree->get_optional<std::string>(key).is_initialized()) {
127  sPtree->put(key, x);
128  auto changed = updateThroughStorageMap(mainkey, subkey, typeid(T), (void*)&x);
129  if (changed) {
130  sValueProvenanceMap->find(key)->second = kRT; // set to runtime
131  }
132  }
133  }

135  // specialized for std::string
136  // which means that the type will be converted internally
137  static void setValue(std::string const& key, std::string const& valuestring)
138  {
139  if (sPtree->get_optional<std::string>(key).is_initialized()) {
140  sPtree->put(key, valuestring);
141  auto changed = updateThroughStorageMapWithConversion(key, valuestring);
142  if (changed) {
143  sValueProvenanceMap->find(key)->second = kRT; // set to runtime
144  }
145  }
